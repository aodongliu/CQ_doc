31 template <>
32 void NRCoreH<dcomplex, dcomplex>::addMagPert(EMPerturbation &pert,
33 std::shared_ptr<PauliSpinorSquareMatrices<dcomplex>> coreH) {
34
35 //Compute the GIAO non-relativistic core Hamiltonian in the CGTO basis
36 //H(S) = 2(T + V) + B * L + sigma * B + 1/4 *(B\timesr)^2
37
38 dcomplex onei = dcomplex(0,1);
39 auto magAmp = pert.getDipoleAmp(Magnetic);
40
41
42 // this part add the angular momentum term
43 for ( auto index = 0 ; index < 3 ; index++ ) {
44 *coreH += -magAmp[index] * onei * (*aoints_.magnetic)[index].matrix();
45 } // for ( auto inde = 0 ; inde < 3 ; inde++ )
46
47 // this part add the length gauge electric quadrupole term
48 const std::array<std::string,3> diagindex =
49 { "XX","YY","ZZ" };
50
51 double diagcoeff[3];
52 diagcoeff[0] = 1.0/8.0*(magAmp[1]*magAmp[1]+magAmp[2]*magAmp[2]);
53 diagcoeff[1] = 1.0/8.0*(magAmp[0]*magAmp[0]+magAmp[2]*magAmp[2]);
54 diagcoeff[2] = 1.0/8.0*(magAmp[0]*magAmp[0]+magAmp[1]*magAmp[1]);
55
56 // add diagonal part
57 for ( size_t index = 0 ; index < 3 ; index++ ) {
58 *coreH += 2.0*diagcoeff[index] * (*aoints_.lenElectric)[diagindex[index]].matrix();
59 }
60
61 const std::array<std::string,3> offindex =
62 { "XY","XZ","YZ" };
63
64 double offcoeff[3];
65 offcoeff[0] = -1.0/4.0*magAmp[0]*magAmp[1];
66 offcoeff[1] = -1.0/4.0*magAmp[0]*magAmp[2];
67 offcoeff[2] = -1.0/4.0*magAmp[1]*magAmp[2];
68
69 // add off diagonal part
70 for ( size_t index = 0 ; index < 3 ; index++ ) {
71 *coreH += 2.0*offcoeff[index] * (*aoints_.lenElectric)[offindex[index]].matrix();
72 }
73
74 // finally spin Zeeman term
75 // z component
76 if(coreH->hasZ())
77 coreH->Z() = magAmp[2] * aoints_.overlap->matrix();
78
79 if(coreH->hasXY()) {
80 // y component
81 coreH->Y() = magAmp[1] * aoints_.overlap->matrix();
82
83 // x coponent
84 coreH->X() = magAmp[0] * aoints_.overlap->matrix();
85 }
86
87
88 }

126 void NRCoreH<MatsT,IntsT>::computeNRCH(EMPerturbation& emPert,
127 std::shared_ptr<PauliSpinorSquareMatrices<MatsT>> coreH) {
128
129 *coreH = 2. * (this->aoints_.kinetic->matrix() + this->aoints_.potential->matrix());
130
131 if( this->hamiltonianOptions_.basisType == COMPLEX_GIAO and pert_has_type(emPert,Magnetic) )
132 addMagPert(emPert,coreH);
133
134

139 coreH->output(std::cout, "Core Ham", true);

144 template <typename MatsT, typename IntsT>
145 std::vector<double> NRCoreH<MatsT,IntsT>::getGrad(EMPerturbation& pert,
146 SingleSlater<MatsT,IntsT>& ss) {
147
148 if (not this->aoints_.gradKinetic)
149 CErr("Kinetic energy gradient integrals missing in NRCoreH::getGrad!");
150 if (not this->aoints_.gradPotential)
151 CErr("Potential energy gradient integrals missing in NRCoreH::getGrad!");
152
153 GradInts<OnePInts,IntsT>& gradK = *this->aoints_.gradKinetic;
154 GradInts<OnePInts,IntsT>& gradV = *this->aoints_.gradPotential;
155
156 //gradK.output(std::cout, "gradKinetic", true);
157 //gradV.output(std::cout, "gradPotential", true);
158
159 size_t NB = ss.basisSet().nBasis;
160 size_t nGrad = 3*ss.molecule().nAtoms;
161
162 std::vector<double> gradient;
163
164 // Allocate scratch (NRCH is SCALAR only)
165 SquareMatrix<MatsT> vdv(ss.memManager, NB);
166 SquareMatrix<MatsT> dvv(ss.memManager, NB);
167 SquareMatrix<MatsT> coreHGrad(ss.memManager,NB);
168
169 // Loop over gradient components
170 for ( auto iGrad = 0; iGrad < nGrad; iGrad++ ) {
171
172 // Assemble core gradient
173 coreHGrad = 2. * (gradK[iGrad]->matrix() + gradV[iGrad]->matrix());
174
175 // Contract
176 double element = ss.template computeOBProperty<DENSITY_TYPE::SCALAR>(
177 coreHGrad.pointer());
178 gradient.emplace_back(0.5*element);
179
180 }
181
182 return gradient;
183
184 }; // std::vector<double> NRCoreH::getGrad
