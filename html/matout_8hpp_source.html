50template <typename T>
51void prettyPrintSmartBase(std::ostream& out, const T* A, const size_t M,
52 const size_t N, const size_t LDA, const size_t colStride = 1,
53 const size_t list = 5, const size_t printWidth = 16) {
54
55 int end,endLT;
56 out << bannerTop;
57
58 out << std::scientific << std::left << std::setprecision(8);
59 for(size_t i = 0; i < N; i += list) {
60 out << std::endl;
61 end = list;
62 out << std::setw(5) << " ";
63 if((i + list) >= N) end = N - i;
64 for(size_t k = i; k < i+end; k++) out << std::setw(printWidth) << k+1;
65 out << std::endl;
66 for(size_t j = 0; j < M; j++) {
67 out << std::setw(5) << std::left << j+1;
68 out << std::right;
69 for(size_t n = i; n < i+end; n++) {
70 T VAL = A[j*colStride + n*LDA];
71 if(std::abs(VAL) > PRINT_SMALL) out << std::setw(printWidth) << VAL;
72 else if(std::isnan(VAL)) out << std::setw(printWidth) << "NAN";
73 else if(std::isinf(VAL)) out << std::setw(printWidth) << "INF";
74 else out << std::setw(printWidth) << 0.;
75 }
76 out << std::endl;
77 };
78 };
79 out << bannerEnd << std::endl;
80
81}; // prettyPrintSmartBase

176void mathematicaPrint(std::ostream& out, std::string str, const T* A,
177 const size_t M, const size_t N, const size_t LDA,
178 const size_t colStride = 1) {
179
180 out << str << ": \n";
181
182 out << "{ \n";
183 for(auto i = 0; i < M; i++){
184 if( N != 1) out << "{ ";
185 for(auto j = 0; j < N; j++){
186 out << std::setprecision(12) << A[colStride*i + j*LDA];
187 if(j != N-1) out << ", ";
188 }
189 if(N != 1) out << "}";
190 if( i != M-1 ) out << ",";
191 }
192 out << "\n}\n";
