56 /*
57 * Build full CASCI Hamiltonian as H(K, L)
58 */
59 template <typename MatsT, typename IntsT>
60 void CASCI<MatsT,IntsT>::buildFullH(MCWaveFunction<MatsT, IntsT> & mcwfn, MatsT * fullH) {
61
62 CASCI_LOOP_INIT(); // check top for variable definitions
63
64 auto & hCoreP = *(mcwfn.moints.template getIntegral<OnePInts,MatsT>("hCoreP_Correlated_Space"));
65 auto & moERI = *(mcwfn.moints.template getIntegral<InCore4indexTPI,MatsT>("ERI_Correlated_Space"));
66
67 // Allocate SCR
68 size_t nSCR = std::max(nStr_a, nStr_b);
69 size_t nThreads = GetLAThreads();
70 MatsT * SCR = mcwfn.memManager.template malloc<MatsT>(nSCR * nThreads);
71
72 // empty CI Hamiltonian
73 std::fill_n(fullH, nStr_a*nStr_a, MatsT(0.));
74 // Alpha Part for 1C or the whole build for 2C and 4C
75 // NOTE: ex_list_ are row-majored c++ objects
76
77 int i, j, k, l, La, Lb, Ka, Kb, Ja, Jb;
78 double signij, signkl;
79 double small_number = std::numeric_limits<double>::epsilon();
80
81 size_t nStr_a_nThread = nStr_a * nThreads;
82
83 MatsT *CIHCol = nullptr, *SCR_ith = nullptr;
84
85 // alpha part
86
87 // fullH as (Ka, La), CIHCol as a column
88#pragma omp parallel default(shared) private(CIHCol, SCR_ith, La, k, l, Ka, signkl, i, j, Ja, signij)
89 {
90 auto iThread = GetThreadID();
91 CIHCol = fullH + nStr_a * iThread;
92 SCR_ith = SCR + nSCR * iThread;
93 for (La = iThread; La < nStr_a; La+=nThreads, CIHCol+=nStr_a_nThread) {
94
95 std::fill_n(SCR_ith, nStr_a, MatsT(0.));
96 const int * exList_La = exList_a->pointerAtDet(La);
97 for (auto Ekl = 0ul; Ekl < nNZa; Ekl++, exList_La+=4) {
98
99 UNPACK_EXCITATIONLIST_4(exList_La, k, l, Ka, signkl);
100 SCR_ith[Ka] += signkl * hCoreP(k, l);
101
102 const int * exList_Ka = exList_a->pointerAtDet(Ka);
103 for (auto Eij = 0ul; Eij < nNZa; Eij++, exList_Ka+=4) {
104
105 UNPACK_EXCITATIONLIST_4(exList_Ka, i, j, Ja, signij);
106 SCR_ith[Ja] += 0.5 * signij * signkl * moERI(i, j, k, l);
107 }
108 }
109
110 // passive screening and updating CHCol
111 for (Ka = 0 ; Ka < nStr_a; Ka++) {
112 if (std::abs(SCR_ith[Ka]) > small_number) CIHCol[Ka] = SCR_ith[Ka];
113 }
114
115 } // La
116 }
117
118 if (nC != 1) {
119
120#ifdef _DEBUG_CIBUILDER_CASCI_IMPL
121 prettyPrintSmart(std::cout,"HH full CASCI Hamiltonian",fullH, NDet, NDet, NDet);
122#endif
123 mcwfn.memManager.free(SCR);
124 return;
125 }
126
127 // 1C Continued: Expand Alpha Part
128 // (Ka, La) -> (Ka, Kb, La, Kb)
129 size_t nStr_a2 = nStr_a * nStr_a;
130 size_t nStr_b2 = nStr_b * nStr_b;
131 MatsT * tmpH = mcwfn.memManager.template malloc<MatsT>(nSCR*nSCR);
132 std::copy_n(fullH, nStr_a2, tmpH);
133 std::fill_n(fullH, NDet*NDet, MatsT(0.));
134
135 // transpose tmpH as (La, Ka)
136 IMatCopy('T', nStr_a, nStr_a, MatsT(1.), tmpH, nStr_a, nStr_a);
137
138 // update fullH as (La, Ka, Kb, Kb)
139 // then transpose back to (Ka, Kb, La, Kb) in each last Kb
140 size_t lastDimOff = nStr_a2*nStr_b;
141 CIHCol = fullH;
142 for (Kb = 0; Kb < nStr_b; Kb++, CIHCol+=lastDimOff) {
143 std::copy_n(tmpH, nStr_a2, CIHCol+Kb*nStr_a2);
144 IMatCopy('T', nStr_a, NDet, MatsT(1.), CIHCol, nStr_a, NDet);
145 }
146
147
148 // 1C Continued: build Beta part as (Kb, Lb)
149 std::fill_n(tmpH, nStr_b2, MatsT(0.));
150 size_t nStr_b_nThread = nStr_b * nThreads;
151#pragma omp parallel default(shared) private(CIHCol, SCR_ith, Lb, k, l, Kb, signkl, i, j, Jb, signij)
152 {
153 auto iThread = GetThreadID();
154 CIHCol = tmpH + nStr_b * iThread;
155 SCR_ith = SCR + nSCR * iThread;
156 for (Lb = iThread; Lb < nStr_b; Lb+=nThreads, CIHCol+=nStr_b_nThread) {
157
158 std::fill_n(SCR_ith, nStr_b, MatsT(0.));
159 const int * exList_Lb = exList_b->pointerAtDet(Lb);
160 for (auto Ekl = 0ul; Ekl < nNZb; Ekl++, exList_Lb+=4) {
161
162 UNPACK_EXCITATIONLIST_4(exList_Lb, k, l, Kb, signkl);
163 SCR_ith[Kb] += signkl * hCoreP(k, l);
164
165 const int * exList_Kb = exList_b->pointerAtDet(Kb);
166 for (auto Eij = 0ul; Eij < nNZb; Eij++, exList_Kb+=4) {
167
168 UNPACK_EXCITATIONLIST_4(exList_Kb, i, j, Jb, signij);
169 SCR_ith[Jb] += 0.5 * signij * signkl * moERI(i, j, k, l);
170
171 }
172 }
173
174 // passive screening and updating CHCol
175 for (Kb = 0 ; Kb < nStr_b; Kb++) {
176 if (std::abs(SCR_ith[Kb]) > small_number) CIHCol[Kb] = SCR_ith[Kb];
177 }
178 } // Lb
179 }
180 // 1C Continued: Expand Beta Part
181 // (Kb, Lb) -> (Ka, Kb, Ka, Lb)
182
183 // TODO: try to used non-transposed algorithm
184 // transpose fullH: (Ka, Kb, La, Lb) -> (Kb, La, Lb, Ka)
185 IMatCopy('T', nStr_a, nStr_b*NDet, MatsT(1.), fullH, nStr_a, nStr_b*NDet);
186
187 // transpose tmpH as (Lb, Kb)
188 IMatCopy('T', nStr_b, nStr_b, MatsT(1.), tmpH, nStr_b, nStr_b);
189
190 // update fullH as (Lb, Kb, Ka, Ka)
191 // then transpose back to (Kb, Ka, Lb, Ka) in each last Ka
192 lastDimOff = nStr_b2*nStr_a;
193 CIHCol = fullH;
194 for (Ka = 0; Ka < nStr_a; Ka++, CIHCol+=lastDimOff) {
195 IMatCopy('T', NDet, nStr_b, MatsT(1.), CIHCol, NDet, nStr_b);
196 MatAdd('N', 'N', nStr_b, nStr_b, MatsT(1.), CIHCol+Ka*nStr_b2, nStr_b,
197 MatsT(1.), tmpH, nStr_b, CIHCol+Ka*nStr_b2, nStr_b);
198 IMatCopy('T', nStr_b, NDet, MatsT(1.), CIHCol, nStr_b, NDet);
199 }
200
201 // transpose fullH: ((Kb, La, Lb, Ka) -> (Ka, Kb, La, Lb)
202 IMatCopy('T', nStr_b*NDet, nStr_a, MatsT(1.), fullH, nStr_b*NDet, nStr_a);
203
204 mcwfn.memManager.free(SCR, tmpH);
205
206 // 1C Continued: Alpha-Beta and Beta-Aphla Part
207#pragma omp parallel for schedule(static) default(shared) \
208 private(CIHCol, Lb, k, l, Kb, signkl, La, i, j, Ka, signij)
209 for(Lb = 0; Lb < nStr_b; Lb++) {
210
211 CIHCol = fullH + NDet*(Lb*nStr_a); // La = 0
212 const int * exList_Lb_head = exList_b->pointerAtDet(Lb);
213
214 for(La = 0; La < nStr_a; La++, CIHCol+=NDet) {
215
216 const int * exList_La_head = exList_a->pointerAtDet(La);
217
218 const int * exList_Lb = exList_Lb_head;
219 for(auto Ekl = 0ul; Ekl < nNZb; Ekl++, exList_Lb+=4) {
220
221 UNPACK_EXCITATIONLIST_4(exList_Lb, k, l, Kb, signkl);
222
223 const int * exList_La = exList_La_head;
224 for (auto Eij = 0ul; Eij < nNZa; Eij++, exList_La+=4) {
225
226 UNPACK_EXCITATIONLIST_4(exList_La, i, j, Ka, signij);
227 CIHCol[Ka + Kb*nStr_a] += signij * signkl * moERI(i, j, k, l);
228 }
229 }
230 }
231 }
232
233#ifdef _DEBUG_CIENGINE_CASCI_IMPL
234 prettyPrintSmart(std::cout,"HH full CASCI Hamiltonian", fullH, NDet, NDet, NDet);
235#endif
236
237 }; // CASCI::buildFullH

239 /*
240 * Build Diagonal CASCI Hamiltonian
241 */
242 template <typename MatsT, typename IntsT>
243 void CASCI<MatsT,IntsT>::buildDiagH(MCWaveFunction<MatsT, IntsT> & mcwfn, MatsT * diagH) {
244
245 CASCI_LOOP_INIT(); // check top for variable definitions
246
247 auto & hCoreP = *(mcwfn.moints.template getIntegral<OnePInts, MatsT>("hCoreP_Correlated_Space"));
248 auto & moERI = *(mcwfn.moints.template getIntegral<InCore4indexTPI, MatsT>("ERI_Correlated_Space"));
249
250 // Allocate SCR
251 MatsT SCR;
252
253 // empty CI Diagonal Hamiltonian
254 std::fill_n(diagH, nStr_a, MatsT(0.));
255 // Alpha Part for 1C or the whole build for 2C and 4C
256
257 int i, j, k, l, La, Lb, Ka, Kb, Ja, Jb;
258 double signij, signkl;
259 double small_number = std::numeric_limits<double>::epsilon();
260
261#pragma omp parallel for schedule(static) default(shared) private(SCR, La, k, l, Ka, signkl, i, j, Ja, signij)
262 for (La = 0; La < nStr_a; La++) {
263
264 const int * exList_La = exList_a->pointerAtDet(La);
265 SCR = MatsT(0.);
266 for (auto Ekl = 0ul; Ekl < nNZa; Ekl++, exList_La+=4) {
267
268 UNPACK_EXCITATIONLIST_4(exList_La, k, l, Ka, signkl);
269 if(Ka == La) SCR += signkl * hCoreP(k, l);
270
271 const int * exList_Ka = exList_a->pointerAtDet(Ka);
272 for (auto Eij = 0ul; Eij < nNZa; Eij++, exList_Ka+=4) {
273
274 UNPACK_EXCITATIONLIST_4(exList_Ka, i, j, Ja, signij);
275 if(Ja == La) SCR += 0.5 * signij * signkl * moERI(i, j, k, l);
276 }
277 }
278
279 // update diagH
280 diagH[La] = SCR;
281
282 } // La
283
284 if (nC != 1) {
285#ifdef _DEBUG_CIENGINE_CASCI_IMPL
286 prettyPrintSmart(std::cout,"HH full CASCI Hamiltonian",diagH, NDet, 1, NDet);
287#endif
288 return;
289 }
290 // 1C Continued: Expand Alpha Part
291 // (Ka) -> (Ka, Kb)
292 MatsT * dH = diagH + nStr_a;
293 for (Kb = 1; Kb < nStr_b; Kb++, dH+=nStr_a) std::copy_n(diagH, nStr_a, dH);
294
295 MatsT * tmpdH = mcwfn.memManager.template malloc<MatsT>(nStr_b);
296
297 // 1C Continued: build Beta part
298 std::fill_n(tmpdH, nStr_b, MatsT(0.));
299#pragma omp parallel for schedule(static) default(shared) private(SCR, Lb, k, l, Kb, signkl, i, j, Jb, signij)
300 for (Lb = 0; Lb < nStr_b; Lb++) {
301
302 SCR = MatsT(0.);
303 const int * exList_Lb = exList_b->pointerAtDet(Lb);
304 for (auto Ekl = 0ul; Ekl < nNZb; Ekl++, exList_Lb+=4) {
305
306 UNPACK_EXCITATIONLIST_4(exList_Lb, k, l, Kb, signkl);
307 if(Kb == Lb) SCR += signkl * hCoreP(k, l);
308
309 const int * exList_Kb = exList_b->pointerAtDet(Kb);
310 for (auto Eij = 0ul; Eij < nNZb; Eij++, exList_Kb+=4) {
311
312 UNPACK_EXCITATIONLIST_4(exList_Kb, i, j, Jb, signij);
313 if(Jb == Lb) SCR += 0.5 * signij * signkl * moERI(i, j, k, l);
314 }
315 }
316
317 // update diagH
318 tmpdH[Lb] = SCR;
319
320 } // Lb
321
322 // 1C Continued: Expand Beta Part
323 // transpose diagH: (Ka, Kb) -> (Kb, Ka)
324 IMatCopy('T', nStr_a, nStr_b, MatsT(1.), diagH, nStr_a, nStr_b);
325
326 // Expand
327 dH = diagH;
328 for (Ka = 0; Ka < nStr_a; Ka++)
329 for (Kb = 0; Kb < nStr_b; Kb++, dH++)
330 *dH += tmpdH[Kb];
331
332 // transpose diagH: (Kb, Ka) -> (Ka, Kb)
333 IMatCopy('T', nStr_b, nStr_a, MatsT(1.), diagH, nStr_b, nStr_a);
334
335 mcwfn.memManager.free(tmpdH);
336
337 // 1C Continued: Alpha-Beta and Beta-Aphla Part
338 size_t nActEa = mcwfn.MOPartition.nCorrEA;
339 size_t nActEb = mcwfn.MOPartition.nCorrEB;
340#pragma omp parallel for schedule(static) default(shared) \
341 private(dH, Lb, k, l, Kb, signkl, La, i, j, Ka, signij)
342 for(Lb = 0; Lb < nStr_b; Lb++) {
343
344 dH = diagH + Lb*nStr_a;
345 const int * exList_Lb_head = exList_b->pointerAtDet(Lb);
346
347 for(La = 0; La < nStr_a; La++, dH++) {
348
349 const int * exList_La_head = exList_a->pointerAtDet(La);
350
351 auto exList_Lb = exList_Lb_head;
352 for(auto Ekl = 0ul; Ekl < nActEa; Ekl++, exList_Lb+=4) {
353
354 UNPACK_EXCITATIONLIST_4(exList_Lb, k, l, Kb, signkl);
355 const int * exList_La = exList_La_head;
356
357 for (auto Eij = 0ul; Eij < nActEb; Eij++, exList_La+=4) {
358
359 UNPACK_EXCITATIONLIST_4(exList_La, i, j, Ka, signij);
360 *dH += signij * signkl * moERI(i, j, k, l);
361 }
362 }
363 }
364 }
365 }; // CASCI::buildDiagH

367 /*
368 * Sigma, Matrix-vector product
369 */
370
371 template <typename MatsT, typename IntsT>
372 void CASCI<MatsT,IntsT>::buildSigma(MCWaveFunction<MatsT, IntsT> & mcwfn,
373 size_t nVec, MatsT * C, MatsT * Sigma) {
374
375 CASCI_LOOP_INIT(); // check top for variable definitions
376
377 auto & hCoreP = *(mcwfn.moints.template getIntegral<OnePInts,MatsT>("hCoreP_Correlated_Space"));
378 auto & moERI = *(mcwfn.moints.template getIntegral<InCore4indexTPI,MatsT>("ERI_Correlated_Space"));
379
380#ifdef DEBUG_CI_SIGMA
381 prettyPrintSmart(std::cout,"HH CASCI Sigma Build -- C", C, NDet, nVec, NDet);
382#endif
383
384 // Allocate SCR
385 size_t nSCR = std::max(nStr_a, nStr_b);
386 size_t nThreads = GetLAThreads();
387 MatsT * SCR = mcwfn.memManager.template malloc<MatsT>(nSCR * nThreads);
388
389 // empty Sigma
390 std::fill_n(Sigma, NDet*nVec, MatsT(0.));
391 // Alpha Part for 1C or the whole build for 2C and 4C
392 // NOTE: exList_ are row-majored c++ objects
393
394 int i, j, k, l, La, Lb, Ka, Kb, Ja, Jb;
395 double signij, signkl;
396 double small_number = std::numeric_limits<double>::epsilon();
397
398 MatsT *HC, *Ci, *SCR_ith;
399
400 // SCR_ith is a row of H instead of a column now
401#pragma omp parallel default(shared) private(HC, Ci, SCR_ith, La, k, l, Ka, signkl, \
402 i, j, Ja, signij, Kb)
403 {
404 auto iThread = GetThreadID();
405 SCR_ith = SCR + nSCR * iThread;
406 for (Ka = iThread; Ka < nStr_a; Ka+=nThreads) {
407
408 std::fill_n(SCR_ith, nStr_a, MatsT(0.));
409 const int * exList_Ka = exList_a->pointerAtDet(Ka);
410 for (auto Ekl = 0ul; Ekl < nNZa; Ekl++, exList_Ka+=4) {
411
412 UNPACK_EXCITATIONLIST_4(exList_Ka, l, k, La, signkl);
413 SCR_ith[La] += signkl * hCoreP(k, l);
414
415 const int * exList_La = exList_a->pointerAtDet(La);
416 for (auto Eij = 0ul; Eij < nNZa; Eij++, exList_La+=4) {
417
418 UNPACK_EXCITATIONLIST_4(exList_La, j, i, Ja, signij);
419 SCR_ith[Ja] += 0.5 * signij * signkl * moERI(i, j, k, l);
420 }
421 }
422
423 // passive screening
424 std::vector<int> SCR_nonZero_ith;
425 for (La = 0 ; La < nStr_a; La++) {
426 if (std::abs(SCR_ith[La]) > small_number)
427 SCR_nonZero_ith.push_back(La);
428 }
429
430 // update Sigma
431 HC = Sigma;
432 Ci = C;
433 if(nC != 1) { // for more than 1C
434 for (auto iVec = 0ul; iVec < nVec; iVec++, HC+=NDet, Ci+=NDet)
435 for (auto iSCR = 0ul; iSCR < SCR_nonZero_ith.size(); iSCR++) {
436 La = SCR_nonZero_ith[iSCR];
437 HC[Ka] += SCR_ith[La] * Ci[La];
438 }
439 } else { // 1C
440 for (auto iVec = 0ul; iVec < nVec; iVec++)
441 for (Kb = 0; Kb < nStr_b; Kb++, HC+=nStr_a, Ci+=nStr_a)
442 for (auto iSCR = 0ul; iSCR < SCR_nonZero_ith.size(); iSCR++) {
443 La = SCR_nonZero_ith[iSCR];
444 HC[Ka] += SCR_ith[La] * Ci[La];
445 }
446 }
447 } // La
448 }
449
450 if (nC != 1) {
451
452#ifdef DEBUG_CI_SIGMA
453 prettyPrintSmart(std::cout,"HH CASCI Sigma Build -- Sigma", Sigma, NDet, nVec, NDet);
454#endif
455 mcwfn.memManager.free(SCR);
456 return;
457 }
458
459 // 1C Continued: Build Beta part
460
461 // transpose sigma and C to make update continuous in inner loop
462 // (a, b) -> (b, a)
463 HC = Sigma;
464 Ci = C;
465 for (auto iVec = 0ul; iVec < nVec; iVec++, HC+=NDet, Ci+=NDet) {
466 IMatCopy('T', nStr_a, nStr_b, MatsT(1.), HC, nStr_a, nStr_b);
467 IMatCopy('T', nStr_a, nStr_b, MatsT(1.), Ci, nStr_a, nStr_b);
468 }
469
470#pragma omp parallel default(shared) private(HC, Ci, SCR_ith, Lb, k, l, Kb, signkl, \
471 i, j, Jb, signij, Ka)
472 {
473 auto iThread = GetThreadID();
474 SCR_ith = SCR + nSCR * iThread;
475 for (Kb = iThread; Kb < nStr_b; Kb+=nThreads) {
476
477 std::fill_n(SCR_ith, nStr_b, MatsT(0.));
478 const int * exList_Kb = exList_b->pointerAtDet(Kb);
479 for (auto Ekl = 0ul; Ekl < nNZb; Ekl++, exList_Kb+=4) {
480
481 UNPACK_EXCITATIONLIST_4(exList_Kb, l, k, Lb, signkl);
482 SCR_ith[Lb] += signkl * hCoreP(k,l);
483
484 const int * exList_Lb = exList_b->pointerAtDet(Lb);
485 for (auto Eij = 0ul; Eij < nNZb; Eij++, exList_Lb+=4) {
486
487 UNPACK_EXCITATIONLIST_4(exList_Lb, j, i, Jb, signij);
488 SCR_ith[Jb] += 0.5 * signij * signkl * moERI(i, j, k, l);
489 }
490 }
491
492 // passive screening
493 std::vector<int> SCR_nonZero_ith;
494 for (Lb = 0 ; Lb < nStr_b; Lb++) {
495 if (std::abs(SCR_ith[Lb]) > small_number)
496 SCR_nonZero_ith.push_back(Lb);
497 }
498
499 // update Sigma
500 HC = Sigma;
501 Ci = C;
502 for (auto iVec = 0ul; iVec < nVec; iVec++)
503 for (Ka = 0; Ka < nStr_a; Ka++, HC+=nStr_b, Ci+=nStr_b)
504 for (auto iSCR = 0ul; iSCR < SCR_nonZero_ith.size(); iSCR++) {
505 Lb = SCR_nonZero_ith[iSCR];
506 HC[Kb] += SCR_ith[Lb] * Ci[Lb];
507 }
508 } // Lb
509 }
510
511 // transpose sigma and C back
512 HC = Sigma;
513 Ci = C;
514 // (b, a) -> (a, b)
515 for (auto iVec = 0ul; iVec < nVec; iVec++, HC+=NDet, Ci+=NDet) {
516 IMatCopy('T', nStr_b, nStr_a, MatsT(1.), HC, nStr_b, nStr_a);
517 IMatCopy('T', nStr_b, nStr_a, MatsT(1.), Ci, nStr_b, nStr_a);
518 }
519
520 mcwfn.memManager.free(SCR);
521
522 // 1C Continued: Alpha-Beta and Beta-Aphla Part
523 // TODO: try to vectorized the loop
524#pragma omp parallel default(shared) private(HC, Ci, SCR_ith, Lb, k, l, Kb, signkl, \
525 i, j, La, signij, Ka)
526 {
527 MatsT tmpH;
528 int KAddr, LAddr;
529 auto iThread = GetThreadID();
530 for(Kb = iThread; Kb < nStr_b; Kb+=nThreads) {
531
532 const int * exList_Kb_head = exList_b->pointerAtDet(Kb);
533
534 for(Ka = 0, KAddr = Kb * nStr_a; Ka < nStr_a; Ka++, KAddr++) {
535
536 const int * exList_Ka_head = exList_a->pointerAtDet(Ka);
537
538 auto exList_Kb = exList_Kb_head;
539 for(auto Ekl = 0ul; Ekl < nNZb; Ekl++, exList_Kb+=4) {
540
541 UNPACK_EXCITATIONLIST_4(exList_Kb, l, k, Lb, signkl);
542
543 auto exList_Ka = exList_Ka_head;
544 for (auto Eij = 0ul; Eij < nNZa; Eij++, exList_Ka+=4) {
545
546 UNPACK_EXCITATIONLIST_4(exList_Ka, j, i, La, signij);
547
548 tmpH = signij * signkl * moERI(i, j, k, l);
549

551 HC = Sigma;
552 Ci = C;
553 LAddr = La + Lb*nStr_a;
554 for (auto iVec = 0ul; iVec < nVec; iVec++, HC+=NDet, Ci+=NDet)
555 HC[KAddr] += tmpH*Ci[LAddr];
556 }
557 }
558 }
559 }
560 }
561 }

577 template <typename MatsT, typename IntsT>
578 void CASCI<MatsT,IntsT>::computeTwoRDM(MCWaveFunction<MatsT, IntsT> & mcwfn,
579 MatsT * C, InCore4indexTPI<MatsT> & twoRDM) {
580
581 CASCI_LOOP_INIT(); // check top for variable definitions
582
583
584 size_t nThreads = GetLAThreads();
585 std::vector<InCore4indexTPI<MatsT>> SCR;
586 auto nDim = twoRDM.nBasis();
587 for (auto i = 0ul; i < nThreads; i++)
588 SCR.emplace_back(mcwfn.memManager, nDim);
589
590 // alpha-alpha part
591 int i, j, k, l, La, Lb, Ka, Kb, Ja, Jb;
592 double signij, signkl;
593#pragma omp parallel default(shared) private(i, j, k, l, La, Ka, Ja, Lb,\
594 signij, signkl)
595 {
596 auto iThread = GetThreadID();
597 auto & tmpRDM = SCR[iThread];
598 tmpRDM.clear();
599 for (La = iThread; La < nStr_a; La+=nThreads) {
600
601 const int * exList_La = exList_a->pointerAtDet(La);
602 for (auto Ekl = 0ul; Ekl < nNZa; Ekl++, exList_La+=4) {
603
604 UNPACK_EXCITATIONLIST_4(exList_La, k, l, Ka, signkl);
605 const int * exList_Ka = exList_a->pointerAtDet(Ka);
606
607 for (auto Eij = 0ul; Eij < nNZa; Eij++, exList_Ka+=4) {
608
609 UNPACK_EXCITATIONLIST_4(exList_Ka, i, j, Ja, signij);
610
611 auto tmp = MatsT(0.);
612 for (auto Lb = 0; Lb < nStr_b; Lb++)
613 tmp += SmartConj(C[Ja + Lb*nStr_a]) * C[La + Lb*nStr_a];
614
615 tmpRDM(i, j, k, l) += tmp * signkl * signij;
616 }
617 }
618 }
619 }
620
621
622 if (nC == 1) {
623
624#pragma omp parallel default(shared) private(i, j, k, l, Lb, Kb, Jb, La, Ka, \
625 signij, signkl)
626 {
627 auto iThread = GetThreadID();
628 auto & tmpRDM = SCR[iThread];
629 for (Lb = iThread; Lb < nStr_b; Lb+=nThreads) {
630
631 const int * exList_Lb = exList_b->pointerAtDet(Lb);
632 for (auto Ekl = 0ul; Ekl < nNZb; Ekl++, exList_Lb+=4) {
633
634 UNPACK_EXCITATIONLIST_4(exList_Lb, k, l, Kb, signkl);
635
636 // beta-beta part
637 const int * exList_Kb = exList_b->pointerAtDet(Kb);
638 for (auto Eij = 0ul; Eij < nNZb; Eij++, exList_Kb+=4) {
639
640 UNPACK_EXCITATIONLIST_4(exList_Kb, i, j, Jb, signij);
641
642 auto tmp = MatsT(0.);
643 for (La = 0; La < nStr_a; La++)
644 tmp += SmartConj(C[La + Jb*nStr_a]) * C[La + Lb*nStr_a];
645
646 tmpRDM(i, j, k, l) += tmp * signkl * signij;
647 }
648
649 // alpha-beta and beta-alpha part
650 for(La = 0; La < nStr_a; La++) {
651 const int * exList_La = exList_a->pointerAtDet(La);
652 for (auto Eij = 0ul; Eij < nNZa; Eij++, exList_La+=4) {
653
654 UNPACK_EXCITATIONLIST_4(exList_La, i, j, Ka, signij);
655 tmpRDM(i, j, k, l) += 2.0 * signkl * signij * SmartConj(C[Ka + Kb*nStr_a]) * C[La + Lb*nStr_a];
656 }
657 }
658
659 }
660 }
661 }
662 }
663
664 twoRDM.clear();
665 auto nDim2 = nDim * nDim;
666 auto nDim4 = nDim2 * nDim2;
667 for (auto i = 0ul; i < nThreads; i++)
668 blas::axpy(nDim4, 1.0, SCR[i].pointer(), 1, twoRDM.pointer(), 1);
669
670 return;
671 } // CASCI::computeTwoRDM

699

705 }
706 }
707 }

710 // beta part
711#pragma omp parallel default(shared) private(La, Lb, k, l, Kb, signkl)
