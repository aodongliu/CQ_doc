37 /*
38 * Generate improved virtual orbials
39 */
40 template <typename MatsT, typename IntsT>
41 void OrbitalRotation<MatsT, IntsT>::generateIVOs(EMPerturbation & pert,
42 SquareMatrix<MatsT> & oneRDM) {
43
44 auto & mopart = mcwfn_.MOPartition;
45 auto & mem = mcwfn_.memManager;
46 auto & ss = mcwfn_.reference();
47 auto & mo = ss.mo[0];
48
49 size_t nC = ss.nC;
50
51 if (nC == 1) CErr("1C IVO NYI");
52
53 size_t nAO = mo.dimension();
54 size_t nTOrb = mopart.nMO;
55 size_t nCorrO = mopart.nCorrO;
56 size_t nCorrE = mopart.nCorrE;
57 size_t nInact = mopart.nInact;
58 size_t nFVirt = mopart.nFVirt;
59 size_t coreOrbOff = nC == 4 ? nAO / 2 : 0;
60 size_t corrOrbOff = coreOrbOff + mopart.nFCore + nInact;
61 size_t virtOrbOff = corrOrbOff + nCorrO;
62
63 /**************************************/
64 /* Step 1: build scaled AO density */
65 /**************************************/
66 SquareMatrix<MatsT> SCR(mem, nAO);
67 SCR.clear();
68
69 // Cas contribution
70 double scale = double(nCorrE - 1) / double(nCorrE);
71 mcwfn_.rdm2pdm(oneRDM, scale);
72
73 /****************************************************/
74 /* Step 2: form and diagonalize virtual fock matrix */
75 /****************************************************/
76
77 // form fock through SingleSlater
78 ss.formFock(pert, false, 1.);
79 SCR = ss.fockMatrix->template spinGather<MatsT>();
80
81 // transform to MO basis only with virtual block
82 SquareMatrix<MatsT> virtualFock(mem, nFVirt);
83 auto off_sizes = mcwfn_.mointsTF->parseMOType("ab");
84 SCR.subsetTransform('N', mo.pointer(), nAO, off_sizes, virtualFock.pointer());
85
86 // diagonalize virtual fock Matrix
87 dcomplex * EIVOs = mem.template malloc<dcomplex>(nFVirt);
88 MatsT * U = mem.template malloc<MatsT>(nFVirt * nFVirt);
89 MatsT * dummy = nullptr;
90 GeneralEigen('N','V', nFVirt, virtualFock.pointer(), nFVirt, EIVOs, dummy, 1, U, nFVirt);
91
92 /*****************************************************************/
93 /* Step 3: transform virtal MO: C'(mu,b) = \sum_a C(mu,a) U(a,b) */
94 /*****************************************************************/
95 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans,
96 nAO, nFVirt, nFVirt, MatsT(1.), mo.pointer() + virtOrbOff * nAO, nAO,
97 U, nFVirt, MatsT(0.), SCR.pointer(), nAO);
98 SetMat('N', nAO, nFVirt, MatsT(1.), SCR.pointer(), nAO,
99 mo.pointer() + virtOrbOff * nAO, nAO);
100
101 // print out IVOs eignevalues
102 std::fill_n(ss.eps1 + coreOrbOff, nTOrb - nFVirt, 0.);
103 for(auto i = 0ul; i < nFVirt; i++) ss.eps1[i + virtOrbOff] = std::real(EIVOs[i]);
104
105 std::cout << std::endl << std::endl;
106 std::cout << " *-------------------------------------------------------*" << std::endl;
107 std::cout << " * Improved Virtual Orbitals (IVOs) have been generated. *" << std::endl;
108 std::cout << " *-------------------------------------------------------*" << std::endl;
109 std::cout << std::endl;
110
111 ss.printEPS(std::cout);
112
113 std::cout << std::endl << std::endl;
114
115 mem.free(EIVOs, U);
116
117 }; // OrbitalRotation<MatsT>::generateIVOs
118
119}; // namespace ChronusQ
