43 template <typename MatsT, typename IntsT>
44 void SingleSlater<MatsT,IntsT>::formDensity() {
45
46 size_t NB = this->nAlphaOrbital() * nC;
47 bool iRO = (std::dynamic_pointer_cast<ROFock<MatsT, IntsT>>(this->fockBuilder) != nullptr);
48
49 // ROHF copy modified orbitals to redundant set
50 if( iRO ){
51 std::copy_n(this->mo[0].pointer(),NB*NB,this->mo[1].pointer());
52 std::copy_n(this->eps1,NB,this->eps2);
53 }
54
55 // Form the 1PDM on the root MPI process as slave processes
56 // do not posses the up-to-date MO coefficients
57 if( MPIRank(comm) == 0 ) {
58
59 if(nC == 1) {
60
61 SquareMatrix<MatsT> DA(memManager, NB);
62
63 //this->mo[0].output(std::cout, "mo1", true);
64
65 // DA = CA * CA**H
66 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans, blas::Op::ConjTrans, NB, NB, this->nOA, MatsT(1.), this->mo[0].pointer(), NB,
67 this->mo[0].pointer(), NB, MatsT(0.), DA.pointer(), NB);
68
69 if(iCS) {
70
71 // DS = 2 * DA
72 *this->onePDM = PauliSpinorSquareMatrices<MatsT>::spinBlockScatterBuild(DA);
73
74 } else {
75
76 SquareMatrix<MatsT> DB(memManager, NB);
77
78 // DB = CB * CB**H
79 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans, blas::Op::ConjTrans, NB, NB, this->nOB, MatsT(1.), this->mo[1].pointer(), NB,
80 this->mo[1].pointer(), NB, MatsT(0.), DB.pointer(), NB);
81
82 // DS = DA + DB
83 // DZ = DA - DB
84 *this->onePDM = PauliSpinorSquareMatrices<MatsT>::spinBlockScatterBuild(DA,DB);
85
86 }
87 } else {
88
89 // 2C or 4C cases
90 SquareMatrix<MatsT> spinBlockForm(memManager, NB);
91
92 if( nC == 2 ) {
93 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans, blas::Op::ConjTrans, NB, NB, this->nO, MatsT(1.), this->mo[0].pointer(), NB,
94 this->mo[0].pointer(), NB, MatsT(0.), spinBlockForm.pointer(), NB);
95 } else if( nC == 4 ) {
96 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans, blas::Op::ConjTrans, NB, NB, this->nO, MatsT(1.), this->mo[0].pointer()+(2*(NB/nC))*NB, NB,
97 this->mo[0].pointer()+(2*(NB/nC))*NB, NB, MatsT(0.), spinBlockForm.pointer(), NB);
98 }
99
100 *this->onePDM = spinBlockForm.template spinScatter<MatsT>();
101
102 }
103 ao2orthoDen();
104 }
105
106
107#ifdef CQ_ENABLE_MPI
108
109 // Broadcast the 1PDM to all MPI processes
110 if( MPISize(comm) > 1 ) {
111 std::cerr << " *** Scattering the 1PDM ***\n";
112 for(auto p : this->onePDM->SZYXPointers())
113 MPIBCast(p,NB*NB/nC/nC,0,comm);
114 }
115
116#endif
117
118
119#if 0
120 print1PDM(std::cerr);
121#endif
122
123 }; // SingleSlater<T>::formDensity

146 template <typename MatsT, typename IntsT>
147 void SingleSlater<MatsT,IntsT>::computeEnergy() {
148
149 ROOT_ONLY(comm);
150
151 // Scalar core hamiltonian contribution to the energy
152 this->OBEnergy =
153 this->template computeOBProperty<DENSITY_TYPE::SCALAR>(
154 coreH->S().pointer());
155
156
157
158 // One body Spin Orbit
159 double SOEnergy = 0.;
160 if (coreH->hasZ())
161 SOEnergy =
162 this->template computeOBProperty<DENSITY_TYPE::MZ>(
163 coreH->Z().pointer());
164 if (coreH->hasXY()) {
165 SOEnergy +=
166 this->template computeOBProperty<DENSITY_TYPE::MY>(
167 coreH->Y().pointer());
168 SOEnergy +=
169 this->template computeOBProperty<DENSITY_TYPE::MX>(
170 coreH->X().pointer());
171 }
172
173
174 this->OBEnergy += SOEnergy;
175
176
177 this->OBEnergy *= 0.5;
178
179 // Compute many-body contribution to energy
180 // *** These calls are safe as proper zeros are returned by
181 // property engine ***
182 this->MBEnergy =
183 this->template computeOBProperty<DENSITY_TYPE::SCALAR>(
184 twoeH->S().pointer());
185
186 if (twoeH->hasZ())
187 this->MBEnergy +=
188 this->template computeOBProperty<DENSITY_TYPE::MZ>(
189 twoeH->Z().pointer());
190 if (twoeH->hasXY()) {
191 this->MBEnergy +=
192 this->template computeOBProperty<DENSITY_TYPE::MY>(
193 twoeH->Y().pointer());
194 this->MBEnergy +=
195 this->template computeOBProperty<DENSITY_TYPE::MX>(
196 twoeH->X().pointer());
197 }
198
199 this->MBEnergy *= 0.25;
200
201 // Assemble total energy
202 this->totalEnergy =
203 this->OBEnergy + this->MBEnergy + this->molecule().nucRepEnergy
204 + this->extraEnergy;
205
206 // Sanity checks
207 assert( not std::isnan(this->OBEnergy) );
208 assert( not std::isnan(this->MBEnergy) );
209 assert( not std::isinf(this->OBEnergy) );
210 assert( not std::isinf(this->MBEnergy) );
211
212 }; // SingleSlater<T>::computeEnergy

221 template <typename MatsT, typename IntsT>
222 void SingleSlater<MatsT,IntsT>::computeMultipole(EMPerturbation &pert) {
223 ROOT_ONLY(comm);
224 // Compute elecric contribution to the dipoles
225 for(auto iXYZ = 0; iXYZ < 3; iXYZ++)
226 this->elecDipole[iXYZ] = -this->template computeOBProperty<SCALAR>((*this->aoints.lenElectric)[iXYZ].pointer());
227
228
229 // Nuclear contributions to the dipoles
230 for(auto &atom : this->molecule().atoms){
231 if (atom.quantum) continue;
232 MatAdd('N','N',3,1,1.,&this->elecDipole[0],3,atom.nucCharge,
233 &atom.coord[0],3,&this->elecDipole[0],3);
234 }
235 // Electric contribution to the quadrupoles
236 for(size_t iXYZ = 0, iX = 0; iXYZ < 3; iXYZ++)
237 for(size_t jXYZ = iXYZ ; jXYZ < 3; jXYZ++, iX++){
238
239 this->elecQuadrupole[iXYZ][jXYZ] = -
240 this->template computeOBProperty<SCALAR>((*this->aoints.lenElectric)[iX+3].pointer());
241
242 this->elecQuadrupole[jXYZ][iXYZ] = this->elecQuadrupole[iXYZ][jXYZ];
243 }
244
245 // Nuclear contributions to the quadrupoles
246 for(auto &atom : this->molecule().atoms){
247
248 if (atom.quantum) continue;
249
250 for(size_t iXYZ = 0; iXYZ < 3; iXYZ++)
251 for(size_t jXYZ = 0; jXYZ < 3; jXYZ++)
252 this->elecQuadrupole[iXYZ][jXYZ] +=
253 atom.nucCharge * atom.coord[iXYZ] * atom.coord[jXYZ];
254 }
255 // Electric contribution to the octupoles
256 for(size_t iXYZ = 0, iX = 0; iXYZ < 3; iXYZ++)
257 for(size_t jXYZ = iXYZ ; jXYZ < 3; jXYZ++)
258 for(size_t kXYZ = jXYZ ; kXYZ < 3; kXYZ++, iX++){
259
260 this->elecOctupole[iXYZ][jXYZ][kXYZ] = -
261 this->template computeOBProperty<SCALAR>(
262 (*this->aoints.lenElectric)[iX+9].pointer());
263
264 this->elecOctupole[iXYZ][kXYZ][jXYZ] = this->elecOctupole[iXYZ][jXYZ][kXYZ];
265
266 this->elecOctupole[jXYZ][iXYZ][kXYZ] = this->elecOctupole[iXYZ][jXYZ][kXYZ];
267
268 this->elecOctupole[jXYZ][kXYZ][iXYZ] = this->elecOctupole[iXYZ][jXYZ][kXYZ];
269
270 this->elecOctupole[kXYZ][iXYZ][jXYZ] = this->elecOctupole[iXYZ][jXYZ][kXYZ];
271
272 this->elecOctupole[kXYZ][jXYZ][iXYZ] = this->elecOctupole[iXYZ][jXYZ][kXYZ];
273 }
274
275 // Nuclear contributions to the octupoles
276 for(auto &atom : this->molecule().atoms){
277
278 if (atom.quantum) continue;
279
280 for(size_t iXYZ = 0; iXYZ < 3; iXYZ++)
281 for(size_t jXYZ = 0; jXYZ < 3; jXYZ++)
282 for(size_t kXYZ = 0; kXYZ < 3; kXYZ++)
283 this->elecOctupole[iXYZ][jXYZ][kXYZ] +=
284 atom.nucCharge * atom.coord[iXYZ] * atom.coord[jXYZ] *
285 atom.coord[kXYZ];
286 }
287 };

290 template <typename MatsT, typename IntsT>
291 void SingleSlater<MatsT,IntsT>::computeSpin() {
292
293 ROOT_ONLY(comm);
294
295 this->SExpect[0] = 0.5 * this->template computeOBProperty<MX>(
296 this->aoints.overlap->pointer());
297 this->SExpect[1] = 0.5 * this->template computeOBProperty<MY>(
298 this->aoints.overlap->pointer());
299 this->SExpect[2] = 0.5 * this->template computeOBProperty<MZ>(
300 this->aoints.overlap->pointer());
301
302 if( not this->onePDM->hasZ() ) this->SSq = 0;
303 else {
304 size_t NB = this->basisSet().nBasis;
305 MatsT * SCR = this->memManager.template malloc<MatsT>(NB*NB);
306 MatsT * SCR2 = this->memManager.template malloc<MatsT>(NB*NB);
307
308
309 // SCR2 = S * D(S) * S
310/*
311 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,NB,NB,NB,T(1.),aoints.overlap,NB,this->onePDM[SCALAR],NB,
312 T(0.),SCR,NB);
313 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,NB,NB,NB,T(1.),SCR,NB,aoints.overlap,NB,T(0.),SCR2,NB);
314*/
315 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
316 this->onePDM->S().pointer(),NB,MatsT(0.),SCR,NB);
317 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
318 SCR,NB,MatsT(0.),SCR2,NB);
319
320
321 this->SSq = 3 * this->nO - (3./2.) *
322 this->template computeOBProperty<SCALAR>(SCR2);
323
324
325 // SCR2 = D(Z) * S * D(Z)
326 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
327 this->onePDM->Z().pointer(),NB,MatsT(0.),SCR,NB);
328 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
329 SCR,NB,MatsT(0.),SCR2,NB);
330
331 this->SSq += 0.5 * this->template computeOBProperty<MZ>(SCR2);
332
333 if( this->onePDM->hasXY() ) {
334
335 // SCR2 = D(Y) * S * D(Y)
336 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
337 this->onePDM->Y().pointer(),NB,MatsT(0.),SCR,NB);
338 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
339 SCR,NB,MatsT(0.),SCR2,NB);
340
341 this->SSq += 0.5 * this->template computeOBProperty<MY>(SCR2);
342
343
344 // SCR2 = D(X) * S * D(X)
345 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
346 this->onePDM->X().pointer(),NB,MatsT(0.),SCR,NB);
347 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
348 SCR,NB,MatsT(0.),SCR2,NB);
349
350 this->SSq += 0.5 * this->template computeOBProperty<MX>(SCR2);
351
352 }
353
354 for(auto i = 0; i < 3; i++)
355 this->SSq += 4 * this->SExpect[i] * this->SExpect[i];
356
357 this->SSq *= 0.25;
358
359 this->memManager.free(SCR,SCR2);
360 }
