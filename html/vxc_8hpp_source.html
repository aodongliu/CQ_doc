58 template <typename MatsT, typename IntsT>
59 void KohnSham<MatsT,IntsT>::formVXC() {
60
61#if VXC_DEBUG_LEVEL >= 1
62 // TIMING
63 auto topMem = std::chrono::high_resolution_clock::now();
64#endif
65 ProgramTimer::tick("Form VXC");
66
67 assert( intParam.nRad % intParam.nRadPerBatch == 0 );
68
69
70
71
72 // Parallelism
73
74 size_t nthreads = GetNumThreads();
75 size_t LAThreads = GetLAThreads();
76 size_t mpiRank = MPIRank(this->comm);
77 size_t mpiSize = MPISize(this->comm);
78
79 // MPI Communicator for numerical integration
80 // *** Assumes that MPI will be done only across atoms in integration
81
82 size_t nAtoms = this->molecule().nAtoms;
83 int color = ((mpiSize < nAtoms) or
84 (mpiRank < nAtoms)) ? 1 : MPI_UNDEFINED;
85
86
87 MPI_Comm intComm = MPICommSplit(this->comm,color,mpiRank);
88
89
90
91
92
93
94
95#ifdef CQ_ENABLE_MPI
96 if( intComm != MPI_COMM_NULL )
97#endif
98 {
99
100
101
102
103
104
105 // Define several useful quantities for later on
106 size_t NPtsMaxPerBatch = intParam.nRadPerBatch * intParam.nAng;
107
108 bool isGGA = std::any_of(functionals.begin(),functionals.end(),
109 [](std::shared_ptr<DFTFunctional> &x) {
110 return x->isGGA();
111 });
112
113 // Turn off LA threads
114 SetLAThreads(1);
115
116 BasisSet &basis = this->basisSet();
117 size_t NB = basis.nBasis;
118 size_t NB2 = NB*NB;
119 size_t NTNB2 = nthreads * NB2;
120 size_t NTNPPB = nthreads*NPtsMaxPerBatch;
121
122 // Clean up all VXC components for a the evaluation for a new
123 // batch of points
124
125 std::vector<std::vector<double*>> integrateVXC;
126 double* intVXC_RAW = (nthreads == 1) ? nullptr :
127 this->memManager.template malloc<double>(VXC->nComponent() * NTNB2);
128
129 std::vector<double*> VXC_SZYX = VXC->SZYXPointers();
130 for(auto k = 0; k < VXC_SZYX.size(); k++) {
131 integrateVXC.emplace_back();
132
133 if( nthreads != 1 )
134 for(auto ith = 0; ith < nthreads; ith++)
135 integrateVXC.back().emplace_back(
136 intVXC_RAW + (ith + k*nthreads) * NB2
137 );
138
139 else
140 integrateVXC.back().emplace_back(VXC_SZYX[k]);
141
142 }
143
144 for(auto &X : integrateVXC) for(auto &Y : X) std::fill_n(Y,NB2,0.);
145
146 std::vector<double> integrateXCEnergy(nthreads,0.);
147
148 // Start Debug quantities
149#if VXC_DEBUG_LEVEL >= 3
150 // tmp VXC submat
151 double *tmpS = this->memManager.template malloc<double>(NB2);
152 std::fill_n(tmpS,basis.nBasis*basis.nBasis,0.);
153#endif
154
155#if VXC_DEBUG_LEVEL >= 2
156 double sumrho = 0.;
157 double sumspin = 0.;
158 double sumgamma = 0.;
159#endif
160 // END Debug quantities
161
162 // Allocating Memory
163 // ----------------------------------------------------------//
164
165 XCEnergy = 0.;
166 double *SCRATCHNBNB =
167 this->memManager.template malloc<double>(NTNB2);
168 double *SCRATCHNBNP =
169 this->memManager.template malloc<double>(NTNPPB*NB);
170
171 double *DenS, *DenZ, *DenX, *DenY, *Mnorm ;
172 double *KScratch;
173 double *HScratch;
174 bool *Msmall;
175 DenS = this->memManager.template malloc<double>(NTNPPB);
176
177 if( this->onePDM->hasZ() )
178 DenZ = this->memManager.template malloc<double>(NTNPPB);
179
180 if( this->onePDM->hasXY() ) {
181 DenY = this->memManager.template malloc<double>(NTNPPB);
182 DenX = this->memManager.template malloc<double>(NTNPPB);
183
184 Mnorm = this->memManager.template malloc<double>(NTNPPB);
185 KScratch = this->memManager.template malloc<double>(3*NTNPPB);
186 Msmall = this->memManager.template malloc<bool>(NTNPPB);
187 }
188
189 double *epsEval = this->memManager.template malloc<double>(NTNPPB);
190
191 // Density U-Variables
192 double *U_n =
193 this->memManager.template malloc<double>(2*NTNPPB);
194 double *dVU_n =
195 this->memManager.template malloc<double>(2*NTNPPB);
196
197 double *ZrhoVar1, *ZgammaVar1, *ZgammaVar2;
198
199 ZrhoVar1 = this->memManager.template malloc<double>(NTNPPB);
200
201 double *GDenS, *GDenZ, *GDenX, *GDenY, *U_gamma, *dVU_gamma;
202 if( isGGA ) {
203 GDenS = this->memManager.template malloc<double>(3*NTNPPB);
204 ZgammaVar1 = this->memManager.template malloc<double>(NTNPPB);
205 ZgammaVar2 = this->memManager.template malloc<double>(NTNPPB);
206
207 if( this->onePDM->hasZ() )
208 GDenZ = this->memManager.template malloc<double>(3*NTNPPB);
209
210 if( this->onePDM->hasXY() ) {
211 GDenY = this->memManager.template malloc<double>(3*NTNPPB);
212 GDenX = this->memManager.template malloc<double>(3*NTNPPB);
213 HScratch = this->memManager.template malloc<double>(3*NTNPPB);
214 }
215
216 // Gamma U-Variables
217 U_gamma = this->memManager.template malloc<double>(3*NTNPPB);
218 dVU_gamma = this->memManager.template malloc<double>(3*NTNPPB);
219 }
220
221 double *epsSCR, *dVU_n_SCR, *dVU_gamma_SCR;
222 if( functionals.size() > 1 ) {
223 epsSCR = this->memManager.template malloc<double>(NTNPPB);
224 dVU_n_SCR = this->memManager.template malloc<double>(2*NTNPPB);
225 if(isGGA)
226 dVU_gamma_SCR =
227 this->memManager.template malloc<double>(3*NTNPPB);
228 }
229
230 // ZMatrix
231 double *ZMAT = this->memManager.template malloc<double>(NTNPPB*NB);
232
233 // Decide if we need to allocate space for real part of the
234 // densities and copy over the real parts
235 std::shared_ptr<PauliSpinorSquareMatrices<double>> Re1PDM;
236 if (std::is_same<MatsT,double>::value)
237 Re1PDM = std::dynamic_pointer_cast<PauliSpinorSquareMatrices<double>>(
238 this->onePDM);
239 else
240 Re1PDM = std::make_shared<PauliSpinorSquareMatrices<double>>(
241 this->onePDM->real_part());
242
243
244 // -------------------------------------------------------------//
245 // End allocating Memory
246
247#if VXC_DEBUG_LEVEL >= 1
248 // TIMING
249 auto botMem = std::chrono::high_resolution_clock::now();
250 std::chrono::duration<double> durevalDen(0.) ;
251 std::chrono::duration<double> durmkAuxVar(0.) ;
252 std::chrono::duration<double> durloadVXCder(0.) ;
253 std::chrono::duration<double> durenergy_vxc(0.) ;
254 std::chrono::duration<double> durconstructZVars(0.) ;
255 std::chrono::duration<double> durformZ_vxc(0.) ;
256 std::chrono::duration<double> durDSYR2K(0.) ;
257 std::chrono::duration<double> durIncBySubMat(0.) ;
258#endif
259
260 auto vxcbuild = [&](size_t &res, std::vector<cart_t> &batch,
261 std::vector<double> &weights, std::vector<size_t> NBE_vec,
262 std::vector<double*> BasisEval_vec,
263 std::vector<std::vector<size_t>>& batchEvalShells_vec,
264 std::vector<std::vector<std::pair<size_t,size_t>>>& subMatCut_vec) {
265
266#if VXC_DEBUG_LEVEL > 3
267 prettyPrintSmart(std::cerr,"BASIS SCR",BasisEval,NBE,
268 4*batch.size(),NBE);
269#endif
270
271 size_t NBE = NBE_vec[0];
272 double* BasisEval = BasisEval_vec[0];
273 std::vector<size_t> & batchEvalShells = batchEvalShells_vec[0];
274 std::vector<std::pair<size_t,size_t>> & subMatCut = subMatCut_vec[0];
275
276 // intParam.epsilon / ntotalpts (NANG * NRAD * NATOMS)
277 double epsScreen = intParam.epsilon / nAtoms /
278 intParam.nAng / intParam.nRad;
279
280 epsScreen = std::max(epsScreen,
281 std::numeric_limits<double>::epsilon());
282
283 size_t NPts = batch.size();
284 size_t IOff = NBE*NPts;
285
286 size_t thread_id = GetThreadID();
287 size_t TIDNPPB = thread_id * NPtsMaxPerBatch;
288
289#if VXC_DEBUG_LEVEL >= 1
290 // TIMING
291 auto topevalDen = std::chrono::high_resolution_clock::now();
292#endif
293
294 // Setup local pointers
295 double * SCRATCHNBNB_loc = SCRATCHNBNB + thread_id * NB2;
296 double * SCRATCHNBNP_loc = SCRATCHNBNP + thread_id * NB*NPtsMaxPerBatch;
297
298 double * DenS_loc = DenS + TIDNPPB;
299 double * DenZ_loc = DenZ + TIDNPPB;
300 double * DenY_loc = DenY + TIDNPPB;
301 double * DenX_loc = DenX + TIDNPPB;
302
303 double * GDenS_loc = GDenS + 3*TIDNPPB;
304 double * GDenZ_loc = GDenZ + 3*TIDNPPB;
305 double * GDenY_loc = GDenY + 3*TIDNPPB;
306 double * GDenX_loc = GDenX + 3*TIDNPPB;
307
308
309 double * epsEval_loc = epsEval + TIDNPPB;
310 double * U_n_loc = U_n + 2*TIDNPPB;
311 double * dVU_n_loc = dVU_n + 2*TIDNPPB;
312 double * U_gamma_loc = U_gamma + 3*TIDNPPB;
313 double * dVU_gamma_loc = dVU_gamma + 3*TIDNPPB;
314
315
316 double * ZrhoVar1_loc = ZrhoVar1 + TIDNPPB;
317 double * ZgammaVar1_loc = ZgammaVar1 + TIDNPPB;
318 double * ZgammaVar2_loc = ZgammaVar2 + TIDNPPB;
319
320 double * epsSCR_loc = epsSCR + TIDNPPB;
321 double * dVU_n_SCR_loc = dVU_n_SCR + 2*TIDNPPB;
322 double * dVU_gamma_SCR_loc = dVU_gamma_SCR + 3*TIDNPPB;
323
324 double *ZMAT_loc = ZMAT + NB * TIDNPPB;
325
326 //2C
327 double * Mnorm_loc = Mnorm + TIDNPPB;
328 double * KScratch_loc = KScratch + 3*TIDNPPB;
329 bool * Msmall_loc = Msmall + TIDNPPB;
330 double * HScratch_loc = HScratch + 3*TIDNPPB;
331
332 // This evaluates the V variables for all components
333 // (Scalar, MZ (UKS) and Mx, MY (2 Comp))
334 evalDen((isGGA ? GRADIENT : NOGRAD), NPts, NBE, NB, subMatCut,
335 SCRATCHNBNB_loc, SCRATCHNBNP_loc, Re1PDM->S().pointer(), DenS_loc,
336 GDenS_loc, GDenS_loc + NPts, GDenS_loc + 2*NPts, BasisEval);
337
338#if VXC_DEBUG_LEVEL < 3
339 // Coarse screen on Density
340 double MaxDenS_loc = *std::max_element(DenS_loc,DenS_loc+NPts);
341 if (MaxDenS_loc < epsScreen) { return; }
342#endif
343
344 if( this->onePDM->hasZ() )
345 evalDen((isGGA ? GRADIENT : NOGRAD), NPts, NBE, NB, subMatCut,
346 SCRATCHNBNB_loc ,SCRATCHNBNP_loc, Re1PDM->Z().pointer(), DenZ_loc,
347 GDenZ_loc, GDenZ_loc + NPts, GDenZ_loc + 2*NPts, BasisEval);
348
349 if( this->onePDM->hasXY() ) {
350 evalDen((isGGA ? GRADIENT : NOGRAD), NPts, NBE, NB, subMatCut,
351 SCRATCHNBNB_loc ,SCRATCHNBNP_loc, Re1PDM->Y().pointer(), DenY_loc,
352 GDenY_loc, GDenY_loc + NPts, GDenY_loc + 2*NPts, BasisEval);
353 evalDen((isGGA ? GRADIENT : NOGRAD), NPts, NBE, NB, subMatCut,
354 SCRATCHNBNB_loc ,SCRATCHNBNP_loc, Re1PDM->X().pointer(), DenX_loc,
355 GDenX_loc, GDenX_loc + NPts, GDenX_loc + 2*NPts, BasisEval);
356 }
357
358#if VXC_DEBUG_LEVEL >= 1
359 // TIMING
360 auto botevalDen = std::chrono::high_resolution_clock::now();
361 auto topmkAuxVar = std::chrono::high_resolution_clock::now();
362#endif
363
364 // V -> U variables for evaluating the kernel derivatives.
365 mkAuxVar(this->onePDM,isGGA,epsScreen,NPts,
366 DenS_loc,DenZ_loc,DenY_loc,DenX_loc,
367 GDenS_loc,GDenS_loc + NPts,GDenS_loc + 2*NPts,
368 GDenZ_loc,GDenZ_loc + NPts,GDenZ_loc + 2*NPts,
369 GDenY_loc,GDenY_loc + NPts,GDenY_loc + 2*NPts,
370 GDenX_loc,GDenX_loc + NPts,GDenX_loc + 2*NPts,
371 Mnorm_loc,
372 KScratch_loc, KScratch_loc + NPts, KScratch_loc + 2* NPts,
373 HScratch_loc, HScratch_loc + NPts, HScratch_loc + 2* NPts,
374 nullptr, nullptr,
375 Msmall_loc,U_n_loc,U_gamma_loc
376 );
377
378#if VXC_DEBUG_LEVEL >= 1
379 // TIMING
380 auto botmkAuxVar = std::chrono::high_resolution_clock::now();
381#endif
382
383
384#if VXC_DEBUG_LEVEL >= 2
385 assert(nthreads == 1);
386 // Debug int
387 for(auto iPt = 0; iPt < NPts; iPt++) {
388 sumrho += weights[iPt] * (U_n[2*iPt] + U_n[2*iPt + 1]);
389 sumspin += weights[iPt] * (U_n[2*iPt] - U_n[2*iPt + 1]);
390 if(isGGA)
391 sumgamma += weights[iPt] *
392 ( U_gamma[3*iPt] + U_gamma[3*iPt+1] + U_gamma[3*iPt+2]);
393 };
394 // end debug
395#endif
396
397#if VXC_DEBUG_LEVEL >= 1
398 // TIMING
399 auto toploadVXCder = std::chrono::high_resolution_clock::now();
400#endif
401
402 // Get DFT Energy derivatives wrt U variables
403 loadVXCder(functionals, NPts, U_n_loc, U_gamma_loc, epsEval_loc,
404 dVU_n_loc, dVU_gamma_loc, epsSCR_loc, dVU_n_SCR_loc,
405 dVU_gamma_SCR_loc);
406
407#if VXC_DEBUG_LEVEL >= 1
408 // TIMING
409 auto botloadVXCder = std::chrono::high_resolution_clock::now();
410 auto topenergy_vxc = std::chrono::high_resolution_clock::now();
411#endif
412
413 // Compute for the current batch the XC energy and increment the
414 // total XC energy.
415 integrateXCEnergy[thread_id] +=
416 energy_vxc(NPts, weights, epsEval_loc, DenS_loc);
417
418#if VXC_DEBUG_LEVEL >= 1
419 // TIMING
420 auto botenergy_vxc =
421 std::chrono::high_resolution_clock::now();
422 auto topconstructZVars =
423 std::chrono::high_resolution_clock::now();
424#endif
425
426 // Construct the required quantities for the formation of the Z
427 // vector (SCALAR) given the kernel derivatives wrt U variables.
428
429 constructZVars(this->onePDM,SCALAR,isGGA,NPts,dVU_n_loc,dVU_gamma_loc,
430 ZrhoVar1_loc, ZgammaVar1_loc, ZgammaVar2_loc);
431
432#if VXC_DEBUG_LEVEL >= 1
433 // TIMING
434 auto botconstructZVars =
435 std::chrono::high_resolution_clock::now();
436 auto topformZ_vxc =
437 std::chrono::high_resolution_clock::now();
438#endif
439
440 // Creating ZMAT (SCALAR) according to
441 // J. Chem. Theory Comput. 2011, 7, 3097–3104 Eq. 15
442 formZ_vxc(this->onePDM, SCALAR, isGGA, NPts, NBE, IOff, epsScreen,
443 weights, ZrhoVar1_loc, ZgammaVar1_loc, ZgammaVar2_loc, DenS_loc,
444 DenZ_loc, DenY_loc, DenX_loc, GDenS_loc, GDenZ_loc, GDenY_loc,
445 GDenX_loc, KScratch_loc, KScratch_loc + NPts,
446 KScratch_loc + 2* NPts, HScratch_loc, HScratch_loc + NPts,
447 HScratch_loc + 2* NPts, BasisEval, ZMAT_loc);
448
449#if VXC_DEBUG_LEVEL >= 1
450 // TIMING
451 auto botformZ_vxc = std::chrono::high_resolution_clock::now();
452#endif
453
454 bool evalZ = true;
455
456#if VXC_DEBUG_LEVEL < 3
457 // Coarse screen on ZMat
458 double MaxBasis = *std::max_element(BasisEval,BasisEval+IOff);
459 double MaxZ = *std::max_element(ZMAT_loc,ZMAT_loc+IOff);
460 evalZ = ( std::abs(2 * MaxBasis * MaxZ) > epsScreen);
461#endif
462
463 if (evalZ) {
464
465 #if VXC_DEBUG_LEVEL >= 1
466 auto topDSYR2K = std::chrono::high_resolution_clock::now();
467 #endif
468 // Creating according to
469 // J. Chem. Theory Comput. 2011, 7, 3097–3104 Eq. 14
470 //
471 // Z -> VXC (submat - SCALAR)
472 blas::syr2k(blas::Layout::ColMajor,blas::Uplo::Lower,blas::Op::NoTrans,NBE,NPts,1.,BasisEval,NBE,ZMAT_loc,NBE,0.,
473 SCRATCHNBNB_loc,NBE);
474
475 #if VXC_DEBUG_LEVEL >= 1
476 // TIMING
477 auto botDSYR2K = std::chrono::high_resolution_clock::now();
478 durDSYR2K += botDSYR2K - topDSYR2K;
479 auto topIncBySubMat = std::chrono::high_resolution_clock::now();
480 #endif
481
482 // Locating the submatrix in the right position given the
483 // subset of shells for the given batch.
484 IncBySubMat(NB,NB,NBE,NBE,integrateVXC[SCALAR][thread_id],NB,
485 SCRATCHNBNB_loc,NBE,subMatCut);
486
487 #if VXC_DEBUG_LEVEL >= 1
488 // TIMING
489 auto botIncBySubMat = std::chrono::high_resolution_clock::now();
490 durIncBySubMat += botIncBySubMat - topIncBySubMat;
491 #endif
492 }
493
494
495
496#if VXC_DEBUG_LEVEL > 3
497 prettyPrintSmart(std::cerr,"Basis ",BasisEval,NBE,NPts,NBE);
498 prettyPrintSmart(std::cerr,"BasisX ",BasisEval+NBE*NPts,NBE,
499 NPts,NBE);
500 prettyPrintSmart(std::cerr,"BasisY ",BasisEval+2*NBE*NPts,
501 NBE,NPts,NBE);
502 prettyPrintSmart(std::cerr,"BasisZ ",BasisEval+3*NBE*NPts,
503 NBE,NPts,NBE);
504 prettyPrintSmart(std::cerr,"ZMAT ",ZMAT_loc,NBE,NPts,NBE);
505#endif
506
507#if VXC_DEBUG_LEVEL >= 1
508 // TIMING
509 durevalDen += botevalDen - topevalDen;
510 durmkAuxVar += botmkAuxVar - topmkAuxVar;
511 durloadVXCder += botloadVXCder - toploadVXCder;
512 durenergy_vxc += botenergy_vxc - topenergy_vxc;
513 durconstructZVars += botconstructZVars - topconstructZVars;
514 durformZ_vxc += botformZ_vxc - topformZ_vxc;
515#endif
516
517#if VXC_DEBUG_LEVEL >= 3
518 // Create Numerical Overlap
519 for(auto iPt = 0; iPt < NPts; iPt++)
520 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,NB,NB,1,weights[iPt],BasisEval + iPt*NB,NB,
521 BasisEval + iPt*NB,NB, 1.,tmpS,NB);
522#endif
523
524 if( not this->onePDM->hasZ() ) return;
525
526 //
527 // --------------- UKS or 2C ------------- Mz ---------------
528 // See J. Chem. Theory Comput. 2017, 13, 2591-2603
529 //
530
531 // Construct the required quantities for the formation of
532 // the Z vector (Mz) given the kernel derivatives wrt U
533 // variables.
534 constructZVars(this->onePDM,MZ,isGGA,NPts,dVU_n_loc,dVU_gamma_loc,ZrhoVar1_loc,
535 ZgammaVar1_loc, ZgammaVar2_loc);
536
537 // Creating ZMAT (Mz) according to
538 // J. Chem. Theory Comput. 2011, 7, 3097–3104 Eq. 15
539 formZ_vxc(this->onePDM, MZ, isGGA, NPts, NBE, IOff, epsScreen, weights,
540 ZrhoVar1_loc, ZgammaVar1_loc, ZgammaVar2_loc, DenS_loc,
541 DenZ_loc, DenY_loc, DenX_loc, GDenS_loc, GDenZ_loc, GDenY_loc,
542 GDenX_loc, KScratch_loc, KScratch_loc + NPts,
543 KScratch_loc + 2* NPts, HScratch_loc, HScratch_loc + NPts,
544 HScratch_loc + 2* NPts, BasisEval, ZMAT_loc);
545
546
547#if VXC_DEBUG_LEVEL < 3
548 MaxZ = *std::max_element(ZMAT_loc,ZMAT_loc+IOff);
549 evalZ = ( std::abs(2 * MaxBasis * MaxZ) > epsScreen);
550#endif
551 // Coarse screen on ZMat
552 if(evalZ) {
553
554 // Creating according to
555 // J. Chem. Theory Comput. 2011, 7, 3097–3104 Eq. 14
556 //
557 // Z -> VXC (submat)
558 blas::syr2k(blas::Layout::ColMajor,blas::Uplo::Lower,blas::Op::NoTrans,NBE,NPts,1.,BasisEval,NBE,ZMAT_loc,NBE,0.,
559 SCRATCHNBNB_loc,NBE);
560
561
562 // Locating the submatrix in the right position given
563 // the subset of shells for the given batch.
564 IncBySubMat(NB,NB,NBE,NBE,integrateVXC[MZ][thread_id],NB,
565 SCRATCHNBNB_loc,NBE,subMatCut);
566
567 }
568
569
570
571 if( not this->onePDM->hasXY() ) return;
572
573 //
574 // --------------- 2C ------------- My ----------------------
575 //
576
577 // Construct the required quantities for the formation of the
578 // Z vector (Mz) given the kernel derivatives wrt U variables.
579 constructZVars(this->onePDM,MY,isGGA,NPts,dVU_n_loc,dVU_gamma_loc,ZrhoVar1_loc,
580 ZgammaVar1_loc, ZgammaVar2_loc);
581
582 // Creating ZMAT (My) according to
583 // J. Chem. Theory Comput. 2011, 7, 3097–3104 Eq. 15
584 formZ_vxc(this->onePDM, MY, isGGA, NPts, NBE, IOff, epsScreen, weights,
585 ZrhoVar1_loc, ZgammaVar1_loc, ZgammaVar2_loc, DenS_loc,
586 DenZ_loc, DenY_loc, DenX_loc, GDenS_loc, GDenZ_loc, GDenY_loc,
587 GDenX_loc, KScratch_loc, KScratch_loc + NPts,
588 KScratch_loc + 2* NPts, HScratch_loc, HScratch_loc + NPts,
589 HScratch_loc + 2* NPts, BasisEval, ZMAT_loc);
590
591
592#if VXC_DEBUG_LEVEL < 3
593 MaxZ = *std::max_element(ZMAT_loc,ZMAT_loc+IOff);
594 evalZ = ( std::abs(2 * MaxBasis * MaxZ) > epsScreen);
595#endif
596 // Coarse screen on ZMat
597 if(evalZ) {
598
599 // Creating according to
600 // J. Chem. Theory Comput. 2011, 7, 3097–3104 Eq. 14
601 //
602 // Z -> VXC (submat)
603 blas::syr2k(blas::Layout::ColMajor,blas::Uplo::Lower,blas::Op::NoTrans,NBE,NPts,1.,BasisEval,NBE,ZMAT_loc,NBE,0.,
604 SCRATCHNBNB_loc,NBE);
605
606
607 // Locating the submatrix in the right position given the
608 // subset of shells for the given batch.
609 IncBySubMat(NB,NB,NBE,NBE,integrateVXC[MY][thread_id],NB,
610 SCRATCHNBNB_loc,NBE,subMatCut);
611
612 }
613
614 //
615 // --------------- 2C ------------- Mx ----------------------
616 //
617
618 // Construct the required quantities for the formation of the
619 // Z vector (Mx) given the kernel derivatives wrt U variables.
620 constructZVars(this->onePDM,MX,isGGA,NPts,dVU_n_loc,dVU_gamma_loc,ZrhoVar1_loc,
621 ZgammaVar1_loc, ZgammaVar2_loc);
622
623 // Creating ZMAT (Mz) according to
624 // J. Chem. Theory Comput. 2011, 7, 3097–3104 Eq. 15
625 formZ_vxc(this->onePDM, MX, isGGA, NPts, NBE, IOff, epsScreen, weights,
626 ZrhoVar1_loc, ZgammaVar1_loc, ZgammaVar2_loc, DenS_loc,
627 DenZ_loc, DenY_loc, DenX_loc, GDenS_loc, GDenZ_loc, GDenY_loc,
628 GDenX_loc, KScratch_loc, KScratch_loc + NPts,
629 KScratch_loc + 2* NPts, HScratch_loc, HScratch_loc + NPts,
630 HScratch_loc + 2* NPts, BasisEval, ZMAT_loc);
631
632
633#if VXC_DEBUG_LEVEL < 3
634 MaxZ = *std::max_element(ZMAT_loc,ZMAT_loc+IOff);
635 evalZ = ( std::abs(2 * MaxBasis * MaxZ) > epsScreen);
636#endif
637 // Coarse screen on ZMat
638 if(evalZ) {
639
640 // Creating according to
641 // J. Chem. Theory Comput. 2011, 7, 3097–3104 Eq. 14
642 //
643 // Z -> VXC (submat)
644 blas::syr2k(blas::Layout::ColMajor,blas::Uplo::Lower,blas::Op::NoTrans,NBE,NPts,1.,BasisEval,NBE,ZMAT_loc,NBE,0.,
645 SCRATCHNBNB_loc,NBE);
646
647
648 // Locating the submatrix in the right position given the
649 // subset of shells for the given batch.
650 IncBySubMat(NB,NB,NBE,NBE,integrateVXC[MX][thread_id],NB,
651 SCRATCHNBNB_loc,NBE,subMatCut);
652 }
653
654 }; // VXC integrate
655
656
657 // Create the BeckeIntegrator object
658 BeckeIntegrator<EulerMac>
659 integrator(intComm,this->memManager,this->molecule(),basis,
660 EulerMac(intParam.nRad), intParam.nAng, intParam.nRadPerBatch,
661 (isGGA ? GRADIENT : NOGRAD), intParam.epsilon);
662
663 // Integrate the VXC
664 integrator.integrate<size_t>(vxcbuild);
665
666
667#if VXC_DEBUG_LEVEL >= 1
668 // TIMING
669 auto toptransform = std::chrono::high_resolution_clock::now();
670#endif
671
672 // Finishing up the VXC
673 // factor in the 4 pi (Lebedev) and built the upper triagolar part
674 // since we create only the lower triangular. For all components
675 for(auto k = 0; k < VXC_SZYX.size(); k++) {
676 if( nthreads == 1 )
677 blas::scal(NB2,4*M_PI,VXC_SZYX[k],1);
678 else
679 for(auto ithread = 0; ithread < nthreads; ithread++)
680 MatAdd('N','N',NB,NB,((ithread == 0) ? 0. : 1.),VXC_SZYX[k],NB,
681 4*M_PI,integrateVXC[k][ithread],NB, VXC_SZYX[k],NB);
682
683 HerMat('L',NB,VXC_SZYX[k],NB);
684 }
685
686 for(auto &X : integrateXCEnergy)
687 XCEnergy += 4*M_PI*X;
688
689
690
691// Combine MPI results
692#ifdef CQ_ENABLE_MPI
693
694 double* mpiScr;
695 if( mpiRank == 0 ) mpiScr = this->memManager.template malloc<double>(NB*NB);
696
697 for(auto &V : VXC_SZYX) {
698
699 mxx::reduce(V,NB*NB,mpiScr,0,std::plus<double>(),intComm);
700
701 if( mpiRank == 0 ) std::copy_n(mpiScr,NB*NB,V);
702
703 }
704
705 if( mpiRank == 0 ) this->memManager.free(mpiScr);
706
707 XCEnergy = mxx::reduce(XCEnergy,0,std::plus<double>(),intComm);
708
709#endif
710
711#if VXC_DEBUG_LEVEL >= 1
712 // TIMING
713 auto bottransform = std::chrono::high_resolution_clock::now();
714#endif
715
716#if VXC_DEBUG_LEVEL >= 3
717 // DebugPrint
718 std::cerr << std::endl;
719 blas::scal(NB2,4*M_PI,tmpS,1);
720 prettyPrintSmart(std::cerr,"Analytic Overlap",
721 this->aoints.overlap,NB,NB,NB);
722 prettyPrintSmart(std::cerr,"Numeric Overlap",tmpS,NB,NB,NB);
723 std::cerr << std::endl;
724 std::cerr << std::endl;
725 for(auto i = 0; i < NB2; i++)
726 tmpS[i] = std::abs(tmpS[i] - this->aoints.overlap[i]);
727 std::cerr << "MAX DIFF OVERLAP = " <<
728 *std::max_element(tmpS,tmpS+NB2) << std::endl;
729#endif
730
731
732
733#if VXC_DEBUG_LEVEL >= 2
734 // DEBUG
735 std::cerr << std::scientific << "\n";
736 std::cerr << "N electrons = " << 4*M_PI*sumrho << "\n";
737 std::cerr << "N unp electrons = " << 4*M_PI*sumspin << "\n";
738 std::cerr << "sum gamma = " << 4*M_PI*sumgamma << "\n";
739 std::cerr << "EXC = " << XCEnergy << "\n";
740
741 prettyPrintSmart(std::cerr,"onePDM Scalar",this->onePDM[SCALAR],
742 NB,NB,NB);
743 prettyPrintSmart(std::cerr,"Numerical Scalar VXC ",
744 integrateVXC[SCALAR][0],NB,NB,NB);
745
746 if( not this->iCS or this->nC > 1 ) {
747 prettyPrintSmart(std::cerr,"onePDM Mz",this->onePDM[MZ],
748 NB,NB,NB);
749 prettyPrintSmart(std::cerr,"Numerical Mz VXC",
750 integrateVXC[MZ][0],NB,NB,NB);
751
752 if( this->onePDM->hasXY() ) {
753 prettyPrintSmart(std::cerr,"onePDM My",this->onePDM[MY],
754 NB,NB,NB);
755 prettyPrintSmart(std::cerr,"Numerical My VXC",
756 integrateVXC[MY][0],NB,NB,NB);
757
758 prettyPrintSmart(std::cerr,"onePDM Mx",this->onePDM[MX],
759 NB,NB,NB);
760 prettyPrintSmart(std::cerr,"Numerical Mx VXC",
761 integrateVXC[MX][0],NB,NB,NB);
762 }
763 }
764#endif
765
766 // Freeing the memory
767 // ------------------------------------------------------------ //
768 this->memManager.free(SCRATCHNBNB,SCRATCHNBNP,DenS,epsEval,U_n,
769 dVU_n, ZrhoVar1,ZMAT);
770 if( isGGA )
771 this->memManager.free(ZgammaVar1,ZgammaVar2,GDenS,U_gamma,
772 dVU_gamma);
773
774 if( this->onePDM->hasZ() ) {
775 this->memManager.free(DenZ);
776 if( isGGA ) this->memManager.free(GDenZ);
777 }
778
779 if( this->onePDM->hasXY() ) {
780 this->memManager.free(DenX,DenY,Mnorm,KScratch,Msmall);
781 if( isGGA ) this->memManager.free(GDenX,GDenY,HScratch);
782 }
783
784 if( functionals.size() > 1 ) {
785 this->memManager.free(epsSCR,dVU_n_SCR);
786 if( isGGA ) this->memManager.free(dVU_gamma_SCR);
787 }
788
789
790 if( nthreads != 1 ) this->memManager.free(intVXC_RAW);
791
792 Re1PDM = nullptr;
793 // ------------------------------------------------------------- //
794 // End freeing the memory
795
796
797#if VXC_DEBUG_LEVEL >= 1
798 // TIMING
799 double d_batch = this->molecule().nAtoms *
800 intParam.nRad / intParam.nRadPerBatch;
801
802 std::chrono::duration<double> durMem = botMem - topMem;
803 std::chrono::duration<double> durtransform =
804 bottransform - toptransform;
805
806 std::cerr << std::scientific << "\n";
807 std::cerr << "Mem " << durMem.count()/d_batch << "\n";
808 std::cerr << "transform " << durtransform.count()/d_batch << "\n";
809 std::cerr << "evalDen " << durevalDen.count()/d_batch << "\n";
810 std::cerr << "mkAuxVar " << durmkAuxVar.count()/d_batch << "\n";
811 std::cerr << "loadVXCder " << durloadVXCder.count()/d_batch << "\n";
812 std::cerr << "energy_vxc " << durenergy_vxc.count()/d_batch << "\n";
813 std::cerr << "constructZVars " << durconstructZVars.count()/d_batch
814 << "\n";
815 std::cerr << "formZ_vxc " << durformZ_vxc.count()/d_batch << "\n";
816 std::cerr << "DSYR2K " << durDSYR2K.count()/d_batch << "\n";
817 std::cerr << "IncBySubMat " << durIncBySubMat.count()/d_batch;
818 std::cerr << "\n\n\n";
819#endif
820
821
822
823 // Turn back on LA threads
824 SetLAThreads(LAThreads);
825
826 MPICommFree(intComm); // Free communicator
827
828 } // Valid intComm
829
830 MPI_Barrier(this->comm); // Syncronize the MPI processes
831
832 ProgramTimer::tock("Form VXC");
833
834 }; // KohnSham::formVXC
