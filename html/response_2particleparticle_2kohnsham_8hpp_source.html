39 template <typename MatsT, typename IntsT>
40 template <typename U>
41 void ParticleParticlePropagator<KohnSham<MatsT,IntsT>>::formLinearTrans_direct_impl(
42 MPI_Comm c, RC_coll<U> x){
43
44
45 SingleSlater<MatsT,IntsT> &ss = dynamic_cast<SingleSlater<MatsT,IntsT>&>(*this->ref_);
46
47 bool doA = this->tdaOp == PP_A;
48 bool doOcc = not this->genSettings.doTDA or this->tdaOp == PP_C;
49
50 const size_t tdOffSet = this->genSettings.doTDA ? 0 : this->aDim_;
51 const size_t N = this->nSingleDim_ ;
52 const size_t chunk = 600;
53
54 std::shared_ptr<TPIContractions<U,IntsT>> TPI =
55 TPIContractions<MatsT,IntsT>::template convert<U>(ss.TPI);
56
57 for(auto &X : x) {
58
59 const size_t nVec = X.nVec;
60
61 if( X.AX ) std::fill_n(X.AX,N*nVec,0.);
62
63 for(size_t k = 0; k < nVec; k += chunk) {
64
65 MPI_Barrier(c); // Sync MPI Processes at begining of each batch of
66 // vectors
67
68 const size_t nDo = std::min( chunk, nVec - k);
69
70 auto *V_c = X.X + k*N;
71 auto *HV_c = X.AX + k*N;
72
73
74 bool scatter = not bool(X.X);
75#ifdef CQ_ENABLE_MPI
76 scatter = mxx::any_of(scatter,c);
77#endif
78
79 // Transform pp vector MO -> AO
80 auto cList = this->genSettings.doTDA ?
81 this->template ppTransitionVecMO2AO<U>(c,scatter,nDo,N,V_c) :
82 this->template ppTransitionVecMO2AO<U>(c,scatter,nDo,N,V_c,
83 V_c + tdOffSet);
84
85 TPI->twoBodyContract(c,cList); // form G[V]
86
87
88
89
90 // Only finish transformation on root process
91 if( MPIRank(c) == 0 ) {
92
93 // Transform pp vector AO -> MO
94 if( this->genSettings.doTDA )
95 this->ppTransitionVecAO2MO(nDo,N,cList,HV_c);
96 else
97 this->ppTransitionVecAO2MO(nDo,N,cList,HV_c,HV_c + tdOffSet);
98
99
100
101
102 // Scale by diagonals
103 if( not this->genSettings.doTDA or doA )
104 this->ppEpsilonScale(true,false,false,nDo,N,V_c,HV_c);
105
106 if( doOcc )
107 this->ppEpsilonScale(true,false,true,nDo,N,V_c + tdOffSet,
108 HV_c + tdOffSet);
109
110 }
111
112
113 // Free up transformation memory
114 this->memManager_.free(cList[0].X);
115
116 } // loop over vectors
117
118 } // loop over groups of vectors
119
120
121 };
