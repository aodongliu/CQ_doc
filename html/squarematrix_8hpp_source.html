184 virtual void broadcast(MPI_Comm comm = MPI_COMM_WORLD, int root = 0) {
185
186#ifdef CQ_ENABLE_MPI
187 // BCast matrix to all MPI processes
188 if( MPISize(comm) > 1 ) {
189 std::cerr << " *** Scattering a matrix ***\n";
190 size_t N_bcast = N_;
191 MPIBCast(N_bcast,root,comm);
192
193 if (N_bcast != N_) {
194 N_ = N_bcast;
195 malloc();
196 }
197
198 MPIBCast(ptr_,N_*N_,root,comm);
199 }

247 void malloc() {
248 if (ptr_) memManager_.free(ptr_);
249 size_t N2 = N_*N_;
250 if (N2 != 0) {
251 try { ptr_ = memManager_.malloc<MatsT>(N2); }
252 catch(...) {

255 << (N2/1e9) * sizeof(double) << " GB)" << std::endl;
