42 template <typename MatsT, typename IntsT>
43 void RASCI<MatsT,IntsT>::reassembleCatDet1RAS(std::vector<size_t> & LCat,
44 std::vector<size_t> & nDet, size_t e, std::vector<size_t> & LCatAlt,
45 std::vector<size_t> & nDetAlt) {
46

52 ii += 1;

61 template <typename MatsT, typename IntsT>
62 void RASCI<MatsT,IntsT>::reassembleCatDet2RAS(std::vector<size_t> & LCat,
63 std::vector<size_t> & nDet, size_t e1, size_t e2, std::vector<size_t> & LCatAlt,
64 std::vector<size_t> & nDetAlt, std::vector<size_t> & Jrs,
65 std::vector<size_t> & LCat2) {
66
67 if (LCat2.size() > 0) {
68 Jrs[0] = LCat2[e1];
69 Jrs[1] = LCat2[e2];
70 } else {
71 Jrs[0] = e1;
72 Jrs[1] = e2;
73 }
74 LCatAlt[0] = LCat[e1];
75 LCatAlt[1] = LCat[e2];
76 nDetAlt[0] = nDet[e1];
77 nDetAlt[1] = nDet[e2];
78 for (auto i = 0ul; i < 3; i++)
79 if (i != e1 and i != e2) {
80 if (LCat2.size() > 0) Jrs[2] = LCat2[i];
81 else Jrs[2] = i;
82 LCatAlt[2] = LCat[i];
83 nDetAlt[2] = nDet[i];
84 }
85 } // RASCI::reassembleCatDet2RAS

93 template <typename MatsT, typename IntsT>
94 void RASCI<MatsT,IntsT>::SigmaD_diag(MatsT * C, MatsT * DBlk, size_t nNZof,
95 size_t nNZB, std::shared_ptr<const ExcitationList> exList, size_t rsB,
96 std::vector<size_t> & LCat,
97 std::vector<size_t> & nDet) {
98
99 size_t l1, l, j23off, j, p, q, j1, j2, j3, iNZ;
100 double sign;
101
102 // re-assemble LCat and nDet into LCatAlt and nDetAlt
103 std::vector<size_t> LCatAlt(3, 0);
104 std::vector<size_t> nDetAlt(3, 0);
105 reassembleCatDet1RAS(LCat, nDet, rsB, LCatAlt, nDetAlt);
106
107 #pragma omp parallel for schedule(static, 1) default(shared) private(j1, j2, j3, iNZ, l1, l, j23off, j, sign, p, q)
108 for (j3 = 0; j3 < nDetAlt[2]; j3++)
109 for (j2 = 0; j2 < nDetAlt[1]; j2++)
110 for (j1 = 0; j1 < nDetAlt[0]; j1++) {
111 j23off = j3 * LCatAlt[2] + j2 * LCatAlt[1];
112 j = j1 * LCatAlt[0] + j23off;
113 const int * exList_j1 = exList->pointerAtDet(j1) + nNZof * 4;
114 for (iNZ = 0; iNZ < nNZB; iNZ++, exList_j1+=4) {
115 UNPACK_EXCITATIONLIST_4(exList_j1, p, q, l1, sign);
116 l = l1 * LCatAlt[0] + j23off;
117 DBlk[iNZ+j*nNZB] += sign * C[l];
118 }
119 }
120 // End openMP
121
122#ifdef _DEBUG_CIBUILDER_RASCI_IMPL
123 prettyPrintSmart(std::cout,"LL RASCI Sigma D diag --", DBlk, nNZB, LCat[3], nNZB);
124#endif
125
126 } //RASCI::SigmaD_diag

134 template <typename MatsT, typename IntsT>
135 void RASCI<MatsT,IntsT>::SigmaD_offdiag(MatsT * C, MatsT * DBlk, size_t nNZof,
136 size_t nNZB, std::shared_ptr<const ExcitationList> exList, size_t rB, size_t sB,
137 std::vector<size_t> & LCat1, std::vector<size_t> & LCat2,
138 std::vector<size_t> & nDet) {
139
140 // re-assemble LCatAlt and NDAlt
141 std::vector<size_t> LCatAlt1(3, 0);
142 std::vector<size_t> LCatAlt2(3, 0);
143 std::vector<size_t> nDetAlt(3, 0);
144 reassembleCatDet2RAS(LCat1, nDet, sB, rB, LCatAlt1, nDetAlt, LCatAlt2, LCat2);
145
146 size_t j, l, l1, l2, p, q, j1, j2, j3, iNZ;
147 double sign;
148
149 #pragma omp parallel for schedule(static, 1) default(shared) private(j1, j2, j3, iNZ, l1, l, l2, j, sign, p, q)
150 for (j3 = 0; j3 < nDetAlt[2]; j3++)
151 for (j1 = 0; j1 < nDetAlt[0]; j1++)
152 for (j2 = 0; j2 < nDetAlt[1]; j2++) {
153 j = j1*LCatAlt1[0] + j2*LCatAlt1[1] + j3*LCatAlt1[2];
154 const int * exList_j = exList->pointerAtDet(j1, j2) + nNZof * 5;
155 for (iNZ = 0; iNZ < nNZB; iNZ++, exList_j+=5) {
156 UNPACK_EXCITATIONLIST_5(exList_j, p, q, l1, l2, sign);
157 l = l1*LCatAlt2[0] + l2*LCatAlt2[1] + j3*LCatAlt2[2];
158 DBlk[iNZ+j*nNZB] += sign * C[l];
159 }
160 }
161 // End openMP
162
163#ifdef _DEBUG_CIBUILDER_RASCI_IMPL
164 prettyPrintSmart(std::cout,"LL RASCI Sigma D off-diag --", DBlk, nNZB, LCat1[3], nNZB);
165#endif
166
167 } // RASCI::SigmaD_offdiag

175 template <typename MatsT, typename IntsT>
176 void RASCI<MatsT,IntsT>::Sigma1e_diag(MatsT * Sigma, MatsT * Sscr, MatsT * DBlk,
177 MatsT * Dscr, OnePInts<MatsT> & hCoreP, MatsT * hCPscr, size_t nNZof,
178 size_t nNZB, std::shared_ptr<const ExcitationList> exList, size_t rsB,
179 std::vector<size_t> & LCat,
180 std::vector<size_t> & nDet, int rsoff) {
181
182 // re-assemble LCatAlt and NDAlt
183 std::vector<size_t> LCatAlt(3, 0);
184 std::vector<size_t> nDetAlt(3, 0);
185 reassembleCatDet1RAS(LCat, nDet, rsB, LCatAlt, nDetAlt);
186
187 size_t n23 = LCat[3] / nDetAlt[0];
188 size_t j1off, j23, j13off, j, r, s, l1;
189 double sign;
190
191 #pragma omp for schedule(static, 1)
192 for (auto j1 = 0ul; j1 < nDetAlt[0]; j1++) {
193 j1off = j1 * LCatAlt[0];
194 // Gathering D, implicitly J_rs = j1;
195 j23 = 0;
196 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
197 j13off = j1off + j3 * LCatAlt[2];
198 for (auto j2 = 0ul; j2 < nDetAlt[1]; j2++, j23++) {
199 j = j13off + j2 * LCatAlt[1];
200 std::copy_n(DBlk+j*nNZB, nNZB, Dscr+j23*nNZB);
201 }
202 }
203 // Gathering HCorePrime
204 const int * exList_j1 = exList->pointerAtDet(j1) + nNZof * 4;
205 for (auto iNZ = 0ul; iNZ < nNZB; iNZ++, exList_j1+=4) {
206 UNPACK_EXCITATIONLIST_4(exList_j1, s, r, l1, sign);
207 s += rsoff;
208 r += rsoff;
209 hCPscr[iNZ] = hCoreP(r, s);
210 }
211 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans,
212 1,n23,nNZB,MatsT(1.),hCPscr,1,Dscr,nNZB,
213 MatsT(0.),Sscr,1);
214 // Scattering back to Sigma
215 j23 = 0;
216 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
217 j13off = j1off + j3 * LCatAlt[2];
218 for (auto j2 = 0ul; j2 < nDetAlt[1]; j2++, j23++) {
219 j = j13off + j2 * LCatAlt[1];
220 Sigma[j] += Sscr[j23];
221 }
222 }
223 }
224 // End omp for
225
226 } //RASCI::Sigma1e_diag

234 template <typename MatsT, typename IntsT>
235 void RASCI<MatsT,IntsT>::Sigma1e_offdiag(MatsT * Sigma, MatsT * Sscr, MatsT * DBlk,
236 MatsT * Dscr, OnePInts<MatsT> & hCoreP, MatsT * hCPscr, size_t nNZof,
237 size_t nNZB, std::shared_ptr<const ExcitationList> exList, size_t rB, size_t sB,
238 std::vector<size_t> & LCat, std::vector<size_t> & nDet,
239 int roff, int soff) {
240
241 // locate the RAS space involved in excitation
242 size_t exBlk1 = std::min(rB, sB);
243 size_t exBlk2 = std::max(rB, sB);
244
245 // re-assemble LCatAlt and nDetAlt
246 std::vector<size_t> Jrs(3, 0);
247 std::vector<size_t> LCatAlt(3, 0);
248 std::vector<size_t> nDetAlt(3, 0);
249 std::vector<size_t> LCat2 = {};
250 reassembleCatDet2RAS(LCat, nDet, exBlk1, exBlk2, LCatAlt, nDetAlt, Jrs, LCat2);
251
252 std::vector<size_t> Jras(3, 0);
253 size_t j, j12off, s, r, l1, l2;
254 double sign;
255
256 #pragma omp for schedule(static, 1)
257 for (auto j1 = 0ul; j1 < nDetAlt[0]; j1++)
258 for (auto j2 = 0ul; j2 < nDetAlt[1]; j2++) {
259 j12off = j2 * LCatAlt[1] + j1 * LCatAlt[0];
260 Jras[Jrs[0]] = j1;
261 Jras[Jrs[1]] = j2;
262 // Gathering D
263 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
264 j = j12off + j3 * LCatAlt[2];
265 std::copy_n(DBlk+j*nNZB, nNZB, Dscr+j3*nNZB);
266 }
267 // Gathering HCorePrime
268 const int * exList_j = exList->pointerAtDet(Jras[sB], Jras[rB]) + nNZof * 5;
269 for (auto iNZ = 0ul; iNZ < nNZB; iNZ++, exList_j+=5) {
270 UNPACK_EXCITATIONLIST_5(exList_j, s, r, l1, l2, sign);
271 s += soff;
272 r += roff;
273 hCPscr[iNZ] = hCoreP(r, s);
274 }
275 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans,
276 1,nDetAlt[2],nNZB,MatsT(1.),hCPscr,1,
277 Dscr,nNZB,MatsT(0.),Sscr,1);
278 // Scattering back to Sigma
279 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
280 j = j12off + j3 * LCatAlt[2];
281 Sigma[j] += Sscr[j3];
282 }
283 }
284
285 } // RASCI::Sigma1e_offdiag

294 template <typename MatsT, typename IntsT>
295 void RASCI<MatsT,IntsT>::SigmaG_dd(MatsT * GBlk, MatsT * Gscr, MatsT * DBlk,
296 MatsT * Dscr, MatsT * ERIscr, InCore4indexTPI<MatsT> & moERI,
297 size_t nNZrsof, size_t nNZrsB, size_t nNZpqof, size_t nNZpqB,
298 std::shared_ptr<const ExcitationList> pqList, std::shared_ptr<const ExcitationList> rsList,
299 size_t rsB, size_t pqB,
300 std::vector<size_t> & LCat, std::vector<size_t> & nDet, int rsoff, int pqoff) {
301
302 // determine excitation cases:
303 // 1. one RAS space is involved in excitation
304 // 2. two RAS spaces are involved in excitation
305
306 size_t exBlk1, exBlk2, j, p, q, r, s;
307 double sign;
308 std::vector<size_t> Jrs(3, 0);
309 std::vector<size_t> LCatAlt(3, 0);
310 std::vector<size_t> nDetAlt(3, 0);
311
312 // Case 1: one RAS space is involved in excitation
313 if (rsB == pqB) {
314 // locate the RAS space involved in excitation
315 exBlk1 = rsB;
316 // re-assemble LCatAlt and nDetAlt
317 reassembleCatDet1RAS(LCat, nDet, exBlk1, LCatAlt, nDetAlt);
318
319 size_t n23 = LCat[3]/nDetAlt[0];
320 size_t j1off, j23, j13off,l1;
321
322 #pragma omp for schedule(static, 1)
323 for (auto j1 = 0ul; j1 < nDetAlt[0]; j1++) {
324 // Gathering D, implicitly jrs = jpq = j1
325 j1off = j1 * LCatAlt[0];
326 j23 = 0;
327 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
328 j13off = j1off + j3 * LCatAlt[2];
329 for (auto j2 = 0ul; j2 < nDetAlt[1]; j2++, j23++) {
330 j = j13off + j2 * LCatAlt[1];
331 std::copy_n(DBlk+j*nNZrsB, nNZrsB, Dscr+j23*nNZrsB);
332 }
333 }
334 // Gathering moERI
335 const int * exList_pq = pqList->pointerAtDet(j1) + nNZpqof * 4;
336 for (auto iNZpq = 0ul; iNZpq < nNZpqB; iNZpq++, exList_pq+=4) {
337 UNPACK_EXCITATIONLIST_4(exList_pq, p, q, l1, sign);
338 p += pqoff;
339 q += pqoff;
340 const int * exList_rs = rsList->pointerAtDet(j1) + nNZrsof * 4;
341 for (auto iNZrs = 0ul; iNZrs < nNZrsB; iNZrs++, exList_rs+=4) {
342 UNPACK_EXCITATIONLIST_4(exList_rs, s, r, l1, sign);
343 s += rsoff;
344 r += rsoff;
345 ERIscr[iNZrs+iNZpq*nNZrsB] = moERI(r, s, p, q);
346 }
347 }
348 blas::gemm(blas::Layout::ColMajor, blas::Op::Trans, blas::Op::NoTrans,
349 nNZpqB,n23,nNZrsB,MatsT(1.),ERIscr,nNZrsB,
350 Dscr,nNZrsB,MatsT(0.),Gscr,nNZpqB);
351 // Scattering back to GBlk
352 j23 = 0;
353 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
354 j13off = j1off + j3 * LCatAlt[2];
355 for (auto j2 = 0ul; j2 < nDetAlt[1]; j2++, j23++) {
356 j = j13off + j2 * LCatAlt[1];
357 MatAdd('N','N',nNZpqB,1,MatsT(1.),Gscr+j23*nNZpqB,nNZpqB,
358 MatsT(1.),GBlk+j*nNZpqB,nNZpqB,GBlk+j*nNZpqB,nNZpqB);
359 }
360 }
361 }
362 // Case 2: two RAS spaces are involved in excitation
363 // implicitly pBlk < rBlk
364 } else {
365 // locate the RAS space involved in excitation
366 exBlk1 = std::min(rsB, pqB);
367 exBlk2 = std::max(rsB, pqB);
368
369 // re-assemble LCatAlt and nDetAlt
370 std::vector<size_t> LCat2 = {};
371 reassembleCatDet2RAS(LCat, nDet, exBlk1, exBlk2, LCatAlt, nDetAlt, Jrs, LCat2);
372
373 std::vector<size_t> Jras(3, 0);
374 size_t j12off, l1;
375
376 #pragma omp for schedule(static, 1)
377 for (auto j1 = 0ul; j1 < nDetAlt[0]; j1++)
378 for (auto j2 = 0ul; j2 < nDetAlt[1]; j2++) {
379 j12off = j2 * LCatAlt[1] + j1 * LCatAlt[0];
380 Jras[Jrs[0]] = j1;
381 Jras[Jrs[1]] = j2;
382 // Gathering D
383 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
384 j = j12off + j3 * LCatAlt[2];
385 std::copy_n(DBlk+j*nNZrsB, nNZrsB, Dscr+j3*nNZrsB);
386 }
387 // Gathering moERI, implicitly pBlk < rBlk
388 const int * exList_pq = pqList->pointerAtDet(Jras[pqB]) + nNZpqof * 4;
389 for (auto iNZpq = 0ul; iNZpq < nNZpqB; iNZpq++, exList_pq+=4) {
390 UNPACK_EXCITATIONLIST_4(exList_pq, p, q, l1, sign);
391 p += pqoff;
392 q += pqoff;
393 const int * exList_rs = rsList->pointerAtDet(Jras[rsB]) + nNZrsof * 4;
394 for (auto iNZrs = 0ul; iNZrs < nNZrsB; iNZrs++, exList_rs+=4) {
395 UNPACK_EXCITATIONLIST_4(exList_rs, s, r, l1, sign);
396 s += rsoff;
397 r += rsoff;
398 ERIscr[iNZrs+iNZpq*nNZrsB] = moERI(r, s, p, q) - moERI(p, s, r, q);
399 }
400 }
401
402 blas::gemm(blas::Layout::ColMajor, blas::Op::Trans, blas::Op::NoTrans,
403 nNZpqB,nDetAlt[2],nNZrsB,MatsT(1.),ERIscr,nNZrsB,
404 Dscr,nNZrsB,MatsT(0.),Gscr,nNZpqB);
405 // Scattering back to GBlk
406 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
407 j = j12off + j3 * LCatAlt[2];
408 MatAdd('N','N',nNZpqB,1,MatsT(1.),Gscr+j3*nNZpqB,nNZpqB,
409 MatsT(1.),GBlk+j*nNZpqB,nNZpqB,GBlk+j*nNZpqB,nNZpqB);
410 }
411 }
412 }
413#ifdef _DEBUG_CIBUILDER_RASCI_IMPL
414 prettyPrintSmart(std::cout,"LL RASCI Sigma G diag-diag --", GBlk, nNZpqB, LCat[3], nNZpqB);
415#endif
416 } // RASCI::SigmaG_dd

425 template <typename MatsT, typename IntsT>
426 void RASCI<MatsT,IntsT>::SigmaG_do(MatsT * GBlk, MatsT * Gscr, MatsT * DBlk,
427 MatsT * Dscr, MatsT * ERIscr, InCore4indexTPI<MatsT> & moERI, size_t nNZrsof,
428 size_t nNZrsB, size_t nNZpqof, size_t nNZpqB,
429 std::shared_ptr<const ExcitationList> pqList,
430 std::shared_ptr<const ExcitationList> rsList, size_t rsB, size_t pB, size_t qB,
431 std::vector<size_t> & LCat, std::vector<size_t> & nDet,
432 int rsoff, int poff, int qoff) {
433
434 std::vector<size_t> exRAS(3, 0);
435 std::vector<size_t> Jrs(3, 0);
436 std::vector<size_t> LCatAlt(3, 0);
437 std::vector<size_t> nDetAlt(3, 0);
438 std::vector<size_t> Jras(3, 0);
439 size_t exBlk1, exBlk2, j, p, q, r, s, l1, l2;
440 double sign;
441
442 // determine excitation cases:
443 // 2. two RAS spaces are involved in excitation
444 // 3. three RAS spaces are involved in excitation
445 exRAS[rsB] = 1;
446 exRAS[pB] = 1;
447 exRAS[qB] = 1;
448 int eCase = std::accumulate(exRAS.begin(), exRAS.end(), 0);
449 // Case 2: two RAS spaces are involved in excitation
450 if (eCase == 2) {
451 // locate the RAS space involved in excitation
452 exBlk1 = std::min({rsB, pB, qB});
453 exBlk2 = std::max({rsB, pB, qB});
454 std::vector<size_t> LCat2 = {};
455 reassembleCatDet2RAS(LCat, nDet, exBlk1, exBlk2, LCatAlt, nDetAlt, Jrs, LCat2);
456
457 #pragma omp for schedule(static, 1)
458 for (auto j1 = 0ul; j1 < nDetAlt[0]; j1++)
459 for (auto j2 = 0ul; j2 < nDetAlt[1]; j2++) {
460 size_t j12off = j2 * LCatAlt[1] + j1 * LCatAlt[0];
461 Jras[Jrs[0]] = j1;
462 Jras[Jrs[1]] = j2;
463 // Gathering D
464 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
465 j = j12off + j3 * LCatAlt[2];
466 std::copy_n(DBlk+j*nNZrsB, nNZrsB, Dscr+j3*nNZrsB);
467 }
468 // Scattering moERI
469 const int * exList_pq = pqList->pointerAtDet(Jras[pB], Jras[qB]) + nNZpqof * 5;
470 for (auto iNZpq = 0ul; iNZpq < nNZpqB; iNZpq++, exList_pq+=5) {
471 UNPACK_EXCITATIONLIST_5(exList_pq, p, q, l1, l2, sign);
472 p += poff;
473 q += qoff;
474 const int * exList_rs = rsList->pointerAtDet(Jras[rsB]) + nNZrsof * 4;
475 for (auto iNZrs = 0ul; iNZrs < nNZrsB; iNZrs++, exList_rs+=4) {
476 UNPACK_EXCITATIONLIST_4(exList_rs, s, r, l1, sign);
477 s += rsoff;
478 r += rsoff;
479 ERIscr[iNZrs+iNZpq*nNZrsB] = moERI(r, s, p, q);
480 }
481 }
482 blas::gemm(blas::Layout::ColMajor, blas::Op::Trans, blas::Op::NoTrans,
483 nNZpqB,nDetAlt[2],nNZrsB,MatsT(1.),ERIscr,nNZrsB,
484 Dscr,nNZrsB,MatsT(0.),Gscr,nNZpqB);
485 // Scattering back to GBlk
486 for (auto j3 = 0ul; j3 < nDetAlt[2]; j3++) {
487 j = j12off + j3 * LCatAlt[2];
488 MatAdd('N','N',nNZpqB,1,MatsT(1.),Gscr+j3*nNZpqB,nNZpqB,
489 MatsT(1.),GBlk+j*nNZpqB,nNZpqB,GBlk+j*nNZpqB,nNZpqB);
490 }
491 }
492 // Case 3: three RAS spaces are involved in excitation
493 // implicitly that pBlk != rBlk, qBlk != sBlk
494 } else if (eCase == 3) {
495 // LCatAlt == LCat, nDetAlt == nDet
496 #pragma omp for schedule(static, 1)
497 for (auto j3 = 0ul; j3 < nDet[2]; j3++)
498 for (auto j2 = 0ul; j2 < nDet[1]; j2++)
499 for (auto j1 = 0ul; j1 < nDet[0]; j1++) {
500 Jras[0] = j1;
501 Jras[1] = j2;
502 Jras[2] = j3;
