49 template <typename MatsT, typename IntsT>
50 void MOIntsTransformer<MatsT,IntsT>::subsetTransformTPISSFockN6(EMPerturbation & pert,
51 const std::vector<std::pair<size_t,size_t>> &off_sizes, MatsT* MOTPI,
52 const std::string & moType, bool cacheIntermediates,
53 TPI_TRANS_DELTA_TYPE delta) {
54
55 size_t poff = off_sizes[0].first;
56 size_t qoff = off_sizes[1].first;
57 size_t roff = off_sizes[2].first;
58 size_t soff = off_sizes[3].first;
59 size_t np = off_sizes[0].second;
60 size_t nq = off_sizes[1].second;
61 size_t nr = off_sizes[2].second;
62 size_t ns = off_sizes[3].second;
63
64 size_t nAO = ss_.nAlphaOrbital() * ss_.nC;
65
66 // Initial Batching if can not hold all the HalfTransTPI
67
68 size_t npq = (delta == KRONECKER_DELTA_PQ or
69 delta == KRONECKER_DELTA_PQ_RS ) ? np: np*nq;
70
71 size_t nrs = (delta == KRONECKER_DELTA_RS or
72 delta == KRONECKER_DELTA_PQ_RS ) ? nr: nr*ns;
73
74 size_t halfTMOTPISize = nAO * nAO * std::min(npq, nrs);
75
76 auto nBatch = memManager_.max_avail_allocatable<MatsT>(halfTMOTPISize);
77
78 // clear cache for more memory
79 if ( nBatch <= 1) {
80 ints_cache_.clear();
81 nBatch = memManager_.max_avail_allocatable<MatsT>(halfTMOTPISize);
82 }
83
84 // No batch at this level if remaining memory can allocate at
85 // least two halfTransTPI
86 if (nBatch > 1) {
87 performSubsetTransformTPISSFockN6(pert, off_sizes, MOTPI,
88 moType, cacheIntermediates, delta);
89 return;
90 }
91
92 // TODO: preserve the rs symmetry if presented
93 // simple batch over s here
94 size_t nsMax = ns;
95 halfTMOTPISize = nAO * nAO * nr * nsMax;
96 nBatch = memManager_.max_avail_allocatable<MatsT>(halfTMOTPISize);
97
98 while (nBatch < 1) {
99
100 nsMax /= 2;
101
102 std::cout << "nsMax = " << nsMax << std::endl;
103
104 if (nsMax < 1) {
105 std::cout << "Memory not enough to batch over last index" << std::endl;
106 double mem_avail = this->memManager_.template max_avail_allocatable<double>()
107 * sizeof(double) / 1e9;
108 double mem_needed = nAO * nAO * nr * sizeof(MatsT) / 1e9;
109
110 std::cout << " - Memory available before batch integral transformation: "
111 << mem_avail << " GB" << std::endl;
112 std::cout << " - Memory needed at least for nsMax = 1: "
113 << mem_needed << " GB" << std::endl;
114
115 throw std::bad_alloc();
116 }
117
118 halfTMOTPISize = nAO * nAO * nr * nsMax;
119 nBatch = memManager_.max_avail_allocatable<MatsT>(halfTMOTPISize);
120 }
121
122 // balance batching inside the work loop
123 if (nsMax > 1) {
124 size_t nMatsTAvail = memManager_.max_avail_allocatable<MatsT>();
125 // with extra memory as 20%
126 size_t fockGDSCRSize = ss_.fockBuilder->formRawGDSCRSizePerBatch(ss_, false, false) * 1.2;
127
128 // try different nsMax and use the smallest total batch
129 size_t nrnsMax = nr * nsMax;
130 size_t maxNBatch = (nMatsTAvail - halfTMOTPISize)/fockGDSCRSize;
131 size_t minTotalBatch = ((nrnsMax - 1)/maxNBatch + 1) * nBatch;
132 size_t bestnsMax = nsMax;
133 size_t curTotalBatch = minTotalBatch;
134
135 for (auto i = 1ul; i < nsMax; i++) {
136 nrnsMax = nr * i;
137 nBatch = (ns - 1) / i + 1;
138 halfTMOTPISize = nAO * nAO * nr * i;
139 maxNBatch = (nMatsTAvail - halfTMOTPISize)/fockGDSCRSize;
140 curTotalBatch = ((nrnsMax - 1)/maxNBatch + 1) * nBatch;
141 if (curTotalBatch < minTotalBatch) {
142 minTotalBatch = curTotalBatch;
143 bestnsMax = i;
144 }
145 }
146
147 // update nsMax
148 nsMax = bestnsMax;
149 halfTMOTPISize = nAO * nAO * nr * nsMax;
150 nBatch = (ns - 1) / nsMax + 1;
151 }
152
153 std::cout << "* Batch over last indice s: with maxNs = " << nsMax << std::endl;
154
155 size_t npqr = np * nq * nr;
156
157 for (auto si = 0ul, sioff = soff, iBatch = 1ul; si < ns; iBatch++) {
158
159 size_t nsi = std::min(nsMax, ns - si);
160
161 std::cout << " - Batch " << iBatch << ": s range from "
162 << std::setw(5) << sioff + 1 << " ~ "
163 << std::setw(5) << sioff + nsi <<std::endl;
164
165 std::vector<std::pair<size_t,size_t>> batch_off_sizes = off_sizes;
166 batch_off_sizes[3].first = sioff;
167 batch_off_sizes[3].second = nsi;
168
169 performSubsetTransformTPISSFockN6(pert, batch_off_sizes, MOTPI + npqr * si,
170 moType, false, delta);
171
172 si += nsi;
173 sioff += nsi;
174 }
175
176 return;
177
178 } // subsetTransformTPISSFockN6

184 template <typename MatsT, typename IntsT>
185 void MOIntsTransformer<MatsT,IntsT>::performSubsetTransformTPISSFockN6(EMPerturbation & pert,
186 const std::vector<std::pair<size_t,size_t>> &off_sizes, MatsT* MOTPI,
187 const std::string & moType, bool cacheIntermediates,
188 TPI_TRANS_DELTA_TYPE delta) {
189
190 size_t poff = off_sizes[0].first;
191 size_t qoff = off_sizes[1].first;
192 size_t roff = off_sizes[2].first;
193 size_t soff = off_sizes[3].first;
194 size_t np = off_sizes[0].second;
195 size_t nq = off_sizes[1].second;
196 size_t nr = off_sizes[2].second;
197 size_t ns = off_sizes[3].second;
198
199 bool pqSymm = (poff == qoff) and (np == nq);
200 std::string moType_cache = "HalfTMOTPI-";
201 moType_cache += getUniqueSymbol(moType[0]);
202 moType_cache += getUniqueSymbol(moType[1]);
203 std::string rs_moType_cache = "HalfTMOTPI-";
204 rs_moType_cache += getUniqueSymbol(moType[2]);
205 rs_moType_cache += getUniqueSymbol(moType[3]);
206
207 if (delta == KRONECKER_DELTA_PQ or delta == KRONECKER_DELTA_PQ_RS)
208 moType_cache += "-delta";
209 if (delta == KRONECKER_DELTA_RS or delta == KRONECKER_DELTA_PQ_RS)
210 rs_moType_cache += "-delta";
211
212 // check pq and rs can be swapped to accelarate the computation
213 bool rsSymm = (roff == soff) and (nr == ns);
214 bool swap_pq_rs = false;
215 auto pqCache = ints_cache_.getIntegral<InCoreRITPI, MatsT>(moType_cache);
216 auto rsCache = ints_cache_.getIntegral<InCoreRITPI, MatsT>(rs_moType_cache);
217
218 if (not pqCache and rsCache) swap_pq_rs = true;
219 if (delta == KRONECKER_DELTA_RS) swap_pq_rs = true;
220
221 //if (rsSymm and not pqSymm) swap_pq_rs = true;
222
223 // swap pq rs when nrs_batch < npq_batch because first half is N6 and second is N5
224 if (not swap_pq_rs) {
225
226 size_t npq_batch = pqSymm ? np * (np + 1) / 2: np*nq;
227 size_t nrs_batch = rsSymm ? nr * (nr + 1) / 2: nr*ns;
228
229 if (delta == KRONECKER_DELTA_PQ or delta == KRONECKER_DELTA_PQ_RS)
230 npq_batch = np;
231 if (delta == KRONECKER_DELTA_RS or delta == KRONECKER_DELTA_PQ_RS)
232 nrs_batch = nr;
233
234 if (npq_batch > nrs_batch) swap_pq_rs = true;
235 }
236
237 if (swap_pq_rs) {
238 std::swap(poff, roff);
239 std::swap(qoff, soff);
240 std::swap(np, nr);
241 std::swap(nq, ns);
242 std::swap(pqSymm, rsSymm);
243 moType_cache = rs_moType_cache;
244
245 if (delta == KRONECKER_DELTA_PQ) delta = KRONECKER_DELTA_RS;
246 else if (delta == KRONECKER_DELTA_RS) delta = KRONECKER_DELTA_PQ;
247 else if (delta == KRONECKER_DELTA_PS) delta = KRONECKER_DELTA_RQ;
248 else if (delta == KRONECKER_DELTA_RQ) delta = KRONECKER_DELTA_PS;
249 }
250
251#ifdef _DEBUG_MOINTSTRANSFORMER_CACHE
252 std::cout << "moType = " << moType << ", kronecker_delta = " << delta
253 << ", swap_pq_rs = " << swap_pq_rs << std::endl;
254 std::cout << "moType_cache = " << std::setw(20) << moType_cache;
255#endif
256
257 size_t npr = np * nr;
258 size_t npq = np * nq;
259 size_t npqr = npq * nr;
260 size_t npqDim = (delta == KRONECKER_DELTA_PQ or
261 delta == KRONECKER_DELTA_PQ_RS) ? np: npq;
262
263 size_t nAO = ss_.nAlphaOrbital() * ss_.nC;
264 size_t NB = ss_.nAlphaOrbital();
265 size_t nAO2 = nAO * nAO;
266
267 MatsT * dummy_ptr = nullptr;
268 std::shared_ptr<InCoreRITPI<MatsT>> halfTMOTPI = nullptr;
269
270 if (cacheIntermediates) {
271 halfTMOTPI = ints_cache_.getIntegral<InCoreRITPI, MatsT>(moType_cache);
272#ifdef _DEBUG_MOINTSTRANSFORMER_CACHE
273 if (halfTMOTPI) std::cout << "----Find cache!!!" << std::endl;
274#endif
275 }
276
277 //
278 // 1/2 transformation to obtain halfTMOTPI(mu, nu, p, q)
279 //
280 if (not halfTMOTPI) {
281
282#ifdef _DEBUG_MOINTSTRANSFORMER_CACHE
283 std::cout << "----Not find cache, do transformation" << std::endl;
284#endif
285
286 ALLOCATE_AND_CLEAR_CACHE_IF_NECESSARY(
287 halfTMOTPI = std::make_shared<InCoreRITPI<MatsT>>(memManager_, nAO, npqDim);
288 )
289
290 SquareMatrix<MatsT> SCR(memManager_, nAO);
291 std::vector<std::shared_ptr<PauliSpinorSquareMatrices<MatsT>>> pq1PDMs, pqMOTPIs, dummy;
292 MatsT *MOTPIpq_ptr = nullptr, *density_ptr = nullptr;
293 bool is4C = ss_.nC == 4;
294 bool is2C = ss_.nC == 2;
295 bool is1C = ss_.nC == 1;
296 size_t pqSCRSize = is4C ? 2*NB: NB;
297
298 // find out maximum batch size base on current memory limit
299
300 // SCR size form fockbuilder
301 size_t fockGDSCRSize = ss_.fockBuilder->formRawGDSCRSizePerBatch(ss_, false, false);
302
303 // SCR size for spinor density and half-transformed integrals
304 fockGDSCRSize += is4C ? pqSCRSize*pqSCRSize: 2*pqSCRSize*pqSCRSize;
305 size_t maxNBatch = 0;
306 double allow_extra = 0.2;
307
308 ALLOCATE_AND_CLEAR_CACHE_IF_NECESSARY(
309 maxNBatch = memManager_.max_avail_allocatable<MatsT>(size_t(fockGDSCRSize*(1.+allow_extra)));
310 if(maxNBatch == 0)
311 CErr(" Memory is not enough for 1 density in subsetTransformTPISSFockN6");
312 )
313
314 std::vector<std::pair<size_t,size_t>> pqJobs;
315
316 if (delta == KRONECKER_DELTA_PQ or delta == KRONECKER_DELTA_PQ_RS) {
317 for (auto p = 0ul; p < np; p++) pqJobs.push_back({p,p});
318 } else {
319 for (auto q = 0ul; q < nq; q++)
320 for (auto p = 0ul; p < np; p++) {
321 pqJobs.push_back({p,q});
322 if (pqSymm and p == q) break;
323 }
324 }
325
326 size_t NJob = pqJobs.size();
327 size_t NJobComplete = 0ul;
328 size_t NJobToDo = 0ul;
329
330#ifdef _DEBUG_MOINTSTRANSFORMER_CACHE
331 std::cout << "First Half transfromation with maxNBatch = " << maxNBatch << std::endl;
332#endif
333
334 while (NJobComplete < NJob) {
335
336 size_t p, q;
337 NJobToDo = std::min(maxNBatch, NJob-NJobComplete);
338
339#ifdef _DEBUG_MOINTSTRANSFORMER_CACHE
340 std::cout << "----" << std::endl;
341 std::cout << "NJob = " << NJob << std::endl;
342 std::cout << "NJobComplete = " << NJobComplete << std::endl;
343 std::cout << "NJobToDo = " << NJobToDo << std::endl;
344#endif
345
346 // build fake densities in batch
347 for (auto i = 0ul; i < NJobToDo; i++) {
348
349 p = pqJobs[i + NJobComplete].first;
350 q = pqJobs[i + NJobComplete].second;
351
352 pq1PDMs.push_back(std::make_shared<PauliSpinorSquareMatrices<MatsT>>(memManager_, pqSCRSize, is4C, is4C));
353
354 // 4C will reuse pq1PDM as densities are component scattered anyways
355 if (is4C) pqMOTPIs.push_back(pq1PDMs.back());
356 else pqMOTPIs.push_back(std::make_shared<PauliSpinorSquareMatrices<MatsT>>(memManager_, pqSCRSize, is4C, is4C));
357
358 if (is1C) {
359 density_ptr = pq1PDMs.back()->S().pointer();
360 } else {
361 density_ptr = SCR.pointer();
362 }
363
364 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::ConjTrans,
365 nAO, nAO, 1, MatsT(1.), ss_.mo[0].pointer() + (q+qoff)*nAO, nAO,
366 ss_.mo[0].pointer() + (p+poff)*nAO, nAO, MatsT(0.), density_ptr, nAO);
367
368 if (not is1C) *pq1PDMs.back() = SCR.template spinScatter<MatsT>(is4C, is4C);
369
370 }
371
372 // do contraction to get half-transformed integrals
373 if (is1C) ss_.fockBuilder->formRawGDInBatches(ss_, pert, false, 0., false, pq1PDMs, pqMOTPIs, dummy, dummy);
374 else ss_.fockBuilder->formRawGDInBatches(ss_, pert, false, 0., false, pq1PDMs, dummy, dummy, pqMOTPIs);
375
376 // copy over to SCR
377 for (auto i = 0ul; i < NJobToDo; i++) {
378
379 if (not is1C) {
380 SCR = pqMOTPIs[i]->template spinGather<MatsT>();
381 MOTPIpq_ptr = SCR.pointer();
382 } else MOTPIpq_ptr = pqMOTPIs[i]->S().pointer();
383
384 p = pqJobs[i + NJobComplete].first;
385
386 if (delta == KRONECKER_DELTA_PQ or delta == KRONECKER_DELTA_PQ_RS) {
387 SetMat('N', nAO, nAO, MatsT(1.), MOTPIpq_ptr, nAO, halfTMOTPI->pointer() + p*nAO2, nAO);
388 } else {
389 q = pqJobs[i + NJobComplete].second;
390 SetMat('N', nAO, nAO, MatsT(1.), MOTPIpq_ptr, nAO, halfTMOTPI->pointer() + (p + q*np)*nAO2, nAO);
391 if (pqSymm and p < q)
392 SetMat('C', nAO, nAO, MatsT(1.), MOTPIpq_ptr, nAO, halfTMOTPI->pointer() + (q + p*np)*nAO2, nAO);
393 }
394 }
395
396 // increment and clear SCR after job done
397 NJobComplete += NJobToDo;
398 pq1PDMs.clear();
399 pqMOTPIs.clear();
400
401 } // main loop
402
403 if (cacheIntermediates) ints_cache_.addIntegral(moType_cache, halfTMOTPI);
404
405 } // if there is no cache
406
407 // allocate intermediate SCR
408 MatsT * SCR = nullptr;
409 size_t SCRSize;
410 if (delta == NO_KRONECKER_DELTA) SCRSize = nAO*npqr;
411 else if (delta == KRONECKER_DELTA_PQ) SCRSize = nAO*np*nr;
412 else if (delta == KRONECKER_DELTA_RS) SCRSize = nAO*npq;
413 else if (delta == KRONECKER_DELTA_PS) SCRSize = nAO*npqr;
414 else if (delta == KRONECKER_DELTA_RQ) SCRSize = nAO*npr;
415 else if (delta == KRONECKER_DELTA_PQ_RS) SCRSize = nAO*np;
416 else if (delta == KRONECKER_DELTA_PS_RQ) SCRSize = nAO*npr;
417
418 SCR = memManager_.malloc<MatsT>(SCRSize);
419
420 if (delta == NO_KRONECKER_DELTA or delta == KRONECKER_DELTA_PQ) {
421
422 char TransMOTPI = swap_pq_rs ? 'N': 'T';
423 size_t ndim = delta == NO_KRONECKER_DELTA ? npq: np;
424
425 PairTransformation('N', ss_.mo[0].pointer(), nAO, roff, soff,
426 'N', halfTMOTPI->pointer(), nAO, nAO, ndim,
427 TransMOTPI, MOTPI, nr, ns, dummy_ptr, SCR, false);
428
429 } else if (delta == KRONECKER_DELTA_RS or delta == KRONECKER_DELTA_PQ_RS) {
430
431 for (auto r = 0ul; r < nr; r++) {
432
433 auto rMO = ss_.mo[0].pointer() + (r + roff) * nAO;
434
435 // SCR(nu p q, r) = (mu, nu p q)^H MO(mu, r)
436 blas::gemm(blas::Layout::ColMajor, blas::Op::ConjTrans, blas::Op::NoTrans,
437 nAO*npqDim, 1, nAO, MatsT(1.), halfTMOTPI->pointer(), nAO,
438 rMO, nAO, MatsT(0.), SCR, nAO*npqDim);
439
440 // MOTPI(p q, r) = SCR(nu, p q)^H MO(nu, r)
441 blas::gemm(blas::Layout::ColMajor, blas::Op::ConjTrans, blas::Op::NoTrans,
442 npqDim, 1, nAO, MatsT(1.), SCR, nAO,
443 rMO, nAO, MatsT(0.), MOTPI + r*npqDim, npqDim);
444
445 }
446
447 if(swap_pq_rs) IMatCopy('T', npqDim, nr, MatsT(1.), MOTPI, npqDim, nr);
448
449 } else if (delta == KRONECKER_DELTA_PS) {
450
451 // SCR(nu p q, r) = (mu, nu p q)^H MO(mu, r)
452 blas::gemm(blas::Layout::ColMajor, blas::Op::ConjTrans, blas::Op::NoTrans,
453 nAO*npq, nr, nAO, MatsT(1.), halfTMOTPI->pointer(), nAO,
454 ss_.mo[0].pointer() + roff*nAO, nAO, MatsT(0.), SCR, nAO*npq);
455
456 MatsT * SCR2 = nullptr;
457 SCR2 = memManager_.malloc<MatsT>(nAO*nq*nr);
458
459 size_t nqr = nq*nr;
460
461 for (auto p = 0ul; p < np; p++) {
462
463 // Gather SCR2(nu, q, r) from SCR(nu p q, r)
464 #pragma omp parallel for
465 for (auto r = 0ul; r < nr; r++)
466 for (auto q = 0ul; q < nq; q++) {
467 std::copy_n(SCR + p*nAO + q*nAO*np + r*nAO*npq,
468 nAO, SCR2 + q*nAO + r*nAO*nq);
469 }
470
471 // MOTPI(q r, p) = SCR2(nu, q r)^H MO(nu, p)
472 blas::gemm(blas::Layout::ColMajor, blas::Op::ConjTrans, blas::Op::NoTrans,
473 nqr, 1, nAO, MatsT(1.), SCR2, nAO,
474 ss_.mo[0].pointer() + (p+poff)*nAO, nAO,
475 MatsT(0.), MOTPI + p*nqr, nqr);
476 }
477
478 // MOTPI(q r, p) -> MOTPI(p,q,r)
479 if (not swap_pq_rs) IMatCopy('T', nqr, np, MatsT(1.), MOTPI, nqr, np);
480 // MOTPI(q r, p) -> MOTPI(r,p,q)
481 else IMatCopy('T', nq, nr*np, MatsT(1.), MOTPI, nq, nr*np);
482
483 if (SCR2) memManager_.free(SCR2);
484
485 } else if (delta == KRONECKER_DELTA_RQ or delta == KRONECKER_DELTA_PS_RQ) {
486
487 size_t nAOp = nAO * np;
488 for (auto r = 0ul; r < nr; r++) {
489 // SCR(nu p, r) = (mu, nu p)^H MO(mu, r)
490 blas::gemm(blas::Layout::ColMajor, blas::Op::ConjTrans, blas::Op::NoTrans,
491 nAOp, 1, nAO, MatsT(1.), halfTMOTPI->pointer() + r*nAO2*np, nAO,
492 ss_.mo[0].pointer() + (r+roff)*nAO, nAO, MatsT(0.), SCR+r*nAOp, nAOp);
493 }
494
495 if (delta == KRONECKER_DELTA_RQ) {
496 // MOTPI(p,r,s) = SCR(nu p, r)^H MO(nu, s)
497 blas::gemm(blas::Layout::ColMajor, blas::Op::ConjTrans, blas::Op::NoTrans,
498 npr, ns, nAO, MatsT(1.), SCR, nAO,
499 ss_.mo[0].pointer() + soff*nAO, nAO, MatsT(0.), MOTPI, npr);
500
501 // MOTPI(p,r,s) -> MOTPI(r,s,p)
502 if (swap_pq_rs) IMatCopy('T', np, ns*nr, MatsT(1.), MOTPI, np, ns*nr);
503
504 } else if (delta == KRONECKER_DELTA_PS_RQ) {
505
506 MatsT * SCR2 = nullptr;
507 SCR2 = memManager_.malloc<MatsT>(nAO*nr);
508
509 for (auto p = 0ul; p < np; p++) {
510
511 // Gather SCR2(nu, r) from SCR(nu, p, r)
512 #pragma omp parallel for
513 for (auto r = 0ul; r < nr; r++) {
514 std::copy_n(SCR + p*nAO + r*nAO*np, nAO, SCR2 + r*nAO);
515 }
516
517 // MOTPI(r,p) = SCR2(nu, r)^H MO(nu, p)
518 blas::gemm(blas::Layout::ColMajor, blas::Op::ConjTrans, blas::Op::NoTrans,
519 nr, 1, nAO, MatsT(1.), SCR2, nAO,
520 ss_.mo[0].pointer() + (p+poff)*nAO, nAO, MatsT(0.), MOTPI + p*nr, nr);
521
522 }
523
524 // MOTPI(r,p) -> MOTPI(p,r)
525 if (not swap_pq_rs) IMatCopy('T', nr, np, MatsT(1.), MOTPI, nr, np);
526
527 if (SCR2) memManager_.free(SCR2);
528 }
529
530 } // delta
531
532 // free memories
533 halfTMOTPI = nullptr;
534 if (SCR) memManager_.free(SCR);
535
536 }; // MOIntsTransformer::perfromSubsetTransformTPISSFockN6
