58 template <typename MatsT, typename IntsT>
59 void InCore4indexTPIContraction<MatsT, IntsT>::twoBodyContract(
60 MPI_Comm comm,
61 const bool,
62 std::vector<TwoBodyContraction<MatsT>> &list,
63 EMPerturbation&) const {
64 ROOT_ONLY(comm);
65
66 if (typeid(*this) == typeid(InCore4indexRelERIContraction<MatsT,IntsT>)
67 and typeid(this->ints_) != typeid(InCore4indexRelERI<IntsT>))
68 CErr("InCore4indexRelTPIContraction expect a InCore4indexRelTPI reference.");
69
70 if (typeid(*this) == typeid(InCore4indexTPIContraction<MatsT,IntsT>))
71 try {
72 dynamic_cast<InCore4indexTPI<IntsT>&>(this->ints_);
73 } catch (const std::bad_cast&) {
74 CErr("InCore4indexTPIContraction expect a InCore4indexTPI reference.");
75 }
76
77 if (typeid(*this) == typeid(InCoreRITPIContraction<MatsT,IntsT>))
78 try {
79 dynamic_cast<InCoreRITPI<IntsT>&>(this->ints_);
80 } catch (const std::bad_cast&) {
81 CErr("InCoreRITPIContraction expect a InCoreRITPI reference.");
82 }
83
84 ProgramTimer::timeOp("Contraction Total", [&](){
85
86 // Loop over matricies to contract with
87 for(auto &C : list) {
88
89 // Coulomb-type (34,12) ERI contraction
90 // AX(mn) = (mn | kl) X(kl)
91 if( C.contType == TWOBODY_CONTRACTION_TYPE::COULOMB ) {
92 JContract(comm,C);
93 // Exchange-type (23,12) ERI contraction
94 // AX(mn) = (mk |ln) X(kl)
95 } else if( C.contType == TWOBODY_CONTRACTION_TYPE::EXCHANGE ) {
96 KContract(comm,C);
97 }
98
99 } // loop over matricies
100
101 });
102
103 } // InCore4indexTPIContraction::twoBodyContract

110 template <typename MatsT, typename IntsT>
111 void InCore4indexTPIContraction<MatsT, IntsT>::JContract(
112 MPI_Comm, TwoBodyContraction<MatsT> &C) const {
113
114 ProgramTimer::tick("J Contract");
115
116 InCore4indexTPI<IntsT> &tpi4I =
117 dynamic_cast<InCore4indexTPI<IntsT>&>(this->ints_);
118 CQMemManager& memManager_ = tpi4I.memManager();
119 size_t NB = tpi4I.nBasis();
120 size_t NB2 = NB*NB;
121 size_t sNB = tpi4I.snBasis();
122 size_t sNB2 = sNB*sNB;
123
124 // need to swap NB and sNB if contraction is done in aux
125 if (this->contractSecond) {
126 std::swap(NB,sNB);
127 std::swap(NB2,sNB2);
128 }
129
130 const bool sameIntsTMatsT = std::is_same<IntsT,MatsT>::value;
131
132 // for Hermitian densities, output Coulomb Matrix is same type as IntsT
133 if (C.HER or sameIntsTMatsT) {
134
135 IntsT *X = reinterpret_cast<IntsT*>(C.X);
136 IntsT *AX = reinterpret_cast<IntsT*>(C.AX);
137
138 // Extract the real part of X if X is Hermetian and if the ints are
139 // real
140 const bool extractRealPartX =
141 C.HER and std::is_same<IntsT,double>::value and
142 std::is_same<MatsT,dcomplex>::value;
143
144 // Allocate scratch if IntsT and MatsT are different
145 const bool allocAXScratch = not std::is_same<IntsT,MatsT>::value;
146
147 if( extractRealPartX ) {
148
149 X = memManager_.malloc<IntsT>(sNB2);
150 for(auto k = 0ul; k < sNB2; k++) X[k] = std::real(C.X[k]);
151 }
152
153 if( allocAXScratch ) {
154
155 AX = memManager_.malloc<IntsT>(NB2);
156 std::fill_n(AX,NB2,0.);
157
158 }
159
160
161 #ifdef _BULLET_PROOF_INCORE
162
163 size_t NB3 = NB * NB2;
164 #pragma omp parallel for
165 for(auto i = 0; i < NB; ++i)
166 for(auto j = 0; j < NB; ++j)
167 for(auto k = 0; k < sNB; ++k)
168 for(auto l = 0; l < sNB; ++l)
169
170 AX[i + j*NB] += tpi4I.pointer()[i+j*NB+k*NB2+l*NB3] * X[l + k*NB];
171
172 #else
173
174 if( std::is_same<IntsT,dcomplex>::value )
175 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB2,1,sNB2,IntsT(1.),tpi4I.pointer(),NB2,X,sNB2,IntsT(0.),AX,NB2);
176 else
177 if (not this->contractSecond)
178 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB2,1,sNB2,IntsT(1.),tpi4I.pointer(),NB2,X,sNB2,IntsT(0.),AX,NB2);
179 else
180 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB2,1,sNB2,IntsT(1.),tpi4I.pointer(),sNB2,X,sNB2,IntsT(0.),AX,NB2);
181
182 // if Complex ints + Hermitian, conjugate
183 if( std::is_same<IntsT,dcomplex>::value and C.HER )
184 IMatCopy('R',NB,NB,IntsT(1.),AX,NB,NB);
185
186 // If non-hermetian, transpose
187 if( not C.HER ) {
188
189 IMatCopy('T',NB,NB,IntsT(1.),AX,NB,NB);
190
191 }
192
193 #endif
194
195 // Cleanup temporaries
196 if( extractRealPartX ) memManager_.free(X);
197 if( allocAXScratch ) {
198
199 std::copy_n(AX,NB2,C.AX);
200 memManager_.free(AX);
201
202 }
203
204 // for non-hermiatin and MatsT = dcomplex, IntsT = double
205 } else {
206 if (not this->contractSecond)
207 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB2,1,sNB2,MatsT(1.),tpi4I.pointer(),NB2,C.X,sNB2,MatsT(0.),C.AX,NB2);
208 else
209 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB2,1,sNB2,MatsT(1.),tpi4I.pointer(),sNB2,C.X,sNB2,MatsT(0.),C.AX,NB2);
210 }
211
212 ProgramTimer::tock("J Contract");
213
214 }; // InCore4indexTPIContraction::JContract

222 InCore4indexTPI<IntsT> &tpi4I =

224 // check to see whether the two basis sets are different

228

246
247 #pragma omp parallel for
248 for(auto nu = 0; nu < NB; nu++)

252
253 #endif
254
255 ProgramTimer::tock("K Contract");

264 template <typename MatsT, typename IntsT>
265 void InCore4indexRelERIContraction<MatsT, IntsT>::JContract(
266 MPI_Comm, TwoBodyContraction<MatsT> &C) const {
267
268 InCore4indexRelERI<IntsT> &tpi4I =
269 dynamic_cast<InCore4indexRelERI<IntsT>&>(this->ints_);
270 size_t NB = tpi4I.nBasis();
271 size_t NB2 = NB * NB;
272 size_t NB3 = NB * NB2;
273
274 if( C.ERI4 == nullptr ) C.ERI4 = reinterpret_cast<double*>(tpi4I.pointer());
275
276 memset(C.AX,0.,NB2*sizeof(MatsT));
277
278 if( C.intTrans == TRANS_KL ) {
279
280 // D(μν) = D(λκ)(μν|[κλ]^T) = D(λκ)(μν|λκ)
281 #pragma omp parallel for
282 for(auto m = 0; m < NB; ++m)
283 for(auto n = 0; n < NB; ++n)
284 for(auto k = 0; k < NB; ++k)
285 for(auto l = 0; l < NB; ++l) {
286
287 C.AX[m + n*NB] += C.ERI4[m + n*NB + l*NB2 + k*NB3] * C.X[l + k*NB];
288
289 }
290
291 } else if (C.intTrans == TRANS_MN_TRANS_KL) {
292
293 // D(μν) = D(λκ)([μν]^T|[κλ]^T) = D(λκ)(νμ|λκ)
294 #pragma omp parallel for
295 for(auto m = 0; m < NB; ++m)
296 for(auto n = 0; n < NB; ++n)
297 for(auto k = 0; k < NB; ++k)
298 for(auto l = 0; l < NB; ++l) {
299
300 C.AX[m + n*NB] += C.ERI4[n + m*NB + l*NB2 + k*NB3]*C.X[l + k*NB];
301
302 }
303
304 } else if (C.intTrans == TRANS_MN) {
305
306 // D(μν) = D(λκ)([μν]^T|κλ) = D(λκ)(νμ|κλ)
307 #pragma omp parallel for
308 for(auto m = 0; m < NB; ++m)
309 for(auto n = 0; n < NB; ++n)
310 for(auto k = 0; k < NB; ++k)
311 for(auto l = 0; l < NB; ++l) {
312
313 C.AX[m + n*NB] += C.ERI4[n + m*NB + k*NB2 + l*NB3]*C.X[l + k*NB];
314
315 }
316
317 } else if( C.intTrans == TRANS_MNKL ) {
318
319 // D(μν) = D(λκ)(μν|κλ)^T = D(λκ)(κλ|μν)
320 #pragma omp parallel for
321 for(auto m = 0; m < NB; ++m)
322 for(auto n = 0; n < NB; ++n)
323 for(auto k = 0; k < NB; ++k)
324 for(auto l = 0; l < NB; ++l) {
325
326 C.AX[m + n*NB] += C.ERI4[k + l*NB + m*NB2 + n*NB3]*C.X[l + k*NB];
327
328 }
329
330 } else if( C.intTrans == TRANS_NONE ) {
331
332 // D(μν) = D(λκ)(μν|κλ)
333 #pragma omp parallel for
334 for(auto m = 0; m < NB; ++m)
335 for(auto n = 0; n < NB; ++n)
336 for(auto k = 0; k < NB; ++k)
337 for(auto l = 0; l < NB; ++l) {
338
339 C.AX[m + n*NB] += C.ERI4[m + n*NB + k*NB2 + l*NB3] * C.X[l + k*NB];
340
341 }
342
343 }
344
345
346 }; // InCore4indexRelERIContraction::JContract

533 template <typename MatsT, typename IntsT>
534 void InCoreRITPIContraction<MatsT, IntsT>::KCoefContract(
535 MPI_Comm comm, size_t NO, MatsT *C, MatsT *AX) const {
536 ROOT_ONLY(comm);
537
538 InCoreRITPI<IntsT> &eri3j =
539 dynamic_cast<InCoreRITPI<IntsT>&>(this->ints_);
540 CQMemManager& memManager_ = eri3j.memManager();
541 size_t NB = eri3j.nBasis();
542 size_t NBRI = eri3j.nRIBasis();
543 size_t NBNBRI = NB*NBRI;
544 size_t NONBRI= NO*NBRI;
545
546 MatsT *Btemp1 = memManager_.malloc<MatsT>(NBNBRI*NO);
547 MatsT *Btemp2 = memManager_.malloc<MatsT>(NBNBRI*NO);
548
549 // 1. Bt1(i, L | nu) = C(lambda, i)^H @ B(L, lambda | nu)^T
550 size_t LAThreads = GetLAThreads();
551 SetLAThreads(1);
552 #pragma omp parallel for
553 for(auto nu = 0ul; nu < NB; nu++)
554 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::Trans,NO,NBRI,NB,MatsT(1.),C,NB,
555 eri3j.pointer()+nu*NBNBRI,NBRI,
556 MatsT(0.),Btemp1+nu*NONBRI,NO);
557 SetLAThreads(LAThreads);
558
559 // 2. Bt2(i, L mu) = C(sigma, i)^T @ B(L mu, sigma)^T
560 blas::gemm(blas::Layout::ColMajor,blas::Op::Trans,blas::Op::Trans,NO,NBNBRI,NB,MatsT(1.),C,NB,
561 reinterpret_cast<MatsT*>(eri3j.pointer()),NBNBRI,
562 MatsT(0.),Btemp2,NO);
563
564 // 3. K(mu, nu) = Bt2(i L, mu)^T @ Bt1(i L, nu)
565 blas::gemm(blas::Layout::ColMajor,blas::Op::Trans,blas::Op::NoTrans,NB,NB,NONBRI,MatsT(1.),Btemp2,NONBRI,Btemp1,NONBRI,
566 MatsT(0.),AX,NB);
567
568 memManager_.free(Btemp1, Btemp2);
569
570 }; // InCoreRIERIContraction::KCoefContract

572 template <>
573 void InCoreRITPIContraction<dcomplex, double>::KCoefContract(
574 MPI_Comm comm, size_t NO, dcomplex *C, dcomplex *AX) const {
575 ROOT_ONLY(comm);
576
577 InCoreRITPI<double> &eri3j =
578 dynamic_cast<InCoreRITPI<double>&>(this->ints_);
579 CQMemManager& memManager_ = eri3j.memManager();
580 size_t NB = eri3j.nBasis();
581 size_t NBRI = eri3j.nRIBasis();
582 size_t NBNBRI = NB*NBRI;
583 size_t NONBRI= NO*NBRI;
584
585 dcomplex *Btemp1 = memManager_.malloc<dcomplex>(NBNBRI*NO);
586 dcomplex *Btemp2 = memManager_.malloc<dcomplex>(NBNBRI*NO);
587 dcomplex *Btemp3 = memManager_.malloc<dcomplex>(NBNBRI*NO);
