30 template <template <typename, typename> class _SSTyp, typename IntsT>
31 template <typename MatsT>
32 void RealTime<_SSTyp,IntsT>::alloc() {
33
34 // XXX: Member functions can't be partially specialized,
35 // so we're just going to cram this all into a single if statement...
36 if( std::is_same<NEOSS<dcomplex,IntsT>,_SSTyp<dcomplex,IntsT>>::value ) {
37 auto prop_c = dynamic_cast<NEOSS<dcomplex,IntsT>*>(&propagator_);
38 auto map = prop_c->getSubsystemMap();
39 auto order = prop_c->getOrder();
40
41 assert( !map.empty() );
42
43 // Loop over all subsystems
44 for( auto& label: order ) {
45
46 // Easier to read name for the subsystem
47 auto& system = map[label];
48
49 // Information for RealTime only
50 size_t NB = system->onePDM->dimension();
51 bool hasZ = system->onePDM->hasZ();
52 bool hasXY= system->onePDM->hasXY();
53
54 systems_.push_back(system.get());
55
56 DOSav.push_back(
57 std::make_shared<PauliSpinorSquareMatrices<dcomplex>>(
58 memManager_, NB, hasXY, hasZ
59 ));
60 UH.push_back(
61 std::make_shared<PauliSpinorSquareMatrices<dcomplex>>(
62 memManager_, NB, hasXY, hasZ
63 ));
64
65 }
66
67 }
68 else {
69
70 size_t NB = propagator_.onePDM->dimension();
71 bool hasZ = propagator_.onePDM->hasZ();
72 bool hasXY= propagator_.onePDM->hasXY();
73
74 systems_.push_back(&propagator_);
75
76 DOSav.emplace_back(
77 std::make_shared<PauliSpinorSquareMatrices<dcomplex>>(
78 memManager_, NB, hasXY, hasZ
79 ));
80 UH.emplace_back(

85 }
86
87 };
