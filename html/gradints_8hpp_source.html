113 // Integral matrix type converter constructor
114 template <typename IntsU>
115 GradInts( const GradInts<IntClass,IntsU> &other, int = 0 ):
116 ParticleIntegrals(other), nAtoms_(other.nAtoms_) {
117
118 if (std::is_same<IntsU, dcomplex>::value
119 and std::is_same<IntsT, double>::value)
120 CErr("Cannot create a Real GradInts from a Complex one.");

122 components_.reserve(other.components_.size());

198 oeiStr = "Gradient Integral";
199 else

201 out << oeiStr << " with " << nAtoms_ << " atoms.";

207 // Conversion to other type
208 template <typename TransT>
209 GradInts<IntClass, typename std::conditional<
210 (std::is_same<IntsT, dcomplex>::value or
211 std::is_same<TransT, dcomplex>::value),
212 dcomplex, double>::type> transform(
213 char TRANS, const TransT* T, int NT, int LDT) const {
214
215 GradInts<IntClass, typename std::conditional<
216 (std::is_same<IntsT, dcomplex>::value or
217 std::is_same<TransT, dcomplex>::value),
218 dcomplex, double>::type> transInts(
219 memManager(), NT, nAtoms_);
220
221 for (size_t i = 0; i < components_.size(); i++)
222 transInts[i] = (*this)[i].transform(TRANS, T, NT, LDT);
223
224 return transInts;
225 }
