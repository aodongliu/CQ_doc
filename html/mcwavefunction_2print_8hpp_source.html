33 template <typename MatsT, typename IntsT>
34 void MCWaveFunction<MatsT,IntsT>::printMOSpacePatition() {
35
36 auto & mopart = this->MOPartition;
37 auto & ref = this->reference();
38
39 std::cout << std::left << std::endl;
40 FormattedLine(std::cout,"* MO Space Partition:");
41 FormattedLine(std::cout," Number of Electronic MOs:", mopart.nElecMO);
42 if (ref.nC == 4)
43 FormattedLine(std::cout," Number of Negative MOs:", mopart.nNegMO);
44 FormattedLine(std::cout," Number of Electrons in Full Space:", ref.nO);
45
46 std::cout << std::endl;
47 FormattedLine(std::cout," Number of Inactive Orbitals:", mopart.nInact);
48 FormattedLine(std::cout," Number of Frozen Virtual Orbitals:", mopart.nFVirt);
49
50 FormattedLine(std::cout," Number of Correlated Orbitals:", mopart.nCorrO);
51 if (mopart.scheme == RAS) {
52 FormattedLine(std::cout," Number of Orbitals in RAS 1:", mopart.nActOs[0]);
53 FormattedLine(std::cout," Number of Orbitals in RAS 2:", mopart.nActOs[1]);
54 FormattedLine(std::cout," Number of Orbitals in RAS 3:", mopart.nActOs[2]);
55 FormattedLine(std::cout," Maximum Number of Holes in RAS 1:", mopart.mxHole);
56 FormattedLine(std::cout," Maximum Number of Electrons in RAS 3:", mopart.mxElec);
57 }
58
59 std::cout << std::endl;
60 FormattedLine(std::cout," Number of Correlated Electrons:", mopart.nCorrE);
61 if (ref.nC == 1) {
62 FormattedLine(std::cout," Number of Correlated Alpha Electrons:", mopart.nCorrEA);
63 FormattedLine(std::cout," Number of Correlated Beta Electrons:", mopart.nCorrEB);
64 }
65
66 std::cout << std::endl;
67 FormattedLine(std::cout," Number of Roots Requested:",this->NStates);
68 FormattedLine(std::cout," Number of Determinants:", this->NDet);
69
70 std::cout << std::endl;
71 std::cout << std::right << std::endl;
72 this->mointsTF->printMORangesSummary();
73 std::cout << std::left << std::endl;
74
75 }; // MCWaveFunction::printMOSpacePatition

77 template <typename MatsT, typename IntsT>
78 void MCWaveFunction<MatsT,IntsT>::print1RDMs() {
79
80 if( printRDMs==0 ) return;
81
82 std::cout << std::endl << "MCWaveFunction 1RDMs:" << std::endl;
83 std::cout << bannerTop << std::endl;
84
85 size_t nCorrO = this->MOPartition.nCorrO;
86 size_t nS = this->NStates;
87 size_t nRDMEle= 0; // used for line breaker
88
89 //Print full 1RDM
90 if( printRDMs==1 ){
91 std::cout << "Printing full 1-RDM" << std::endl;
92 for (auto i = 0ul; i < nS; i++) {
93
94 prettyPrintSmart(std::cout, "State " + std::to_string(i),
95 this->oneRDM[i].pointer(), nCorrO, nCorrO, nCorrO);
96
97 }
98 } else if( printRDMs==2 ){
99 std::cout.precision(2);
100 std::cout << "Printing large (>" << std::fixed << rdmCut << ") diagonal elements of real 1-RDM" << std::endl;
101
102 for (auto i = 0ul; i < nS; i++) {
103
104 std::cout << std::endl << "State " << i+1 << ": ";
105 nRDMEle = 0;
106
107 for( auto ipp = 0ul; ipp < nCorrO; ipp++){
108
109 if( std::real(this->oneRDM[i].pointer()[ipp*nCorrO+ipp]) > rdmCut ){
110 if( nRDMEle > 6 ){
111 std::cout << std::endl << " ";
112 nRDMEle = 0;
113 }
114 std::cout << std::setw(3) << ipp+1 << "(" << std::real(this->oneRDM[i].pointer()[ipp*nCorrO+ipp]) << ")" << " ";
115 nRDMEle++;
116 }
117 }
118
119 std::cout << std::endl;
120
121 }
122 }
123
124 std::cout << bannerTop << std::endl;
125
126 }; //MCWaveFunction::print1RDMs
