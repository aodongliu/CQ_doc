111 // Computation interfaces
112 virtual void computeAOInts(BasisSet &basisSet, Molecule &mol,
113 EMPerturbation &emPert, OPERATOR op, const HamiltonianOptions &hamiltonianOptions) {
114
115 if(hamiltonianOptions.Libcint) {
116
117 // Use Libcint to compute nonrelativistic and DCB integrals
118 computeERICINT(basisSet, mol, emPert, op, hamiltonianOptions);
119
120 //if( hamiltonianOptions.Gauge )
121 //computeERIGauge(basisSet, mol, emPert, op, hamiltonianOptions);
122
123 } else {
124
125 // Use Libint to compute nonrelativistic
126 InCore4indexTPI<IntsT>::computeAOInts(basisSet, mol, emPert, op, hamiltonianOptions);
127
128 // Use Libint to compute DCB integrals
129 if (hamiltonianOptions.DiracCoulomb or hamiltonianOptions.Gaunt)
130 computeERIDCB(basisSet, mol, emPert, op, hamiltonianOptions);
131
132 // use in house code to compute gauge integral
133 if (hamiltonianOptions.Gauge)
134 computeERIGauge(basisSet, mol, emPert, op, hamiltonianOptions);
135 }
136
137 }

179 virtual void broadcast(MPI_Comm comm = MPI_COMM_WORLD, int root = 0) override {
180 InCore4indexTPI<IntsT>::broadcast(comm, root);
181
182#ifdef CQ_ENABLE_MPI
183 if( MPISize(comm) > 1 ) {
184 size_t nRel = components_.size();
185 MPIBCast(nRel,root,comm);
186
187 if (components_.size() != nRel) {
188 components_.clear();
189 components_.reserve(nRel);
190 for (size_t i = 0; i < nRel; i++) {
191 components_.emplace_back(this->memManager_, this->NB);
192 }
193 }
194
195 for (InCore4indexTPI<IntsT>& comp : components_)
196 comp.broadcast(comm, root);
197 }
198#endif
199 }

205 template <typename TransT>
206 InCore4indexRelERI<typename std::conditional<
207 (std::is_same<IntsT, dcomplex>::value or
208 std::is_same<TransT, dcomplex>::value),
209 dcomplex, double>::type> transform(
210 char TRANS, const TransT* T, int NT, int LDT) const {
211 InCore4indexRelERI<typename std::conditional<
212 (std::is_same<IntsT, dcomplex>::value or
213 std::is_same<TransT, dcomplex>::value),
214 dcomplex, double>::type> transInts(
215 this->ParticleIntegrals::memManager(), NT, nRelComp());
216 transInts[REL_INTS_COMPS::O] =
217 (*this)[REL_INTS_COMPS::O].transform(TRANS, T, NT, LDT);
218 for (size_t i = 0; i < nRelComp(); i++)
219 transInts[i] = (*this)[i].transform(TRANS, T, NT, LDT);
220 return transInts;
221 }
