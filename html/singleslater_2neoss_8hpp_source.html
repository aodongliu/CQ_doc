228 // Properties
229 void computeEnergy() {
230
231 this->totalEnergy = 0.;
232 applyToEach([&](SubSSPtr& ss){
233 ss->computeEnergy();
234 this->totalEnergy += ss->totalEnergy - this->molecule().nucRepEnergy;
235 });
236
237 // If we're doing EPC, we've double counted the energy, so subtract it
238 // for all non-electronic systems here...
239 // XXX: If we ever have particle-particle correlation that isn't
240 // between electrons and nuclei, this will no longer work
241 applyToEach([&](SubSSPtr& ss) {
242 auto ks = std::dynamic_pointer_cast<KohnSham<MatsT,IntsT>>(ss);
243 if( ss->particle.charge < 0 || !ks ) return;
244 this->totalEnergy -= ks->XCEnergy;
245 });
246
247 this->totalEnergy += this->molecule().nucRepEnergy;
248
249 }

251 void computeMultipole(EMPerturbation& emPert) {
252 // Zeroing our Dipole, Quadrupole, and Octopole
253 for (auto iXYZ = 0; iXYZ < 3; iXYZ++) {
254
255 this->elecDipole[iXYZ] = 0.;
256
257 for (auto jXYZ = 0; jXYZ < 3; jXYZ++){
258
259 this->elecQuadrupole[iXYZ][jXYZ] = 0.;
260
261 for (auto kXYZ = 0; kXYZ < 3; kXYZ++){
262
263 this->elecOctupole[iXYZ][jXYZ][kXYZ] = 0.;
264
265 }
266 }
267 }
268
269 applyToEach([&](SubSSPtr& ss){
270 // Computing the Multipole for each subsystem and then adding to the overall multipoles
271 ss->computeMultipole(emPert);
272 for (auto iXYZ = 0; iXYZ < 3; iXYZ++) {
273 this->elecDipole[iXYZ] += ss->elecDipole[iXYZ];
274
275 for (auto jXYZ = 0; jXYZ < 3; jXYZ++){
276
277 this->elecQuadrupole[iXYZ][jXYZ] += ss->elecQuadrupole[iXYZ][jXYZ];
278
279 for (auto kXYZ = 0; kXYZ < 3; kXYZ++){
280
281 this->elecOctupole[iXYZ][jXYZ][kXYZ] += ss->elecOctupole[iXYZ][jXYZ][kXYZ];
282
283 }
284 }
285 }
286 });
287 // Nuclear contributions to the dipoles
288 for(auto &atom : this->molecule().atoms){
289
290 if (atom.quantum){
291 continue;
292 }
293 for (int iXYZ = 0; iXYZ < 3; iXYZ++){
294 this->elecDipole[iXYZ] -= atom.nucCharge*atom.coord[iXYZ]*(subsystems.size()-1);
295 }
296
297 }
298
299 // Nuclear contributions to the quadrupoles
300 for(auto &atom : this->molecule().atoms){
301 if (atom.quantum) {
302 continue;
303 }
304
305 for(size_t iXYZ = 0; iXYZ < 3; iXYZ++)
306 for(size_t jXYZ = 0; jXYZ < 3; jXYZ++)
307 this->elecQuadrupole[iXYZ][jXYZ] -=
308 (subsystems.size()-1) * atom.nucCharge * atom.coord[iXYZ] * atom.coord[jXYZ];
309 }
310
311 // Nuclear contributions to the octupoles
312 for(auto &atom : this->molecule().atoms){
313
314 if (atom.quantum){
315 continue;
316 }
317
318 for(size_t iXYZ = 0; iXYZ < 3; iXYZ++)
319 for(size_t jXYZ = 0; jXYZ < 3; jXYZ++)
320 for(size_t kXYZ = 0; kXYZ < 3; kXYZ++)
321 this->elecOctupole[iXYZ][jXYZ][kXYZ] -=
322 (subsystems.size()-1) * atom.nucCharge * atom.coord[iXYZ] * atom.coord[jXYZ] *
323 atom.coord[kXYZ];
324 }
325
326 };
