57 template <template <typename> class AddT, typename IntsT>
58 void addIntegral(const std::string & name, std::shared_ptr<AddT<IntsT>> integral) {
59
60 if (this->integrals.find(name) != integrals.end()) {
61 this->integrals.erase(name);
62 }
63
64 try {
65 this->integrals.emplace(name, std::dynamic_pointer_cast<ParticleIntegrals>(integral));
66 } catch (...) {
67 CErr( name + " is not an ParticleIntegrals");
68 }
69
70 }; // add to ints
71
72 template <template <typename> class GetT, typename IntsT>
73 std::shared_ptr<GetT<IntsT>> getIntegral(const std::string & name) const {
74
75 if(this->integrals.find(name) == integrals.end()) return nullptr;
76
77 std::shared_ptr<GetT<IntsT>> integral;
78
79 try {
80 integral = std::dynamic_pointer_cast<GetT<IntsT>>(this->integrals.at(name));
81 } catch (...) {
82 return nullptr;
83 }
84
85 return integral;
86 }; // get from ints
87
88 void clear() { integrals.clear(); }
89
90 }; // struct IntegralsCollection
