47 TPIContractions<MatsT,IntsT>::convert(const std::shared_ptr<TPIContractions<MatsT,IntsT>>& ch) {

49 if (not ch) return nullptr;
50

52
53 if (tID == typeid(InCore4indexTPIContraction<MatsT,IntsT>)) {
54 return std::make_shared<InCore4indexTPIContraction<MatsU,IntsT>>(
55 *std::dynamic_pointer_cast<InCore4indexTPIContraction<MatsT,IntsT>>(ch));
56
57 } else if (tID == typeid(GTODirectTPIContraction<MatsT,IntsT>)) {
58 return std::make_shared<GTODirectTPIContraction<MatsU,IntsT>>(
59 *std::dynamic_pointer_cast<GTODirectTPIContraction<MatsT,IntsT>>(ch));
60
61 } else if (tID == typeid(GIAODirectERIContraction)) {
62 return std::dynamic_pointer_cast<TPIContractions<MatsU,IntsT>>(ch);
63
64 } else if (tID == typeid(InCoreRITPIContraction<MatsT,IntsT>)) {
65 return std::make_shared<InCoreRITPIContraction<MatsU,IntsT>>(
66 *std::dynamic_pointer_cast<InCoreRITPIContraction<MatsT,IntsT>>(ch));
67
68 } else {
69 std::stringstream errMsg;
70 errMsg << "TPIContractions implementation \"" << tID.name() << "\" not registered in convert." << std::endl;
71 CErr(errMsg.str(),std::cout);
72 }
73
74 return nullptr;
