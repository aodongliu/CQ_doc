79 OptimizeOrbitals<MatsT>(sC, comm, modOpt, mem),rotOpt(nrrot) {

81 sanityChecks();
82
83 // Compute the total number of nonredundant parameters
84 nParam = 0;
85 for( auto& rO : rotOpt )
86 nParam += rO.rotIndices.first.size() * rO.rotIndices.second.size();

88 // Alloc Quasi-Newton matrices
89 alloc();
90 }

92 // Destructor
93 ~NewtonRaphsonSCF() {
94 this->memManager.free(orbRot, orbGrad, orbDiagHess);
95
96 if( this->scfControls.nrAlg == QUASI_BFGS or this->scfControls.nrAlg == QUASI_SR1 ) {
97 for( auto* p : qnOrbRot )
98 this->memManager.free(p);
99 for( auto* p : qnOrbGrad )
100 this->memManager.free(p);
101 }
102 }
