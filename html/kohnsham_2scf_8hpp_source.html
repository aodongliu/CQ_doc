33template<typename MatsT, typename IntsT>
34void KohnSham<MatsT, IntsT>::buildModifyOrbitals() {
35 // Modify SCFControls
36 this->scfControls.printLevel = this->printLevel;
37 this->scfControls.refLongName_ = this->refLongName_;
38 this->scfControls.refShortName_ = this->refShortName_;
39
40 // Initialize ModifyOrbitalOptions
41 ModifyOrbitalsOptions<MatsT> modOrbOpt;
42
43 // Bind Lambdas to std::functions
44 modOrbOpt.printProperties = [this]() { this->printProperties(); };
45 modOrbOpt.saveCurrentState = [this]() { this->saveCurrentState(); };
46 modOrbOpt.formFock = [this](EMPerturbation& pert) { this->formFock(pert); };
47 modOrbOpt.computeProperties = [this](EMPerturbation& pert) { this->computeProperties(pert); };
48 modOrbOpt.formDensity = [this]() { this->formDensity(); };
49 modOrbOpt.getFock = [this]() { return this->getFock(); };
50 modOrbOpt.getOnePDM = [this]() { return this->getOnePDM(); };
51 modOrbOpt.getOrtho = [this]() { return this->getOrtho(); };
52 modOrbOpt.getTotalEnergy = [this]() { return this->getTotalEnergy(); };
53
54 // Make ModifyOrbitals based on scfControls
55 if( this->scfControls.scfAlg == _CONVENTIONAL_SCF ) {
56 this->modifyOrbitals = std::dynamic_pointer_cast<ModifyOrbitals<MatsT>>(
57 std::make_shared<ConventionalSCF<MatsT>>(this->scfControls, this->comm, modOrbOpt, this->memManager));
58 } else if( this->scfControls.scfAlg == _NEWTON_RAPHSON_SCF ) {
59
60 bool iRO = (std::dynamic_pointer_cast<ROFock<MatsT,IntsT>>(this->fockBuilder) != nullptr);
61 // The orbital gradient is wrong for ROHF. Also, it neglects the mixing
62 // between the doubly occupied orbitals and the singly occupied orbitals
63 if( iRO ) CErr("NewtonRaphson SCF is not yet implemented for Restricted Open Shell");
64
65 // Generate NRRotationOptions
66 std::vector<NRRotOptions> rotOpt = this->buildRotOpt();
67
68 this->modifyOrbitals = std::dynamic_pointer_cast<ModifyOrbitals<MatsT>>(
69 std::make_shared<NewtonRaphsonSCF<MatsT>>(rotOpt, this->scfControls, this->comm, modOrbOpt, this->memManager));
70 } else {
71 this->scfControls.doExtrap = false;
72 this->modifyOrbitals = std::dynamic_pointer_cast<ModifyOrbitals<MatsT>>(
73 std::make_shared<SkipSCF<MatsT>>(this->scfControls, this->comm, modOrbOpt, this->memManager));
74 }
75}; // KohnSham<MatsT,IntsT> :: buildModifyOrbitals
