137 /*
138 * Determine the sign for nonzero matrix elements.
139 * WARNING: it's assumed that p and q will generate
140 * a nonzero matrix element
141 */
142 int sign1e(const size_t p, const size_t q) const {
143
144 bool sign = true;
145 size_t l = std::min(p,q);
146 size_t r = std::max(p,q);
147 for (auto k = l+1; k < r; k++){
148 sign ^= ptr_[k]; // using exclusive or
149 }
150
151 return (sign)? 1: -1;
152 } // sign1e

158 void excitation(const size_t p, const size_t q) {
159// if (not ptr_.[q] or (ptr_[p] and p != q)) return;
160 ptr_[q] = not ptr_[q];
161 ptr_[p] = not ptr_[p];
162 } // excitation

165 void bitInfo(const bool bitType, const std::pair<size_t, size_t> & occRange,
166 InputIt bitList) const {
167 auto bitList_ptr = bitList;
168 for(auto i = occRange.first; i < occRange.second; i++) {
169 if(bitType == ptr_[i]) {
170 *bitList_ptr = i;
171 bitList_ptr++;
172 }
173 }
174 }
