122 /*
123 * Brief: The variables are optimized using a set of unconstrained
124 * variables. coeffs[i] = xOpt[i]*xOpt[i] / (\sum_k xOpt[k]*xOpt[k])
125 */
126 template <typename T>
127 void ENERGYDIIS<T>::computeCoeffs()
128 {
129 T sumC = T(0.);
130 for (size_t i = 0; i < nInter; i++)
131 {
132 coeffs[i] = xOpt[i] * xOpt[i];
133 sumC += coeffs[i];
134 }
135
136 for (size_t i = 0; i < nInter; i++)
137 coeffs[i] /= sumC;
138 }
139
140
141 /*
142 * Brief: This function computes the equation we are optimizing
143 * val = E^T*coeffs - 1/2 coeffs^T B coeffs
144 */
145 template <typename T>
146 double ENERGYDIIS<T>::objFunc()
147 {
148 computeCoeffs();
149
150 double val = T(0.);
151 for (size_t i = 0; i < nInter; i++)
152 {
153 val += coeffs[i] * energy[i];
154 for (size_t j = 0; j < nInter; j++)
155 val -= T(0.5) * coeffs[i] * B[j + i * nDimB] * coeffs[j];
156 }
157
158 return val;
159 }

161 /*
162 * Brief: This is the gradient for the objective function with respect
163 * to the unconstrained optimization variables(xOpt).
164 *
165 */
166 template <typename T>
167 void ENERGYDIIS<T>::gradFunc()
168 {
169 computeCoeffs();
170
171 // Renormalize parameters
172 // This helps keep the gradient consistent during the optimization
173 for (size_t i = 0; i < nInter; i++)
174 this->xOpt[i] = std::sqrt(std::abs(coeffs[i]));
175
176 T *lag = memManager.template malloc<T>(nInter);
177 T sumC = T(0.);
178 for (size_t i = 0; i < nInter; i++)
179 {
180 sumC += this->xOpt[i] * this->xOpt[i];
181 lag[i] = energy[i];
182 for (size_t j = 0; j < nInter; j++)
183 lag[i] -= B[j + i * nDimB] * coeffs[j];
184 }
185
186 for (size_t k = 0; k < nInter; k++)
187 {
188 T cpartial;
189 this->gOpt[k] = T(0.);
190 for (size_t i = 0; i < nInter; i++)
191 {
192 cpartial = T(0.);
193 if (i == k)
194 {
195 cpartial = T(2.) * this->xOpt[i] / sumC * (T(1.) - coeffs[i]);
196 }
197 else
198 {
199 cpartial = T(-2.) * coeffs[i] * this->xOpt[k] / sumC;
200 }
201 this->gOpt[k] += cpartial * lag[i];
202 }
203 }
204
205#ifdef INTERPOLATE_DEBUG_GRADIENT
206 this->printGrad = true;
207 std::cout << "Coeffs:" << std::endl;
208 for (size_t i = 0; i < nInter; i++)
209 std::cout << coeffs[i] << std::endl;
210 std::cout << std::endl;
211#endif
212
213 memManager.free(lag);
214 }

217 /*
218 * Brief: This routine drives the optimization and returns the result
219 */
220
221 template <typename T>
222 bool ENERGYDIIS<T>::interpolate()
223 {
224#ifdef INTERPOLATE_DEBUG
225 std::cout << "Energies:" << std::endl;
226 for (size_t i = 0; i < nInter; i++)
227 std::cout << energy[i] << std::endl;
228 std::cout << std::endl;
229 prettyPrintSmart(std::cout, "B Matrix", B, nInter, nInter, nDimB);
230 std::cout << std::endl;
231 this->printOptOutput = true;
232 this->printSearchDirection = true;
233 this->printLineSearch = true;
234 //this->lineAlg = SteapestDescent; // This gives a good test of the gradient
235 //this->sdMaxDisp = 0.05;
236 this->updateInvHess = true;
237#endif
238
239 // Make the target guess twice as large as the rest unless nInter == 2
240 T targetGuess = nInter == 2 ? T(1./double(nInter)) : T( 2./double(nInter) );
241 T restGuess = std::sqrt((1.-targetGuess) / T(nInter-1));
242 targetGuess = std::sqrt(targetGuess);
243 for (size_t i = 0; i < nInter; i++)
244 this->xOpt[i] = T(restGuess);
245 this->xOpt[target] = targetGuess;
246 double guessE = this->objFunc();
247
248#ifdef INTERPOLATE_PRINT_COEFFS
249 std::cout << "EDIIS Guess Coefficients: " << guessE << std::endl;
250 for (size_t i = 0; i < nInter; i++)
251 {
252 std::cout << coeffs[i];
253 if (i == target)
254 std::cout << "***";
255 std::cout << std::endl;
256 }
257 std::cout << std::endl;
258#endif
259
260 bool conv = BFGS<T> :: optimize();
261 computeCoeffs();
262
263#ifdef INTERPOLATE_PRINT_COEFFS
264 std::cout << "EDIIS Coefficients:" << std::endl;
265 for (size_t i = 0; i < nInter; i++)
266 {
267 std::cout << coeffs[i];
268 if (i == target)
269 std::cout << "***";
270 std::cout << std::endl;
271 }
272 std::cout << std::endl;
273#endif
274
275 // To force progress, make sure target coeff is greater than 1E-8
276 if (coeffs[target] < 1E-4)
277 {
278#ifdef INTERPOLATE_PRINT_WARN
279 std::cout << "EDIIS Warning: found solution with latest coefficient zero." << std::endl;
280#endif
281
282#ifndef INTERPOLATE_USE_GAUSSIAN_ALG
283 this->xOpt[target] = std::sqrt(1E-4);
284 this->objFunc();
285#endif
286
287 // NOTE: The following is an implementation of what Kudin et al. describe to do
288 // when the target is zero. However, I found that making the target 1E-4
289 // does not change the coeffs a lot and they are still close to the optimal
290 // ones.
291 //
292 // Mix the other degrees of freedom with the targe state to find local minimum
293 // This is Gaussian's method and provided in:
294 // Kudin, K. N.; Scuseria, G. E.; Cancs, E. A Black-Box Self-Consistent Field Convergence Algorithm: One Step Closer. J. Chem. Phys. 2002, 116 (19), 82558261.
295
296#ifdef INTERPOLATE_USE_GAUSSIAN_ALG
297 // Start with target as unity
298 std::fill_n(this->xOpt, nInter, T(0.));
299 this->xOpt[target] = 1.;
300
301 std::vector<size_t> vopt = {target};
302 std::vector<size_t> vmix;
303 for (size_t i = 0; i < nInter; i++)
304 if (i != target)
305 vmix.push_back(i);
306
307 T *dx = memManager.template malloc<T>(nInter);
308 T *initX = memManager.template malloc<T>(nInter);
309 double ymin;
310 T amin;
311 for (auto &iMix : vmix)
312 {
313 // Compute dx
314 std::fill_n(dx, nInter, T(0.));
315 for (auto &iOpt : vopt)
316 dx[iOpt] = T(-1.);
317 dx[iMix] = T(1.);
318
319 // Perform Line Search
320 std::copy_n(xOpt, nInter, initX);
321 GoldenSectionSearch<T>::optimize(initX, dx, T(1E-8), T(0.9), ymin, amin);
322
323 vopt.push_back(iMix);
324 }
325 memManager.free(dx, initX);
326#endif
327
328#ifdef INTERPOLATE_PRINT_COEFFS
329 std::cout << "EDIIS Coefficients:" << std::endl;
330 for (size_t i = 0; i < nInter; i++)
331 {
332 std::cout << coeffs[i];
333 if (i == target)
334 std::cout << "***";
335 std::cout << std::endl;
336 }
337 std::cout << std::endl;
338#endif
339 }
340 return conv;
341 }
