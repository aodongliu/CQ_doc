40template<typename MatsT, typename IntsT>
41void SingleSlater<MatsT, IntsT>::saveCurrentState() {
42
43 ROOT_ONLY(comm);
44
45 // Checkpoint if file exists
46 if( savFile.exists() ) {
47
48 size_t NB = this->nAlphaOrbital();
49 size_t NBC = this->nC * NB;
50
51 size_t t_hash = std::is_same<MatsT, double>::value ? 1 : 2;
52
53 // Save Field type
54 std::string prefix = "SCF/";
55 if( this->particle.charge == 1.0 ) prefix = "PROT_" + prefix;
56
57 savFile.safeWriteData(prefix + "FIELD_TYPE", &t_hash, {1});
58
59 savFile.safeWriteData(prefix + "1PDM", *this->onePDM);
60
61 savFile.safeWriteData(prefix + "FOCK", *fockMatrix);
62
63 savFile.safeWriteData(prefix + "1PDM_ORTHO", *onePDMOrtho);
64
65 savFile.safeWriteData(prefix + "FOCK_ORTHO", *fockMatrixOrtho);
66
67 savFile.safeWriteData("SCF/ORTHO", orthoSpinor->forwardPointer()->pointer(), {NB, NB});
68
69 savFile.safeWriteData("SCF/ORTHO_INV", orthoSpinor->backwardPointer()->pointer(), {NB, NB});
70
71 // Save MOs
72 savFile.safeWriteData(prefix + "MO1", this->mo[0].pointer(), {NBC, NBC});
73 if( this->nC == 1 and not this->iCS ) savFile.safeWriteData(prefix + "MO2", this->mo[1].pointer(), {NBC, NBC});
74
75 // Save Energies
76 savFile.safeWriteData(prefix + "TOTAL_ENERGY", &this->totalEnergy, {1});
77 savFile.safeWriteData(prefix + "ONE_BODY_ENERGY", &this->OBEnergy, {1});
78 savFile.safeWriteData(prefix + "MANY_BODY_ENERGY", &this->MBEnergy, {1});
79
80 // Save Multipoles
81 savFile.safeWriteData(prefix + "LEN_ELECTRIC_DIPOLE", &this->elecDipole[0], {3});
82 savFile.safeWriteData(prefix + "LEN_ELECTRIC_QUADRUPOLE", &this->elecQuadrupole[0][0], {3, 3});
83 savFile.safeWriteData(prefix + "LEN_ELECTRIC_OCTUPOLE", &this->elecOctupole[0][0][0], {3, 3, 3});
84
85 // Save Spin
86 savFile.safeWriteData(prefix + "S_EXPECT", &this->SExpect[0], {3});
87 savFile.safeWriteData(prefix + "S_SQUARED", &this->SSq, {1});
88 }

92 template <typename MatsT, typename IntsT>
93 bool SingleSlater<MatsT, IntsT>::checkStability() {
94
95 double W;
96 MatsT* J;
97 std::tie(W,J) = this->getStab();
98 std::cout << " * LOWEST STABILITY EIGENVALUE " <<
99 std::scientific << W << "\n";
100
101 if( W < 0. and std::abs(W) > 1e-08 )
102 std::cout << " * LOWEST STABILITY EIGENVALUE NEGATIVE: "
103 << "PERFORMING THOULESS ROTATION\n";
104 else {
105
106 std::cout << " * LOWEST STABILITY EIGENVALUE POSITIVE: "
107 << "WAVE FUNCTION 2nd ORDER STABLE\n";
108
109 this->memManager.free(J); return true;
110
111 }
112
113 const size_t NB = this->nAlphaOrbital();
114 const size_t NB2 = NB * NB;
115 const size_t NBC = nC * NB;
116 const size_t NBC2 = NBC * NBC;
117
118 const size_t NO = (nC == 2) ? this->nO : this->nOA;
119 const size_t NV = (nC == 2) ? this->nV : this->nVA;
120 const size_t nOAVA = this->nOA * this->nVA;
121 const size_t nOBVB = this->nOB * this->nVB;
122
123
124 MatsT* ROT = this->memManager.template malloc<MatsT>(NBC2);
125 MatsT* EXPROT = this->memManager.template malloc<MatsT>(NBC2);
126 std::fill_n(ROT,NBC2,0.);
127
128 for(auto i = 0ul, ai = 0ul; i < NO; i++)
129 for(auto a = NO; a < NBC; a++, ai++) {
130
131 ROT[a + i*NBC] = J[ai];
132 ROT[i + a*NBC] = -SmartConj(J[ai]);
133
134 }
135
136 // FIXME: need to generalize MatExp to take non-hermetian and real
137 // matricies
138 //MatExp('D',NBC,T(-1.),ROT,NBC,EXPROT,NBC,this->memManager);
139
140 // Taylor
141 MatsT s = 1.;
142 std::copy_n(ROT,NBC2,EXPROT); // n = 1
143 blas::scal(NBC2,-s,EXPROT,1);
144 for(auto j = 0; j < NBC; j++) EXPROT[j*(NBC+1)] += 1.; // n = 0
145
146 MatsT* SCR = this->memManager.template malloc<MatsT>(NBC2);
147 MatsT* SCR2 = this->memManager.template malloc<MatsT>(NBC2);
148 std::copy_n(ROT,NBC2,SCR);
149
150 size_t tayMax = 30;
151 for(auto n = 2; n <= tayMax; n++) {
152
153 MatsT* M = nullptr;
154 if( n % 2 ) {
155 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NBC,NBC,NBC,MatsT(1.),ROT,NBC,SCR2,NBC,MatsT(0.),SCR,NBC);
156 M = SCR;
157 } else {
158 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NBC,NBC,NBC,MatsT(1.),ROT,NBC,SCR,NBC,MatsT(0.),SCR2,NBC);
159 M = SCR2;
160 }
161
162 MatsT fact = std::pow(-s,n)/factorial(n);
163 MatAdd('N','N',NBC,NBC,MatsT(1.),EXPROT,NBC,fact,M,NBC,
164 EXPROT,NBC);
165
166 }
167
168 /*
169 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NBC,NBC,NBC,T(1.),EXPROT,NBC,EXPROT,NBC,T(0.),SCR,NBC);
170 // prettyPrintSmart(std::cerr,"ROT",ROT,NBC,NBC,NBC);
171 // prettyPrintSmart(std::cerr,"EXPROT",EXPROT,NBC,NBC,NBC);
172 prettyPrintSmart(std::cout,"SCR",SCR,NBC,NBC,NBC);
173 // CErr();
174 */
175
176
177 // MO1 = MO1 * EXPROT
178 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NBC,NBC,NBC,MatsT(1.),this->mo[0].pointer(),NBC,EXPROT,NBC,MatsT(0.),
179 ROT,NBC);
180 std::copy_n(ROT,NBC2,this->mo[0].pointer());
181
182
183
184
185 orthoAOMO();
186 this->formDensity();
187 this->memManager.free(J,ROT,EXPROT);
188
189 return false;
190
191 }

205 template <typename MatsT, typename IntsT>
206 void SingleSlater<MatsT,IntsT>::diagOrthoFock() {
207
208 ROOT_ONLY(comm);
209 size_t NB = this->nAlphaOrbital() * nC;
210 size_t NB2 = NB*NB;
211 bool iRO = (std::dynamic_pointer_cast<ROFock<MatsT,IntsT>>(fockBuilder) != nullptr);
212
213
214 // Copy over the fockMatrixOrtho into MO storage
215 if(nC == 1 and iCS)
216 this->mo = fockMatrixOrtho->template spinGatherToBlocks<MatsT>(false,false);
217 else if(iRO)
218 this->mo[0] = fockMatrixOrtho->S();
219 else if(nC == 1)
220 this->mo = fockMatrixOrtho->template spinGatherToBlocks<MatsT>(false);
221 else {
222 this->mo[0] = fockMatrixOrtho->template spinGather<MatsT>();
223 }
224
225 // Diagonalize the Fock Matrix
226 int INFO = HermetianEigen('V', 'L', NB, this->mo[0].pointer(), NB, this->eps1,
227 memManager );
228 if( INFO != 0 ) CErr("HermetianEigen failed in Fock1",std::cout);
229
230 if(iRO) {
231 this->mo[1] = this->mo[0]; // for ROHF
232 std::copy_n(this->eps1, NB, this->eps2);
233 } else if(nC == 1 and not iCS) {
234 INFO = HermetianEigen('V', 'L', NB, this->mo[1].pointer(), NB, this->eps2,
235 memManager );
236 if( INFO != 0 ) CErr("HermetianEigen failed in Fock2",std::cout);
237 }
238
239#if 0
240 printMO(std::cout);
241#endif
242
243}; // SingleSlater<MatsT>::diagOrthoFock

421template<typename MatsT, typename IntsT>
422std::vector<NRRotOptions> SingleSlater<MatsT,IntsT>::buildRotOpt(){
423
424 // Generate NRRotationOptions
425 std::vector<NRRotOptions> rotOpt;
426 size_t NB = this->nC*this->basisSet().nBasis;
427 if( this->nC == 1 and this->iCS ){
428 rotOpt.emplace_back();
429 rotOpt[0].spaceIndex = 0;
430 rotOpt[0].rotIndices = ssNRRotIndices(this->nOA, NB);
431 } else if( this->nC == 1 ) {
432 rotOpt.emplace_back();
433 rotOpt[0].spaceIndex = 0;
434 rotOpt[0].rotIndices = ssNRRotIndices(this->nOA, NB);
435 rotOpt.emplace_back();
436 rotOpt[1].spaceIndex = 1;
437 rotOpt[1].rotIndices = ssNRRotIndices(this->nOB, NB);
438 } else if( this->nC == 2 ){
439 rotOpt.emplace_back();
440 rotOpt[0].spaceIndex = 0;
441 rotOpt[0].rotIndices = ssNRRotIndices(this->nO, NB);
442 } else if( this->nC == 4 ){
443 size_t N = NB/2;
444 rotOpt.emplace_back();
445 rotOpt[0].spaceIndex = 0;
446 rotOpt[0].rotIndices = ssNRRotIndices(this->nO, N, N); // only rotate the positive energy orbitals
447 }
448 return rotOpt;
449}

451template<typename MatsT, typename IntsT>
452void SingleSlater<MatsT, IntsT>::runModifyOrbitals(EMPerturbation& pert) {
453
454 bool iRO = (std::dynamic_pointer_cast<ROFock<MatsT, IntsT>>(this->fockBuilder) != nullptr);
455
456 // Initialize properties
457 if( not std::dynamic_pointer_cast<SkipSCF<MatsT>>(this->modifyOrbitals) ) getNewOrbitals();
458 this->computeProperties(pert);
459
460 // Setup MO reference vector
461 std::vector<std::reference_wrapper<SquareMatrix<MatsT>>> moRefs;
462 if( iRO ) {
463 moRefs.emplace_back(this->mo[0]);
464 } else {
465 for( auto& m : this->mo )
466 moRefs.emplace_back(m);
467 }
468
469 // Setup Eigenvalue vector
470 std::vector<double*> epsVec;
471 if( this->nC == 1 and not(iCS or iRO) ) {
472 epsVec = {this->eps1, this->eps2};
473 } else {
474 epsVec = {this->eps1};
475 }
476
477 // Run modify orbitals
478 this->modifyOrbitals->runModifyOrbitals(pert, moRefs, epsVec);
479
480 // Orthogonalize MO's for NewtonRaphsonSCF and check Stability
481 if( std::dynamic_pointer_cast<NewtonRaphsonSCF<MatsT>>(this->modifyOrbitals) ){
482 this->orthoAOMO();
483 if( this->scfControls.nrAlg == FULL_NR ){
484 bool converged = this->checkStability();
485 if( not converged ) {
486 this->modifyOrbitals->runModifyOrbitals(pert, moRefs, epsVec);
487 this->orthoAOMO();
488 converged = this->checkStability();
489 if( not converged ) CErr("Newton-Raphson SCF failed to converge to a minimum");
490 }
491 }
492 }
493
494 this->ao2orthoFock(); // SCF Does not update the fockMatrixOrtho
495 this->MOFOCK();
496 saveCurrentState();
497
498#ifdef CQ_ENABLE_MPI
499
500 // Broadcast the updated MOs to all MPI processes
501 if( MPISize(comm) > 1 ) {
502
503 std::cerr << " *** Scattering the AO-MOs ***\n";
504 size_t Nmo = this->mo[0].dimension();
505 MPIBCast(this->mo[0].pointer(),Nmo*Nmo,0,comm);
506 if( nC == 1 and not iCS )
507 MPIBCast(this->mo[1].pointer(),Nmo*Nmo,0,comm);
508
509 std::cerr << " *** Scattering EPS ***\n";
510 MPIBCast(this->eps1,Nmo,0,comm);
511 if( nC == 1 and not iCS )
512 MPIBCast(this->eps2,Nmo,0,comm);
513
514 std::cerr << " *** Scattering FOCK ***\n";
515 size_t fockDim = fockMatrix->dimension();
516 for(MatsT *mat : fockMatrix->SZYXPointers())
517 MPIBCast(mat,fockDim*fockDim,0,comm);
518
519 std::cerr << " *** Scattering the 1PDM ***\n";
520 size_t denDim = this->onePDM->dimension();
521 for(auto p : this->onePDM->SZYXPointers())
522 MPIBCast(p,denDim*denDim,0,comm);
523 }
524
525#endif
526}; // SingleSlater<MatsT,IntsT> :: runModifyOrbitals

528/*
529 * Brief: Function to generate shared pointers to Fock Matrix for modify orbitals
530 * Here, the ModifyOrbitals object cannot modify the fockMatrix in SingleSlater
531 * since they are temporary objects. Additionally, the interface assumes the
532 * matrices are spin gathered.
533 */
534template<typename MatsT, typename IntsT>
535std::vector<std::shared_ptr<SquareMatrix<MatsT>>> SingleSlater<MatsT, IntsT>::getFock() {
536
537 bool iRO = (std::dynamic_pointer_cast<ROFock<MatsT, IntsT>>(fockBuilder) != nullptr);
538 if( this->nC == 1 and iCS ) {
539 return {std::make_shared<SquareMatrix<MatsT>>(MatsT(0.5) * fockMatrix->S())};
540 } else if( this->nC == 1 and iRO ) {
541 return {std::make_shared<SquareMatrix<MatsT>>(MatsT(1.) * fockMatrix->S())};
542 } else if( this->nC == 1 ) {
543 std::shared_ptr<SquareMatrix<MatsT>> fA = std::make_shared<SquareMatrix<MatsT>>(
544 MatsT(0.5) * fockMatrix->S() + MatsT(0.5) * fockMatrix->Z()
545 );
546 std::shared_ptr<SquareMatrix<MatsT>> fB = std::make_shared<SquareMatrix<MatsT>>(
547 MatsT(0.5) * fockMatrix->S() - MatsT(0.5) * fockMatrix->Z()
548 );
549 return {fA, fB};
550 } else {
551 return {std::make_shared<SquareMatrix<MatsT>>(fockMatrix->template spinGather<MatsT>())};
552 }
553}; // SingleSlater<MatsT,IntsT> :: getFock

555/*
556 * Brief: Function to generate shared pointers to Fock Matrix for modify orbitals
557 * Here, the ModifyOrbitals object cannot modify the fockMatrix in SingleSlater
558 * since they are copies of the objects. Additionally, the interface assumes the
559 * matrices are spin gathered.

563
564 bool iRO = (std::dynamic_pointer_cast<ROFock<MatsT, IntsT>>(fockBuilder) != nullptr);
565 if( this->nC == 1 and iCS ) {
566 return {std::make_shared<SquareMatrix<MatsT>>(MatsT(0.5) * this->onePDM->S())};
567 } else if( this->nC == 1 and iRO ) {
568 return {std::make_shared<SquareMatrix<MatsT>>(MatsT(0.5) * this->onePDM->S() + MatsT(0.5)*this->onePDM->Z())};
569 } else if( this->nC == 1 ) {
570 std::shared_ptr<SquareMatrix<MatsT>> dA = std::make_shared<SquareMatrix<MatsT>>(

574 MatsT(0.5) * this->onePDM->S() - MatsT(0.5) * this->onePDM->Z()
575 );
576 return {dA, dB};
577 } else {
578 return {std::make_shared<SquareMatrix<MatsT>>(this->onePDM->template spinGather<MatsT>())};
