119 template <typename MatsT, typename IntsT>
120 void MCWaveFunction<MatsT,IntsT>::alloc() {
121
122#ifdef DEBUG_MULTISTATEWFN_IMPL
123 std::cout << "MCWaveFunction::alloc (this = " << this << ")" << std::endl;
124#endif
125
126 if (reference().nC == 4 and not this->FourCompNoPair)
127 CErr("NYI for Four Component without NoPair Approximation");
128
129 size_t NS = this->NStates;
130 size_t NDet = this->NDet;
131 size_t nCorrO = this->MOPartition.nCorrO;
132
133 CIVecs = std::vector<MatsT*>(NS);
134 oneRDM.reserve(NS);
135
136 if (MOPartition.scheme == CAS) {
137 ciBuilder = std::make_shared<CASCI<MatsT,IntsT>>();
138 } else if (MOPartition.scheme == RAS) {
139 ciBuilder = std::make_shared<RASCI<MatsT,IntsT>>();
140 } else {
141 CErr();
142 }
143
144 try {
145 for (auto i = 0ul; i < NS; i++) {
146 CIVecs[i] = this->memManager.template malloc<MatsT>(NDet);
147 oneRDM.emplace_back(SquareMatrix<MatsT>(this->memManager, nCorrO));
148 }
149 } catch (...) {
150 CErr("Not enough Memory to allocate CIVector for the specified number of determiants");
151 }
152
153 // computing list for detstring
154 std::cout << std::endl;
155 FormattedLine(std::cout, "Compute Excitation List(s) ...");
156 if (detStr) detStr->computeList();
157 if (detStrBeta) detStrBeta->computeList();
158 std::cout << std::endl;
159
160 }; // MCWaveFunction<T>::alloc

180 void MCWaveFunction<MatsT,IntsT>::saveCurrentStates() {
181
182 ROOT_ONLY(comm);
183
184 // Checkpoint if file exists
185 if( savFile.exists() ) {
186
187 size_t NS = this->NStates;
188 savFile.safeWriteData("MCWFN/NSTATES", &NS, {1});
189 savFile.safeWriteData("MCWFN/INACT_ENERGY", &(this->InactEnergy), {1});
190 savFile.safeWriteData("MCWFN/STATE_ENERGY", this->StateEnergy.data(), {NS});
191
192 auto & mopart = this->MOPartition;
193 savFile.safeWriteData("MCWFN/ORB_INDEX", & (mopart.orbIndices[0]),{mopart.nMO});
194
195 // Save oscillator strength
196 if(NosS1) {
197 savFile.safeWriteData("MCWFN/OSC_STR", osc_str, {NosS1, NS});
198 }
199 }
