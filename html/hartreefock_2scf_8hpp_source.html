62template<typename MatsT, typename IntsT>
63void HartreeFock<MatsT, IntsT>::buildModifyOrbitals() {
64 // Modify SCFControls
65 this->scfControls.printLevel = this->printLevel;
66 this->scfControls.refLongName_ = this->refLongName_;
67 this->scfControls.refShortName_ = this->refShortName_;
68
69 // Initialize ModifyOrbitalOptions
70 ModifyOrbitalsOptions<MatsT> modOrbOpt;
71
72 // Bind Lambdas to std::functions
73 modOrbOpt.printProperties = [this]() { this->printProperties(); };
74 modOrbOpt.saveCurrentState = [this]() { this->saveCurrentState(); };
75 modOrbOpt.formFock = [this](EMPerturbation& pert) { this->formFock(pert,false,1.); };
76 modOrbOpt.computeProperties = [this](EMPerturbation& pert) { this->computeProperties(pert); };
77 modOrbOpt.formDensity = [this]() { this->formDensity(); };
78 modOrbOpt.getFock = [this]() { return this->getFock(); };
79 modOrbOpt.getOnePDM = [this]() { return this->getOnePDM(); };
80 modOrbOpt.getOrtho = [this]() { return this->getOrtho(); };
81 modOrbOpt.getTotalEnergy = [this]() { return this->getTotalEnergy(); };
82 modOrbOpt.computeFullNRStep = [this](MatsT* dx) { this->computeFullNRStep(dx); };
83
84 // Make ModifyOrbitals based on scfControls
85 if( this->scfControls.scfAlg == _CONVENTIONAL_SCF ) {
86 // Conventional SCF
87
88 this->modifyOrbitals = std::dynamic_pointer_cast<ModifyOrbitals<MatsT>>(
89 std::make_shared<ConventionalSCF<MatsT>>(this->scfControls, this->comm, modOrbOpt, this->memManager));
90
91 } else if( this->scfControls.scfAlg == _NEWTON_RAPHSON_SCF ) {
92 // Newton-Raphson SCF
93
94 bool iRO = (std::dynamic_pointer_cast<ROFock<MatsT,IntsT>>(this->fockBuilder) != nullptr);
95 // The orbital gradient is wrong for ROHF. Also, it neglects the mixing
96 // between the doubly occupied orbitals and the singly occupied orbitals
97 if( iRO ) CErr("NewtonRaphson SCF is not yet implemented for Restricted Open Shell");
98
99 std::vector<NRRotOptions> rotOpt = this->buildRotOpt();
100
101 this->modifyOrbitals = std::dynamic_pointer_cast<ModifyOrbitals<MatsT>>(
102 std::make_shared<NewtonRaphsonSCF<MatsT>>(rotOpt,this->scfControls, this->comm, modOrbOpt, this->memManager));
103
104 } else {
105 // SKIP SCF
106 this->scfControls.doExtrap = false;
107 this->modifyOrbitals = std::dynamic_pointer_cast<ModifyOrbitals<MatsT>>(
108 std::make_shared<SkipSCF<MatsT>>(this->scfControls, this->comm, modOrbOpt, this->memManager));
109 }
110}; // HartreeFock<MatsT,IntsT> :: buildModifyOrbitals

112template<typename MatsT, typename IntsT>
113std::pair<double, MatsT*> HartreeFock<MatsT, IntsT>::getStab() {
114
115 MatsT* CCPY = nullptr;
116 double W = 0;
117#if 1
118 this->MOFOCK(); // Update MOFock
119 auto ptr = getPtr();
120 PolarizationPropagator<HartreeFock<MatsT, IntsT>> resp(this->comm, RESIDUE, ptr);
121 resp.doStab = true;
122
123 std::cout << " * STARTING HESSIAN DIAG FOR STABILITY CHECK\n";
124 resp.run();
125 std::cout << "\n\n";
126
127 // Make sure that the Hessian gets deallocated
128 MatsT* FM = resp.fullMatrix();
129 this->memManager.free(FM);
130
131 MatsT* C = resp.resResults.VR;
132
133 size_t N = resp.getNSingleDim(false);
134 CCPY = this->memManager.template malloc<MatsT>(N);
135 std::copy_n(C, N, CCPY);
136
137 W = resp.resResults.W[0];
138#else
139 CErr();
140#endif
141
142 return {W, CCPY};
143};
