41 std::shared_ptr<SingleSlater<MatsT,IntsT>> makeNewSS(const std::shared_ptr<SingleSlater<MatsU,IntsT>>& old_ss) {
42 std::shared_ptr<SingleSlater<MatsT,IntsT>> new_ss = nullptr;
43
44 TRY_REF(KohnSham, old_ss, new_ss);

54 template <typename MatsU>
55 NEOSS<MatsT,IntsT>::NEOSS(const NEOSS<MatsU,IntsT>& other, int dummy) :
56 SingleSlater<MatsT,IntsT>(dynamic_cast<const SingleSlater<MatsU,IntsT>&>(other), dummy),
57 WaveFunctionBase(dynamic_cast<const WaveFunctionBase&>(other)),
58 QuantumBase(dynamic_cast<const QuantumBase&>(other)),
59 order_(other.order_), functionals(other.functionals)
60 {
61 // Loop over all old subsystems
62 for( auto& x: other.subsystems ) {
63
64 // Easier to read names
65 auto& label = x.first;
66 auto& old_sys = x.second;
67
68 // Find a non-NEO fockbuilder
69 auto baseFock = old_sys->fockBuilder.get();
70 if( auto p = dynamic_cast<NEOFockBuilder<MatsU,IntsT>*>(baseFock) ) {
71 baseFock = p->getNonNEOUpstream();
72 }
73
74 // Assume just non-relativistic Fock here
75 // TODO: Make this general to all fockbuilders
76 auto newFock = std::make_shared<FockBuilder<MatsT,IntsT>>(*baseFock);
77
78 // Get new SingleSlater object
79 SubSSPtr new_sys = makeNewSS<MatsT>(old_sys);
80 new_sys->fockBuilder = newFock;
81
82 // If we have EPC functionals and this is a nuclear subsystem, add the
83 // functionals (only works for electron/proton systems)
84 if( old_sys->particle.charge > 0 ) {
85 if( auto ks = std::dynamic_pointer_cast<KohnSham<MatsU,IntsT>>(old_sys) ) {
86 auto new_ks = std::dynamic_pointer_cast<KohnSham<MatsT,IntsT>>(new_sys);
87 new_ks->functionals.insert(new_ks->functionals.end(), functionals.begin(), functionals.end());
88 }
89 }
90
91 // Add the new subsystem to the current object
92 addSubsystem(label, new_sys, other.interIntegrals.at(label));
93 }
94
95 // Copy gradient integrals
96 for( auto& gradSys1: other.gradInterInts ) {
97 for( auto& gradSys2: gradSys1.second ) {
98
99 auto label1 = gradSys1.first;
100 auto label2 = gradSys2.first;
101 auto intPtr = gradSys2.second.second;
102 auto contractSecond = gradSys2.second.first;
103
104 addGradientIntegrals(label1, label2, intPtr, contractSecond);
105 }
106 }
107
108 }; // Other type copy constructor

122 // Add a subsystem to the NEO object
123 //
124 // label Unique name for this subsystem
125 // ss SingleSlater object representing this subsystem
126 // ints Two particle integrals for interacting with all other subsystems
127 template <typename MatsT, typename IntsT>
128 void NEOSS<MatsT,IntsT>::addSubsystem(
129 std::string label,
130 std::shared_ptr<SingleSlater<MatsT,IntsT>> ss,
131 const LabeledMap<std::pair<bool,std::shared_ptr<TwoPInts<IntsT>>>>& ints)
132 {
133
134 // Check that we have at least the same number of integrals as other systems
135 if(ints.size() < subsystems.size())
136 CErr("Subsystem and integral number mismatch in addSubsystem!");
137
138 auto NB = ss->basisSet().nBasis;
139
140 bool this_nuclear = ss->particle.charge > 0;
141 auto ks = std::dynamic_pointer_cast<KohnSham<MatsT,IntsT>>(ss);
142
143 // Add a FockBuilder and Coulomb matrix:
144 // - For the new system's interaction with each other system
145 // - For each other system's interaction with the new system
146
147 // Add the base fockBuilder to make sure it has a persistent lifetime
148 fockBuilders.insert({label, {ss->fockBuilder}});
149 // New coulomb matrices to be added to the _new_ system
150 std::unordered_map<std::string, SquareMatrix<MatsT>> newCoulombs;
151
152 // Loop over other subsystems
153 for( auto& x: subsystems ) {
154
155 // first = label;
156 // second = SingleSlater;
157 auto other_NB = x.second->basisSet().nBasis;
158
159 // Add a new coulomb matrix to the new system
160 newCoulombs.insert({x.first, SquareMatrix<MatsT>(ss->memManager, NB)});
161 // Add a new coulomb matrix to the other system
162 interCoulomb.at(x.first).insert({label, SquareMatrix<MatsT>(ss->memManager, other_NB)});
163
164 HamiltonianOptions this_options = ss->aoints.options_;
165 HamiltonianOptions other_options = x.second->aoints.options_;
166
167 // New fock builders
168 auto this_newFock = std::make_shared<NEOFockBuilder<MatsT,IntsT>>(this_options);
169 auto other_newFock = std::make_shared<NEOFockBuilder<MatsT,IntsT>>(other_options);
170
171 this_newFock->setAux(x.second.get());
172 this_newFock->setOutput(&newCoulombs.at(x.first));
173 this_newFock->setUpstream(fockBuilders[label].back().get());
174
175 other_newFock->setAux(ss.get());
176 other_newFock->setOutput(&interCoulomb.at(x.first).at(label));
177 other_newFock->setUpstream(fockBuilders[x.first].back().get());
178
179 // Add integrals to other system (mostly for consistency)
180 auto& contractSecond = ints.at(x.first).first;
181 auto& tpi = ints.at(x.first).second;
182 interIntegrals[x.first].insert({label, {not contractSecond, tpi}});
183
184 // Contractions
185 std::shared_ptr<TPIContractions<MatsT,IntsT>> this_cont;
186 std::shared_ptr<TPIContractions<MatsT,IntsT>> other_cont;
187 if( auto tpi_t = std::dynamic_pointer_cast<DirectTPI<IntsT>>(tpi) ) {
188 this_cont = std::make_shared<GTODirectTPIContraction<MatsT,IntsT>>(*tpi_t);
189 other_cont = std::make_shared<GTODirectTPIContraction<MatsT,IntsT>>(*tpi_t);
190 }
191 else if( auto tpi_t = std::dynamic_pointer_cast<InCore4indexTPI<IntsT>>(tpi) ) {
192 this_cont = std::make_shared<InCore4indexTPIContraction<MatsT,IntsT>>(*tpi_t);
193 other_cont = std::make_shared<InCore4indexTPIContraction<MatsT,IntsT>>(*tpi_t);
194 }
195 else {
196 CErr("Invalid TwoPInts type for NEO!");
197 }
198 // Set contractSecond
199 this_cont->contractSecond = contractSecond;
200 other_cont->contractSecond = not contractSecond;
201
202 this_newFock->setContraction(this_cont);
203 other_newFock->setContraction(other_cont);
204
205 fockBuilders[label].push_back(this_newFock);
206 fockBuilders[x.first].push_back(other_newFock);
207
208 // KS specific fock builders
209 // XXX: This assumes that there is no correlation functional between
210 // nuclear subsystems
211 auto other_ks = std::dynamic_pointer_cast<KohnSham<MatsT,IntsT>>(x.second);
212 bool other_nuclear = x.second->particle.charge > 0;
213 std::shared_ptr<NEOKohnShamBuilder<MatsT,IntsT>> this_newKSBuilder = nullptr;
214 std::shared_ptr<NEOKohnShamBuilder<MatsT,IntsT>> other_newKSBuilder = nullptr;
215 if( other_ks && other_nuclear && !this_nuclear ) {
216 this_newKSBuilder = std::make_shared<NEOKohnShamBuilder<MatsT,IntsT>>(this_options);
217 other_newKSBuilder = std::make_shared<NEOKohnShamBuilder<MatsT,IntsT>>(other_options);
218
219 this_newKSBuilder->setFunctionals(other_ks->functionals);
220 other_newKSBuilder->setFunctionals(other_ks->functionals);
221
222 // Get EPC out of the functional list of the nuclear subsystem
223 other_ks->functionals.clear();
224 }
225 else if( ks && this_nuclear && !other_nuclear ) {
226 this_newKSBuilder = std::make_shared<NEOKohnShamBuilder<MatsT,IntsT>>(this_options);
227 other_newKSBuilder = std::make_shared<NEOKohnShamBuilder<MatsT,IntsT>>(other_options);
228
229 this_newKSBuilder->setFunctionals(ks->functionals);
230 other_newKSBuilder->setFunctionals(ks->functionals);
231
232 // Get EPC out of the functional list of the nuclear subsystem
233 this->functionals.insert(functionals.end(),
234 ks->functionals.begin(), ks->functionals.end());
235 ks->functionals.clear();
236 }
237
238 if( this_newKSBuilder ) {
239 this_newKSBuilder->setAux(x.second.get());
240 this_newKSBuilder->setUpstream(fockBuilders[label].back().get());
241
242 other_newKSBuilder->setAux(ss.get());
243 other_newKSBuilder->setUpstream(fockBuilders[x.first].back().get());
244
245 fockBuilders[label].push_back(this_newKSBuilder);
246 fockBuilders[x.first].push_back(other_newKSBuilder);
247 }
248
249 }
250
251 // Add the other system's coulomb matrices to the coulomb matrix storage
252 interCoulomb.insert({label, std::move(newCoulombs)});
253
254 // Add the single slater object to the list of systems
255 subsystems[label] = ss;
256
257 // Add the integrals to this system
258 interIntegrals.insert({label, ints});
259
260 // Update all FockBuilders used to the most recent version
261 for( auto& x: subsystems ) {
262 x.second->fockBuilder = fockBuilders.at(x.first).back();
263 }
264
265 };

268 template <typename MatsT, typename IntsT>
269 void NEOSS<MatsT,IntsT>::addGradientIntegrals(
270 std::string label1, std::string label2,
271 std::shared_ptr<GradInts<TwoPInts,IntsT>> ints, bool contractSecond) {
272
273 gradInterInts.insert({label1, {}});
274 gradInterInts.insert({label2, {}});
275
276 gradInterInts[label1].insert({label2, {contractSecond, ints}});
277 gradInterInts[label2].insert({label1, {not contractSecond, ints}});
278
279 // TODO: Make this work with nested systems. Right now there is no
280 // guarantee that it will be placed on the right fockBuilder for more
281 // than two systems. May require labeling of nested fockBuilders.
282 auto setGradInts = [&](std::shared_ptr<FockBuilder<MatsT,IntsT>>& fock) {
283 if(auto neofock = std::dynamic_pointer_cast<NEOFockBuilder<MatsT,IntsT>>(fock)) {
284 neofock->setGradientIntegrals(ints.get());
285 }
286 else {
287 CErr("Can't set gradient integrals on a non-NEOFockBuilder");
288 }
289 };
290
291 setGradInts(fockBuilders[label1].back());
292 setGradInts(fockBuilders[label2].back());
293
294 };

297 template <typename MatsT, typename IntsT>
298 std::vector<double> NEOSS<MatsT,IntsT>::getGrad(EMPerturbation& pert,
299 bool equil, bool saveInts) {
300
301 // Constants and return value
302 size_t nAtoms = this->molecule().nAtoms;
303 size_t nGrad = 3*nAtoms;
304 std::vector<double> gradient(nGrad);
305
306 // Initialize with classical nuclear repulsion
307 for( auto iGrad = 0; iGrad < nGrad; iGrad++ )
308 gradient[iGrad] = this->molecule().nucRepForce[iGrad/3][iGrad%3];
309
310 // Determine the names of the systems to loop over (so we can loop
311 // over system1 < system2)
312 std::vector<std::string> systemList;
313 if( order_.size() == subsystems.size() ) {
314 systemList.insert(systemList.end(), order_.begin(), order_.end());
315 }
316 else {
317 for( auto& system: subsystems )
318 systemList.push_back(system.first);
319 }
320
321 // Compute required integrals (intrasystem integrals are handled with
322 // the subsystem's getGrad call)
323 for( auto iSys = 0; iSys < systemList.size(); iSys++ ) {
324 for( auto jSys = 0; jSys < iSys; jSys++ ) {
325
326 auto& sys1Label = systemList[iSys];
327 auto& sys2Label = systemList[jSys];
328
329 auto& sys1Basis = subsystems[sys1Label]->basisSet();
330 auto& sys2Basis = subsystems[sys2Label]->basisSet();
331
332 bool sys1Left = not gradInterInts[sys1Label][sys2Label].first;
333 auto& gradInt12 = gradInterInts[sys1Label][sys2Label].second;
334
335 HamiltonianOptions options;
336 options.OneEScalarRelativity = false;
337
338 if( sys1Left )
339 gradInt12->computeAOInts(sys1Basis, sys2Basis, this->molecule(),
340 pert, EP_ATTRACTION, options);
341 else
342 gradInt12->computeAOInts(sys2Basis, sys1Basis, this->molecule(),
343 pert, EP_ATTRACTION, options);
344 }
345 }
346
347
348 applyToEach([&](SubSSPtr& ss) {
349 auto localGrad = ss->getGrad(pert, equil, saveInts);
350 for( auto iGrad = 0; iGrad < nGrad; iGrad++ ) {
351 gradient[iGrad] += localGrad[iGrad] - this->molecule().nucRepForce[iGrad/3][iGrad%3];
352 }
353 });
354
355 return gradient;
356
357 };

359 template <typename MatsT, typename IntsT>
360 void NEOSS<MatsT,IntsT>::buildModifyOrbitals() {
361 // Modify SCFControls
362 this->scfControls.printLevel = this->printLevel;
363 this->scfControls.refLongName_ = this->refLongName_;
364 this->scfControls.refShortName_ = this->refShortName_;
365
366 // Initialize ModifyOrbitalOptions
367 ModifyOrbitalsOptions<MatsT> modOrbOpt;
368
369 // Register functions
370 modOrbOpt.printProperties = [this]() { this->printProperties(); };
371 modOrbOpt.saveCurrentState = [this]() { this->saveCurrentState(); };
372 modOrbOpt.formFock = [this](EMPerturbation& pert) { this->formFock(pert,false,1.); };
373 modOrbOpt.computeProperties = [this](EMPerturbation& pert) { this->computeProperties(pert); };
374 modOrbOpt.formDensity = [this]() { this->formDensity(); };
375 modOrbOpt.getFock = [this]() { return this->getFock(); };
376 modOrbOpt.getOnePDM = [this]() { return this->getOnePDM(); };
377 modOrbOpt.getOrtho = [this]() { return this->getOrtho(); };
378 modOrbOpt.getTotalEnergy = [this]() { return this->getTotalEnergy(); };
379
380 // Make ModifyOrbitals based on scfControls
381 if( this->scfControls.scfAlg == _CONVENTIONAL_SCF ) {
382 // Conventional SCF
383 this->modifyOrbitals = std::dynamic_pointer_cast<ModifyOrbitals<MatsT>>(
384 std::make_shared<ConventionalSCF<MatsT>>(this->scfControls, this->comm, modOrbOpt, this->memManager));
385 } else if( this->scfControls.scfAlg == _NEWTON_RAPHSON_SCF ) {
386 // Newton-Raphson SCF
387 CErr("Newton-Raphson SCF NYI for NEO methods!");
388 } else {
389 // SKIP SCF
390 this->scfControls.doExtrap = false;
391 this->modifyOrbitals = std::dynamic_pointer_cast<ModifyOrbitals<MatsT>>(
392 std::make_shared<SkipSCF<MatsT>>(this->scfControls, this->comm, modOrbOpt, this->memManager));
393 }
394
395 }
