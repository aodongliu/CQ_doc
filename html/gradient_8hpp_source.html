36 /*
37 * Build gradient block-wisely:
38 * IN -> inactive
39 * CO -> correlated
40 * FV -> frozen virtual
41 * NA -> negative MO for 4C only
42 */
43 template <typename MatsT, typename IntsT>
44 double OrbitalRotation<MatsT, IntsT>::computeOrbGradient(EMPerturbation & pert,
45 SquareMatrix<MatsT> & oneRDM, InCore4indexTPI<MatsT> & twoRDM) {
46
47 ProgramTimer::tick("Form Gradient");
48
49 auto & mopart = mcwfn_.MOPartition;
50 auto & mem = mcwfn_.memManager;
51 if (not orbitalGradient_)
52 orbitalGradient_ = std::make_shared<SquareMatrix<MatsT>>(mem, mopart.nMO);
53
54 size_t nTOrb = mopart.nMO;
55 size_t nCorrO = mopart.nCorrO;
56 size_t nInact = mopart.nInact;
57 size_t nFVirt = mopart.nFVirt;
58 size_t nINCO = nInact + nCorrO;
59 size_t SCRDim = std::max(nCorrO, std::max(nInact, nFVirt));
60
61 // Allocate SCR
62 MatsT * SCR = mem.template malloc<MatsT>(SCRDim*SCRDim);
63 MatsT * SCR2 = mem.template malloc<MatsT>(nCorrO*nInact);
64
65 // zero out G
66 std::fill_n(orbitalGradient_->pointer(), nTOrb * nTOrb, MatsT(0.));
67
68 // point to start of the positive energy part
69 MatsT * G = orbitalGradient_->pointer() + mopart.nNegMO * (nTOrb + 1);
70
71 // CO-CO block
72 if (settings.rotate_within_correlated) {
73 CErr("rotate MOs within correlated orbitals not implemented");
74 }
75
76 // IN-CO block
77 if (settings.rotate_inact_correlated) {
78 // g_it = -2 * (F1_it - F2_ti)
79 // populate SCR as F1_it and SCR2 as F2_ti
80 this->formGeneralizedFock1(pert, oneRDM, SCR, "it");
81 this->formGeneralizedFock2(pert, oneRDM, twoRDM, SCR2, "i");
82
83 IMatCopy('T', nCorrO, nInact, MatsT(1.), SCR2, nCorrO, nInact);
84 MatAdd('N', 'N', nInact, nCorrO, MatsT(1.), SCR, nInact,
85 MatsT(-1.), SCR2, nInact, SCR, nInact);
86 // blas::scal(nInact * nCorrO, MatsT(2.), SCR, 1);
87
88 SetMat('N', nInact, nCorrO, -MatsT(1.), SCR, nInact, G + nTOrb * nInact, nTOrb);
89 SetMat('C', nInact, nCorrO, MatsT(1.), SCR, nInact, G + nInact, nTOrb);
90 }
91
92
93 // IN-FV block
94 if (settings.rotate_inact_virtual) {
95 // g_ia = -2 * F1_ia
96 // pupulate SCR as F1_ia
97 this->formGeneralizedFock1(pert, oneRDM, SCR, "ia");
98 // blas::scal(nInact * nFVirt, MatsT(2.), SCR, 1);
99 SetMat('N', nInact, nFVirt, -MatsT(1.), SCR, nInact, G + nTOrb * nINCO, nTOrb);
100 SetMat('C', nInact, nFVirt, MatsT(1.), SCR, nInact, G + nINCO, nTOrb);
101 }
102
103
104 // CO-FV block
105 if (settings.rotate_correlated_virtual) {
106 // g_ta = -2 * F2_ta^*
107 // pupulate SCR as F2_ta
108 this->formGeneralizedFock2(pert, oneRDM, twoRDM, SCR, "a");
109 // blas::scal(nCorrO * nFVirt, MatsT(2.), SCR, 1);
110 SetMat('R', nCorrO, nFVirt, -MatsT(1.), SCR, nCorrO, G + nTOrb * nINCO + nInact, nTOrb);
111 SetMat('T', nCorrO, nFVirt, MatsT(1.), SCR, nCorrO, G + nTOrb * nInact + nINCO, nTOrb);
112 }
113
114 mem.free(SCR, SCR2);
115
116 // repointing to the begining
117 G = orbitalGradient_->pointer();
118
119 // NA-PO block
120 // which is the formualed similarly as IN-FV block and CO-FV block
121 if (settings.rotate_negative_positive) {
122 size_t nNegMO = mopart.nNegMO;
123 SCR = mem.template malloc<MatsT>(nNegMO * std::max(nInact, nCorrO));
124
125 // IN-NA block
126 if (nInact > 0) {
127 this->formGeneralizedFock1(pert, oneRDM, SCR, "in");
128 SetMat('N', nInact, nNegMO, -MatsT(1.), SCR, nInact, G + nNegMO, nTOrb);
129 SetMat('C', nInact, nNegMO, MatsT(1.), SCR, nInact, G + nNegMO * nTOrb, nTOrb);
130 }
131
132 // CO-NA Block
133 this->formGeneralizedFock2(pert, oneRDM, twoRDM, SCR, "n");
134 SetMat('R', nCorrO, nNegMO, -MatsT(1.), SCR, nCorrO, G + nNegMO + nInact, nTOrb);
135 SetMat('T', nCorrO, nNegMO, MatsT(1.), SCR, nCorrO, G + (nNegMO + nInact) * nTOrb, nTOrb);
136
137 mem.free(SCR);
138 }
139
140 double orbitalGradientNorm = lapack::lange(lapack::Norm::Fro, nTOrb, nTOrb, G, nTOrb);
141
142#ifdef DEBUG_ORBITALROTATION_GRADIENT
143 prettyPrintSmart(std::cout, " 1RDM ", oneRDM.pointer(), nCorrO, nCorrO, nCorrO);
144 prettyPrintSmart(std::cout, " OR Orbital Gradient ", G, nTOrb, nTOrb, nTOrb);
145 std::cout << "OR Orbital Gradient Norm = " << orbitalGradientNorm << std::endl;
146#endif
147
148 ProgramTimer::tock("Form Gradient");
149
150 return orbitalGradientNorm;
151 }; // OrbitalRotation<MatsT>::computeOrbGradient
152
