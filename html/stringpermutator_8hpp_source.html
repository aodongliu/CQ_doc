34 /*
35 * Brief class to generate all permutation of strings
36 * with n permuted bits within m total bits
37 */
38 class StringPermutator {
39
40 size_t m_; // number of total bits
41 size_t n_; // number of occupied bits
42 size_t nStr_; // number of string in total
43 std::vector<size_t> pos_; // permuted bits locations
44
45 size_t str_index_; // starting from 0
46 size_t cur_ptr_;
47
48 public:
49
50 // disable default constructors
51 StringPermutator() = delete;
52 StringPermutator(const StringPermutator &) = default;
53 StringPermutator(StringPermutator && ) = default;
54
55 StringPermutator(size_t m, size_t n): m_(m), n_(n) {
56 nStr_ = Comb(m,n);
57 pos_.resize(n);
58 reset_seed();
59 }
60
61 void reset_seed() {
62 for(auto i = 0ul; i < n_; i++) pos_[i] = i;
63 cur_ptr_ = n_ - 1;
64 str_index_ = 0;
65 }
66
67 size_t stringIndex() const { return str_index_; }
68 size_t maxNString() const { return nStr_; }
69 std::vector<size_t> permutedPositions() const { return pos_; }
70 std::vector<size_t> & permutedPositions() { return pos_; }
71
72 size_t next() {
73
74 if (str_index_ == nStr_ - 1) return nStr_;
75
76 if (cur_ptr_ == n_ - 1) {
77 pos_[cur_ptr_] += 1;
78 } else {
79 auto k = pos_[cur_ptr_];
80 for (auto i = cur_ptr_, j = 1ul; i < n_; i++, j++)
81 pos_[i] = k + j;
82 cur_ptr_ = n_ - 1;
83 }
84
85 while (pos_[cur_ptr_] + (n_ - cur_ptr_) == m_) cur_ptr_--;
86
87 str_index_++;
88
89 return str_index_;
90 }
91
92 void fullStringList(size_t * strList) {
93
94 reset_seed();
95 auto strList_ptr = strList;
96 std::copy_n(pos_.begin(), n_, strList_ptr);
97
98 while (next() < nStr_) {
99 strList_ptr += n_;
100 std::copy_n(pos_.begin(), n_, strList_ptr);
101 }
102 }
103
104 }; // ChronusQ::DetString
