76 /*
77 * Build Diagonal RASCI Hamiltonian
78 */
79 template <typename MatsT, typename IntsT>
80 void RASCI<MatsT,IntsT>::buildDiagH(MCWaveFunction<MatsT, IntsT> & mcwfn, MatsT * diagH) {
81
82#ifdef _DEBUG_CIBUILDER_RASCI_IMPL
83 std::cout << "LL RAS diagonal Hamilton build." << std::endl;
84#endif
85 auto RASdiagHSt = tick();
86
87 RASCI_LOOP_INIT(); // check top for variable definitions
88 auto & hCore = *(mcwfn.moints.template getIntegral<OnePInts, MatsT>("hCore_Correlated_Space"));
89
90 // empty CI Diagonal Hamiltonian
91 std::fill_n(diagH, nStr, MatsT(0.));
92 std::vector<size_t> j(3, 0);
93 std::vector<size_t> nEras(3, 0);
94 std::vector<size_t> nDras(3, 0);
95 std::vector<size_t> elecPos(nCorrE, 0);
96 int_matrix empty = {{}};
97
98 // Alpha Part for 1C or the whole build for 2C and 4C
99 size_t iCatJ = 0;
100 for (auto iEJ = 0ul; iEJ <= mxElec; iEJ++)
101 for (auto iHJ = 0ul; iHJ <= mxHole; iHJ++, iCatJ++) {
102
103 nEras[0] = nActO[0] - iHJ;
104 nEras[2] = iEJ;
105 if (nEras[0] + nEras[2] > nCorrE or nActO[1] + nEras[0] + nEras[2] < nCorrE)
106 continue;
107
108 nEras[1] = nCorrE - nEras[0] - nEras[2];
109 nDras[0] = LCat[iCatJ][1];
110 nDras[1] = LCat[iCatJ][2]/LCat[iCatJ][1];
111 nDras[2] = LCat[iCatJ][3]/LCat[iCatJ][2];
112
113 MatsT* diagH_J = diagH + fCat[iCatJ];
114
115 std::vector<int_matrix> addr_array;
116 std::vector<int_matrix> de_addr_array;
117 for (auto i = 0ul; i < 3; i++){
118 if (nActO[i] == 0) {
119 addr_array.push_back(empty);
120 de_addr_array.push_back(empty);
121 } else {
122 addr_array.push_back(RASString->buildAddressingArray(nEras[i], nActO[i]));
123 de_addr_array.push_back(RASString->buildDeAddressingArray(addr_array[i]));
124 }
125 }
126
127 for (j[2] = 0; j[2] < nDras[2]; j[2]++)
128 for (j[1] = 0; j[1] < nDras[1]; j[1]++)
129 for (j[0] = 0; j[0] < nDras[0]; j[0]++) {
130 size_t j_addr = j[0] * LCat[iCatJ][0] + j[1] * LCat[iCatJ][1] + j[2] * LCat[iCatJ][2];
131 size_t orb_offset = 0;
132 size_t e_offset = 0;
133 for (auto i = 0ul; i < 3; i++){
134 if (nActO[i] == 0) continue;
135 std::vector<size_t> eP = RASString->address2DetString(j[i], addr_array[i],
136 de_addr_array[i]);
137 for (auto pos = 0ul; pos < nEras[i]; pos++)
138 elecPos[pos+e_offset] = eP[pos] + orb_offset;
139 orb_offset += nActO[i];
140 e_offset += nEras[i];
141 }
142 // Update 1e part
143 // hCore_ii
144 for (auto pos_i = 0ul; pos_i < nCorrE; pos_i++) {
145 diagH_J[j_addr] += hCore(elecPos[pos_i], elecPos[pos_i]);
146 // Update 2e part
147 // moERI: (ii|jj) or -(ij|ji)
148 for (auto pos_j = 0ul; pos_j < nCorrE; pos_j++) {
149 diagH_J[j_addr] += MatsT(0.5) * moERI(elecPos[pos_i], elecPos[pos_i],
150 elecPos[pos_j], elecPos[pos_j]);
151 diagH_J[j_addr] -= MatsT(0.5) * moERI(elecPos[pos_i], elecPos[pos_j],
152 elecPos[pos_j], elecPos[pos_i]);
153 }
154 }
155 }
156 }
157
158 if (nC != 1) {
159#ifdef _DEBUG_CIBUILDER_RASCI_IMPL
160 prettyPrintSmart(std::cout,"LL RASCI diag Hamiltonian", diagH, nStr, 1, nStr);
161#endif
162
163 double RASdiagHdur = tock(RASdiagHSt);
164 std::cout << "\nRASDiagH - DURATION = " << std::setprecision(8)
165 << RASdiagHdur << " s." << std::endl;
166
167 } else CErr("Only 2C RASCI diag Hamiltonian is implemented");
168
169 } // RASCI::buildDiagH

172 /*
173 * Build RASCI Sigma, Matrix-vector product
174 */
175 template <typename MatsT, typename IntsT>
176 void RASCI<MatsT,IntsT>::buildSigma(MCWaveFunction<MatsT, IntsT> & mcwfn,
177 size_t nVec, MatsT * C, MatsT * Sigma) {
178
179// auto RASSigmaTSt = tick();
180
181 RASCI_LOOP_INIT(); // check top for variable definitions
182
183#ifdef _DEBUG_CIBUILDER_RASCI_IMPL
184 std::cout<<"Sigma build starts"<<std::endl;
185 prettyPrintSmart(std::cout,"LL RASCI Sigma Build -- C", C, nStr, nVec, nStr);
186#endif
187
188 // Allocate SCR for D and G subblocks
189 size_t maxdim = 0; // max dimension of D and G subblocks
190 size_t nNZatMD = 0; // number of nonzero at maxdim
191 size_t nDetatMD = 0; // number of determinants at maxdim
192 size_t maxnNZ = 0; // max number of nonzero elements
193 size_t maxdSCR = 0; // max dimension of D and G scratch array
194 size_t mindSCR = 0; // min dimension of D and G scratch array (when nNonzero = 1)
195 size_t maxDetCat = 0; // max number of determinants in one catergory
196
197 std::vector<size_t> nDetL(3, 0);
198 std::vector<size_t> nDetJ(3, 0);
199 std::vector<size_t> nDetK(3, 0);
200
201 for (auto p = 0ul; p < 3; p++)
202 for (auto q = 0ul; q < 3; q++)
203 for (auto iCatL = 0ul; iCatL < nCat; iCatL++)
204 for (auto iCatK = 0ul; iCatK < nCat; iCatK++) {
205 if (auto exList = RASString->find_exlist(p, q, iCatL, iCatK)) {
206 size_t nNonzero = exList->nNonZero();
207 if (nNonzero * LCat[iCatL][3] > maxdim) {
208 maxdim = nNonzero * LCat[iCatL][3];
209 nNZatMD = nNonzero;
210 nDetatMD = LCat[iCatL][3];
211 }
212 if (nNonzero > maxnNZ) maxnNZ = nNonzero;
213
214 nDetL[0] = LCat[iCatL][1];
215 nDetL[1] = LCat[iCatL][2]/LCat[iCatL][1];
216 nDetL[2] = LCat[iCatL][3]/LCat[iCatL][2];
217
218 size_t nDetSub = (p == q) ? LCat[iCatL][3]/nDetL[p]
219 : LCat[iCatL][3]/(nDetL[p]*nDetL[q]);
220 if (nNonzero * nDetSub > maxdSCR) maxdSCR = nNonzero * nDetSub;
221 if (nDetSub > mindSCR) mindSCR = nDetSub;
222 }
223 }
224 std::cout<<"maxdim: "<<maxdim<<std::endl;
225 std::cout<<"nNZatMD: "<<nNZatMD<<", nDetatMD: "<<nDetatMD<<std::endl;
226 std::cout<<"maxdSCR: "<<maxdSCR<<", mindSCR: "<<mindSCR<<std::endl;
227
228 for (auto iCatL = 0ul; iCatL < nCat; iCatL++)
229 if (LCat[iCatL][3] > maxDetCat) maxDetCat = LCat[iCatL][3];
230 std::cout<<"maxDetCat: "<<maxDetCat<<std::endl;
231
232 bool DoBatch = false;
233 MatsT* Ci = C;
234 MatsT* HC = Sigma;
235 size_t maxdimD = maxdim;
236 size_t maxdimG = maxdim;
237 size_t ndSCR = maxdSCR;
238 size_t nrSCR= int((maxdim-1)/ndSCR) + 1; // ratio of maxdim over maxdSCR
239 size_t maxAvlMem, nNZAvl, nBatch;
240 int nNZG = 0;
241 int nNZD = 0;
242
243 // Determine the number of OpenMP threads
244 int nthreads = GetNumThreads();
245// size_t totalSCR = (maxnNZ*maxnNZ + maxdSCR + maxdSCR) * nthreads;
246
247 /*-------------------------------------*/
248 /* Determine whether there is enough */
249 /* memory to store the maxdim block D */
250 /* and G. If not, do batching. */
251 /*-------------------------------------*/
252
253// int check_mem = 0;
254 MatsT* DBlk = nullptr;
255 MatsT* GBlk = nullptr;
256 MatsT* ERIscr = nullptr;
257 MatsT* Dscr = nullptr;
258 MatsT* Gscr = nullptr;
259
260// std::cout << mem << std::endl;
261 try{
262 ERIscr = mem.template malloc<MatsT>(maxnNZ*maxnNZ*nthreads);
263 Dscr = mem.template malloc<MatsT>(ndSCR*nthreads);
264 Gscr = mem.template malloc<MatsT>(ndSCR*nthreads);
265 DBlk = mem.template malloc<MatsT>(maxdimD);
266 GBlk = mem.template malloc<MatsT>(maxdimG);
267 } catch (std::bad_alloc& ba) {
268 if (not ERIscr) CErr ("Not enough memory for RAS sigma.");
269 std::cout<<"---- modifying batching ----"<<std::endl;
270 if (Dscr) mem.free(Dscr);
271 if (Gscr) mem.free(Gscr);
272 if (DBlk) mem.free(DBlk);
273 if (GBlk) mem.free(GBlk);
274
275// mem.print_free();
276 maxAvlMem = mem.template max_avail_allocatable<MatsT>();
277 // to determine ndSCR
278 if ((maxAvlMem / (ndSCR*2)) < (nrSCR + nthreads))
279 ndSCR = std::max(maxAvlMem / ((nrSCR + nthreads)*2), mindSCR);
280 Dscr = mem.template malloc<MatsT>(ndSCR*nthreads);
281 Gscr = mem.template malloc<MatsT>(ndSCR*nthreads);
282 // to determine maxdimD and maxdimG
283 maxAvlMem = mem.template max_avail_allocatable<MatsT>();
284 std::cout<<"maxAvlMem: "<<maxAvlMem<<std::endl;
285 if (maxAvlMem < 2 * maxDetCat) CErr("Not enough memory for RAS sigma.");
286 nNZAvl = maxAvlMem / nDetatMD;
287 nBatch = nNZatMD / nNZAvl;
288 while (nNZG < 1 or (2*nNZG) < nNZD) {
289 nBatch += 1;
290 nNZD = nNZatMD / nBatch + 1;
291 nNZG = nNZAvl - nNZD;
292 }
293// std::cout<<"nNZD: "<<nNZD<<", nNZG: "<<nNZG<<std::endl;
294 maxdimG = nDetatMD * nNZG;
295 GBlk = mem.template malloc<MatsT>(maxdimG);
296 maxdimD = mem.template max_avail_allocatable<MatsT>();
297 if (maxdimD < maxDetCat) CErr("Not enough memory for RAS sigma.");
298 DBlk = mem.template malloc<MatsT>(maxdimD);;
299// std::cout<<"maxdimG: "<<maxdimG<<", maxdimD: "<<maxdimD<<std::endl;
300 DoBatch = true;
301// std::cout<<"Batching done!"<<std::endl;
302 }
303
304 if (DoBatch) std::cout<<"Doing batch"<<std::endl;
305
306 // Compute nBatch
307 std::vector<size_t> nBatchrs(9*nCat, 1);
308 std::vector<size_t> nBatchpq(9*nCat, 1);
309 std::shared_ptr<const ExcitationList> rasList;
310 size_t nNZL, ndCat, ndim;
311 if (DoBatch) {
312 for (auto I = 0ul; I < 3; I++)
313 for (auto J = 0ul; J < 3; J++)
314 for (auto L = 0ul; L < nCat; L++) {
315 for (auto iCat = 0ul; iCat < nCat; iCat++) {
316 rasList = RASString->find_exlist(I, J, L, iCat);
317 if (rasList) break;
318 }
319 if (rasList == nullptr) continue;
320 nNZL = rasList->nNonZero();
321 ndCat = LCat[L][3];
322 ndim = nNZL * ndCat;
323 nDetL[0] = LCat[L][1];
324 nDetL[1] = LCat[L][2]/LCat[L][1];
325 nDetL[2] = LCat[L][3]/LCat[L][2];
326 size_t nDetSub = (I == J) ? ndCat/nDetL[I] : ndCat/(nDetL[I] * nDetL[J]);
327
328 nNZAvl = ndSCR / nDetSub;
329 nBatchrs[L+J*nCat+I*3*nCat] = 1 + int((nNZL - 1) / nNZAvl);
330 nBatchpq[L+J*nCat+I*3*nCat] = 1 + int((nNZL - 1) / nNZAvl);
331 if (ndim > maxdimD) {
332 nNZAvl = maxdimD / ndCat;
333 size_t prenBrs = nBatchrs[L+J*nCat+I*3*nCat];
334 nBatchrs[L+J*nCat+I*3*nCat] = std::max(1 + ((nNZL - 1) / nNZAvl), prenBrs);
335 }
336 if (ndim > maxdimG) {
337 nNZAvl = maxdimG / ndCat;;
338 size_t prenBpq = nBatchpq[L+J*nCat+I*3*nCat];
339 nBatchpq[L+J*nCat+I*3*nCat] = std::max(1 + ((nNZL - 1) / nNZAvl), prenBpq);
340 }
341 }
342 }
343
344
345 // Initalize Sigma
346 std::fill_n(Sigma, nStr*nVec, MatsT(0.));
347
348 size_t nNZrs, iCatJ, iCatL, nrsBatch, nNZrsof, nNZrsB, nNZpq, npqBatch, nNZpqB, nNZpqof;
349 double symmFc;
350 std::shared_ptr<const ExcitationList> LrsList, LpqList;
351
352 auto nLAThreads = GetLAThreads();
353 SetLAThreads(1);
354 for (auto iVec = 0ul; iVec < nVec; iVec++, HC+=nStr, Ci+=nStr) {
355
356// auto RASSigmaVSt = tick();
357
358 // Alpha part for 1C or the whole build for 2C and 4C
359 for (auto rBlk = 0ul; rBlk < 3; rBlk++)
360 for (auto sBlk = 0ul; sBlk < 3; sBlk++) {
361 if (nActO[rBlk] == 0 or nActO[sBlk] == 0) continue;
362 iCatJ = 0;
363 for (auto iEJ = 0ul; iEJ <= mxElec; iEJ++)
364 for (auto iHJ = 0ul; iHJ <= mxHole; iHJ++, iCatJ++) {
365 if (LCat[iCatJ][3] == 0) continue;
366 for (iCatL = 0; iCatL < nCat; iCatL++) {
367 LrsList = RASString->find_exlist(sBlk, rBlk, iCatJ, iCatL);
368 if (LrsList) break;
369 }
370 if (fCat[iCatL] < 0 or LrsList == nullptr) continue;
371 nDetJ[0] = LCat[iCatJ][1];
372 nDetJ[1] = LCat[iCatJ][2]/LCat[iCatJ][1];
373 nDetJ[2] = LCat[iCatJ][3]/LCat[iCatJ][2];
374 nNZrs = LrsList->nNonZero();
375 MatsT * Ci_L = Ci + fCat[iCatL];
376
377 // Batch rs block
378 nrsBatch = nBatchrs[iCatJ + rBlk*nCat + sBlk*3*nCat];
379 nNZrsB = 1 + ((nNZrs-1)/nrsBatch);
380 for (auto irsB = 0ul; irsB < nrsBatch; irsB++) {
381 nNZrsof = irsB * nNZrsB;
382 if (nNZrsof >= nNZrs) break;
383 else {
384 nNZrsB = std::min(nNZrsB, nNZrs-nNZrsof);
385// std::cout<<std::endl;
386// std::cout<<"irsBatch: "<<irsB<<", nNZrsBatch: "<<nNZrsB<<std::endl;
387
388
389 /*-------------------------------------*/
390 /* Build Block D as [nNZrs, J] */
391 /* Use excitation list as<J|E_sr|L> */
392 /*-------------------------------------*/
393 //auto RASSigma1St = tick();
394 std::fill_n(DBlk, nNZrsB*LCat[iCatJ][3], MatsT(0.));
395 if (rBlk == sBlk) // diagonal
396 SigmaD_diag(Ci_L, DBlk, nNZrsof, nNZrsB, LrsList, rBlk, LCat[iCatJ], nDetJ);
397 else // off-diagonal
398 SigmaD_offdiag(Ci_L, DBlk, nNZrsof, nNZrsB, LrsList, rBlk, sBlk,
399 LCat[iCatJ], LCat[iCatL], nDetJ);
400 //double RASSigma1dur = tock(RASSigma1St);
401 //std::cout << "\nRAS Sigma Step 1 - DURATION = " << std::setprecision(8)
402 // << RASSigma1dur << " s." << std::endl;
403
404 /*-------------------------------------*/
405 /* Update Sigma_1e[K] = */
406 /* \sum_pq hCorP_pq * D[pq, K]; */
407 /*-------------------------------------*/
408 //auto RASSigma2St = tick();
409 MatsT * HC_J = HC + fCat[iCatJ];
410 int thread_id;
411 MatsT *ERISCR, *DSCR, *GSCR;
412 #pragma omp parallel default(shared) firstprivate(ERISCR, DSCR, GSCR) private(thread_id)
413 {
414 thread_id = GetThreadID();
415 ERISCR = ERIscr + maxnNZ*maxnNZ*thread_id;
416 DSCR = Dscr + ndSCR*thread_id;
417 GSCR = Gscr + ndSCR*thread_id;
418
419 if (rBlk == sBlk) { // diagonal
420 int rsoff = 0;
421 for (auto i = 0ul; i < rBlk; i++) rsoff += nActO[i];
422 Sigma1e_diag(HC_J, GSCR, DBlk, DSCR, hCoreP, ERISCR, nNZrsof,
423 nNZrsB, LrsList, rBlk, LCat[iCatJ], nDetJ, rsoff);
424 }
425 else { // off-diagonal
426 int roff = 0;
427 int soff = 0;
428 for (auto i = 0ul; i < rBlk; i++) roff += nActO[i];
429 for (auto i = 0ul; i < sBlk; i++) soff += nActO[i];
430 Sigma1e_offdiag(HC_J, GSCR, DBlk, DSCR, hCoreP, ERISCR, nNZrsof,
431 nNZrsB, LrsList, rBlk, sBlk, LCat[iCatJ], nDetJ, roff, soff);
432 }
433 } // End omp parallel
434 //double RASSigma2dur = tock(RASSigma2St);
435 //std::cout << "\nRAS Sigma Step 2 - DURATION = " << std::setprecision(8)
436 // << RASSigma2dur << " s." << std::endl;
437
438#ifdef _DEBUG_CIBUILDER_RASCI_IMPL
439 prettyPrintSmart(std::cout,"LL Sigma Hamiltonian -- Sigma1e", Sigma, nStr, nVec, nStr);
440#endif
441
442 /*-------------------------------------*/
443 /* Update Sigma 2e parts */
444 /* 2 steps: */
445 /* a. Gemm: for nonzero pq, rs */
446 /* G(pq, J) = (pq|rs) * D(rs,J)*/
447 /* b. Update: Sigma_2e(K) = */
448 /* \sum_pq <K|E_pq|J> * G(pq,J)*/
449 /*-------------------------------------*/
450
451 for (auto pBlk = 0ul; pBlk < 3; pBlk++)
452 for (auto qBlk = 0ul; qBlk < 3; qBlk++) {
453 if (nActO[pBlk] == 0 or nActO[qBlk] == 0) continue;
454 size_t iCatK;
455 for (iCatK = 0; iCatK < nCat; iCatK++) {
456 LpqList = RASString->find_exlist(pBlk, qBlk, iCatJ, iCatK);
457 if (LpqList) break;
458 }
459 if (LpqList == nullptr) continue;
460 if ((pBlk+3*qBlk) > (rBlk+3*sBlk)) continue;
461 if (pBlk != rBlk and qBlk != sBlk and pBlk > rBlk) continue;
462 if ((pBlk+3*qBlk) == (rBlk+3*sBlk)) symmFc = 0.5;
463 else symmFc = 1.;
464
465 nDetK[0] = LCat[iCatK][1];
466 nDetK[1] = LCat[iCatK][2]/LCat[iCatK][1];
467 nDetK[2] = LCat[iCatK][3]/LCat[iCatK][2];
468 nNZpq = LpqList->nNonZero();
469 MatsT * HC_K = HC + fCat[iCatK];
470
471 // Batch pq block
472 npqBatch = nBatchpq[iCatJ+qBlk*nCat+pBlk*3*nCat];
473 nNZpqB = 1 + (nNZpq - 1) / npqBatch;
474 for (auto ipqB = 0ul; ipqB < npqBatch; ipqB++) {
475 nNZpqof = ipqB * nNZpqB;
476 if (nNZpqof >= nNZpq) break;
477 else {
478 nNZpqB = std::min(nNZpqB, nNZpq - nNZpqof);
479// std::cout<<"r: "<<rBlk<<", s: "<<sBlk<<", p: "<<pBlk<<", q: "<<qBlk<<std::endl;
480// std::cout<<"ipqBatch: "<<ipqB<<", nNZpqBatch: "<<nNZpqB<<std::endl;
481
482
483 // a. G(pq, J) = (pq|rs) * D(rs,J)
484 //auto RASSigma3St = tick();
485 std::fill_n(GBlk, nNZpqB*LCat[iCatJ][3], MatsT(0.));
486
487 int poff = 0;
488 int qoff = 0;
489 int roff = 0;
490 int soff = 0;
491 for (auto i = 0ul; i < rBlk; i++) roff += nActO[i];

575 void RASCI<MatsT,IntsT>::computeTDM(MCWaveFunction<MatsT, IntsT> & mcwfn, MatsT * Cm,
576 MatsT * Cn, SquareMatrix<MatsT> & TDM) {

578#ifdef _DEBUG_CIBUILDER_RASCI_IMPL

580#endif
581 auto RAS1rdmSt = tick();

586 // 2C or 1C alpha part
587 std::shared_ptr<const ExcitationList> LrsList;
588 size_t iCatJ, iCatL, r, s, l, l1, l2, j, j1, j2, j3, j23off, iNZ;
589 double sign;

592 std::vector<size_t> LCatAlt2(3, 0);
593 std::vector<size_t> nDetAlt(3, 0);
594
595 for (auto rBlk = 0ul; rBlk < 3; rBlk++)
596 for (auto sBlk = 0ul; sBlk < 3; sBlk++) {
597 if (nActO[rBlk] == 0 or nActO[sBlk] == 0) continue;
598 iCatJ = 0;
599 for (auto iEJ = 0ul; iEJ <= mxElec; iEJ++)
600 for (auto iHJ = 0ul; iHJ <= mxHole; iHJ++, iCatJ++) {

611 if (rBlk == sBlk) {// diagonal

615 #pragma omp parallel for schedule(static, 1) default(shared) private(j1, j2, j3, iNZ, l1, l, j23off, j, sign, r, s)
616 for (j3 = 0; j3 < nDetAlt[2]; j3++)

626 }
627 } // End openMP
628 } else { // off-diagonal
629 reassembleCatDet2RAS(LCat[iCatJ], nDetJ, rBlk, sBlk, LCatAlt, nDetAlt, LCatAlt2, LCat[iCatL]);

638 for (j1 = 0; j1 < nDetAlt[0]; j1++)
