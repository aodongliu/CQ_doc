30 template <typename IntsT>
31 class DirectTPI : public TwoPInts<IntsT> {
32
33 template <typename IntsU>
34 friend class DirectTPI;
35
36 protected:
37 BasisSet &basisSet_;
38 BasisSet &basisSet2_;
39 double threshSchwarz_ = 1e-12;
40 double* schwarz_ = nullptr;
41 double* schwarz2_ = nullptr;
42 Molecule &molecule_;
43
44 public:
45
46 // Constructor
47 DirectTPI() = delete;
48 DirectTPI(CQMemManager &mem, BasisSet &basis, BasisSet &basis2, Molecule &mol, double threshSchwarz):
49 TwoPInts<IntsT>(mem, basis.nBasis, basis2.nBasis),
50 basisSet_(basis), basisSet2_(basis2),molecule_(mol),
51 threshSchwarz_(threshSchwarz) {}
52 DirectTPI( const DirectTPI &other ):
53 DirectTPI(other.memManager(), other.nBasis(), other.nBasis2(), other.threshSchwarz_) {
54 std::copy_n(other.schwarz_, basisSet_.nShell*basisSet_.nShell, schwarz_);
55 std::copy_n(other.schwarz2_, basisSet2_.nShell*basisSet2_.nShell, schwarz2_);
56 }
57 template <typename IntsU>
58 DirectTPI( const DirectTPI<IntsU> &other, int = 0 ):
59 DirectTPI(other.memManager_, other.basisSet_, other.basisSet2_, other.molecule_, other.threshSchwarz_) {
60 if (other.schwarz_) {
61 size_t NS = basisSet().nShell;
62 schwarz_ = this->memManager().template malloc<double>(NS*NS);
63 std::copy_n(other.schwarz_, NS*NS, schwarz_);
64 }
65 if (other.schwarz2_) {
66 size_t NS = basisSet2().nShell;
67 schwarz2_ = this->memManager().template malloc<double>(NS*NS);
68 std::copy_n(other.schwarz2_, NS*NS, schwarz2_);
69 }
70 }
71 DirectTPI( DirectTPI &&other ): TwoPInts<IntsT>(std::move(other)),
72 basisSet_(other.basisSet_), basisSet2_(other.basisSet2_),
73 threshSchwarz_(other.threshSchwarz_),
74 molecule_(other.molecule_),
75 schwarz_(other.schwarz_), schwarz2_(other.schwarz2_) {
76 other.schwarz_ = nullptr;
77 other.schwarz2_ = nullptr;
78 }
79
80 BasisSet& basisSet() { return basisSet_; }
81 BasisSet& basisSet2() { return basisSet2_; }
82 Molecule& molecule() { return molecule_; }
83 double threshSchwarz() const { return threshSchwarz_; }
84 double*& schwarz() { return schwarz_; }
85 double*& schwarz2() { return schwarz2_; }

102 OPERATOR, const HamiltonianOptions&) {}

108 virtual void output(std::ostream &out, const std::string &s = "",
109 bool printFull = false) const {
110 if (s == "")
111 out << " Two particle integral:" << std::endl;
112 else
113 out << " TPI[" << s << "]:" << std::endl;
114 out << " * Contraction Algorithm: ";
115 out << "DIRECT";
116 out << std::endl;
117
118 out << " * Schwarz Screening Threshold = "
119 << threshSchwarz() << std::endl;
120
121 if (printFull) {
122 CErr("Printing Full ERI tensor for Direct contraction NYI.", out);
123 }
124 }

126 virtual void broadcast(MPI_Comm comm = MPI_COMM_WORLD, int root = 0) override {
127 TwoPInts<IntsT>::broadcast(comm, root);
128#ifdef CQ_ENABLE_MPI

130#endif
131 }

135 if(schwarz2_) this->memManager().free(schwarz2_);
136 }
