75 * \brief the OrbitalRotation class. The class can perform
76 * post-SCF orbital rotation using Newton-Ralphson method
77 *
78 * \warning:
79 * make sure the dimension of input are correct
80 * make sure the input RDMs are in same definition.
81 *
82 * Exponential parametrization of the MO coefficient:
83 * |\Psi>_new = exp(X) |\Psi>_old
84 *
85 */

100 OrbitalRotation(
101 MCWaveFunction<MatsT,IntsT> & mcwfn,
102 OrbitalRotationSettings & input_settings):
103 mcwfn_(mcwfn), settings(input_settings) {
104
105 auto & mopart = mcwfn_.MOPartition;
106
107 if (mopart.nCorrO == 0) CErr("the correlated space cannot be empty in orbital rotation");
108
109 if (mopart.nInact == 0) {
110 std::cout << " No Inactive in Orbital Rotations" << std::endl;
111 settings.rotate_inact_correlated = false;
112 settings.rotate_inact_virtual = false;
113 }
114
115 if (mopart.nFVirt == 0) {
116 std::cout << " No Frozen Virtual in Orbital Rotations" << std::endl;
117 settings.rotate_correlated_virtual = false;
118 settings.rotate_inact_virtual = false;
119 }
