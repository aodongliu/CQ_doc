71 InCore4indexTPI& operator=( const InCore4indexTPI &other ) {
72 if (this != &other) { // self-assignment check expected
73 if (this->nBasis() != other.nBasis() or this->snBasis() != other.snBasis()) {
74 this->NB = other.NB;
75 this->sNB = other.sNB;
76 NB2 = other.NB2;
77 sNB2 = other.sNB2;
78 NB3 = other.NB3;
79 malloc(); // reallocate memory
80 }
81 std::copy_n(other.TPI, NB2*sNB2, TPI);
82 }
83 return *this;
84 }
85 InCore4indexTPI& operator=( InCore4indexTPI &&other ) {
86 if (this != &other) { // self-assignment check expected
87 this->memManager().free(TPI);
88 this->NB = other.NB;
89 this->sNB = other.sNB;
90 NB2 = other.NB2;
91 sNB2 = other.sNB2;
92 NB3 = other.NB3;
93 TPI = other.TPI;
94 other.TPI = nullptr;
95 }
96 return *this;
97 }

155 virtual void output(std::ostream &out, const std::string &s = "",
156 bool printFull = false) const {
157 if (s == "")
158 out << " Two particle integral:" << std::endl;
159 else
160 out << " TPI[" << s << "]:" << std::endl;
161 out << " * Contraction Algorithm: ";
162 out << "INCORE (Gemm)";
163 out << std::endl;
164 if (printFull) {
165 out << bannerTop << std::endl;
166 size_t NB = this->nBasis();
167 size_t sNB = this->snBasis();
168 out << std::scientific << std::left << std::setprecision(8);
169 for(auto i = 0ul; i < NB; i++)
170 for(auto j = 0ul; j < NB; j++)
171 for(auto k = 0ul; k < sNB; k++)
172 for(auto l = 0ul; l < sNB; l++){
173 if (std::abs(operator()(i,j,k,l)) > PRINT_SMALL) {
174 out << " (" << i << "," << j << "|" << k << "," << l << ") ";
175 out << operator()(i,j,k,l) << std::endl;
176 }
177 };
178 out << bannerEnd << std::endl;
179 }

182 virtual void broadcast(MPI_Comm comm = MPI_COMM_WORLD, int root = 0) {
183 TwoPInts<IntsT>::broadcast(comm, root);
184
185#ifdef CQ_ENABLE_MPI
186 if( MPISize(comm) > 1 ) {
187 size_t NB2_bcast = NB2;
188 size_t sNB2_bcast = sNB2;
189 MPIBCast(NB2_bcast,root,comm);
190 MPIBCast(sNB2_bcast,root,comm);
191
192 if (NB2_bcast != NB2 or sNB2_bcast != sNB2) {
193 NB2 = this->nBasis()*this->nBasis();
194 sNB2 = this->snBasis()*this->snBasis();
195 NB3 = NB2 * this->snBasis();
196 malloc();
197 }
198
199 MPIBCast(TPI,NB2*sNB2,root,comm);
200 }
201#endif
202
203 }

205 void malloc() {
206 if(TPI) this->memManager().free(TPI);
207 size_t NB4 = NB2*sNB2;
208 try { TPI = this->memManager().template malloc<IntsT>(NB4); }
209 catch(...) {
210 std::cout << std::fixed;
211 std::cout << "Insufficient memory for the full TPI tensor ("
212 << (NB4/1e9) * sizeof(double) << " GB)" << std::endl;
213 std::cout << std::endl << this->memManager() << std::endl;
214 CErr();
215 }
