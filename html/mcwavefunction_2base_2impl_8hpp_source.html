31 void MCWaveFunctionBase::partitionMOSpace(std::vector<size_t> nActO, size_t nCorrE) {
32
33 auto & mopart = this->MOPartition;
34 auto & wfn = referenceWaveFunction();
35
36 if (wfn.nC == 4 and not this->FourCompNoPair)
37 CErr("4C without no-pair approximation is not implemented.");
38
39 if (wfn.nC == 1) {
40 mopart.nElecMO = wfn.nAlphaOrbital();
41 } else {
42 mopart.nElecMO = wfn.nV + wfn.nO;
43 }
44
45
46 if (wfn.nC == 4) {
47 mopart.nMO = 2 * mopart.nElecMO;
48 mopart.nNegMO = mopart.nElecMO;
49 } else {
50 mopart.nMO = mopart.nElecMO;
51 }
52
53 size_t nCorrO = std::accumulate(nActO.begin(), nActO.end(), 0);
54
55 if( nCorrO > mopart.nElecMO or nCorrO <=0 or (nCorrE > wfn.nO) or nCorrE <= 0)
56 CErr("This is not a valid active space, please modify the input!");
57
58 if (wfn.nC == 1) {
59 if (wfn.iCS) {
60 mopart.nCorrEA = nCorrE / 2;
61 mopart.nCorrEB = mopart.nCorrEA;
62 } else {
63 int singleE = wfn.nOA - wfn.nOB;
64 mopart.nCorrEA = (nCorrE + singleE) / 2;
65 mopart.nCorrEB = nCorrE - mopart.nCorrEA;
66 }
67
68 if (nCorrO < mopart.nCorrEA)
69 CErr("The specified number of correlated orbitals can not hold all the correlated electrons");
70
71 mopart.nInact = wfn.nOA - mopart.nCorrEA;
72 } else { // other than 1C
73
74 if (nCorrO < nCorrE)
75 CErr("The specified number of correlated orbitals can not hold all the correlated electrons");
76
77 mopart.nInact = wfn.nO - nCorrE;
78 }
79
80 mopart.nFVirt = mopart.nElecMO - nCorrO - mopart.nInact;
81 mopart.nCorrO = nCorrO;
82 mopart.nCorrE = nCorrE;
83
84 if( mopart.nElecMO < (nCorrO + mopart.nInact) )
85 CErr("This is not a valid active space, please modify the input!");
86
87
88 #define CONSTRUCT_CASSTRINGMANAGER(M_, N_) \
89 std::dynamic_pointer_cast<DetStringManager>( \
90 std::make_shared<CASStringManager>(this->memManager, M_, N_))
91
92 // Initialize String Engine
93 if (mopart.scheme == CAS) {
94 if (wfn.nC == 1) {
95 this->NDet = Comb(nCorrO, mopart.nCorrEA) *
96 Comb(nCorrO, mopart.nCorrEB);
97 this->detStr = CONSTRUCT_CASSTRINGMANAGER(nCorrO, mopart.nCorrEA);
98
99 this->detStrBeta = (wfn.iCS) ? detStr:
100 CONSTRUCT_CASSTRINGMANAGER(nCorrO, mopart.nCorrEB);
101 } else {
102 this->NDet = Comb(nCorrO, nCorrE);
103 this->detStr = CONSTRUCT_CASSTRINGMANAGER(nCorrO, nCorrE);
104 }
105 } else if (mopart.scheme == RAS) {
106 size_t mhRas1 = std::min(nActO[0],mopart.mxHole);
107 size_t meRas3 = std::min(nActO[2],mopart.mxElec);
108 if(wfn.nC ==1) {
109 if(mopart.mxHole > nActO[0]*2 or mopart.mxElec > nActO[2]*2)
110 CErr("Invalid Input for RAS Specification");
111 if((mopart.nCorrEA + mhRas1) > nCorrO)
112 CErr("RAS3 is too small to generate full MxHole.");
113 if(mopart.nCorrEA < meRas3)
114 CErr("RAS1 is too small to generate full MxElec.");
115 std::shared_ptr<RASStringManager> rasStr = std::make_shared<RASStringManager>(
116 this->memManager, nActO, mopart.nCorrEA, mopart.mxHole, mopart.mxElec);
117 std::shared_ptr<RASStringManager> rasStrBeta = (wfn.iCS) ? rasStr:
118 std::make_shared<RASStringManager>(this->memManager, nActO, mopart.nCorrEB,
119 mopart.mxHole, mopart.mxElec);
120 mopart.fCat = genfCat(rasStr->LCategory(), rasStrBeta->LCategory(),
121 rasStr->nCategory(), rasStrBeta->nCategory());
122 this->NDet = rasStr->nString() * rasStrBeta->nString();
123 this->detStr = std::dynamic_pointer_cast<DetStringManager>(rasStr);
124
125 this->detStrBeta = (wfn.iCS) ? detStr:
126 std::dynamic_pointer_cast<DetStringManager>(rasStrBeta);
127 } else { // nC ==2 or 4
128 if(mopart.mxHole > nActO[0] or mopart.mxElec > nActO[2])
129 CErr("Invalid Input for RAS Specification");
130 if((mopart.nCorrE + mhRas1) > nCorrO)
131 CErr("RAS3 is too small to generate full MxHole.");
132 if(mopart.nCorrE < meRas3)
133 CErr("RAS1 is too small to generate full MxElec.");
134 std::shared_ptr<RASStringManager> rasStr = std::make_shared<RASStringManager>(
135 this->memManager, nActO, nCorrE, mopart.mxHole, mopart.mxElec);
136 mopart.fCat = genfCat(rasStr->LCategory(), rasStr->nCategory());
137 this->NDet = rasStr->nString();
138 this->detStr = std::dynamic_pointer_cast<DetStringManager>(rasStr);
139 }
140 mopart.nActOs = nActO; // Number of orbitals for each RAS space.
141 } else
142 CErr("DetStringManager other than CAS or RAS is not implemented");
143
144 // initialize orbital index
145 setActiveSpaceAndReOrder();
146
147 // initalize indices in MOIntsTransformer
148 setMORanges();
149
150 }; // partiation MOSpace

153 void MCWaveFunctionBase::turnOnStateAverage(const std::vector<double> & weight) {
154
155 size_t NS = this->NStates;
156
157 if( weight.size() != NS)
158 CErr("MCSCF needs "+std::to_string(NS)+" weights for state average" );
159
160 this->StateAverage = true;
161 this->SAWeight = std::vector<double>(NS);
162 std::copy_n(weight.begin(), NS, this->SAWeight.begin());

165 // Generate the category offset for nC=1: fCat[iCata+iCatb*nCat]
166 std::vector<int> MCWaveFunctionBase::genfCat(std::vector<std::vector<size_t>> LCata,
167 std::vector<std::vector<size_t>> LCatb, size_t nCata, size_t nCatb) {
168
169 std::vector<int> fCat(nCata * nCatb, 0);
170 size_t nDet = 0;
171
172 size_t iCatb = 0;
173 for (auto iEb = 0; iEb <= this->MOPartition.mxElec; iEb++)
174 for (auto iHb = 0; iHb <= this->MOPartition.mxHole; iHb++, iCatb++) {
175 size_t iCata = 0;
176 for (auto iEa = 0; iEa <= this->MOPartition.mxElec; iEa++)
177 for (auto iHa = 0; iHa <= this->MOPartition.mxHole; iHa++, iCata++) {
178 if (LCata[iCata][3] == 0 or LCatb[iCatb][3] == 0) fCat[iCata + iCatb*nCata] = -2;
179 else if ((iEb+iEa) > this->MOPartition.mxElec or (iHb+iHa) > this->MOPartition.mxHole)
180 fCat[iCata + iCatb*nCata] = -1;
181 else {
182 fCat[iCata + iCatb*nCata] = nDet;
183 nDet += LCata[iCata][3] * LCatb[iCatb][3];
184 }
185 }
186 }
187
188 return fCat;
189
190 } // MCWaveFunctionBase::genfCat

192 // Generate the category offset for nC>1: fCat[iCat]
193 std::vector<int> MCWaveFunctionBase::genfCat(std::vector<std::vector<size_t>> LCat,
194 size_t nCat) {
195
196 std::vector<int> fCat(nCat, 0);
197 size_t nDet = 0;
198
199 for (auto iCat = 0; iCat < nCat; iCat++) {
200 if (LCat[iCat][3] == 0) fCat[iCat] = -2;
201 else {
202 fCat[iCat] = nDet;
203 nDet += LCat[iCat][3];
204 }
205 }
206
207 return fCat;
208
209 }; // MCWaveFunctionBase::genfCat

212 /*
213 * 'N' -> Negative Energy MO, only for 4C
214 * 'I' -> Inactive / inactive
215 * 'A' -> Active "1, 2, 3" for RAS
216 * 'S' -> Secondary / frozen virtual
217 */
218 void MCWaveFunctionBase::setActiveSpaceAndReOrder() {
219
220 std::vector<size_t> nOrbs;
221 std::vector<char> orbIdentifiers;
222 size_t totalElecMO = MOPartition.nElecMO;
223
224 if (referenceWaveFunction().nC == 4) {
225 nOrbs.push_back(MOPartition.nNegMO);
226 orbIdentifiers.push_back('N');
227 }
228
229 // set up numbers
230 nOrbs.push_back(MOPartition.nInact);
231 orbIdentifiers.push_back('I');
232
233 if (MOPartition.scheme != RAS) {
234 nOrbs.push_back(MOPartition.nCorrO);
235 orbIdentifiers.push_back('A');
236 } else {
237 nOrbs.push_back(MOPartition.nActOs[0]);
238 orbIdentifiers.push_back('1');
239 nOrbs.push_back(MOPartition.nActOs[1]);
240 orbIdentifiers.push_back('2');
241 nOrbs.push_back(MOPartition.nActOs[2]);
242 orbIdentifiers.push_back('3');
243 }
244
245 nOrbs.push_back(MOPartition.nFVirt);
246 orbIdentifiers.push_back('S');
247
248 auto & orbIndices = MOPartition.orbIndices;
249 size_t scan_start = 0, scan_end = 0;
250
251 if (orbIndices.size() == 0) {
252 // initialize the MO indices
253 for(auto i = 0; i < nOrbs.size(); i++) {
254 scan_end = scan_start + nOrbs[i];
255 char & scan_id = orbIdentifiers[i];
256 for(size_t j = scan_start; j < scan_end; j++) {
257 orbIndices.push_back(scan_id);
258 }
259 scan_start = scan_end;
260 }
261 } else if (orbIndices.size() == totalElecMO) {
262
263 // generating swapping pairs if necessary
264 std::vector<std::vector<std::pair<size_t, size_t>>> moPairs;
265 moPairs.resize(2, {});
266
267 for(auto i = 0; i < nOrbs.size(); i++) {
268 scan_end = scan_start + nOrbs[i];
269 char & scan_id = orbIdentifiers[i];
270
271 for (size_t j = scan_start; j < scan_end; j++) {
272 if (orbIndices[j] != scan_id) {
273 bool found = false;
274 for (size_t k = scan_end; k < totalElecMO; k++) {
275 if (orbIndices[k] == scan_id) {
276 found = true;
277 moPairs[0].push_back({j+1, k+1});
278 orbIndices[k] = orbIndices[j];
279 orbIndices[j] = scan_id;
280 break;
281 }
282 }
283 if (not found) CErr("inconsistency between number of orbitals and orbial indices");
284 }
285 }
286 scan_start = scan_end;
287 }
288
289 this->swapMOs(moPairs,isAlpha);
290 } else {
291 CErr("wrong sizes set in MOPartition.orbIndices");
292 }
293
294 }; // MCWaveFunctionBase::setActiveSpaceAndReOrder
