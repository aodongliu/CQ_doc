57 /*
58 * \brief rotate MO for one step size using Newton-Raphson:
59 *
60 * steps:
61 * 1. compute Gradient if not computed
62 * 2. compute (approximated) Hessian and inverse it
63 * 3. compute rotation paramemter X = - g / H
64 * 4. compute rotation matrix by matrix exponential U = exp(X)
65 * 5. rotate MO for one step size as C' = C U
66 */
67
68 template <typename MatsT, typename IntsT>
69 void OrbitalRotation<MatsT, IntsT>::rotateMO(EMPerturbation & pert,
70 SquareMatrix<MatsT> & oneRDM, InCore4indexTPI<MatsT> & twoRDM) {
71
72 auto & mopart = mcwfn_.MOPartition;
73 auto & mem = mcwfn_.memManager;
74 auto & mo = mcwfn_.reference().mo[0];
75 size_t nTOrb = mopart.nMO;
76 size_t nTOrb2 = nTOrb * nTOrb;
77 size_t nAO = mo.dimension();
78
79 // offset it by inactive core index
80 auto mo_pointer = mo.pointer();
81
82 // get gradient
83 if (not orbitalGradient_) computeOrbGradient(pert, oneRDM, twoRDM);
84 MatsT * G = orbitalGradient_->pointer();
85
86 // allocate memory
87 MatsT * X = mem.template malloc<MatsT>(nTOrb2);
88 MatsT * U = mem.template malloc<MatsT>(nTOrb2);
89
90 MatsT * H = nullptr;
91 if (settings.alg == ORB_ROT_2ND_ORDER) {
92 H = mem.template malloc<MatsT>(nTOrb2*nTOrb2);
93 } else {
94 H = mem.template malloc<MatsT>(nTOrb2);
95 }
96
97 ProgramTimer::tick("Form Hessian");
98 // X = - H ^{-1} * G
99 if (settings.alg == ORB_ROT_2ND_ORDER) {
100 // this->computeOrbOrbHessian(H);
101 CErr("2nd Order orbital rotation not implemented");
102 } else {
103 this->computeOrbOrbHessianDiag(pert, oneRDM, twoRDM, H);
104 for (auto i = 0ul; i < nTOrb2; i++) X[i] = - G[i] / H[i];
105 }
106 ProgramTimer::tock("Form Hessian");
107
108 ProgramTimer::tick("Rotate MO");
109
110 // Damp X to avoid Large Rotation
111 auto X_max = std::max_element(X, X+nTOrb2,
112 [&] (MatsT a, MatsT b) { return std::norm(a) < std::norm(b);} );
113
114 if(std::abs(*X_max) > settings.XDampTol) {
115 double XDampFc = settings.XDampTol / std::abs(*X_max);
116 auto pos = std::distance(X, X_max);
117 auto j = int(pos / nTOrb);
118 auto i = pos - j*nTOrb;
119 std::cout << " WARNING!: Large Rotation! XMax at X["
120 << i << ", " << j << "] = " << *X_max
121 << ", step scaled by " << XDampFc
122 << "\n" << std::endl;
123
124 blas::scal(nTOrb2, MatsT(XDampFc), X, 1);
125 }
126
127 // U = exp(X)
128 MatExp(nTOrb, X, nTOrb, U, nTOrb, mem);
129
130#ifdef DEBUG_ORBITALROTATION_IMPL
131 double HNorm = lapack::lange(lapack::Norm::Fro, nTOrb, nTOrb, H, nTOrb);
132 prettyPrintSmart(std::cout, " OR H ", H, nTOrb, nTOrb, nTOrb);
133 double XNorm = lapack::lange(lapack::Norm::Fro, nTOrb, nTOrb, X, nTOrb);
134 prettyPrintSmart(std::cout, " OR X ", X, nTOrb, nTOrb, nTOrb);
135 double UNorm = lapack::lange(lapack::Norm::Fro, nTOrb, nTOrb, U, nTOrb);
136 prettyPrintSmart(std::cout, " OR U ", U, nTOrb, nTOrb, nTOrb);
137 std::cout << "OR H Norm = " << HNorm << std::endl;
138 std::cout << "OR X Norm = " << XNorm << std::endl;
139 std::cout << "OR U Norm = " << UNorm << std::endl;
140#endif
141
142 // Orthonormalized U and disable GramSchmidt printining
143 std::cout.setstate(std::ios_base::failbit);
144 size_t NUOrtho = GramSchmidt(nTOrb, 0, nTOrb, U, nTOrb, mem);
145 std::cout.clear();
146
147 if(NUOrtho != nTOrb) CErr("Failed at Orthonormalizing Rotation U Matix.");
148
149 // Rotate MO Coefficient C' = C * U
150 // use X as SCR
151
152 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans,
153 nAO, nTOrb, nTOrb, MatsT(1.), mo_pointer,
154 nAO, U, nTOrb, MatsT(0.), X, nAO);
155 SetMat('N', nAO, nTOrb, MatsT(1.), X, nAO, mo_pointer, nAO);
156
157 ProgramTimer::tock("Rotate MO");
158
159 // free memory
160 if(orbitalGradient_) orbitalGradient_ = nullptr;
161 if(H) mem.free(H);
162 if(X) mem.free(X);
163 if(U) mem.free(U);
164
165 }; // OrbitalRotation::rotateOrbitals()
