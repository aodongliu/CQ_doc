69 // Getter/Setters for overlap
70 inline std::shared_ptr<SquareMatrix<MatsT>> overlapPointer() const {
71 if( not overlap ) CErr("Overlap has not been initialized");
72 return overlap;
73 };
74
75 inline bool hasOverlap() const { return (overlap != nullptr); };
76
77 // Change out the Overlap Matrix
78 void setOverlap(SquareMatrix<MatsT>& s) {
79 if (not overlap or s.dimension() != overlap->dimension()) {
80 size_t nDim = s.dimension();
81 if (overlap and nDim != overlap->dimension()) {
82 std::cout << std::endl;
83 std::cout << "WARNING: The overlap matrix given to setOverlap is a different dimension" << std::endl;
84 std::cout << "WARNING: Changing the dimension of overlap and setting the new matrix" << std::endl << std::endl;
85 }
86 overlap = std::make_shared<SquareMatrix<MatsT>>(s.memManager(), nDim);
87 forwardTrans = std::make_shared<SquareMatrix<MatsT>>(s.memManager(), nDim);
88 backwardTrans = std::make_shared<SquareMatrix<MatsT>>(s.memManager(), nDim);
89 }
90 *overlap = s;
91 computeOrtho();
92 }

96 inline ORTHO_TYPE getOrthoType() const {return orthoType; };
