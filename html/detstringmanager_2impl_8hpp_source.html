44 int_matrix DetStringManager::buildAddressingArray(size_t nE, size_t nOrb) const {
45
46 assert (nE <= nOrb and nOrb != 0 );
47 int_matrix addr_array;
48
49 if (nE == 0) addr_array = {{}};
50 else {
51 addr_array = this->init_int_matrix(nE, nOrb, 0);
52
53 for ( auto k = 1ul; k < nE; k++)
54 for ( auto l = k ; l < nOrb-nE+k+1; l++)
55 for ( auto m = nOrb-l+1; m < nOrb-k+1; m++) {
56 addr_array[k-1][l-1] += Comb(m-1, nE-k);
57 }
58
59 for ( auto l = nE; l < nOrb+1; l++) {
60 addr_array[nE-1][l-1] = l - nE;
61 }
62 }
63
64 return addr_array;
65 }; // DetStringManager::buildAddressingArray

67 int_matrix DetStringManager::buildDeAddressingArray(int_matrix & addr_array) const {
68
69 size_t nE = addr_array.size();
70 size_t nOrb = addr_array[0].size();
71 if (nE == 0) return {{}};
72
73 int_matrix de_addr_array = this->init_int_matrix(nE, nOrb, 0);
74
75 std::copy_n(addr_array[nE - 1].begin(), nOrb, de_addr_array[nE - 1].begin());
76
77 for (int iE = nE - 2; iE >= 0; iE--)
78 for (int iOrb = 1; iOrb < nOrb - 1; iOrb++)
79 de_addr_array[iE][iOrb] = addr_array[iE][iOrb] + de_addr_array[iE + 1][iOrb + 1];
80
81 return de_addr_array;
82
83 }; // DetStringManager::buildDeAddressingArray(int_matrix)

107 size_t DetStringManager::detString2Address(bool * detStr,
108 int_matrix & addr_array) const {
109
110 size_t nE = addr_array.size();
111 size_t nOrb = addr_array[0].size();
112 if (nOrb == 0) return 0;
113
114 std::vector<size_t> elecPos;
115
116 for (auto iE = 0ul, iOrb = 0ul; iOrb < nOrb, iE < nE; iOrb++) {
117 if (detStr[iOrb]) {
118 elecPos.push_back(iOrb);
119 iE ++;
120 }
121 }
122
123 return detString2Address(elecPos, addr_array);
124

127 std::vector<size_t> DetStringManager::address2DetString(size_t addr,
128 int_matrix & addr_array, int_matrix & de_addr_array) const {
129
130 size_t nE = addr_array.size();
131 if (nE == 0) return {};
132
133 size_t nOrb = addr_array[0].size();
134 std::vector<size_t> elecPos;
135 std::pair<int, int> iOrb_search = {nOrb - nE, 0};
136
137 for (int iE = 0; iE < nE; iE++)
138 for (int iOrb = iOrb_search.first; iOrb >= iOrb_search.second; iOrb--) {
139 if (addr >= de_addr_array[iE][iOrb]) {
140 elecPos.push_back(iOrb);
141 addr -= addr_array[iE][iOrb];
142 iOrb_search.first ++;
143 iOrb_search.second = iOrb + 1;
144 break;
145 }
146 }
147
148 return elecPos;
149 }; // DetStringManager::address2DetString

158 /*
159 *
160 * compute all <K|Epq|L> for a given L
161 * Configuration Driven 1e Excitation List: <K|a_i^\dagger a_j |L>
162 * (1) nonzero excitation i,
163 * (2) nonzero excitation j,
164 * (3) address of excited string |K>
165 * (4) sign (+/- 1)
166 */
167 void DetStringManager::computeSingleString1eExcitationList(
168 DetString & detStr, int * exList, size_t nOrb,
169 size_t nE, int_matrix & addrArray) const {
170
171 std::vector<size_t> LOcc = detStr.occupationInfo();
172 std::vector<size_t> LVir = detStr.virtualInfo();
173 /*
174 std::cout << "LOcc.size() = " << LOcc.size() << std::endl;
175 for(auto & i: LOcc) std::cout << " " << i;
176 std::cout << std::endl;
177 std::cout << "LVir.size() = " << LVir.size() << std::endl;
178 for(auto & i: LVir) std::cout << " " << i;
179 std::cout << std::endl;
180 */
181
182 int * exList_ptr = exList;
183 size_t p, q;
184 size_t LAddr = detString2Address(detStr.pointer(), addrArray);
185
186 // case 1: self-excitation, p=q == occ
187 for (auto pOcc = 0ul ; pOcc < nE; pOcc++) {
188 p = LOcc[pOcc];
189 exList_ptr[0] = p;
190 exList_ptr[1] = p;
191 exList_ptr[2] = LAddr;
192 exList_ptr[3] = 1;
193 exList_ptr += 4;
194 }
195
196 // Case 2: q -> p, p=virtual, q=occupied in L
197 for (auto pVir = 0ul; pVir < nOrb - nE; pVir++)
198 for (auto qOcc = 0ul; qOcc < nE; qOcc++) {
199 p = LVir[pVir];
200 q = LOcc[qOcc];
201 exList_ptr[0] = p;
202 exList_ptr[1] = q;
203 detStr.excitation(q,p);
204 exList_ptr[2] = detString2Address(detStr.pointer(), addrArray);
205 exList_ptr[3] = detStr.sign1e(p,q);
206 detStr.excitation(p,q);
207 exList_ptr += 4;
208 }
209
210 //prettyPrintSmart(std::cout, "iStr = " + std::to_string(LAddr),
211 // exList, 4, (nOrb - nE + 1)*nE, 4);//, 1, 10, 8);
212
213 }; // DetStringManager::computeSingleString1eExcitationList

216 /*
217 * Compute the excitation list <K|Eij|L>
218 * L is formed from two DetStrings in difference spaces,
219 * i.e. detsp and detsq
220 * Configuration Driven 1e Excitation List:
221 * <K|a_i^\dagger a_j |L>
222 * (1) nonzero excitation i,
223 * (2) nonzero excitation j,
224 * (3) address of excited string |K> in p block,
225 * (4) address of excited string |K> in q block,
226 * (5) sign (+/- 1)
227 */
228 void DetStringManager::computeTwoString1eExcitationList(
229 DetString & detsp, DetString & detsq, int * exList,
230 int Sgnoff, bool pqseq,
231 int_matrix & addrArrayp, int_matrix & addrArrayq) const {
232
233 std::vector<size_t> qOcc = detsq.occupationInfo();
234 std::vector<size_t> pVir = detsp.virtualInfo();
235
236 int * exList_ptr = exList;
237 size_t i, j;
238
239 for (auto jq = 0ul; jq < detsq.n1s(); jq++)
240 for (auto ip = 0ul; ip < detsp.n0s(); ip++) {
241 j = qOcc[jq];
242 i = pVir[ip];
243 detsq.flip(j);
244 detsp.flip(i);
245
246 int Sign1e;
247 if (pqseq) {
248 DetString tmp = detsp + detsq;
249 Sign1e = Sgnoff * tmp.sign1e(i, j + detsp.size());
250 } else {
251 DetString tmp = detsq + detsp;
252 Sign1e = Sgnoff * tmp.sign1e(j, i + detsq.size());
253 }
254
255 exList_ptr[0] = i;
256 exList_ptr[1] = j;
257 exList_ptr[2] = detString2Address(detsp.pointer(), addrArrayp);
258 exList_ptr[3] = detString2Address(detsq.pointer(), addrArrayq);
259 exList_ptr[4] = Sign1e;
260 exList_ptr += 5;
261
262 detsq.flip(j);
263 detsp.flip(i);
264 }
265 } // DetStringManager::computeTwoString1eExcitationList

268 ExcitationList DetStringManager::computeIntraCASExcitationList(
269 size_t nOrb, size_t nE, int_matrix & addrArray) const {
270
271 size_t nNZ = nE * (nOrb - nE + 1);
272 size_t nStr = Comb(nOrb, nE);
273 CQMemManager & mem = this->memManager_;
274 ExcitationList exList(mem, 4, nNZ, nStr);
275
276 StringPermutator strList(nOrb, nE);
277
278 size_t iStr = 0;
279
280 // TODO: Parallel this loop
281 while (iStr != nStr) {
282
283 const std::vector<size_t> & eLocs = strList.permutedPositions();
284 DetString L(mem, nOrb, eLocs);
285
286 computeSingleString1eExcitationList(L,
287 exList.pointerAtDet(iStr), nOrb, nE, addrArray);
288
289 iStr = strList.next();
290 }
291
292#ifdef _DEBUG_DETSTRING
293 exList.output(std::cout);

295 return exList;

299 /*
300 * Compute excitation list between spaces
301 * e.g off-diagonal block in RAS string
302 * exList dimensions: nStr2_: q block |L> address,
303 * nStr1_: p block |L> address.
304 */
305 ExcitationList DetStringManager::computeInterCASExcitationList(
306 size_t nOp, size_t nEp, size_t nOq, size_t nEq,
307 int Sgnoff, bool pqseq, int_matrix & addr_array_pK,
308 int_matrix & addr_array_qK) const {
309
310 CQMemManager & mem = this->memManager_;
311 size_t nVp = nOp - nEp;
312 size_t nNZ = nEq * nVp;
313 size_t nStrp = Comb(nOp, nEp);
314 size_t nStrq = Comb(nOq, nEq);
315
316 ExcitationList exList(mem, 5, nNZ, nStrp, nStrq);
317
318 StringPermutator strList_qL(nOq, nEq); // SOqL
319 StringPermutator strList_pL(nOp, nEp); // SOpL
320
321 size_t iStr_qL = 0;
322 size_t iStr_pL = 0;
323
324 // TODO: Parallel this loop
325 while (iStr_qL != nStrq) {
326 const std::vector<size_t> & eLocs_qL = strList_qL.permutedPositions();
327 DetString tmpQ(mem, nOq, eLocs_qL);
328
329 strList_pL.reset_seed();
330 iStr_pL = 0;
331
332 while (iStr_pL != nStrp) {
333 const std::vector<size_t> & eLocs_pL = strList_pL.permutedPositions();
334 DetString tmpP(mem, nOp, eLocs_pL);
335
336 computeTwoString1eExcitationList(tmpP, tmpQ,
337 exList.pointerAtDet(iStr_pL, iStr_qL),
338 Sgnoff, pqseq, addr_array_pK, addr_array_qK);
339
340 iStr_pL = strList_pL.next();
341 }
342 iStr_qL = strList_qL.next();
343 }
344
345#ifdef _DEBUG_DETSTRING
346 exList.output(std::cout);
347#endif
348 return exList;
349
350 } // DetStringManager::computeInterCASExcitationList
