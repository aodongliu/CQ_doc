40 template <typename MatsT, typename IntsT>
41 void NEOFockBuilder<MatsT,IntsT>::formepJ(SingleSlater<MatsT,IntsT>& ss,
42 bool increment)
43 {
44 // Validate internal pointers
45 if( this->aux_ss == nullptr )
46 CErr("aux_ss uninitialized in formepJ!");
47 if( this->outMat == nullptr )
48 CErr("outMat uninitialized in formepJ!");
49 if( contraction == nullptr )
50 CErr("contraction uninitialized in formepJ!");
51
52 // Decide onePDM to use
53 PauliSpinorSquareMatrices<MatsT>& contract1PDM
54 = increment ? *this->aux_ss->deltaOnePDM : *this->aux_ss->onePDM;
55
56 size_t NB = ss.basisSet().nBasis;
57
58 // Zero out J and K[i]
59 if(not increment)
60 this->outMat->clear();
61
62 std::vector<TwoBodyContraction<MatsT>> contract =
63 { {contract1PDM.S().pointer(), this->outMat->pointer(), true, COULOMB} };
64
65 EMPerturbation pert;
66 contraction->twoBodyContract(ss.comm, contract, pert);
67 }

69 template <typename MatsT, typename IntsT>
70 std::vector<double> NEOFockBuilder<MatsT,IntsT>::formepJGrad(
71 SingleSlater<MatsT,IntsT>& ss, EMPerturbation& pert, double xHFX) {
72
73 if( this->aux_ss == nullptr )
74 CErr("aux_ss uninitialized in formepJGrad!");
75 if( gradTPI == nullptr )
76 CErr("gradTPI uninitialized in formepJGrad!");
77
78 size_t NB = ss.basisSet().nBasis;
79 size_t nGrad = 3*ss.molecule().nAtoms;
80 CQMemManager& mem = ss.memManager;
81
82 // Form contraction
83 std::unique_ptr<GradContractions<MatsT,IntsT>> contract = nullptr;
84 if ( std::dynamic_pointer_cast<InCore4indexTPI<IntsT>>((*gradTPI)[0]) ) {
85 contract = std::make_unique<InCore4indexGradContraction<MatsT,IntsT>>(*gradTPI);
86 }
87 else if ( std::dynamic_pointer_cast<DirectTPI<IntsT>>((*gradTPI)[0]) ) {
88 contract = std::make_unique<DirectGradContraction<MatsT,IntsT>>(*gradTPI);
89 }
90 else
91 CErr("Gradients of RI NYI!");
92 // Assume that the order of the TPI is the same between the regular and
93 // gradient integrals
94 contract->contractSecond = contraction->contractSecond;
95
96 // Create contraction list
97 std::vector<std::vector<TwoBodyContraction<MatsT>>> cList;
98
99 std::vector<SquareMatrix<MatsT>> JList;
100 JList.reserve(nGrad);
101
102 for( auto iGrad = 0; iGrad < nGrad; iGrad++ ) {
103 std::vector<TwoBodyContraction<MatsT>> tempCont;
104
105 // Coulomb
106 JList.emplace_back(mem, NB);
107 JList.back().clear();
108 tempCont.push_back(
109 {this->aux_ss->onePDM->S().pointer(), JList.back().pointer(), true, COULOMB}
110 );
111
112 cList.push_back(tempCont);
113 }
114
115 // Contract to J/K
116 contract->gradTwoBodyContract(MPI_COMM_WORLD, true, cList, pert);
117
118 // Contract to gradient
119 std::vector<double> gradient;
120 PauliSpinorSquareMatrices<MatsT> twoEGrad(mem, NB, false, false);
121
122 for( auto iGrad = 0; iGrad < nGrad; iGrad++ ) {
123
124 // G[S] = -2 * J
125 // TODO: The negative here is implicitly taking care of the
126 // electron/proton charges.
127 twoEGrad.S() = -2. * JList[iGrad];
128
129 double gradVal = ss.template computeOBProperty<SCALAR>(
130 twoEGrad.S().pointer()
131 );
132 gradient.push_back(0.25*gradVal);
133
134 }
135
136 return gradient;
137
138 }

140 template <typename MatsT, typename IntsT>
141 void NEOFockBuilder<MatsT,IntsT>::formFock(
142 SingleSlater<MatsT,IntsT>& ss, EMPerturbation& empert, bool increment,
143 double xHFX)
144 {
145 if( this->upstream == nullptr )
146 CErr("Upstream FockBuilder uninitialized in formFock!");
147
148 // Call all upstream FockBuilders
149 this->upstream->formFock(ss, empert, increment, xHFX);
150
151 formepJ(ss, increment);
152
153 *ss.twoeH -= 2. * *this->outMat;
154 *ss.fockMatrix -= 2. * *this->outMat;
155 }

157 template <typename MatsT, typename IntsT>
158 std::vector<double> NEOFockBuilder<MatsT,IntsT>::getGDGrad(
159 SingleSlater<MatsT,IntsT>& ss, EMPerturbation& pert, double xHFX) {
160 if( this->upstream == nullptr )
161 CErr("Upstream FockBuilder uninitialized in getGDGrad!");
162
163 size_t nGrad = 3*ss.molecule().nAtoms;
164
165 std::vector<double> gradient = this->upstream->getGDGrad(ss, pert, xHFX);
166 std::vector<double> epjGrad = formepJGrad(ss, pert, xHFX);
167
168 std::transform(gradient.begin(), gradient.end(), epjGrad.begin(),
169 gradient.begin(), std::plus<double>());
170
171 return gradient;
172
173 };

207#ifdef CQ_ENABLE_MPI

219 return x->isGGA();

278 std::vector<double> integrateEPCEnergy(nthreads,0.);

281 // ----------------------------------------------------------//

283 XCEnergy = 0.;
284 double *SCRATCHNBNB =
285 ss.memManager.template malloc<double>(NTNB2);
286 double *SCRATCHNBNP =
287 ss.memManager.template malloc<double>(NTNPPB*NB);

289 double *DenS, *DenZ, *DenX ;
290 DenS = ss.memManager.template malloc<double>(NTNPPB);
291
292 if( ss.onePDM->hasZ() )
293 DenZ = ss.memManager.template malloc<double>(NTNPPB);
294
295 if( ss.onePDM->hasXY() )
296 CErr("Relativistic NEO-Kohn-Sham NYI!", std::cout);

311
312 // These quantities are only used for GGA functionals

359 ss.memManager.template malloc<double>(NTNPPB*aux_NB);

361 // Density pointers for auxiliary system

363 aux_DenS = ss.memManager.template malloc<double>(NTNPPB);

385 aux_ZgammaVar1 = ss.memManager.template malloc<double>(NTNPPB);
386 aux_ZgammaVar2 = ss.memManager.template malloc<double>(NTNPPB);

389 aux_GDenZ = ss.memManager.template malloc<double>(3*NTNPPB);
390
391 if( this->aux_ss->onePDM->hasXY() )
392 CErr("Relativistic NEO-Kohn-Sham NYI!", std::cout);

394 // Gamma U-Variables
395 aux_U_gamma = ss.memManager.template malloc<double>(3*NTNPPB);
396 aux_dVU_gamma = ss.memManager.template malloc<double>(3*NTNPPB);
397 }
398

401 aux_epsSCR = ss.memManager.template malloc<double>(NTNPPB);

406 }

414 this->aux_ss->onePDM->real_part());
415
416
417 // ---------------end NEO------------------------------------------------
418
419 // --------------Cross Memory--------------------------------------------

452 double epsScreen = intParam.epsilon / nAtoms /
453 intParam.nAng / intParam.nRad;
454
455 epsScreen = std::max(epsScreen,
456 std::numeric_limits<double>::epsilon());
457
458 size_t NPts = batch.size();
459 size_t IOff = NBE*NPts;
460
461 size_t thread_id = GetThreadID();
462 size_t TIDNPPB = thread_id * NPtsMaxPerBatch;
463
464#if VXC_DEBUG_LEVEL >= 1
465 // TIMING
466 auto topevalDen = std::chrono::high_resolution_clock::now();

468
469 // --------------Main System------------------------------------------
470 // Setup local pointers
471 double * SCRATCHNBNB_loc = SCRATCHNBNB + thread_id * NB2;

481
482 // local vxc energy and U, V pointer

498 double * dVU_gamma_SCR_loc = dVU_gamma_SCR + 3*TIDNPPB;

502
503 // ---------------End Main System------------------------------------
504
505
506 // ---------------Auxiliary System-----------------------------------
507 // aux local scratch pointers
508 double * AUX_SCRATCHNBNB_loc, * AUX_SCRATCHNBNP_loc;
509 AUX_SCRATCHNBNB_loc = AUX_SCRATCHNBNB + thread_id * aux_NB2;
510 AUX_SCRATCHNBNP_loc = AUX_SCRATCHNBNP + thread_id * aux_NB*NPtsMaxPerBatch;
511
512 // aux local density pointers
513 double * aux_DenS_loc, * aux_DenZ_loc;
514 aux_DenS_loc = aux_DenS + TIDNPPB;
515 aux_DenZ_loc = aux_DenZ + TIDNPPB;
516
517 // aux local density gradient pointers
518 double *aux_GDenS_loc, *aux_GDenZ_loc;
519 aux_GDenS_loc = aux_GDenS + 3*TIDNPPB;
520 aux_GDenZ_loc = aux_GDenZ + 3*TIDNPPB;
521
522 // aux local U and V pointers
523 double * aux_U_n_loc, * aux_dVU_n_loc, * aux_U_gamma_loc, * aux_dVU_gamma_loc;
524 aux_U_n_loc = aux_U_n + 2*TIDNPPB;
525 aux_dVU_n_loc = aux_dVU_n + 2*TIDNPPB;
526 aux_U_gamma_loc = aux_U_gamma + 3*TIDNPPB;
527 aux_dVU_gamma_loc = aux_dVU_gamma + 3*TIDNPPB;
528
529 // aux local Z pointers
530 double * aux_ZrhoVar1_loc, * aux_ZgammaVar1_loc, * aux_ZgammaVar2_loc;
531 aux_ZrhoVar1_loc = aux_ZrhoVar1 + TIDNPPB;
532 aux_ZgammaVar1_loc = aux_ZgammaVar1 + TIDNPPB;
533 aux_ZgammaVar2_loc = aux_ZgammaVar2 + TIDNPPB;
534
535 // aux local multiple functional pointers
536 double * aux_epsSCR_loc, * aux_dVU_n_SCR_loc, * aux_dVU_gamma_SCR_loc;
537 aux_epsSCR_loc = aux_epsSCR + TIDNPPB;
538 aux_dVU_n_SCR_loc = aux_dVU_n_SCR + 2*TIDNPPB;
539 aux_dVU_gamma_SCR_loc = aux_dVU_gamma_SCR + 3*TIDNPPB;
540
541 // ---------------End Auxiliary System--------------------------------
542
543 // ---------------Cross Between Main and Auxiliary system-------------
544 double * cross_U_gamma_loc, *cross_dVU_gamma_loc, *ZgammaVar3_loc;
545 if (epcisGGA) {
546 cross_U_gamma_loc = cross_U_gamma + 4*TIDNPPB;
547 cross_dVU_gamma_loc = cross_dVU_gamma + 4*TIDNPPB;
548 ZgammaVar3_loc = ZgammaVar3 + TIDNPPB;
549 }
550 // ---------------End Cross-------------------------------------------
551
552 // This evaluates the V variables for all components of the main system
553 // (Scalar, MZ (UKS) and Mx, MY (2 Comp))
554 evalDen((isGGA || epcisGGA ? GRADIENT : NOGRAD), NPts, NBE, NB, subMatCut,
555 SCRATCHNBNB_loc, SCRATCHNBNP_loc, Re1PDM->S().pointer(), DenS_loc,
556 GDenS_loc, GDenS_loc + NPts, GDenS_loc + 2*NPts, BasisEval);
557
558 // evaluate density for auxiliary components
559 evalDen((epcisGGA ? GRADIENT : NOGRAD), NPts, aux_NBE,
560 aux_NB, aux_subMatCut,
561 AUX_SCRATCHNBNB_loc, AUX_SCRATCHNBNP_loc, aux_Re1PDM->S().pointer(), aux_DenS_loc,
562 aux_GDenS_loc, aux_GDenS_loc + NPts, aux_GDenS_loc + 2*NPts, aux_BasisEval);
563
564
565 if ( ss.onePDM->hasZ() )
566 evalDen((isGGA || epcisGGA ? GRADIENT : NOGRAD), NPts, NBE, NB, subMatCut,
567 SCRATCHNBNB_loc ,SCRATCHNBNP_loc, Re1PDM->Z().pointer(), DenZ_loc,
568 GDenZ_loc, GDenZ_loc + NPts, GDenZ_loc + 2*NPts, BasisEval);
569
570 if ( this->aux_ss->onePDM->hasZ() )
571 evalDen((epcisGGA ? GRADIENT : NOGRAD), NPts, aux_NBE,
572 aux_NB, aux_subMatCut,
573 AUX_SCRATCHNBNB_loc, AUX_SCRATCHNBNP_loc, aux_Re1PDM->Z().pointer(),
574 aux_DenZ_loc,
575 aux_GDenZ_loc, aux_GDenZ_loc + NPts, aux_GDenZ_loc + 2*NPts, aux_BasisEval);
576
577 if ( ss.onePDM->hasXY() )

585 mkAuxVar(this->aux_ss->onePDM,

587 aux_DenS_loc,aux_DenZ_loc,nullptr,nullptr,
588 aux_GDenS_loc,aux_GDenS_loc + NPts,aux_GDenS_loc + 2*NPts,
589 aux_GDenZ_loc,aux_GDenZ_loc + NPts,aux_GDenZ_loc + 2*NPts,
590 nullptr,nullptr,nullptr,
591 nullptr,nullptr,nullptr,
592 nullptr,

594 nullptr, nullptr, nullptr,

596 aux_U_n_loc, aux_U_gamma_loc

601 mkAuxVar(ss.onePDM,
602 isGGA || epcisGGA,epsScreen,NPts,
603 DenS_loc,DenZ_loc,nullptr,nullptr,

640 std::fill_n(dVU_n_loc, NPts, 0.);
641 if( isGGA || epcisGGA ) std::fill_n(dVU_gamma_loc, NPts, 0.);
642 if( ks && ks->functionals.size() > 1 ) {
643 std::fill_n(epsSCR_loc, NPts, 0.);
644 std::fill_n(dVU_n_SCR_loc, NPts, 0.);
645 if( isGGA || epcisGGA ) std::fill_n(dVU_gamma_SCR_loc, NPts, 0.);

649
650 loadEPCVXCder(ss.particle.charge < 0,
651 epc_functionals,
652 NPts, U_n_loc, U_gamma_loc, aux_U_n_loc,
653 aux_U_gamma_loc, cross_U_gamma_loc, epsEval_loc, dVU_n_loc,
654 dVU_gamma_loc, cross_dVU_gamma_loc, epsSCR_loc, dVU_n_SCR_loc,
655 dVU_gamma_SCR_loc, cross_dVU_gamma_loc, epcEval_loc);
656
657
658 // Compute for the current batch the XC energy and increment the
659 // total XC energy.
660 integrateXCEnergy[thread_id] +=
661 energy_vxc(NPts, weights, epsEval_loc, DenS_loc);
662 //integrateEPCEnergy[thread_id] +=
663 // energy_vxc(NPts, weights, epcEval_loc, DenS_loc);
664
665
666 // Construct the required quantities for the formation of the Z
667 // vector (SCALAR) given the kernel derivatives wrt U variables.
668 constructZVars(ss.onePDM, SCALAR, isGGA || epcisGGA,
669 NPts,dVU_n_loc,dVU_gamma_loc, ZrhoVar1_loc, ZgammaVar1_loc,
670 ZgammaVar2_loc);
671
672 // Construct the required quantities for the formation of the Z
673 // vector for EPC-19 functional
674 if (epcisGGA)
675 constructEPCZVars(ss.particle.charge < 0,
676 SCALAR,NPts,cross_dVU_gamma_loc, ZgammaVar3_loc);
677
678 // Creating ZMAT (SCALAR) according to
679 // J. Chem. Theory Comput. 2011, 7, 30973104 Eq. 15
680 if ( epcisGGA )
681 formZ_vxc_epc(ss.onePDM, this->aux_ss->onePDM,
682 SCALAR, epcisGGA, NPts, NBE, IOff, epsScreen, weights,

685 nullptr, aux_DenS_loc, aux_DenZ_loc, nullptr, nullptr,
686 aux_GDenS_loc, aux_GDenZ_loc, nullptr, nullptr,
687 BasisEval, ZMAT_loc);
688 else
689 formZ_vxc(ss.onePDM, SCALAR,isGGA, NPts, NBE, IOff, epsScreen, weights,
690 ZrhoVar1_loc, ZgammaVar1_loc, ZgammaVar2_loc, DenS_loc,
691 DenZ_loc, nullptr, nullptr, GDenS_loc, GDenZ_loc, nullptr,
692 nullptr, nullptr, nullptr,
693 nullptr, nullptr, nullptr,
694 nullptr, BasisEval, ZMAT_loc);

696 bool evalZ = true;

698 if (evalZ) {

700 // J. Chem. Theory Comput. 2011, 7, 30973104 Eq. 14
701 //
702 // Z -> VXC (submat - SCALAR)
703 blas::syr2k(blas::Layout::ColMajor,blas::Uplo::Lower,blas::Op::NoTrans,NBE,NPts,1.,BasisEval,NBE,ZMAT_loc,NBE,0.,
704 SCRATCHNBNB_loc,NBE);
705
706 // Locating the submatrix in the right position given the
707 // subset of shells for the given batch.
708 IncBySubMat(NB,NB,NBE,NBE,integrateVXC[SCALAR][thread_id],NB,
709 SCRATCHNBNB_loc,NBE,subMatCut);
710 }
711
712
713 if( not ss.onePDM->hasZ() ) return;
714
715 //
716 // --------------- UKS or 2C ------------- Mz ---------------
717 // See J. Chem. Theory Comput. 2017, 13, 2591-2603
718 //
719
720 // Construct the required quantities for the formation of
721 // the Z vector (Mz) given the kernel derivatives wrt U
722 // variables.
723 constructZVars(ss.onePDM, MZ,isGGA || epcisGGA,NPts,dVU_n_loc,dVU_gamma_loc,ZrhoVar1_loc,
724 ZgammaVar1_loc, ZgammaVar2_loc);
725
726 // Construct the required quantities for the formation of the Z
727 // vector for EPC-19 functional
728 if (epcisGGA)
729 constructEPCZVars(ss.particle.charge < 0,
730 MZ,NPts,cross_dVU_gamma_loc, ZgammaVar3_loc);
731
732 // Creating ZMAT (Mz) according to
733 // J. Chem. Theory Comput. 2011, 7, 30973104 Eq. 15
734 if (epcisGGA)
735 formZ_vxc_epc(ss.onePDM, this->aux_ss->onePDM,
736 MZ, epcisGGA, NPts, NBE, IOff, epsScreen, weights,
737 ZrhoVar1_loc, ZgammaVar1_loc, ZgammaVar2_loc, ZgammaVar3_loc,
738 DenS_loc, DenZ_loc, nullptr, nullptr, GDenS_loc, GDenZ_loc, nullptr,
739 nullptr, aux_DenS_loc, aux_DenZ_loc, nullptr, nullptr,
740 aux_GDenS_loc, aux_GDenZ_loc, nullptr, nullptr,
741 BasisEval, ZMAT_loc);
742 else
743 formZ_vxc(ss.onePDM,MZ,isGGA, NPts, NBE, IOff, epsScreen, weights,
744 ZrhoVar1_loc, ZgammaVar1_loc, ZgammaVar2_loc, DenS_loc,
745 DenZ_loc, nullptr, nullptr, GDenS_loc, GDenZ_loc, nullptr,
746 nullptr, nullptr, nullptr,
747 nullptr, nullptr, nullptr,
748 nullptr, BasisEval, ZMAT_loc);
749

751 // Coarse screen on ZMat
752 if(evalZ) {
753
754 // Creating according to
755 // J. Chem. Theory Comput. 2011, 7, 30973104 Eq. 14
756 //
757 // Z -> VXC (submat)
758 blas::syr2k(blas::Layout::ColMajor,blas::Uplo::Lower,blas::Op::NoTrans,NBE,NPts,1.,BasisEval,NBE,ZMAT_loc,NBE,0.,
759 SCRATCHNBNB_loc,NBE);

762 // Locating the submatrix in the right position given
763 // the subset of shells for the given batch.
764 IncBySubMat(NB,NB,NBE,NBE,integrateVXC[MZ][thread_id],NB,
765 SCRATCHNBNB_loc,NBE,subMatCut);
766
767 }
768
769
770
771 if( not ss.onePDM->hasXY() ) return;
772
773 CErr("Relativistic NEO-Kohn-Sham NYI!",std::cout);
774
775 }; // VXC integrate
776
777
778
779 // Create the BeckeIntegrator object
780 if( ks ) intParam = ks->intParam;
781 else if( auto aux_ks = dynamic_cast<KohnSham<MatsT,IntsT>*>( this->aux_ss ) )
782 intParam = aux_ks->intParam;
783 BeckeIntegrator<EulerMac>
784 integrator(intComm,ss.memManager,ss.molecule(),basis,aux_basis,
785 EulerMac(intParam.nRad), intParam.nAng, intParam.nRadPerBatch,
786 (isGGA ? GRADIENT : NOGRAD), (epcisGGA ? GRADIENT : NOGRAD),
787 intParam.epsilon);
788
789 // Integrate the VXC
790 integrator.integrate<size_t>(vxcbuild);
791
792
793
794 // Finishing up the VXC
795 // factor in the 4 pi (Lebedev) and built the upper triagolar part
796 // since we create only the lower triangular. For all components
797 for(auto k = 0; k < VXC_SZYX.size(); k++) {
798 if( nthreads == 1 )
799 blas::scal(NB2,4*M_PI,VXC_SZYX[k],1);
800 else
801 for(auto ithread = 0; ithread < nthreads; ithread++)
802 MatAdd('N','N',NB,NB,((ithread == 0) ? 0. : 1.),VXC_SZYX[k],NB,
803 4*M_PI,integrateVXC[k][ithread],NB, VXC_SZYX[k],NB);
804
805 HerMat('L',NB,VXC_SZYX[k],NB);
806 }
807
808 for(auto &X : integrateXCEnergy) {
809 XCEnergy += 4*M_PI*X;
810 }
811
812 if( ks )
813 ks->XCEnergy = XCEnergy;
814 else
815 ss.extraEnergy = XCEnergy;
816
817
818// Combine MPI results
819#ifdef CQ_ENABLE_MPI
820
821 double* mpiScr;
822 if( mpiRank == 0 ) mpiScr = ss.memManager.template malloc<double>(NB*NB);
823
824 for(auto &V : VXC_SZYX) {
825
826 mxx::reduce(V,NB*NB,mpiScr,0,std::plus<double>(),intComm);
827
828 if( mpiRank == 0 ) std::copy_n(mpiScr,NB*NB,V);
829
830 }
831
832 if( mpiRank == 0 ) ss.memManager.free(mpiScr);
833
834 if(ks) ks->XCEnergy = mxx::reduce(ks->XCEnergy,0,std::plus<double>(),intComm);
835
836#endif
837
838
839 // Freeing the memory
840 // ----------------Main System-------------------------------------------- //
841 ss.memManager.free(SCRATCHNBNB,SCRATCHNBNP,DenS,epsEval,epcEval,U_n,
842 dVU_n, ZrhoVar1,ZMAT);
843 if( isGGA || epcisGGA )
844 ss.memManager.free(ZgammaVar1,ZgammaVar2,GDenS,U_gamma,
845 dVU_gamma);
846
847 if( ss.onePDM->hasZ() ) {
848 ss.memManager.free(DenZ);
849 if( isGGA || epcisGGA ) ss.memManager.free(GDenZ);
850 }
851
852 if( ss.onePDM->hasXY() )
853 CErr("Relativistic NEO-Kohn-Sham NYI!", std::cout);
854
855 if( ks && ks->functionals.size() > 1 ) {
856 ss.memManager.free(epsSCR,dVU_n_SCR);
857 if( isGGA || epcisGGA ) ss.memManager.free(dVU_gamma_SCR);
858 }
859
860 if( nthreads != 1 ) ss.memManager.free(intVXC_RAW);
861
862 Re1PDM = nullptr;
863
864 // -----------------End Main System-------------------------------------------- //
865
866 // ----------------Auxiliary System-------------------------------------------- //
867 ss.memManager.free(AUX_SCRATCHNBNB,AUX_SCRATCHNBNP,aux_DenS,aux_U_n,aux_dVU_n,aux_ZrhoVar1);
868 if(epcisGGA )
869 ss.memManager.free(aux_ZgammaVar1,aux_ZgammaVar2,aux_GDenS,aux_U_gamma,
870 aux_dVU_gamma);
871
872 if( this->aux_ss->onePDM->hasZ() ) {
873 ss.memManager.free(aux_DenZ);
874 if ( epcisGGA ) ss.memManager.free(aux_GDenZ);
875 }
876
877 if( this->aux_ss->onePDM->hasXY() )
878 CErr("Relativistic NEO-Kohn-Sham NYI!", std::cout);
879
880 if( epc_functionals.size() > 1 ) {
881 ss.memManager.free(aux_epsSCR,aux_dVU_n_SCR);
882 if( epcisGGA ) ss.memManager.free(aux_dVU_gamma_SCR);
883 }
884
885 aux_Re1PDM = nullptr;
886
887 // -----------------End Aux-------------------------------------------- //
888 // -----------------Cross---------------------------------------------- //
889 if(epcisGGA)
890 ss.memManager.free(cross_U_gamma, cross_dVU_gamma, ZgammaVar3);
891 // -----------------End Cross------------------------------------------ //
892 // End freeing the memory
893
894 // Turn back on LA threads
895 SetLAThreads(LAThreads);
896
897 MPICommFree(intComm); // Free communicator
898
899 } // Valid intComm
900
901 MPI_Barrier(ss.comm); // Syncronize the MPI processes
902
903 }
