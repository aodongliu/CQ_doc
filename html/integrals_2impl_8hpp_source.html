46 template <typename IntsT>
47 void Integrals<IntsT>::computeAOOneP(CQMemManager &mem,
48 Molecule &mol, BasisSet &basis, EMPerturbation &emPert,
49 const std::vector<std::pair<OPERATOR,size_t>> &ops,
50 const HamiltonianOptions &options) {
51
52 const std::array<std::string,3> dipoleList =
53 { "X","Y","Z" };
54 const std::array<std::string,6> quadrupoleList =
55 { "XX","XY","XZ","YY","YZ","ZZ" };
56 const std::array<std::string,10> octupoleList =
57 { "XXX","XXY","XXZ","XYY","XYZ","XZZ","YYY",
58 "YYZ","YZZ","ZZZ" };
59 const std::array<std::string,9> quadrupoleListAsymm =
60 { "XX","XY","XZ","YX","YY","YZ","ZX","ZY","ZZ" };
61
62 size_t NB = basis.nBasis;
63
64 bool electron = (options.particle.charge < 0.);
65
66 std::string prefix = electron ? "INTS/" : "PINTS/";
67
68#if 0
69 std::cout << std::endl<<"Molecular Geometry in ComputeOneP: (Bohr)"<<std::endl;
70 size_t i = 0;
71 for( Atom& atom : mol.atoms ) {
72
73 std::cout << std::right <<"AtomicNumber = " << std::setw(4) << atom.atomicNumber
74 << std::right <<" X= "<< std::setw(16) << atom.coord[0]
75 << std::right <<" Y= "<< std::setw(16) << atom.coord[1]
76 << std::right <<" Z= "<< std::setw(16) << atom.coord[2] <<std::endl;
77 i += 3;
78
79 }
80
81 for(auto& shell: basis.shells)
82 for(i = 0; i < 3; i++)
83 std::cout << shell.O[i] << std::endl;
84#endif
85
86 for (const std::pair<OPERATOR,size_t> &op : ops)
87 switch (op.first) {
88 case OVERLAP:
89 overlap = std::make_shared<OnePInts<IntsT>>(mem, NB);
90 overlap->computeAOInts(basis, mol, emPert, OVERLAP, options);
91 if( savFile.exists() )
92 savFile.safeWriteData(prefix + "OVERLAP", overlap->pointer(), {NB,NB});
93 break;
94
95 case KINETIC:
96 kinetic = std::make_shared<OnePInts<IntsT>>(mem, NB);
97 kinetic->computeAOInts(basis, mol, emPert, KINETIC, options);
98 if( savFile.exists() )
99 savFile.safeWriteData(prefix + "KINETIC", kinetic->pointer(), {NB,NB});
100 break;
101
102 case NUCLEAR_POTENTIAL:
103 // Use Libint for point nuclei, in-house for Gaussian nuclei
104 if (options.OneEScalarRelativity)
105 potential = std::make_shared<OnePRelInts<IntsT>>(
106 mem, NB, options.OneESpinOrbit);
107 else
108 potential = std::make_shared<OnePInts<IntsT>>(mem, NB);
109 potential->computeAOInts(basis, mol, emPert, NUCLEAR_POTENTIAL, options);
110 if( savFile.exists() ) {
111 std::string potentialTag = options.finiteWidthNuc ? "_FINITE_WIDTH" : "";
112 savFile.safeWriteData(prefix + "POTENTIAL" + potentialTag,potential->pointer(),{NB,NB});
113 }
114 break;
115
116 case LEN_ELECTRIC_MULTIPOLE:
117 lenElectric = std::make_shared<MultipoleInts<IntsT>>(mem, NB, op.second, true);
118 lenElectric->computeAOInts(basis, mol, emPert, LEN_ELECTRIC_MULTIPOLE, options);
119 if( savFile.exists() ) {
120 // Length Gauge electric dipole
121 for(auto i = 0; i < 3; i++)
122 savFile.safeWriteData(prefix + "ELEC_DIPOLE_LEN_" +
123 dipoleList[i], (*lenElectric)[dipoleList[i]].pointer(), {NB,NB} );
124
125 // Length Gauge electric quadrupole
126 if(op.second >= 2)
127 for(auto i = 0; i < 6; i++)
128 savFile.safeWriteData(prefix + "ELEC_QUADRUPOLE_LEN_" +
129 quadrupoleList[i], (*lenElectric)[quadrupoleList[i]].pointer(), {NB,NB} );
130
131 // Length Gauge electric octupole
132 if(op.second >= 3)
133 for(auto i = 0; i < 10; i++)
134 savFile.safeWriteData(prefix + "/ELEC_OCTUPOLE_LEN_" +
135 octupoleList[i], (*lenElectric)[octupoleList[i]].pointer(), {NB,NB} );
136 }
137 break;
138
139 case VEL_ELECTRIC_MULTIPOLE:
140 velElectric = std::make_shared<MultipoleInts<IntsT>>(mem, NB, op.second, true);
141 velElectric->computeAOInts(basis, mol, emPert, VEL_ELECTRIC_MULTIPOLE, options);
142 if( savFile.exists() ) {
143 // Velocity Gauge electric dipole
144 for(auto i = 0; i < 3; i++)
145 savFile.safeWriteData(prefix + "ELEC_DIPOLE_VEL_" +
146 dipoleList[i], (*velElectric)[dipoleList[i]].pointer(), {NB,NB} );
147
148 // Velocity Gauge electric quadrupole
149 if(op.second >= 2)
150 for(auto i = 0; i < 6; i++)
151 savFile.safeWriteData(prefix + "ELEC_QUADRUPOLE_VEL_" +
152 quadrupoleList[i], (*velElectric)[quadrupoleList[i]].pointer(), {NB,NB} );
153
154 // Velocity Gauge electric octupole
155 if(op.second >= 3)
156 for(auto i = 0; i < 10; i++)
157 savFile.safeWriteData(prefix + "ELEC_OCTUPOLE_VEL_" +
158 octupoleList[i], (*velElectric)[octupoleList[i]].pointer(), {NB,NB} );
159 }
160 break;
161
162 case MAGNETIC_MULTIPOLE:
163 magnetic = std::make_shared<MultipoleInts<IntsT>>(mem, NB, op.second, false);
164 magnetic->computeAOInts(basis, mol, emPert, MAGNETIC_MULTIPOLE, options);
165 if( savFile.exists() ) {
166 // Magnetic Dipole
167 for(auto i = 0; i < 3; i++)
168 savFile.safeWriteData(prefix + "MAG_DIPOLE_" +
169 dipoleList[i], (*magnetic)[dipoleList[i]].pointer(), {NB,NB} );
170
171 // Magnetic Quadrupole
172 if(op.second >= 2)
173 for(auto i = 0; i < 6; i++)
174 savFile.safeWriteData(prefix + "MAG_QUADRUPOLE_" +
175 quadrupoleList[i], (*magnetic)[i+3].pointer(), {NB,NB} );
176 }
177 break;
178
179 case ELECTRON_REPULSION:
180 CErr("Electron repulsion integrals are not implemented in computeAOOneP,"
181 " they are implemented in TwoPInts",std::cout);
182 break;
183 }
184
185 }; // AOIntegrals<IntsT>::computeAOOneP

189 // Computes the integrals necessary for the gradients
190 template <typename IntsT>
191 void Integrals<IntsT>::computeGradInts(CQMemManager &mem,
192 Molecule &mol, BasisSet &basis, EMPerturbation &emPert,
193 const std::vector<std::pair<OPERATOR,size_t>> &ops,
194 const HamiltonianOptions &options) {
195
196 size_t NB = basis.nBasis;
197 size_t NAt = mol.nAtoms;
198
199 auto computeOneE = [&](std::shared_ptr<GradInts<OnePInts,IntsT>>& p, OPERATOR o) {
200 if (p == nullptr)
201 p = std::make_shared<GradInts<OnePInts,IntsT>>(mem, NB, NAt);
202 else
203 p->clear();
204
205 p->computeAOInts(basis, mol, emPert, o, options);
206 };
207
208 for ( auto& op: ops ) {
209
210
211 switch (op.first) {
212
213 case OVERLAP:
214 computeOneE(gradOverlap, op.first);
215 break;
216
217 case KINETIC:
218 computeOneE(gradKinetic, op.first);
219 break;
220
221 case NUCLEAR_POTENTIAL:
222 if ( options.OneEScalarRelativity )
223 CErr("Relativistic gradients not yet implemented!");
224 computeOneE(gradPotential, op.first);
225 break;
226
227 case MAGNETIC_MULTIPOLE:
228 case LEN_ELECTRIC_MULTIPOLE:
229 case VEL_ELECTRIC_MULTIPOLE:
230 CErr("Gradients of multipoles are not yet implemented");
231 break;
232
233 case ELECTRON_REPULSION:
234 if ( gradERI == nullptr )
235 CErr("ERI gradients must be allocated outside of computeGradInts!");
236
237 gradERI->computeAOInts(basis, mol, emPert, op.first, options);
238 break;
239 }
240
241 // TODO: Write code to save gradient integrals to binary file. Should be
242 // toggleable with default off to keep binary file to reasonable size.
243
244 }
245
246 }; // AOIntegrals<IntsT>::computeGradInts
