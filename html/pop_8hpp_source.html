30 template <typename MatsT, typename IntsT>
31 void SingleSlater<MatsT,IntsT>::populationAnalysis() {
32
33 const size_t NB = this->basisSet().nBasis;
34 MatsT* SCR = this->memManager.template malloc<MatsT>(NB*NB);
35 MatsT* SCR2 = this->memManager.template malloc<MatsT>(NB*NB);
36
37 // Molecule object to use
38 Molecule inputMol = this->molecule();
39 if (this->particle.charge > 0)
40 inputMol = inputMol.retainQNuc();
41
42 // Mulliken population analysis
43 mullikenCharges.clear();
44 if (nC != 4) {
45 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
46 this->onePDM->S().pointer(),NB,MatsT(0.),SCR,NB);
47 }
48 else {
49 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),this->aoints.kinetic->pointer(),NB,
50 this->onePDM->S().pointer()+2*NB*NB+NB,2*NB,MatsT(0.),SCR,NB);
51 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),this->aoints.overlap->pointer(),NB,
52 this->onePDM->S().pointer(),2*NB,MatsT(1./(2.*SpeedOfLight*SpeedOfLight)),SCR,NB);
53 }
54
55 for(auto iAtm = 0; iAtm < inputMol.nAtoms; iAtm++) {
56
57 size_t iEnd;
58 if( iAtm == inputMol.nAtoms-1 )
59 iEnd = NB;
60 else
61 iEnd = this->basisSet().mapCen2BfSt[iAtm+1];
62
63 size_t iSt = this->basisSet().mapCen2BfSt[iAtm];
64
65 mullikenCharges.emplace_back(inputMol.atoms[iAtm].nucCharge);
66 for(auto i = iSt; i < iEnd; i++)
67 mullikenCharges.back() -= std::real(SCR[i*(NB+1)]);
68 }
69
70
71 // Loewdin population analysis
72 lowdinCharges.clear();
73
74/*
75 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,T(1.),this->aoints.ortho1,NB,this->onePDM[SCALAR],NB,
76 T(0.),SCR2,NB);
77 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,NB,NB,NB,T(1.),this->aoints.ortho1,NB,SCR2,NB,T(0.),SCR,NB);
78*/
79
80 for(auto iAtm = 0; iAtm < inputMol.nAtoms; iAtm++) {
81
82 size_t iEnd;
83 if( iAtm == inputMol.nAtoms-1 ){
84 if( nC == 4 )
85 iEnd = 2*NB;
86 else
87 iEnd = NB;
88 }
89 else
90 iEnd = this->basisSet().mapCen2BfSt[iAtm+1];
91
92 size_t iSt = this->basisSet().mapCen2BfSt[iAtm];
93
94 lowdinCharges.emplace_back(inputMol.atoms[iAtm].nucCharge);
95 for(auto i = iSt; i < iEnd; i++)
96 lowdinCharges.back() -= std::real(this->onePDMOrtho->S()(i,i));
97 }
98
99
100 this->memManager.free(SCR,SCR2);
101
102
103 };
