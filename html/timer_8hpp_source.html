25// Basically this timer is a graph of nodes that have up children separated
26// by thread. All children only have one parent.
27//
28// A known issue with this design is that it does not allow for two overlapping
29// timed regions with the same nested region - you will need to assign it to
30// one or the other.
31//
32// For example, if the execution graph looked like this:
33//
34// Call level: 0 1
35// --------- foo tick
36// |
37// | ----- subfoo tick
38// | |
39// | ----- subfoo tock
40// |
41// |-------- bar tick
42// |
43// | ----- baz tick
44// | |
45// | ----- baz tock
46// |
47// |-------- foo tock
48// |
49// --------- bar tock
50//
51// You would need to assign the baz subblock to the foo or the bar context,
52// even though it is really part of both.
53//
54// In quantum chemistry programs, there aren't many reasons to have overlapping
55// timing regions, so this issue shouldn't come up often.
56//
57// This timer also doesn't account for overhead. Experimentally, the overhead is
58// in the tens of microseconds per tick/tock pair, but the exact number depends
59// on your system and software environmental factors. In short, if you are
60// optimizing/measuring something that takes microseconds and the overhead is
61// substantial compared to your run time, you should use a timer more optimized
62// for that case.
63//
64// Right now this timer can't handle nested OpenMP parallelism or other
65// versions of SMP parallelism, but since the thread ids don't need to be
66// sequential in each section, it can be extended to handle those cases.
67

142 // Threadsafe wrapper around an unordered_map that also offers threadsafe
143 // Id incrementing.
144 //
145 // 0 is treated as a sentinel value by the Timer class, so incrementing
146 // starts at 1.
147 template <typename IdT, typename ValT>
148 class Registry {
149
150 IdT nextId_ = 1;
151 std::unordered_map<IdT,ValT> registry_;
152
153 std::mutex id_mutex_;
154 std::mutex reg_mutex_;
155
156 public:
157
158 IdT getId() {
159 const std::lock_guard<std::mutex> lock(id_mutex_);
160 IdT temp = nextId_++;
161 return temp;
162 }
163
164 void add(IdT id, ValT val) {
165 const std::lock_guard<std::mutex> lock(reg_mutex_);
166 registry_.insert({id, val});
167 }
168
169 ValT operator[](IdT id) {
170 const std::lock_guard<std::mutex> lock(reg_mutex_);
171 ValT retval = registry_[id];
172 return retval;
173 };
174
175 };

177 // Basic node that represents a slice of time with a set of threaded sections
178 // underneath
179 class TimeSection {
180
181 using IdType = size_t;
182 using Clock = std::chrono::steady_clock;
183 using TimePoint = std::chrono::time_point<Clock>;
184 using RegType = Registry<IdType,TimeSection*>;
185
186 // Human readable label for this section
187 std::string label_;
188 // Unique Id for this section
189 IdType id_;
190
191 // Threads (outside) and subsections (inside)
192 // The value is a unique_ptr so references to the object don't get
193 // invalidated on reallocation of the unordered map.
194 // Multiple shared_ptr would cause cyclical references, so we're not going to
195 // do that
196 std::unordered_map<size_t,std::unordered_map<IdType,std::unique_ptr<TimeSection>>> children_;
197 TimeSection* parent_;
198
199 // Registry to get ids and add subsections
200 RegType* registry_;
201
202 // Wall times for this section
203 TimePoint start_;
204 TimePoint stop_;
205
206 bool stopped_ = false;
207
208 std::mutex mutex_;
209
210 friend class Timer;
211
212 public:
213
214 TimeSection(): label_(""), id_(0), parent_(nullptr), registry_(nullptr) {
215 start_ = Clock::now();
216 };
217
218 TimeSection(
219 std::string label,
220 IdType id,
221 RegType* reg,
222 TimeSection* parent) :
223 label_(label),
224 id_(id),
225 registry_(reg),
226 parent_(parent)
227 {
228 start_ = Clock::now();
229 }
230
231 //
232 // Timing control
233 //
234
235 // Stop this section
236 void stop() {
237
238 if (stopped_) {
239 return;
240 }
241
242 // Stop all subsections
243 // Locked because tick can cause reallocation of children_
244 // on another thread - this is _very_ unlikely to happen during
245 // normal usage, but is possible.
246 {
247 const std::lock_guard<std::mutex> lock(mutex_);
248 for ( auto& thread: children_ )
249 for ( auto& section: thread.second )
250 section.second->stop();
251 }
252
253 stop_ = Clock::now();
254 stopped_ = true;
255
256 }
257
258 // Start a new subsection
259 IdType tick(std::string label, size_t threadId = 0) {
260
261 IdType id = registry_->getId();
262
263 auto child = std::make_unique<TimeSection>(label, id, registry_, this);
264
265 registry_->add(id, child.get());
266
267 // Locked due to insertion
268 {
269 const std::lock_guard<std::mutex> lock(mutex_);
270 if (children_.find(threadId) == children_.end()) {
271 std::unordered_map<IdType,std::unique_ptr<TimeSection>> temp;
272 children_.insert(std::move(std::make_pair(threadId, std::move(temp))));
273 }
274 children_[threadId].insert(std::move(std::make_pair(id, std::move(child))));
275 }
276
277 return id;
278 }
279
280 // Stop a subsection
281 void tock(size_t id, size_t threadId = 0) {
282 // Locked because tick can cause reallocation of children_ on another
283 // thread
284 const std::lock_guard<std::mutex> lock(mutex_);
285 children_[threadId].find(id)->second->stop();
286 }
287
288 //
289 // Reporting
290 //
291
292 template <typename DurT>
293 DurT duration() {
294 return std::chrono::duration_cast<DurT>(stop_ - start_);
295 }
296
297 //
298 // Searching ancestors (no recursion)
299 //
300
301 bool isAncestor(IdType id) {
302 TimeSection* curr = this;
303 while ( curr->parent_ != nullptr ) {
304 curr = curr->parent_;
305 if ( curr->id_ == id )
306 return true;
307 }
308 return false;
309 };
310
311 TimeSection* findAncestor(std::string label) {
312 TimeSection* curr = this;
313 while ( curr->parent_ != nullptr ) {
314 curr = curr->parent_;
315 if ( curr->label_ == label )
316 return curr;
317 }
318 return nullptr;
319 };
320
321 };

372 // Start a new section
373 IdType tick(std::string label, IdType parentId = 0, bool addContext = true) {
374
375 size_t threadId = omp_get_thread_num();
376
377 TimeSection* section = contexts_[threadId];
378 if (parentId != 0)
379 section = registry_[parentId];
380
381 // Increment call level
382 levels_[threadId]++;
383
384 // Add the new section
385 auto newId = section->tick(label, threadId);
386
387 if (addContext)
388 contexts_[threadId] = registry_[newId];
389
390 return newId;
391
392 };
393
394 // Stop a section by id
395 void tock(IdType id = 0) {
396
397 size_t threadId = omp_get_thread_num();
398
399 TimeSection* section = contexts_[threadId];
400 if (id != 0)
401 section = registry_[id];
402 section->stop();
403
404 size_t s_id = section->id_;
405
406 if ( section->parent_ != nullptr )
407 if (contexts_[threadId]->id_ == s_id or
408 contexts_[threadId]->isAncestor(s_id))
409 contexts_[threadId] = registry_[section->parent_->id_];
410
411 // Possibly print
412 // This is bad for performance, but it is useful for debugging
413 if ( levels_[threadId] < options.dbgPrint ) {
414 std::stringstream message;
415 std::string name = section->label_ + " duration: ";
416 message << " || " << std::left << std::setw(40) << name;
417 message << std::right << std::setw(14) << std::fixed << std::setprecision(4);
418 message << section->duration<CQMillisecond>().count() << " ms ||\n";
419 std::cout << message.str();
420 }
421
422 // Decrement call level
423 levels_[threadId]--;
424
425 };
426
427 // Stop a section by the best guess at which label it's referring to
428 //
429 // If current context matches label, stop that section, otherwise find the
430 // first section that matches the label in the history
431 void tock(std::string label) {
432
433 size_t threadId = omp_get_thread_num();
434
435 TimeSection* section = contexts_[threadId];
436 if ( section->label_ != label ) {
437 section = section->findAncestor(label);
438 assert(section != nullptr);
439 }
440
441 tock(section->id_);
442 };

454 // Direct manipulation of current context

461 // Get FIRST instance of label's Id, starting from base
462 IdType getLabelId(std::string label, IdType base = 0) {
463
464 std::queue<TimeSection*> working;
465
466 std::function<IdType(std::queue<TimeSection*>&)> tryNextId;
467 tryNextId = [&](std::queue<TimeSection*>& next) -> IdType {
468 TimeSection* curr = next.front();
469 next.pop();
470 if ( curr->label_ == label )
471 return curr->id_;
472
473 for ( auto& thread: curr->children_ )
474 for ( auto& child: thread.second ) {
475 next.push(child.second.get());
476 }
477
478 return 0;
479 };
480
481 size_t threadId = omp_get_thread_num();
482 TimeSection* section = contexts_[threadId];
483 if (base != 0)
484 section = registry_[base];
485
486 working.push(section);
487 IdType id = 0;
488
489 while ( !working.empty() ) {
490 id = tryNextId(working);
491 if ( id != 0 )
492 break;
493 }
494
495 return id;
496
497 };

552 // Get (total, average) of all sections that match label
553 template <typename DurT>
554 std::pair<DurT,DurT> getDurationSummary(std::string label, IdType base = 0,
555 std::string filter = "") {
556
557 auto durations = getDurations<DurT>(label, base, filter);
558
559 if (durations.size() == 0)
560 return {DurT(0.), DurT(0.)};
561
562 DurT sum(0);
563 for (auto& dur: durations)
564 sum += dur;
565
566 return {sum, sum/durations.size()};
567 }
568
569 template <typename DurT>
570 std::pair<DurT,DurT> getAggDurSummary(std::string label,
571 std::string parentLabel, IdType base = 0) {
572
573 auto durations = getAggregateDurations<DurT>(label, parentLabel, base);
574
575 if (durations.size() == 0)
576 return {DurT(0.), DurT(0.)};
577
578 DurT sum(0.);
579 for (auto& section: durations) {
580 sum += *std::max_element(section.begin(), section.end());
581 }
582
583 return {sum, sum/durations.size()};
584
585 }

605 // Main get duration method
606 //
607 // If filterLabel is not "", it will only add durations that have an
608 // ancestor with that label
609 template <typename DurT>
610 std::vector<std::vector<DurT>> getThreadDurations(std::string label,
611 IdType base = 0, std::string filterLabel = "") {
612
613 std::function<void(size_t, size_t, bool, TimeSection*,
614 std::vector<DurT>&)> getTDurRecur;
615 getTDurRecur =
616 [&](size_t select, size_t tId, bool addDur, TimeSection* sect,
617 std::vector<DurT>& results) {
618
619 if (sect->label_ == label && select == tId && addDur) {
620 results.push_back(sect->duration<DurT>());
621 }
622
623 addDur |= sect->label_ == filterLabel;
624
625 for ( auto& thread: sect->children_ ) {
626
627 if ( thread.first != select )
628 continue;
629
630 for ( auto& child: thread.second ) {
631 getTDurRecur(select, thread.first, addDur, child.second.get(),
632 results);
633 }
634
635 }
636
637 };
638
639 std::vector<std::vector<DurT>> results;
640
641 size_t threadId = omp_get_thread_num();
642 TimeSection* sect = contexts_[threadId];
643 if ( base != 0 )
644 sect = registry_[base];
645
646 for ( auto iTh = 0; iTh < contexts_.size(); iTh++ ) {
647 std::vector<DurT> tempRes;
648
649 // Handle the fact that we don't know which thread this section came
650 // from by putting it on the calling thread stack
651 getTDurRecur(iTh, threadId, filterLabel == "", sect, tempRes);
652
653 results.push_back(std::move(tempRes));
654 }
655
656 return results;
657
658 }
659
660 // Aggregate the durations directly underneath parentLabel that match
661 // label.
662 //
663 // This can be useful in highly looped regions where you want to know
664 // how much time is spent in different parts of the loops but want to
665 // separate it by entrances to the loop (e.g. direct J/K builds:
666 // integral formation vs. density contraction)
667 //
668 // This does NOT give threads in the right order, so result[section[0]] may
669 // not correspond to thread 0.
670 template <typename DurT>
671 std::vector<std::vector<DurT>> getAggregateDurations(std::string label,
672 std::string parentLabel, IdType base = 0) {
673
674 std::function<void(TimeSection*,
675 std::vector<std::vector<DurT>>&)> getAggDurRecur;
676 getAggDurRecur =
677 [&](TimeSection* sect, std::vector<std::vector<DurT>>& results) {
678
679 bool doAgg = sect->label_ == parentLabel;
680 std::vector<DurT> current;
681
682 for ( auto& thread: sect->children_ ) {
683 DurT aggDur(0.);
684 size_t tId = thread.first;
685 for ( auto& child: thread.second ) {
686 if ( doAgg && child.second->label_ == label ) {
687 aggDur += child.second->duration<DurT>();
688 }
689 getAggDurRecur(child.second.get(), results);
690 }
691 current.push_back(aggDur);
692 }
693
694 if ( doAgg )
695 results.push_back(current);
696
697 };
698
699 std::vector<std::vector<DurT>> results;
700
701 size_t threadId = omp_get_thread_num();
702 TimeSection* sect = contexts_[threadId];
703 if ( base != 0 )
704 sect = registry_[base];
705
706 getAggDurRecur(sect, results);
707 return results;
708
709 }

711 //
712 // Printing
713 //
714
715 template <typename DurT>
716 void printTimings(std::ostream& out, std::string unit = "", IdType id = 0,
717 int maxDepth = 3) {
718
719 TimeSection* printBase = contexts_[0];
720 if (id != 0)
721 printBase = registry_[id];
722 auto total = printBase->duration<DurT>();
723
724 // Print header
725 std::string first = "Section Label";
726 std::string second = " Duration (";
727 second += unit;
728 second += ") Percent";
729
730 // Use a "max rational section label length" of 20. If it's longer than
731 // that, the printing won't be pretty but will still work
732 size_t fieldWidth = 20;
733 size_t width = maxDepth*2 + fieldWidth;
734
735 // Size of the data fields
736 size_t fixed = 22;
737 size_t dataw = std::max(fixed, second.size());
738
739 // Size of max width
740 size_t outw = width + dataw;
741
742 //
743 // Recursive printing definition
744 //
745 std::function<void(TimeSection*, size_t)> recursivePrint;
746 recursivePrint = [&](TimeSection* sect, size_t level) {
747 if ( level > maxDepth )
748 return;
749
750 DurT dur = sect->duration<DurT>();
751
752 // Print this section
753 std::stringstream ss;
754 for ( auto i = 0; i < level; i++ )
755 ss << "| ";
756 ss << sect->label_ << " ";
757 out << std::setw(width);
758 out << std::left << std::setfill('-') << ss.str();
759
760 out << std::right << std::setfill(' ');
761 out << std::setw(outw - width - fixed) << "";
762
763 out << ' ' << std::setw(12) << std::fixed << std::setprecision(4);
764 out << dur.count();
765
766 out << " " << std::setw(7) << std::fixed << std::setprecision(3);
767 out << dur.count()/total.count()*100;
768
769 out << std::endl;
770
771
772 auto sortByStart = [](TimeSection* a, TimeSection* b) {
773 return (a->start_ < b->start_);
774 };
775
776 // Recurse into children
777 bool threaded = sect->children_.size() > 1;
778
779 std::vector<TimeSection*> children;
780
781 for ( auto iTh = 0; iTh < contexts_.size(); iTh++ ) {
782
783 if (sect->children_.find(iTh) == sect->children_.end())
784 continue;
785
786 children.clear();
787
788 for (auto& child: sect->children_[iTh] ) {
789 children.push_back(child.second.get());
790 }
791
792 sort(children.begin(), children.end(), sortByStart);
793
794 if (threaded) {
795 std::stringstream ss;
796 for ( auto i = 0; i < level+1; i++ )
797 ss << "| ";
798 ss << "*** Thread " << iTh << " ";
799 out << std::left << std::setfill('*');
800 out << std::setw(outw) << ss.str() << std::endl;
801 }
802
803 for ( auto child: children ) {
804 recursivePrint(child, level+1);
805 }
806 }
807 }; // Recursive printing definition
808
809 out << std::setw(width) << std::left << std::setfill(' ') << first;
810 out << std::right << std::setw(dataw-10) << second << std::endl;
811 out << std::setw(outw) << std::setfill('=') << "" << std::endl;
812 recursivePrint(printBase, 0);
813 out << std::setw(outw) << std::setfill('=') << "" << std::endl;
814 out << std::setfill(' ');
815
816 } // printTimings
817
818 void printTimings(std::ostream& out, IdType id = 0) {
819 if ( options.unit == MILLISECONDS )
820 printTimings<CQMillisecond>(out, "ms", id, options.dbgPrint);
821 else if ( options.unit == SECONDS )
822 printTimings<CQSecond>(out, "s", id, options.dbgPrint);
823 else if ( options.unit == MINUTES )
824 printTimings<CQMinute>(out, "min", id, options.dbgPrint);
825 };
