77 template <typename... Args>
78 KohnSham(std::string funcName,
79 std::vector<std::shared_ptr<DFTFunctional>> funclist,
80 MPI_Comm c, IntegrationParam ip,
81 CQMemManager &mem, Molecule &mol, BasisSet &basis,
82 Integrals<IntsT> &aoi, Args... args) :
83 SingleSlater<MatsT,IntsT>(c,mem,mol,basis,aoi,args...),
84 WaveFunctionBase(c,mem,mol,basis,args...),
85 QuantumBase(c,mem,args...), isGGA_(false),
86 functionals(std::move(funclist)),intParam(ip){
87
88 // Append HF tags to reference names
89 if(this->nC == 1) {
90 if(this->iCS) {
91 this->refLongName_ += "Restricted " + funcName;
92 this->refShortName_ += "R" + funcName;
93 } else {
94 this->refLongName_ += "Unrestricted " + funcName;
95 this->refShortName_ += "U" + funcName;
96 }
97 } else {
98 this->refLongName_ += "Generalized " + funcName;
99 this->refShortName_ += "G" + funcName;
100 }
101
102 size_t NB = this->basisSet().nBasis;
103 if(this->nC > 1)
104 VXC = std::make_shared<PauliSpinorSquareMatrices<double>>(this->memManager, NB, true);
105 else if (not this->iCS)
106 VXC = std::make_shared<PauliSpinorSquareMatrices<double>>(this->memManager, NB, false);
107 else
108 VXC = std::make_shared<PauliSpinorSquareMatrices<double>>(this->memManager, NB, false, false);
109
110
111 }; // KohnSham constructor

155 virtual void formFock(EMPerturbation &pert, bool increment = false, double HFX = 0.) {
156
157 double xHFX = functionals.size() != 0 ? functionals.back()->xHFX : 1.;
158
159 SingleSlater<MatsT,IntsT>::formFock(pert,increment,xHFX);
160
161 if( doVXC_ ) {
162 formVXC();
163
164 ROOT_ONLY(this->comm);
165
166 // Add VXC in Fock matrix
167 *this->fockMatrix += *VXC;
168 }
169
170 }; // formFock

178
