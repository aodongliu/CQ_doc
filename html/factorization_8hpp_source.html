58 template <typename _F>
59 inline int QR(int M, int N, _F *A, int LDA, CQMemManager &mem) {
60
61 int LDR = std::min(M,N);
62 _F *R = mem.template malloc<_F>(LDR*LDR);
63
64 int INFO = QR(M,N,A,LDA,R,LDR,mem);
65
66 mem.free(R);
67
68 return INFO;
69
70 }
