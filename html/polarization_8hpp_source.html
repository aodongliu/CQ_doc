74 template <typename U>
75 inline void blockTransform(size_t N, size_t nVec, double fact,
76 U* V1, size_t LDV1, U* V2, size_t LDV2) {
77
78
79 for( auto j = 0ul; j < nVec; j++ )
80 for( auto i = 0ul; i < N ; i++ ) {
81
82 U tmp = V1[i + j*LDV1];
83
84 V1[i + j*LDV1] = fact * ( tmp + V2[i + j*LDV2] );
85 V2[i + j*LDV2] = fact * ( tmp - V2[i + j*LDV2] );
86
87 }
88
89
90 }
91 inline virtual size_t getNSingleDim(const bool doTDA = false) {
92
93 size_t N = 0;
94
95 auto &ss = *this->ref_;
96 size_t nOAVA = ss.nOA * ss.nVA;
97 size_t nOBVB = ss.nOB * ss.nVB;
98 size_t nOV = ss.nO * ss.nV;
99
100 N = (ss.nC == 1) ? nOAVA + nOBVB : nOV;
101
102 if( not doTDA and not doReduced ) N *= 2;
103
104 assert(N != 0);
105
106 return N;
107
108 }

211 void printResMO(std::ostream &out) {
212
213 size_t N = this->nSingleDim_;
214 size_t nRoots = this->resSettings.nRoots;
215
216 auto *VR = this->resResults.VR;
217 auto *VL = this->resResults.VL;
218 if( not doReduced ) VL = VR + N/2;
219
220 // Transform X+Y/X-Y to X/Y for print
221 if( doAPB_AMB ) blockTransform(nRoots,nRoots,std::sqrt(0.5),VR,N,VL,N);
222
223 printResMO(out,this->resSettings.nRoots,this->resResults.W,
224 {{"f = ",this->resObs.oscStrength}},VL,VR);
225
226 // Transform back to X+Y/X-Y from X/Y
227 if( doAPB_AMB ) blockTransform(nRoots,nRoots,std::sqrt(0.5),VR,N,VL,N);
228
229 }
