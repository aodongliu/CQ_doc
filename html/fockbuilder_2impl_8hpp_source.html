74 template <typename MatsT, typename IntsT>
75 void FockBuilder<MatsT,IntsT>::formGD(SingleSlater<MatsT,IntsT> &ss,
76 EMPerturbation &pert, bool increment, double xHFX, bool HerDen) {
77
78 // Decide list of onePDMs to use
79 PauliSpinorSquareMatrices<MatsT> &contract1PDM
80 = increment ? *ss.deltaOnePDM : *ss.onePDM;
81
82 std::vector<std::shared_ptr<PauliSpinorSquareMatrices<MatsT>>>
83 onePDMs, coulombMatrices, exchangeMatrices, twoeHs;
84
85 // setup pointers
86 if (increment) onePDMs.push_back(ss.deltaOnePDM);
87 else onePDMs.push_back(ss.onePDM);
88
89 exchangeMatrices.push_back(ss.exchangeMatrix);
90 twoeHs.push_back(ss.twoeH);
91
92 coulombMatrices.push_back(
93 std::make_shared<PauliSpinorSquareMatrices<MatsT>>(
94 ss.memManager, ss.coulombMatrix->dimension(), false, false)
95 );
96
97 formRawGDInBatches(ss, pert, increment, xHFX, HerDen, onePDMs, coulombMatrices, exchangeMatrices, twoeHs);
98
99 * ss.coulombMatrix = coulombMatrices[0]->S();
100
101 } // FockBuilder::formGD

108 template <typename MatsT, typename IntsT>
109 void FockBuilder<MatsT,IntsT>::formRawGDInBatches(SingleSlater<MatsT,IntsT> &ss,
110 EMPerturbation &pert, bool increment, double xHFX, bool HerDen,
111 std::vector<std::shared_ptr<PauliSpinorSquareMatrices<MatsT>>> & onePDMs,
112 std::vector<std::shared_ptr<PauliSpinorSquareMatrices<MatsT>>> & coulombMatrices,
113 std::vector<std::shared_ptr<PauliSpinorSquareMatrices<MatsT>>> & exchangeMatrices,
114 std::vector<std::shared_ptr<PauliSpinorSquareMatrices<MatsT>>> & twoeHs) {
115
116 size_t NB = ss.basisSet().nBasis;
117 size_t nBatch = onePDMs.size();
118 bool computeCoulomb = coulombMatrices.size() > 0;
119 bool computeExchange = (std::abs(xHFX) > 1e-12) and exchangeMatrices.size() > 0;
120 bool computeTwoeHs = twoeHs.size() > 0;
121
122 if (not computeCoulomb and not computeExchange and not computeTwoeHs) {
123 CErr("Nothing specified to compute in FockBuilder::formRawGDInBatches");
124 }
125
126 if( ss.nC == 4 )
127 CErr("4C formGD is implemented in class FourCompFock.");
128
129 // Zero out J and K[i]
130 if(not increment) {
131 for (auto i = 0ul; i < nBatch; i++) {
132 if (computeCoulomb) coulombMatrices[i]->clear();
133 if (computeExchange) exchangeMatrices[i]->clear();
134 if (computeTwoeHs) twoeHs[i]->clear();
135 }
136 }
137
138 std::vector<TwoBodyContraction<MatsT>> contract;
139
140 if (computeCoulomb or computeTwoeHs) {
141 auto & coulombContainers = computeCoulomb ? coulombMatrices: twoeHs;
142 for (auto i = 0ul; i < nBatch; i++) {
143 contract.push_back(
144 {onePDMs[i]->S().pointer(), coulombContainers[i]->S().pointer(), HerDen, COULOMB}
145 );
146 }
147 }
148
149 // Determine how many (if any) exchange terms to calculate
150 if( std::abs(xHFX) > 1e-12 and not increment and ss.nC == 1 and
151// (ss.scfControls.guess != SAD or ss.modifyOrbitals->scfConv.nSCFIter != 0) and
152 std::dynamic_pointer_cast<InCoreRITPIContraction<MatsT, IntsT>>(ss.TPI)) {
153 ROOT_ONLY(ss.comm);
154 auto ritpi = std::dynamic_pointer_cast<InCoreRITPIContraction<MatsT, IntsT>>(ss.TPI);
155
156 SquareMatrix<MatsT> AAblock(exchangeMatrices[0]->memManager(), NB);
157 ritpi->KCoefContract(ss.comm, ss.nOA, ss.mo[0].pointer(), AAblock.pointer());
158 if(ss.iCS) {
159
160 for (auto i = 0ul; i < nBatch; i++)
161 *exchangeMatrices[i] = PauliSpinorSquareMatrices<MatsT>::spinBlockScatterBuild(AAblock);
162
163 } else {
164 SquareMatrix<MatsT> BBblock(exchangeMatrices[0]->memManager(), NB);
165 ritpi->KCoefContract(ss.comm, ss.nOB, ss.mo[1].pointer(), BBblock.pointer());
166
167 for (auto i = 0ul; i < nBatch; i++)
168 *exchangeMatrices[i] = PauliSpinorSquareMatrices<MatsT>::spinBlockScatterBuild(AAblock, BBblock);
169 }
170
171 } else if(computeExchange) {
172 for (auto i = 0ul; i < nBatch; i++) {
173 contract.push_back(
174 {onePDMs[i]->S().pointer(), exchangeMatrices[i]->S().pointer(), HerDen, EXCHANGE}
175 );
176 if (exchangeMatrices[i]->hasZ())
177 contract.push_back(
178 {onePDMs[i]->Z().pointer(), exchangeMatrices[i]->Z().pointer(), HerDen, EXCHANGE}
179 );
180 if (exchangeMatrices[i]->hasXY()) {
181 contract.push_back(
182 {onePDMs[i]->Y().pointer(), exchangeMatrices[i]->Y().pointer(), HerDen, EXCHANGE}
183 );
184 contract.push_back(
185 {onePDMs[i]->X().pointer(), exchangeMatrices[i]->X().pointer(), HerDen, EXCHANGE}
186 );
187 }
188 }
189 }
190
191 ss.TPI->twoBodyContract(ss.comm, contract, pert);
192
193 ROOT_ONLY(ss.comm); // Return if not root (J/K only valid on root process)
194
195 if (computeTwoeHs) {
196 for (auto i = 0ul; i < nBatch; i++) {
197 // G[D] += 2*J[D]
198 if (computeCoulomb) {
199 *twoeHs[i] += 2.0 * *coulombMatrices[i];
200 } else {
201 *twoeHs[i] *= 2.0;
202 }
203
204 // Form GD: G[D] = 2.0*J[D] - K[D]
205 if (computeExchange) {
206 *twoeHs[i] -= xHFX * *exchangeMatrices[i];
207 }
208 }
209 }
210#if 0
211 //printJ(std::cout);
212 printK(std::cout);
213 //printGD(std::cout);
214#endif
215
216 } // FockBuilder::formGD

219 /* Compute memory requirement for build 4C GD in Batches */

222 /* IMPORTANT HERE: size are all in MatsT */

224 template <typename MatsT, typename IntsT>
225 size_t FockBuilder<MatsT,IntsT>::formRawGDSCRSizePerBatch(SingleSlater<MatsT,IntsT> &ss,
226 bool computeExchange, bool HerDen) const {
227
228 size_t SCRSize = 0ul;
229
230 if( std::dynamic_pointer_cast<GTODirectTPIContraction<MatsT,IntsT>>(ss.TPI) ) {
231
232 GTODirectTPIContraction<MatsT,IntsT> &ERICon =
233 *std::dynamic_pointer_cast<GTODirectTPIContraction<MatsT,IntsT>>(ss.TPI);
234
235 size_t nConPerBatch = computeExchange ? 5: 1;
236
237 SCRSize += ERICon.directScaffoldNewSCRSize() * nConPerBatch;
238 }
239
240 return SCRSize;
241 } // FockBuilder::formRawGDSCRSizePerBatch

252 template <typename MatsT, typename IntsT>
253 void FockBuilder<MatsT,IntsT>::formFock(SingleSlater<MatsT,IntsT> &ss,
254 EMPerturbation &pert, bool increment, double xHFX) {
255
256 auto GDStart = tick(); // Start time for G[D]
257
258 // Form G[D]
259 formGD(ss,pert,increment,xHFX);
260
261 ss.GDDur = tock(GDStart); // G[D] Duraction
262 //std::cout <<"formGD time = "<<ss.GDDur <<std::endl;
263
264 ROOT_ONLY(ss.comm);
265
266 // Form Fock
267 *ss.fockMatrix = *ss.coreH + *ss.twoeH;
268
269 // Add in the electric field contributions
270 // FIXME: the magnetic field contribution should go here as well to allow for RT
271 // manipulation
272
273 if( pert_has_type(pert,Electric) ) {
274
275 auto dipAmp = pert.getDipoleAmp(Electric);
276
277 for(auto i = 0; i < 3; i++)
278 ss.fockMatrix->S() -=
279 2. * dipAmp[i] * (*ss.aoints.lenElectric)[i].matrix();
280
281 }
282
283#if 0
284 ss.printFock(std::cout);
285#endif
286 }

308 template <typename MatsU>
309 std::shared_ptr<FockBuilder<MatsU,IntsT>>
310 FockBuilder<MatsT,IntsT>::convert(const std::shared_ptr<FockBuilder<MatsT,IntsT>>& fb) {
311
312 if (not fb) return nullptr;
313
314 const std::type_info &tID(typeid(*fb));
315
316 if (tID == typeid(ROFock<MatsT,IntsT>)) {
317 return std::make_shared<ROFock<MatsU,IntsT>>(
318 *std::dynamic_pointer_cast<ROFock<MatsT,IntsT>>(fb));
319 }
320 else if (tID == typeid(NEOFockBuilder<MatsT,IntsT>)) {
321 return std::make_shared<NEOFockBuilder<MatsU,IntsT>>(
322 *std::dynamic_pointer_cast<NEOFockBuilder<MatsT,IntsT>>(fb));
323 }
324 else if (tID == typeid(FourCompFock<MatsT,IntsT>)) {
325 return std::make_shared<FourCompFock<MatsU,IntsT>>(
326 *std::dynamic_pointer_cast<FourCompFock<MatsT,IntsT>>(fb));
327
328 } else if (tID == typeid(MatrixFock<MatsT,IntsT>)) {
329 return std::make_shared<MatrixFock<MatsU,IntsT>>(
330 *std::dynamic_pointer_cast<MatrixFock<MatsT,IntsT>>(fb));
331
332 } else {
333 return std::make_shared<FockBuilder<MatsU,IntsT>>(
334 *std::dynamic_pointer_cast<FockBuilder<MatsT,IntsT>>(fb));
335 }
336
337 } // FockBuilder<MatsT,IntsT>::convert

340 template <typename MatsT, typename IntsT>
341 std::vector<double> FockBuilder<MatsT,IntsT>::getGDGrad(
342 SingleSlater<MatsT,IntsT>& ss, EMPerturbation& pert, double xHFX) {
343
344 size_t NB = ss.basisSet().nBasis;
345 size_t nGrad = 3*ss.molecule().nAtoms;
346 CQMemManager& mem = ss.memManager;
347
348 bool hasXY = ss.exchangeMatrix->hasXY();
349 bool hasZ = ss.exchangeMatrix->hasZ();
350
351 if( not ss.aoints.gradERI )
352 CErr("Gradient ERI missing in FockBuilder::getGDGrad!");
353
354 GradInts<TwoPInts,IntsT>& gradERI = *ss.aoints.gradERI;
355
356 // Form contraction
357 // TODO: There's gotta be a better way to do this...
358 std::unique_ptr<GradContractions<MatsT,IntsT>> contract = nullptr;
359 if ( std::dynamic_pointer_cast<InCore4indexTPI<IntsT>>(gradERI[0]) ) {
360 contract = std::make_unique<InCore4indexGradContraction<MatsT,IntsT>>(gradERI);
361 }
362 else if ( std::dynamic_pointer_cast<DirectTPI<IntsT>>(gradERI[0]) ) {
363 contract = std::make_unique<DirectGradContraction<MatsT,IntsT>>(gradERI);
364 }
365 else
366 CErr("Gradients of RI NYI!");
367
368 // Create contraction list
369 std::vector<std::vector<TwoBodyContraction<MatsT>>> cList;
370
371 std::vector<SquareMatrix<MatsT>> JList;
372 std::vector<PauliSpinorSquareMatrices<MatsT>> KList;
373
374 JList.reserve(nGrad);
375 KList.reserve(nGrad);
376
377 for( auto iGrad = 0; iGrad < nGrad; iGrad++ ) {
378 std::vector<TwoBodyContraction<MatsT>> tempCont;
379
380 // Coulomb
381 JList.emplace_back(mem, NB);
382 JList.back().clear();
383 tempCont.push_back(
384 {ss.onePDM->S().pointer(), JList.back().pointer(), true, COULOMB}
385 );
386
387 // Exchange
388 if( std::abs(xHFX) > 1e-12 ) {
389
390 KList.emplace_back(mem, NB, hasXY, hasZ);
391 KList.back().clear();
392
393 tempCont.push_back(
394 {ss.onePDM->S().pointer(), KList.back().S().pointer(), true, EXCHANGE}
395 );
396
397 if (hasZ) {
398 tempCont.push_back(
399 {ss.onePDM->Z().pointer(), KList.back().Z().pointer(), true, EXCHANGE}
400 );
401 }
402 if (hasXY) {
403 tempCont.push_back(
404 {ss.onePDM->Y().pointer(), KList.back().Y().pointer(), true, EXCHANGE}
405 );
406 tempCont.push_back(
407 {ss.onePDM->X().pointer(), KList.back().X().pointer(), true, EXCHANGE}
408 );
409 }
410 }
411
412
413 cList.push_back(tempCont);
414 }
415
416 // Contract to J/K
417 contract->gradTwoBodyContract(MPI_COMM_WORLD, true, cList, pert);
418
419 // Contract to gradient
420 std::vector<double> gradient;
421 PauliSpinorSquareMatrices<MatsT> twoEGrad(mem, NB, hasXY, hasZ);
422
423 for( auto iGrad = 0; iGrad < nGrad; iGrad++ ) {
424
425 // Scale K by alpha
426 twoEGrad = -xHFX * KList[iGrad];
427
428 // G[S] = 2 * J[S] + alpha * K[S]
429 twoEGrad.S() += 2. * JList[iGrad];
430
431 double gradVal = ss.template computeOBProperty<SCALAR>(
432 twoEGrad.S().pointer()
433 );
434 if( hasZ )
435 gradVal += ss.template computeOBProperty<MZ>(
436 twoEGrad.Z().pointer()
437 );
438 if( hasXY ) {
439 gradVal += ss.template computeOBProperty<MY>(
440 twoEGrad.Y().pointer()
441 );
442 gradVal += ss.template computeOBProperty<MX>(
443 twoEGrad.X().pointer()
444 );
445 }
446 gradient.push_back(0.25*gradVal);
447
448 }
449
450 return gradient;
451
452 } // FockBuilder::getGDGrad
