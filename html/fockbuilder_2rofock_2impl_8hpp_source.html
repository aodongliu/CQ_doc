38 template <typename MatsT, typename IntsT>
39 void ROFock<MatsT,IntsT>::rohfFock(SingleSlater<MatsT,IntsT> &ss) {
40
41 size_t NB = ss.basisSet().nBasis;
42 size_t NB2 = NB*NB;
43
44 ROOT_ONLY(ss.comm);
45
46 //construct focka and fockb
47 std::vector<SquareMatrix<MatsT>> SCR = ss.fockMatrix->template spinGatherToBlocks<MatsT>(false);
48
49 //construct projectors for closed, open, virtual
50 //pc = dmb * S
51 //po = (dma - dmb) * S
52 //pv = I - dma * S
53 MatsT* pc = ss.memManager.template malloc<MatsT>(NB*NB);
54 MatsT* po = ss.memManager.template malloc<MatsT>(NB*NB);
55 MatsT* pv = ss.memManager.template malloc<MatsT>(NB*NB);
56 SquareMatrix<MatsT> tmp(ss.memManager,NB);
57 MatsT* tmp2 = ss.memManager.template malloc<MatsT>(NB*NB);
58
59 //overlap matrix
60 tmp = 0.5 * (ss.onePDM->S() - ss.onePDM->Z());
61 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),tmp.pointer(),NB,
62 SquareMatrix<MatsT>(ss.aoints.overlap->matrix()).pointer(),NB,
63 MatsT(0.),pc,NB);
64 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),ss.onePDM->Z().pointer(),NB,
65 SquareMatrix<MatsT>(ss.aoints.overlap->matrix()).pointer(),NB,
66 MatsT(0.),po,NB);
67 tmp = 0.5 * (ss.onePDM->S() + ss.onePDM->Z());
68 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(-1.),tmp.pointer(),NB,
69 SquareMatrix<MatsT>(ss.aoints.overlap->matrix()).pointer(),NB,
70 MatsT(0.),pv,NB);
71 for(auto j = 0; j < NB; j++) pv[j*NB+j] = MatsT(1.) + pv[j*NB+j];
72 /*
73 * construct Roothaan's effective fock
74 ======== ======== ====== =========
75 space closed open virtual
76 ======== ======== ====== =========
77 closed Fc Fb Fc
78 open Fb Fc Fa
79 virtual Fc Fa Fc
80 ======== ======== ====== =========
81 where Fc = (Fa + Fb) / 2
82 */
83 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(0.5),ss.fockMatrix->S().pointer(),NB,
84 pc,NB,MatsT(0.),tmp2,NB);
85 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(0.5),pc,NB,tmp2,NB,MatsT(0.),tmp.pointer(),NB);
86 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),pv,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
87 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(0.5),ss.fockMatrix->S().pointer(),NB,
88 po,NB,MatsT(0.),tmp2,NB);
89 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(0.5),po,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
90 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(0.5),ss.fockMatrix->S().pointer(),NB,
91 pv,NB,MatsT(0.),tmp2,NB);
92 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(0.5),pv,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
93 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),SCR[1].pointer(),NB,pc,NB,MatsT(0.),tmp2,NB);
94 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),po,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
95 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),SCR[0].pointer(),NB,pv,NB,MatsT(0.),tmp2,NB);
96 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),po,NB,tmp2,NB,MatsT(1.),tmp.pointer(),NB);
97 MatAdd('C','N',NB,NB,MatsT(1.),tmp.pointer(),NB,MatsT(1.),
98 tmp.pointer(),NB,ss.fockMatrix->S().pointer(),NB);
99
100 ss.memManager.free(pc,po,pv,tmp2);
101 }; // ROFock<MatsT, IntsT>::rohfFock
