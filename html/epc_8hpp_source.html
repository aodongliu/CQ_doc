83 void mkCrossAuxVar(
84 bool check_aux, bool electron,
85 std::shared_ptr<PauliSpinorSquareMatrices<MatsT>> onePDM1,
86 std::shared_ptr<PauliSpinorSquareMatrices<MatsT>> onePDM2,
87 double epsScreen, size_t NPts,
88 double *dndX1, double *dndY1, double *dndZ1,
89 double *dMxdX1, double *dMxdY1, double *dMxdZ1,
90 double *dMydX1, double *dMydY1, double *dMydZ1,
91 double *dMzdX1, double *dMzdY1, double *dMzdZ1,
92 double *dndX2, double *dndY2, double *dndZ2,
93 double *dMxdX2, double *dMxdY2, double *dMxdZ2,
94 double *dMydX2, double *dMydY2, double *dMydZ2,
95 double *dMzdX2, double *dMzdY2, double *dMzdZ2,
96 double *gammaColl){
97
98 // GGA Contributions
99 // GAMMA(++) = 0.25 * (GSCALAR(e).GSCALAR(p) + GMZ(e).GMZ(p) + GSCALAR(e).GZ(p) + GZ(e).GSCALAR(p))
100 // GAMMA(+-) = 0.25 * (GSCALAR.GSCALAR - GMZ.GMZ)
101 // GAMMA(-+) =
102 // GAMMA(--) = 0.25 * (GSCALAR.GSCALAR + GMZ.GMZ) - 0.5 * GSCALAR.GMZ
103 //2C : not yet implemented
104
105 // RKS part
106 for(auto iPt = 0; iPt < NPts; iPt++) {
107 gammaColl[4*iPt] = 0.25 * (dndX1[iPt]*dndX2[iPt] + dndY1[iPt]*dndY2[iPt] + dndZ1[iPt]*dndZ2[iPt]);
108 gammaColl[4*iPt+1] = gammaColl[4*iPt];
109 gammaColl[4*iPt+2] = gammaColl[4*iPt];
110 gammaColl[4*iPt+3] = gammaColl[4*iPt];
111 }
112
113 // main system is electron
114 if (electron) {
115
116 // restricted
117 for(auto iPt = 0; iPt < NPts; iPt++) {
118 double gammaTmp = 0.25 * (dndX1[iPt]*dMzdX2[iPt] + dndY1[iPt]*dMzdY2[iPt] + dndZ1[iPt]*dMzdZ2[iPt]);
119 gammaColl[4*iPt] += gammaTmp;
120 gammaColl[4*iPt+1] -= gammaTmp;
121 gammaColl[4*iPt+2] += gammaTmp;
122 gammaColl[4*iPt+3] -= gammaTmp;
123 }
124
125 // UKS
126 if( onePDM1->hasZ() ) {
127 for(auto iPt = 0; iPt < NPts; iPt++) {
128 double gammaTmp1 = 0.25 * (dMzdX1[iPt]*dMzdX2[iPt] + dMzdY1[iPt]*dMzdY2[iPt] + dMzdZ1[iPt]*dMzdZ2[iPt]);
129 double gammaTmp2 = 0.25 * (dMzdX1[iPt]*dndX2[iPt] + dMzdY1[iPt]*dndY2[iPt] + dMzdZ1[iPt]*dndZ2[iPt]);
130
131 // aa
132 gammaColl[4*iPt] += gammaTmp1;
133 gammaColl[4*iPt] += gammaTmp2;
134 // ab
135 gammaColl[4*iPt+1] -= gammaTmp1;
136 gammaColl[4*iPt+1] += gammaTmp2;
137 // ba
138 gammaColl[4*iPt+2] -= gammaTmp1;
139 gammaColl[4*iPt+2] -= gammaTmp2;
140 // bb
141 gammaColl[4*iPt+3] += gammaTmp1;
142 gammaColl[4*iPt+3] -= gammaTmp2;
143
144 }
145 }
146 }
147 else { // main system is proton
148
149 // restricted
150 for(auto iPt = 0; iPt < NPts; iPt++) {
151 double gammaTmp = 0.25 * (dMzdX1[iPt]*dndX2[iPt] + dMzdY1[iPt]*dndY2[iPt] + dMzdZ1[iPt]*dndZ2[iPt]);
152
153 gammaColl[4*iPt] += gammaTmp;
154 gammaColl[4*iPt+1] += gammaTmp;
155 gammaColl[4*iPt+2] -= gammaTmp;
156 gammaColl[4*iPt+3] -= gammaTmp;
157
158 }
159
160 // UKS
161 if( onePDM2->hasZ() ) {
162 for(auto iPt = 0; iPt < NPts; iPt++) {
163 double gammaTmp1 = 0.25 * (dMzdX1[iPt]*dMzdX2[iPt] + dMzdY1[iPt]*dMzdY2[iPt] + dMzdZ1[iPt]*dMzdZ2[iPt]);
164 double gammaTmp2 = 0.25 * (dndX1[iPt]*dMzdX2[iPt] + dndY1[iPt]*dMzdY2[iPt] + dndZ1[iPt]*dMzdZ2[iPt]);
165 // aa
166 gammaColl[4*iPt] += gammaTmp1;
167 gammaColl[4*iPt] += gammaTmp2;
168 // ab
169 gammaColl[4*iPt+1] -= gammaTmp1;
170 gammaColl[4*iPt+1] -= gammaTmp2;
171 // ba
172 gammaColl[4*iPt+2] -= gammaTmp1;
173 gammaColl[4*iPt+2] += gammaTmp2;
174 // bb
175 gammaColl[4*iPt+3] += gammaTmp1;
176 gammaColl[4*iPt+3] -= gammaTmp2;
177 }
178 }
179 }

300 template <typename MatsT>
301 void formZ_vxc_epc(
302 std::shared_ptr<PauliSpinorSquareMatrices<MatsT>> onePDM1,
303 std::shared_ptr<PauliSpinorSquareMatrices<MatsT>> onePDM2,
304 DENSITY_TYPE denTyp,
305 bool isGGA, size_t NPts, size_t NBE, size_t IOff,
306 double epsScreen, std::vector<double> &weights, double *ZrhoVar1,
307 double *ZgammaVar1, double *ZgammaVar2, double *ZgammaVar3,
308 double *DenS, double *DenZ, double *DenY, double *DenX,
309 double *GDenS, double *GDenZ, double *GDenY, double *GDenX,
310 double *aux_DenS, double *aux_DenZ, double *aux_DenY, double *aux_DenX,
311 double *aux_GDenS, double *aux_GDenZ, double *aux_GDenY, double *aux_GDenX,
312 double *BasisScratch, double *ZMAT){
313
314 double Fg;
315 // Fx,y,z ^m(all batch) in J. Chem. Theory Comput. 2011, 7, 3097â€“3104 Eq. 17 (changed for Total and Magn)
316 double FgX;
317 double FgY;
318 double FgZ;
319
320 memset(ZMAT,0,IOff*sizeof(double));
321
322 if( not onePDM1->hasXY() ) {
323 for(auto iPt = 0; iPt < NPts; iPt++) {
324 // LDA part -> Eq. 15 and 16 (see constructZVars docs for the missing factor of 0.5)
325 Fg = weights[iPt] * ZrhoVar1[iPt];
326
327#if VXC_DEBUG_LEVEL < 3
328 if(std::abs(Fg) > epsScreen)
329#endif
330 blas::axpy(NBE,Fg,BasisScratch + iPt*NBE,1,ZMAT+iPt*NBE,1);
331
332 // GGA part -> Eq. 15 and 17 (see constructZVars docs for the missing factor of 2)
333 if( isGGA ) {
334 FgX = weights[iPt] * ZgammaVar1[iPt] * GDenS[iPt];
335 // add contribution of the epc part
336 FgX += 0.5 * weights[iPt] * ZgammaVar3[iPt] * aux_GDenS[iPt];
337
338 if( onePDM1->hasZ() )
339 FgX += weights[iPt] * ZgammaVar2[iPt] * GDenZ[iPt];
340
341 // add contribution of the epc part
342 if( onePDM2->hasZ() )
343 FgX += 0.5 * weights[iPt] * ZgammaVar3[iPt] * aux_GDenZ[iPt];
344
345 FgY = weights[iPt] * ZgammaVar1[iPt] * GDenS[iPt + NPts];
346 // add contribution of the epc part
347 FgY += 0.5 * weights[iPt] * ZgammaVar3[iPt] * aux_GDenS[iPt + NPts];
348 if( onePDM1->hasZ() )
349 FgY += weights[iPt] * ZgammaVar2[iPt] * GDenZ[iPt + NPts];
350
351 // add contribution of the epc part
352 if( onePDM2->hasZ() )
353 FgY += 0.5 * weights[iPt] * ZgammaVar3[iPt] * aux_GDenZ[iPt + NPts];
354
355 FgZ = weights[iPt] * ZgammaVar1[iPt] * GDenS[iPt + 2*NPts];
356 // add contribution of the epc part
357 FgZ += 0.5 * weights[iPt] * ZgammaVar3[iPt] * aux_GDenS[iPt + 2*NPts];
358 if( onePDM1->hasZ() )
359 FgZ += weights[iPt] * ZgammaVar2[iPt] * GDenZ[iPt + 2*NPts];
360
361 // add contribution of the epc part
362 if( onePDM2->hasZ() )
363 FgZ += 0.5 * weights[iPt] * ZgammaVar3[iPt] * aux_GDenZ[iPt + 2*NPts];
364
365#if VXC_DEBUG_LEVEL < 3
366 if(std::abs(FgX) > epsScreen)
367#endif
368 blas::axpy(NBE,FgX,BasisScratch + iPt*NBE + IOff,1,ZMAT+iPt*NBE,1);
369
370#if VXC_DEBUG_LEVEL < 3
371 if(std::abs(FgY) > epsScreen)
372#endif
373 blas::axpy(NBE,FgY,BasisScratch + iPt*NBE + 2*IOff,1,ZMAT+iPt*NBE,1);
374
375#if VXC_DEBUG_LEVEL < 3
376 if(std::abs(FgZ) > epsScreen)
377#endif
378 blas::axpy(NBE,FgZ,BasisScratch + iPt*NBE + 3*IOff,1,ZMAT+iPt*NBE,1);
379 }
380 }
381
382 } else {
383 CErr("EPC-19 functional is not implemented for 2-component systems (GHF or X2C)!", std::cout);
384 } // 2C
