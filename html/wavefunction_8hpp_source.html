88 WaveFunction(MPI_Comm c, CQMemManager &mem, Molecule &mol, BasisSet &basis,
89 Integrals<IntsT> &aoi, size_t _nC, bool iCS, Particle p = {-1.0,1.0}) :
90 QuantumBase(c, mem,_nC,iCS,p),
91 WaveFunctionBase(c, mem, mol, basis,_nC,iCS,p),
92 Quantum<MatsT>(c, mem,_nC,iCS,p,basis.nBasis),
93 //molecule_(mol), basisSet_(basis),
94 eps1(nullptr), eps2(nullptr), aoints(aoi) {
95
96 // Compute meta data
97 size_t nBasis = basis.nBasis;
98
99 if (p.charge < 0.) {
100 this->nO = molecule_.nTotalE;
101 this->nV = 2*nBasis - nO;
102
103 if( this->iCS ) {
104 this->nOA = this->nO / 2; this->nOB = this->nO / 2;
105 this->nVA = this->nV / 2; this->nVB = this->nV / 2;
106 } else {
107 size_t nSingleE = molecule_.multip - 1;
108 this->nOB = (this->nO - nSingleE) / 2;
109 this->nOA = this->nOB + nSingleE;
110 this->nVA = nBasis - this->nOA;
111 this->nVB = nBasis - this->nOB;
112 }
113 } else {
114 this->nO = molecule_.nTotalP;
115 this->nV = 2*nBasis - nO;
116
117 if( this->iCS ) {
118 this->nOA = this->nO / 2; this->nOB = this->nO / 2;
119 this->nVA = this->nV / 2; this->nVB = this->nV / 2;
120 } else {
121 size_t nSingleP = molecule_.multip_proton - 1;
122 this->nOB = (this->nO - nSingleP) / 2;
123 this->nOA = this->nOB + nSingleP;
124 this->nVA = nBasis - this->nOA;
125 this->nVB = nBasis - this->nOB;
126 }
127
128 }
129
130 // Allocate internal memory
131 if(nBasis != 0) alloc();
132
