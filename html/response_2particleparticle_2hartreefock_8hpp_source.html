37 template <typename MatsT, typename IntsT>
38 template <typename U>
39 void ParticleParticlePropagator<HartreeFock<MatsT, IntsT>>::formLinearTrans_direct_impl(
40 MPI_Comm c, RC_coll<U> x){
41
42
43 SingleSlater<MatsT,IntsT> &ss = dynamic_cast<SingleSlater<MatsT,IntsT>&>(*this->ref_);
44
45 bool doA = this->tdaOp == PP_A;
46 bool doOcc = not this->genSettings.doTDA or this->tdaOp == PP_C;
47
48 const size_t tdOffSet = this->genSettings.doTDA ? 0 : this->aDim_;
49 const size_t N = this->nSingleDim_ ;
50 const size_t chunk = 600;
51
52
53
54#if 1
55/****************************/
56
57
58 size_t NB = ss.nAlphaOrbital();
59 size_t NO = ss.nOA;
60
61 MatsT* MMat = this->memManager_.template malloc<MatsT>(NB*NB);
62 MatsT* JMMat = this->memManager_.template malloc<MatsT>(NB*NB);
63 MatsT* KMMat = this->memManager_.template malloc<MatsT>(NB*NB);
64
65 MatsT* MO = ss.mo[0].pointer();
66
67 size_t indmo1 = NO;
68 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,NB,NB,NB,MatsT(1.),MO + NO*NB,NB,MO + NO*NB,NB,MatsT(0.),MMat,NB);
69
70 std::fill_n(JMMat,NB*NB,0.);
71 std::fill_n(KMMat,NB*NB,0.);
72
73 std::vector<TwoBodyContraction<MatsT>> in =
74 {
75 { MMat,JMMat,true, COULOMB },
76 { MMat,KMMat,true, EXCHANGE }
77 };
78
79 ss.TPI->twoBodyContract(c,in);
80
81 MatAdd('N','N',NB,NB,MatsT(1.),JMMat, NB, MatsT(-0.5), KMMat, NB, JMMat, NB);
82
83 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),JMMat,NB,MO ,NB,MatsT(0.),KMMat,NB);
84 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,NB,NB,NB,MatsT(1.),MO ,NB,KMMat,NB,MatsT(0.),MMat ,NB);
85
86/****************************/
87#endif
88
89 std::shared_ptr<TPIContractions<U,IntsT>> TPI =
90 TPIContractions<MatsT,IntsT>::template convert<U>(ss.TPI);
91
92 for(auto &X : x) {
93
94 const size_t nVec = X.nVec;
95
96 if( X.AX ) std::fill_n(X.AX,N*nVec,0.);
97
98 for(size_t k = 0; k < nVec; k += chunk) {
99
100 MPI_Barrier(c); // Sync MPI Processes at begining of each batch of
101 // vectors
102
103 const size_t nDo = std::min( chunk, nVec - k);
104
105 auto *V_c = X.X + k*N;
106 auto *HV_c = X.AX + k*N;
107
108
109 bool scatter = not bool(X.X);
110#ifdef CQ_ENABLE_MPI
111 scatter = mxx::any_of(scatter,c);
112#endif
113
114 // Transform pp vector MO -> AO
115 auto cList = this->genSettings.doTDA ?
116 this->template ppTransitionVecMO2AO<U>(c,scatter,nDo,N,V_c) :
117 this->template ppTransitionVecMO2AO<U>(c,scatter,nDo,N,V_c,
118 V_c + tdOffSet);
119
120 TPI->twoBodyContract(c,cList); // form G[V]
121
122
123
124
125 // Only finish transformation on root process
126 if( MPIRank(c) == 0 ) {
127
128 // Transform pp vector AO -> MO
129 if( this->genSettings.doTDA )
130 this->ppTransitionVecAO2MO(nDo,N,cList,HV_c);
131 else
132 this->ppTransitionVecAO2MO(nDo,N,cList,HV_c,HV_c + tdOffSet);
133
134
135
136
137 // Scale by diagonals
138 if( not this->genSettings.doTDA or doA )
139 this->ppEpsilonScale(true,false,false,nDo,N,V_c,HV_c);
140
141 if( doOcc )
142 this->ppEpsilonScale(true,false,true,nDo,N,V_c + tdOffSet,
143 HV_c + tdOffSet);
144
145
146 // SCF* reference
147 if( this->doStarRef ) {
148
149 const size_t NV_t = this->getNV1(this->doStarRef);
150 const size_t NV2_t = this->getNV2(this->doStarRef);
151 const size_t NO_t = this->getNO1(this->doStarRef);
152 const size_t NO2_t = this->getNO2(this->doStarRef);
153
154 const bool doLT = (this->ref_->nC == 2) or (this->spinSepProp != PP_AB);
155 auto bmax = [&](size_t a){ return doLT ? a : NV2_t; };
156 auto jmax = [&](size_t i){ return doLT ? i : NO2_t; };
157
158 for(auto iVec = 0; iVec < nDo; iVec++) {
159
160 auto * HV_cc = HV_c + iVec*N;
161 auto * V_cc = V_c + iVec*N;
162
163 for(auto a = 0ul, ab = 0ul; a < NV_t; a++ )
164 for(auto b = 0ul ; b < bmax(a); b++, ab++){
165
166 const size_t A = a + NO_t;
167 const size_t B = b + NO_t;
168
169 for(auto c = 0ul, cd = 0ul; c < NV_t; c++ )
170 for(auto d = 0ul ; d < bmax(c); d++, cd++){
171
172 const size_t C = c + NO_t;
173 const size_t D = d + NO_t;
174
175 if(a == c) HV_cc[ab] -= V_cc[cd] * MMat[B + D*NB];
176 if(b == d) HV_cc[ab] -= V_cc[cd] * MMat[A + C*NB];
177 if(a == d) HV_cc[ab] += V_cc[cd] * MMat[B + C*NB];
178 if(b == c) HV_cc[ab] += V_cc[cd] * MMat[A + D*NB];
179
180 }
181
182
183 }
184
185 for(size_t i = 0ul, ij = this->aDim_; i < NO_t; i++ )
186 for(size_t j = 0ul ; j < jmax(i); j++, ij++){
187
188 for(size_t k = 0ul, kl = this->aDim_; k < NO_t; k++ )
189 for(size_t l = 0ul ; l < jmax(k); l++, kl++){
190
191 if(i == k) HV_cc[ij] += V_cc[kl] * MMat[j + l*NB];
192 if(j == l) HV_cc[ij] += V_cc[kl] * MMat[i + k*NB];
193 if(i == l) HV_cc[ij] -= V_cc[kl] * MMat[j + k*NB];
194 if(j == k) HV_cc[ij] -= V_cc[kl] * MMat[i + l*NB];
195
196 }
197
198
199 }
200
201
202
203
204
205 }
206
207 }
208 }
209
210
211 // Free up transformation memory
212 this->memManager_.free(cList[0].X);
213
214 } // loop over vectors
215
216 } // loop over groups of vectors
217
218
219
220 };
