73 template <typename MatsT, typename IntsT>
74 std::vector<int> SingleSlater<MatsT,IntsT>::fchkToCQMO() {
75
76 std::ifstream fchkFile;
77 fchkFile.open(fchkFileName);
78 std::vector<int> sl;
79
80 if ( not fchkFile.good() ) CErr("Could not find fchkFile. Use -s flag.");
81
82 // dimension of mo1 and mo2
83 auto NB = this->basisSet().nBasis;
84 auto NBC = this->nC * NB;
85 auto NB2 = NB*NB;
86 auto NBC2 = NBC*NBC;
87
88 // Boolean for if fchk entries are found
89 bool isBeta=false, readAlpha=false, readBeta=false;
90 bool readShell=false;
91 // Various integers
92 int mo1Counter = 0, mo2Counter = 0, compCounter = 0, maxLfchk;
93 // Various double
94 double prevValue = 0.0;
95 // Double pointer for handling complex mo1
96 double* dptr = reinterpret_cast<double*>(this->mo[0].pointer());
97
98 // Parse fchk file
99 while( not fchkFile.eof() ) {
100
101 std::string line;
102 std::getline(fchkFile,line);
103
104 // Determine position of first and last non-space character
105 size_t firstNonSpace = line.find_first_not_of(" ");
106
107 // Skip blank lines
108 if( firstNonSpace == std::string::npos ) continue;
109
110 // Strip trailing spaces
111 trim_right(line);
112 line =
113 line.substr(firstNonSpace,line.length()-firstNonSpace);
114
115 // Split the line into tokens, trim spaces
116 std::vector<std::string> tokens;
117 split(tokens,line," ");
118 for(auto &X : tokens) { trim(X); }
119
120 // Some sanity checks on fchk data
121 // Checking NB
122 if ( tokens.size() > 5 ){
123 if ( tokens[2] == "basis" and tokens[3] == "functions" ){
124 if ( std::stoi(tokens[5]) != NB ){
125 std::cout << " NB from fchk is " << tokens[5] << " and NB in Chronus is " << NB << "\n";
126 CErr("Basis functions do not agree between fchk and Chronus!");
127 }
128 }
129 }
130
131 // Checking highest angular momentum
132 if ( tokens.size() > 4 ){
133 if ( tokens[0] == "Highest" and tokens[1] == "angular" ){
134 maxLfchk = std::stoi(tokens[4]);
135 if( maxLfchk > 5 ){
136 std::cout << "Max L in fchk is " << maxLfchk << ", but greater than " << 5 << " is NYI" << "\n";
137 CErr("Angular momentum too high for fchk parser!");
138 }
139 }
140 }
141
142 // Checking size of mo1
143 if ( tokens.size() > 5 ){
144 if ( tokens[0] == "Alpha" and tokens[1] == "MO" ){
145 if ( this->nC == 1 ){
146 if ( std::stoi(tokens[5]) != NBC2 ){
147 std::cout << " MO size from fchk is " << tokens[5] << " and MO size in Chronus is " << NBC2 << "\n";
148 CErr("MO coefficient size do not agree between fchk and ChronusQ!");
149 }
150 } else if ( this->nC == 2 ){
151 // Factor of two here since real and imaginary are separate in fchk
152 if ( std::stoi(tokens[5]) != 2*NBC2 ){
153 std::cout << " MO size from fchk is " << tokens[5] << " but expected size is " << 2*NBC2 << "\n";
154 CErr("MO coefficient size on fchk is not compatible!");
155 }
156 } else{
157 CErr("fchk functionality only implemented for 1c and 2c!");
158 }
159 // Found alpha MO block
160 readAlpha=true;
161 continue;
162 }
163 }
164
165 // Check if unrestricted
166 if ( tokens.size() > 5 ){
167 if ( tokens[0] == "Beta" and tokens[1] == "MO" ){
168 if ( this->nC == 2 or this->iCS ) CErr("Beta MOs present in fchk file but Chronus is 2c or restricted");
169 isBeta = true;
170 readBeta = true;
171 readAlpha = false;
172 continue;
173 }
174 }
175
176 // Check if shell list started
177 if ( tokens.size() > 4 ){
178 if ( tokens[0] == "Shell" and tokens[1] == "types" ){
179 readShell = true;
180 continue;
181 }
182 }
183
184 // Check if shell list ended
185 if ( tokens.size() > 5 ){
186 if ( tokens[2] == "primitives" and tokens[3] == "per" ){
187 readShell = false;
188 continue;
189 }
190 }
191
192 // Check if alpha or beta block ended
193 if ( tokens.size() == 5 ){
194 if ( tokens[0] == "Orthonormal" and tokens[1] == "basis" ){
195 if( this-> nC == 1){
196 if( isBeta ) readBeta = false;
197 else readAlpha = false;
198 continue;
199 }
200 if( this-> nC == 2){
201 readAlpha = false;
202 continue;
203 }
204 }
205 }
206
207 // Read in Alpha MO coeffients
208 if ( readAlpha ){
209 for(int i=0; i<tokens.size(); i++){
210 dptr[mo1Counter]=std::stod(tokens[i]);
211 mo1Counter=mo1Counter+1;
212 }
213 }
214
215 // Read in Beta MO coeffients
216 if ( readBeta ){
217 for(int i=0; i<tokens.size(); i++){
218 this->mo[1].pointer()[mo2Counter]=std::stod(tokens[i]);
219 mo2Counter=mo2Counter+1;
220 }
221 }
222
223 // Read in shell types
224 if ( readShell ){
225 for(int i=0; i<tokens.size(); i++){
226 sl.push_back(std::stoi(tokens[i]));
227 }
228 }
229
230 }// end of eof loop
231
232 if ( not isBeta and this->nC == 1 and not this->iCS ) CErr("Could not find beta MOs on fchk file!");
233
234 fchkFile.close();
235
236 return sl;
237
238 } // SingleSlater<T>::fchkToCQMO()

245 template <typename MatsT, typename IntsT>
246 void SingleSlater<MatsT,IntsT>::reorderAngMO(std::vector<int> sl, MatsT* tmo, int sp) {
247
248 // Dimension of mo
249 auto NB = this->basisSet().nBasis;
250 auto NBC = this->nC * NB;
251 auto NBC2 = NBC*NBC;
252
253 if( this->nC == 4 ) CErr("reorderAngMO NYI for 4c",std::cout);
254
255 // Skip past beta part if 2c
256 int skipl = this->nC;
257
258 // Obtain map for angular momentum ordering between Gaussian and CQ
259 std::unordered_map<int,std::vector<int>> angReorder = returnAngReorder();
260
261 // Loop over each MO
262 for( auto iMO=0; iMO<NBC; iMO++){
263
264 // Loop over each shell entry in sl
265 for( auto iSh=0, iAO=0; iSh<sl.size(); iSh++){
266
267 // Check that shell entry is implemented
268 bool contains = angReorder.find(sl[iSh]) != angReorder.end();
269 if( not contains ){
270 std::cout << "Cannot find shell entry: " << sl[iSh] << std::endl;
271 CErr("Shell value for FCHKMO NYI!",std::cout);
272 }
273
274 const std::vector<int>& reorder = angReorder[sl[iSh]];
275
276 // Loop over each function per shell
277 for( auto iML=0; iML<reorder.size(); iML++){
278
279 this->mo[sp].pointer()[iMO*NBC+iAO+skipl*(iML+reorder[iML])] = tmo[iMO*NBC+iAO+iML*skipl];
280 // Beta part
281 if( this->nC==2 )
282 this->mo[sp].pointer()[iMO*NBC+1+iAO+skipl*(iML+reorder[iML])] = tmo[iMO*NBC+1+iAO+iML*skipl];
283
284 }
285
286 iAO=iAO+reorder.size()*skipl;
287
288 }
289
290 }
291
292 } // SingleSlater<T>::reorderAngMO()

301 template <typename MatsT, typename IntsT>
302 void SingleSlater<MatsT,IntsT>::reorderSpinMO() {
303
304 // Dimension of mo1
305 auto NB = this->basisSet().nBasis;
306 auto NBC = this->nC * NB;
307 auto NBC2 = NBC*NBC;
308
309 // Counters for spin reordering
310 int acount = 0, bcount = NB - 1;
311
312 // mo scratch
313 MatsT* mo1tmp = memManager.malloc<MatsT>(NBC2);
314 SetMat('N',NBC,NBC,MatsT(1.),this->mo[0].pointer(),NBC,mo1tmp,NBC);
315
316 // Loop over mo1
317 for( int i=0; i<NBC2; i++){
318
319 // Reset acount and bcount for each MO
320 if( i % NBC == 0 ){
321 acount = 0;
322 bcount = NB - 1;
323 }
324
325 // If alpha
326 if( i % 2 == 0 ){
327 this->mo[0].pointer()[i-acount] = mo1tmp[i];
328 acount = acount + 1;
329 }else{ // If beta
330 this->mo[0].pointer()[i+bcount] = mo1tmp[i];
331 bcount = bcount - 1;
332 }
333 }
334
335 memManager.free(mo1tmp);
336
337 } // SingleSlater<T>::reorderSpinMO()
