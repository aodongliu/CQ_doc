34 template <typename MatsT>
35 template <DENSITY_TYPE DenTyp, typename Op>
36 double Quantum<MatsT>::OperatorSpinCombine(const Op &op) {
37 double rZero(0.);
38 bool isReal = std::is_same<double,MatsT>::value;
39
40 size_t NB = onePDM->dimension();
41 size_t DSize = NB * NB;
42
43 // Scalar trace option always valid
44
45 // Zero eval if 1C, closed shell and not scalar trace
46 bool isZero = (nC == 1 and iCS) and (DenTyp != DENSITY_TYPE::SCALAR);
47
48 // Zero eval if 1C, open shell and MX / MY trace
49 isZero = isZero or (
50 (nC == 1 and not iCS) and
51 (DenTyp == DENSITY_TYPE::MY or DenTyp == DENSITY_TYPE::MX)
52 );
53
54 // Zero eval is 2C, real and MY trace
55 isZero = isZero or (
56 (nC > 1) and (DenTyp == DENSITY_TYPE::MY and isReal)
57 );
58
59 // Catch zero evals
60 if(isZero)
61 return rZero;
62
63 // Sanity checks
64 assert( memManager.template getSize(op) == DSize );
65
66 // Perform proper trace
67 MatsT *Dptr = (*onePDM)[static_cast<PAULI_SPINOR_COMPS>(DenTyp)].pointer();
68
69 return OperatorTrace(DSize,Dptr,op);
70
71 }; // Quantum<T>::OperatorSpinCombine
