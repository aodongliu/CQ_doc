40/*
41 * Brief: Object to holds the overlapping functions for orbital optimization
42 * such as printing the header and iteration progress. Thus, getNewOrbitals
43 * is modified depending on which step is used. ie. SCF or Newton-Raphson
44 */
45template<typename MatsT>
46class OptimizeOrbitals : public ModifyOrbitals<MatsT> {
47 private:
48 std::vector<SquareMatrix<MatsT>> prevOnePDM;
49 double prevEnergy;
50
51 public:
52 const SCFControls scfControls;
53 SCFConvergence scfConv;
54 bool doingDamp;
55
56 // Constructor
57 OptimizeOrbitals(SCFControls sC, MPI_Comm comm, ModifyOrbitalsOptions<MatsT> modOpt, CQMemManager& mem):
58 scfControls(sC), ModifyOrbitals<MatsT>(comm, modOpt, mem) {
59
60 // Allocate prevOnePDM
61 VecShrdPtrMat<MatsT> den = this->modOrbOpt.getOnePDM();
62 for( size_t a = 0; a < den.size(); a++ ) {
63 prevOnePDM.emplace_back(den[a]->memManager(), den[a]->dimension());
64 prevOnePDM[a] = *den[a];
65 }
66 };
67
68 // Destructor
69 ~OptimizeOrbitals() {};
70
71 // Perform an SCF procedure (see include/singleslater/scf.hpp for docs)
72 void runModifyOrbitals(EMPerturbation&, VecMORef<MatsT>&, VecEPtr&);
73
74 // Evaluate convergence
75 bool evalProgress(EMPerturbation&);
76 double computeDensityConv();
77 virtual double computeFDCConv() { return 0.; };
78
79 // Print SCF header, footer and progress
80 void printRunHeader(std::ostream& out, EMPerturbation&) const;
81 void printHeaderFinal(std::ostream& out) const;
82 void printIteration(std::ostream& out = std::cout, bool printDiff = true) const;
83
84 // Common SCF Functions
85 void computeEigenvalues(VecMORef<MatsT>&, VecEPtr&);
86};
87}; // namespace ChronusQ
