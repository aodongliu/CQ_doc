35template <typename MatsT>
36void OptimizeOrbitals<MatsT>::runModifyOrbitals(EMPerturbation& pert, VecMORef<MatsT>& mo, VecEPtr& eps) {
37
38 // Perform Sanity Check of MO's and Eigenvalues
39 if( mo.size() != eps.size() ) CErr("Number of sets of MO's and Eigenvalues are not the same");
40
41 ProgramTimer::tick("SCF Total");
42
43 // Copy Guess onePDM
44 VecShrdPtrMat<MatsT> den = this->modOrbOpt.getOnePDM();
45 for( size_t a=0; a<den.size(); a++)
46 prevOnePDM[a] = *den[a];
47 prevEnergy = this->modOrbOpt.getTotalEnergy();
48 doingDamp = scfControls.doDamp;
49
50 // Initialize type independent parameters
51 bool isConverged = false;
52
53 // Compute initial properties
54 this->modOrbOpt.computeProperties(pert);
55
56 if( scfControls.printLevel > 0 ) {
57 printRunHeader(std::cout, pert);
58 printHeaderFinal(std::cout);
59 printIteration(std::cout, false);
60 }
61
62 for( this->scfConv.nSCFIter = 0; this->scfConv.nSCFIter < scfControls.maxSCFIter; this->scfConv.nSCFIter++ ) {
63
64 // Save current state of the wave function (method specific)
65 this->modOrbOpt.saveCurrentState();
66
67 // Exit loop on convergence
68 if( isConverged ) break;
69
70 ProgramTimer::tick("SCF Iter");
71
72 // Get new orbtials and densities from current state:
73 // C/D(k) -> C/D(k + 1)
74 this->getNewOrbitals(pert, mo, eps);
75
76 // Evaluate convergence
77 isConverged = evalProgress(pert);
78
79 // FIXME: handle COMM
80 // Print out iteration information
81 if( scfControls.printLevel > 0 and (MPIRank(this->comm) == 0) ) printIteration(std::cout);
82
83#ifdef CQ_ENABLE_MPI
84
85 // Broadcast the updated MO's/Eigenvalues to all MPI processes
86 if( MPISize(this->comm) > 1 ) {
87 for( auto& m : mo ){
88 std::cerr << " *** Scattering the MOs ***\n";
89 size_t Nmo = m.get().dimension();
90 MPIBCast(m.get().pointer(),Nmo*Nmo,0,this->comm);
91 }
92
93 std::cerr << " *** Scattering EPS ***\n";
94 size_t cnt = 0;
95 for( auto* e : eps ){
96 size_t Nmo = mo[cnt].get().dimension();
97 MPIBCast(e,Nmo,0,this->comm);
98 ++cnt;
99 }
100 }
101#endif
102
103 ProgramTimer::tock("SCF Iter");
104
105 }; // Iteration loop
106
107 // Save current state of the wave function (method specific)
108 this->modOrbOpt.saveCurrentState();
109
110 // Compute initial properties
111 this->modOrbOpt.computeProperties(pert);
112
113 // printSCFFooter(isConverged);
114 if( not isConverged )
115 CErr(std::string("SCF Failed to converged within ") + std::to_string(scfControls.maxSCFIter) + std::string(" iterations"));
116 else if( scfControls.printLevel > 0 ) {
117 std::cout << std::endl
118 << "SCF Completed: E(" << scfControls.refShortName_ << ") = " << std::fixed << std::setprecision(10) << this->modOrbOpt.getTotalEnergy() << " Eh after "
119 << this->scfConv.nSCFIter << " SCF Iterations" << std::endl;
120 }
121
122 if( scfControls.printLevel > 0 ) std::cout << BannerEnd << std::endl;
123
124 if( scfControls.printLevel > 1 ) this->modOrbOpt.printProperties();
125
126 ProgramTimer::tock("SCF Total");
127
128}; // OptimizeOrbitals<MatsT,IntsT>::SCF()

250template <typename MatsT>
251bool OptimizeOrbitals<MatsT>::evalProgress(EMPerturbation& pert) {
252
253 bool isConverged;
254
255 // Compute all SCF convergence information on root process
256 if( MPIRank(this->comm) == 0 ) {
257
258 // Compute new energy (with new Density)
259 this->modOrbOpt.computeProperties(pert);
260 this->scfConv.deltaEnergy = this->modOrbOpt.getTotalEnergy() - prevEnergy;
261 prevEnergy = this->modOrbOpt.getTotalEnergy();
262
263 bool energyConv = std::abs(this->scfConv.deltaEnergy) < scfControls.eneConvTol;
264
265 // Check density convergence
266 this->scfConv.RMSDen = computeDensityConv();
267
268 bool denConv = this->scfConv.RMSDen < scfControls.denConvTol;
269
270 // Check FP convergence
271 this->scfConv.maxFDC = computeFDCConv();
272 bool FDCConv = this->scfConv.maxFDC < scfControls.FDCConvTol;
273
274 isConverged = FDCConv and energyConv and denConv;
275
276 // Toggle damping based on energy difference
277 if(scfControls.doDamp ) {
278 bool largeEDiff = std::abs(this->scfConv.deltaEnergy) > scfControls.dampError;
279
280 if( doingDamp and not largeEDiff and scfControls.dampParam > 0. ) {
281
282 if( scfControls.printLevel > 0 )
283 std::cout << " *** Damping Disabled - Energy Difference Fell Below " << scfControls.dampError << " ***" << std::endl;
284
285 doingDamp = false;
286
287 } else if( not doingDamp and largeEDiff and scfControls.dampParam <= 0. ) {
288
289 if( scfControls.printLevel > 0 )
290 std::cout << " *** Damping Enabled Due to " << scfControls.dampError << " Oscillation in Energy ***" << std::endl;
291
292 doingDamp = true;
293 }
294 }
295 }
296
297#ifdef CQ_ENABLE_MPI
298 // Broadcast whether or not we're converged to ensure that all
299 // MPI processes exit the SCF simultaneously
300 if( MPISize(this->comm) > 1 ) MPIBCast(isConverged, 0, this->comm);
301#endif
302
303 return isConverged;
304
305}; // OptimizeOrbitals<MatsT>::evalConver

307template <typename MatsT>
308double OptimizeOrbitals<MatsT> :: computeDensityConv() {
309 VecShrdPtrMat<MatsT> currDen = this->modOrbOpt.getOnePDM();
310
311 // Compute RMS change in Density
312 double rmsDen = 0.;
313 for( size_t a=0; a<currDen.size(); a++ ) {
314 SquareMatrix<MatsT> dDen = *currDen[a] - prevOnePDM[a];
315 size_t NB = currDen[a]->dimension();
316 rmsDen += blas::nrm2(NB*NB,dDen.pointer(),1) / NB;
317 prevOnePDM[a] = *currDen[a];
318 }
319 return rmsDen;
320}; // OptimizeOrbitals<MatsT> :: computeDensityConv

326template<typename MatsT>
327void OptimizeOrbitals<MatsT>::computeEigenvalues(VecMORef<MatsT>& mo, VecEPtr& eps) {
328
329 VecShrdPtrMat<MatsT> fock = this->modOrbOpt.getFock();
330
331 for( size_t i = 0; i < mo.size(); i++ ) {
332 size_t NB = fock[i]->dimension();
333
334 SquareMatrix<MatsT> moFock = fock[i]->transform('N', mo[i].get().pointer(), NB, NB);
335 for( size_t a = 0; a < NB; a++ )
336 eps[i][a] = std::real(moFock(a, a));
337 }
338};
