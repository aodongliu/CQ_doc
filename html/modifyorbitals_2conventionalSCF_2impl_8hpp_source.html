35template<typename MatsT>
36void ConventionalSCF<MatsT>::getNewOrbitals(EMPerturbation& pert, VecMORef<MatsT>& mo,
37 VecEPtr& eps) {
38
39 // Form the Fock matrix D(k) -> F(k)
40 ProgramTimer::timeOp("Form Fock", [&]() { this->modOrbOpt.formFock(pert); });
41
42 // Transform AO fock into the orthonormal basis (on root MPI process)
43 ao2orthoFock();
44
45 // Compute the orbital gradient for convergence tests and DIIS
46 computeOrbGradient(orbGrad);
47
48 // Modify fock matrix if requested (on root MPI process)
49 if( this->scfControls.doExtrap ) modifyFock(pert);
50
51 // Diagonalize the orthonormal fock Matrix (on root MPI process)
52 diagOrthoFock(mo, eps);
53
54 ortho2aoMOs(mo);
55
56 this->modOrbOpt.formDensity();
57
58}; // NewtonRaphsonSCF<MatsT,IntsT>::getNewOrbitals

60template<typename MatsT>
61void ConventionalSCF<MatsT>::ao2orthoFock(VecShrdPtrMat<MatsT> fock) {
62
63 ROOT_ONLY(this->comm);
64
65 if( fock.empty() ) fock = this->modOrbOpt.getFock();
66
67 VecShrdPtrOrtho<MatsT> ortho = this->modOrbOpt.getOrtho();
68 for( size_t i = 0; i < fock.size(); i++ ) {
69 if( ortho[i]->hasOverlap() ) {
70 fockMatrixOrtho[i] = ortho[i]->nonortho2ortho(*fock[i]);
71 } else {
72 fockMatrixOrtho[i] = *(fock[i]);
73 }
74 }
75};

94template<typename MatsT>
95void ConventionalSCF<MatsT>::diagOrthoFock(VecMORef<MatsT>& mo, VecEPtr& eps) {
96
97 ROOT_ONLY(this->comm);
98
99 for( size_t i = 0; i < mo.size(); i++ ) {
100 size_t NB = mo[i].get().dimension();
101 if( NB != fockMatrixOrtho[i].dimension() ) CErr("Ortho Fock and MO dimensions do not match");
102 std::copy_n(fockMatrixOrtho[i].pointer(),NB*NB,mo[i].get().pointer());
103 int INFO = HermetianEigen('V', 'L', NB, mo[i].get().pointer(), NB, eps[i], this->memManager);
104 if( INFO != 0 ) {
105 std::cout << "Attempted to diagonalize " << i << "th Fock Matrix" << std::endl;
106 CErr("HermetianEigen failed in Fock", std::cout);
107 }
108 }
109}
