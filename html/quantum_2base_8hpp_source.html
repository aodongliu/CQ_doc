146 void computeEnergy(EMPerturbation &pert){
147
148 ROOT_ONLY(comm);
149
150 computeEnergy();
151
152
153 // Field contributions to the energy
154 double field_delta(0.);
155
156 // Purely magnetic field contributions here
157 auto magDipoleField = pert.getDipoleAmp(Magnetic);
158
159
160 // Purely electric field contributions here
161 auto elecDipoleField = pert.getDipoleAmp(Electric);
162 field_delta +=
163 elecDipoleField[0] * elecDipole[0] +
164 elecDipoleField[1] * elecDipole[1] +
165 elecDipoleField[2] * elecDipole[2];
166
167
168 totalEnergy += field_delta; // Increment total energy
169
170 };
