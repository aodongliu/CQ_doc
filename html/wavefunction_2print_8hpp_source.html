60 template <typename T, typename ValManipOp>
61 void prettyMOPrint(std::ostream &out, size_t NB, size_t NOrb, double *EPS,
62 T* MO, size_t LDM, Molecule &mol, BasisSet &basis,
63 ValManipOp op ){
64
65 constexpr size_t maxLPrint = 6 + 1;
66
67 std::array< std::vector<std::string>, maxLPrint > cartLabel, sphLabel;
68 std::array< std::string, maxLPrint > angLabel =
69 { "S", "P", "D", "F", "G", "H", "I" };
70
71 std::array< std::string, 3 > axes = { "X", "Y", "Z" };
72
73 // S Functions
74 cartLabel[0] = { angLabel[0] };
75
76 for(auto i = 0; i < 3; i++) {
77
78 // P Functions
79 cartLabel[1].emplace_back( angLabel[1] + axes[i] );
80
81 for(auto j = i; j < 3; j++) {
82
83 // D Functions
84 cartLabel[2].emplace_back( angLabel[2] + axes[i] + axes[j] );
85
86 for(auto k = j; k < 3; k++) {
87
88 // F Functions
89 cartLabel[3].emplace_back( angLabel[3] + axes[i] + axes[j] + axes[k] );
90
91 for(auto l = k; l < 3; l++) {
92
93 // G Functions
94 cartLabel[4].emplace_back( angLabel[4] + axes[i] + axes[j] + axes[k]
95 + axes[l] );
96
97 for(auto m = l; m < 3; m++) {
98
99 // H Functions
100 cartLabel[5].emplace_back( angLabel[5] + axes[i] + axes[j] + axes[k]
101 + axes[l] + axes[m] );
102
103 for(auto n = m; n < 3; n++) {
104
105 // I Functions
106 cartLabel[6].emplace_back( angLabel[6] + axes[i] + axes[j] + axes[k]
107 + axes[l] + axes[m] + axes[n] );
108
109 }}}}}}
110
111 // Spherical Labels
112 auto pm_string = [](int x) -> std::string {
113
114 if( x > 0 ) return "+" + std::to_string(x) ;
115 else if (x < 0 ) return "" + std::to_string(x) ;
116 else return " " + std::to_string(x) ;
117
118 };
119
120 sphLabel[0] = cartLabel[0];
121 sphLabel[1] = cartLabel[1];
122 for(auto i = 2; i < maxLPrint; i++) {
123
124 for(auto ml = -i; ml <= i; ml++)
125 sphLabel[i].emplace_back( angLabel[i] + pm_string(ml) );
126
127 }
128
129 // Printing format setting
130 size_t list = 4; // #mo per line
131 size_t printWidth = 14;
132 size_t eValOff = 24;
133
134 out << std::endl << bannerTop << std::endl;
135
136 out << std::scientific << std::left << std::setprecision(5);
137 for(size_t p = 0; p < NOrb; p += list) {
138
139 out << std::left << std::endl;
140 int end = list;
141
142 if( (p + list) >= NOrb ) end = NOrb - p;
143
144 out << std::left << std::setw(eValOff) << " ";
145 out << std::right;
146 for(size_t k = p; k < p + end; k++)
147 out << std::setw(printWidth) << k + 1;
148 out << std::endl;
149
150 out << std::left << std::setw(eValOff) << " EigV --";
151 out << std::right;
152 for(size_t k = p; k < p + end; k++)
153 out << std::setw(printWidth) << EPS[k];
154 out << std::endl << std::endl;
155
156 // Function to get the atom info
157 auto getAtmSymb = [&](size_t iAtm) -> std::string {
158
159 std::map<std::string,Atom>::const_iterator it =
160 std::find_if(atomicReference.begin(),atomicReference.end(),
161 [&](const std::pair<std::string,Atom> &st){
162 return (st.second.atomicNumber == mol.atoms[iAtm].atomicNumber) and
163 (st.second.massNumber == mol.atoms[iAtm].massNumber);}
164 );
165
166 return (it == atomicReference.end() ? "X" : it->first);
167
168 };
169
170 size_t iAtm = 0;
171 std::string atmSymb = getAtmSymb(iAtm);
172 size_t iShellAtm = 0;
173 size_t nShell = basis.shells.size();
174
175 for(size_t iShell = 0; iShell < nShell ; iShell++, iShellAtm++) {
176
177 size_t bfst = basis.mapSh2Bf[iShell];
178 size_t sz = basis.shells[iShell].size();
179 size_t L = basis.shells[iShell].contr[0].l;
180
181 size_t curCen = basis.mapSh2Cen[iShell];
182 bool newAtm = false;
183 if( curCen != iAtm ) {
184 newAtm = true;
185 iAtm++;
186 atmSymb = getAtmSymb(iAtm);
187 iShellAtm = 0;
188 }
189
190 bool lastAtm = iAtm == (mol.atoms.size() - 1);
191 bool firstAtm = iShell == 0;
192
193 for(size_t mu = bfst, ml = 0 ; ml < sz; mu++, ml++) {
194
195 // BF number
196 out << " " << std::setw(5) << std::left << mu + 1; // 6
197
198 // atom info
199 if( (newAtm or firstAtm) ) {
200 out << std::setw(3) << iAtm;
201 out << std::setw(7) << atmSymb;
202 } else out << std::setw(10) << " "; // 16
203
204 // #shell and spherical label
205 out << std::setw(2) << iShellAtm ;
206 out << std::setw(4) << sphLabel[L][ml]; //22
207
208 out << std::setw(2) << " "; // 24
209
210 // MO coefficients
211 for(auto q = p; q < p + end; q++) {
212
213 double VAL = op(MO[mu + q*LDM]);
214 out << std::right << std::setw(printWidth);
215
216 if(std::abs(VAL) > PRINT_SMALL) out << VAL;
217 else if(std::isnan(std::abs(VAL))) out << "NAN";
218 else if(std::isinf(std::abs(VAL))) out << "INF";
219 else out << 0.;
220
221 }
222
223 out << std::endl;
224
225 bool nextAtmNew = lastAtm ? false : (mu+1) >= basis.mapCen2BfSt[iAtm+1];
226
227 if( nextAtmNew ) out<<std::endl;
228 }
229 }
230 out << std::endl;
231 }
232 out << bannerEnd << std::endl;
233
234 }; // prettyMOPrint

249 template <typename T, typename IntsT>
250 void analyzeMOPrint(std::ostream &out, size_t NB, size_t NOrb, IntsT* S, T* MO,
251 size_t LDM, Molecule &mol, BasisSet &basis, CQMemManager &mem,
252 bool groupAtm = false, T* MO2 = nullptr) {
253
254 out << "\nMO components projected to basis functions of different angular momentum";
255
256 constexpr size_t maxLPrint = 6 + 1;
257 T* SCR = mem.template malloc<T>(NB*NOrb);
258 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,
259 NB,NOrb,NB,T(1.),S,NB,MO,LDM,T(0.),SCR,NB);
260 // If alpha and beta parts are summed together
261 T *SCR2;
262 if (MO2) {
263 SCR2 = mem.template malloc<T>(NB*NOrb);
264 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,
265 NB,NOrb,NB,T(1.),S,NB,MO2,LDM,T(0.),SCR2,NB);
266 }
267
268 std::array< std::string, maxLPrint > angLabel =
269 { "S", "P", "D", "F", "G", "H", "I" };
270
271 // Printing format setting
272 size_t list = 7;
273 size_t printWidth = 10;
274 size_t eValOff = 10;
275
276 out << std::endl << bannerTop << std::endl;
277
278 auto getAtmSymb = [&](size_t iAtm) -> std::string {
279
280 std::map<std::string,Atom>::const_iterator it =
281 std::find_if(atomicReference.begin(),atomicReference.end(),
282 [&](const std::pair<std::string,Atom> &st){
283 return (st.second.atomicNumber == mol.atoms[iAtm].atomicNumber) and
284 (st.second.massNumber == mol.atoms[iAtm].massNumber);}
285 );
286
287 return (it == atomicReference.end() ? "X" : it->first);
288
289 };
290
291
292 for(size_t p = 0; p < NOrb; p += list) {
293 out << std::left << std::endl;
294 int end = list;
295 if( (p + list) >= NOrb ) end = NOrb - p;
296 std::vector<std::vector<T>> angWeight(end,
297 std::vector<T>(maxLPrint, T(0.0)));
298 out << std::left << std::setw(eValOff) << " ";
299 out << std::right;
300 for(size_t k = p; k < p + end; k++)
301 out << std::setw(printWidth) << k + 1;
302 out << std::endl;
303
304 size_t iAtm = 0;
305 std::string atmSymb = getAtmSymb(iAtm);
306
307 size_t nShell = basis.shells.size();
308 size_t maxL = 0;
309 for(size_t iShell = 0; iShell < nShell ; iShell++) {
310 size_t bfst = basis.mapSh2Bf[iShell];
311 size_t sz = basis.shells[iShell].size();
312 size_t L = basis.shells[iShell].contr[0].l;
313 if (L > maxL) maxL = L;
314 size_t curCen = basis.mapSh2Cen[iShell];
315 bool newAtm = false;
316 if( curCen != iAtm ) {
317 iAtm++;
318 if( !groupAtm or (groupAtm and (atmSymb != getAtmSymb(iAtm))) ) {
319 newAtm = true;
320 atmSymb = getAtmSymb(iAtm);
321 for(size_t i = 0; i < end; i++)
322 std::fill(angWeight[i].begin(), angWeight[i].end(), 0.0);
323 }
324 }
325 bool lastAtm = iAtm == (mol.atoms.size() - 1);
326 bool firstAtm = iShell == 0;
327 bool nextAtmNew;
328
329 for(size_t mu = bfst; mu < bfst + sz; mu++) {
330 for(auto q = p; q < p + end; q++) {
331 angWeight[q-p][L] += SmartConj(MO[mu + q*LDM]) * SCR[mu + q*NB];
332 if (MO2)
333 angWeight[q-p][L] += SmartConj(MO2[mu + q*LDM]) * SCR2[mu + q*NB];
334
335 }
336 nextAtmNew = lastAtm ? false : (mu+1) >= basis.mapCen2BfSt[iAtm+1];
337 if (groupAtm and (atmSymb == getAtmSymb(iAtm+1))) nextAtmNew = false;
338 }
339
340 if( (newAtm or firstAtm) ) {
341 out << std::setw(6) << " ";
342 out << std::setw(3) << std::left << iAtm;
343 out << std::setw(7) << std::left << atmSymb;
344 out << std::endl;
345 }
346 if( nextAtmNew or (iShell + 1 == nShell) ) {
347 for (size_t i = 0; i <= maxL; i++) {
348 out << " " << std::left << std::setw(eValOff-1) << angLabel[i];
349 out << std::right;
350 for (auto q = p; q < p+end; q++) {
351 double VAL = std::real(angWeight[q-p][i]);
352 std::cout << std::fixed << std::right<< std::setprecision(4);
353 std::cout.fill(' ');
354
355 if(std::abs(VAL) >= 1e-4) out << std::setw(printWidth) << VAL;
356 else if(std::isnan(std::abs(VAL))) out << std::setw(printWidth) << "NAN";
357 else if(std::isinf(std::abs(VAL))) out << std::setw(printWidth) << "INF";
358 else out << std::setw(printWidth) << "-";
359 }
360 out << std::endl;
361 }
362 }
363 }
364 out << std::endl;
365 }
366
367 mem.free(SCR);
368 out << bannerEnd << std::endl;
369
370 } // analyzeMOPrint

373 template <typename MatsT, typename IntsT>
374 void WaveFunction<MatsT,IntsT>::printMOInfo(std::ostream &out, size_t printLevel) {
375
376 size_t NB = this->nAlphaOrbital();
377 size_t NOrb = NB * this->nC;
378
379 bool MOcoeffs = printLevel % 2 == 1;
380 bool analyzeMO = printLevel / 2 > 0;
381
382 if (MOcoeffs) {
383
384 printLevel -= 1;
385
386 // Pretty MO print
387 std::function<double(MatsT)> printOp1 = [](MatsT x) { return std::real(x); };
388 std::function<double(MatsT)> printOp2 = printOp1;
389
390 if( std::is_same<MatsT,dcomplex>::value ) {
391 printOp1 = [](MatsT x) { return std::abs(x); };
392 printOp2 = [](MatsT x) { return std::arg(x); };
393 }
394
395 if( this->nC >= 2 )
396 out << " *** NOTICE: Alpha and Beta Coefficients refer to the SAME "
397 << "Canonical MOs ***\n";
398
399 out << "\n\nCanonical Molecular Orbital Coefficients (Alpha)";
400 if( std::is_same<MatsT,dcomplex>::value )
401 out << " Magnitude";
402 if( this->nC == 4 ) out << " for Large component";
403
404 prettyMOPrint(out,NB,NOrb,this->eps1,this->mo[0].pointer(),NOrb,
405 molecule(),basisSet(),printOp1);
406
407 if( std::is_same<MatsT,dcomplex>::value ) {
408 out << "\n\nCanonical Molecular Orbital Coefficients (Alpha) Phase";
409 if( this->nC == 4 ) out << " for Large component";
410 prettyMOPrint(out,NB,NOrb,this->eps1,this->mo[0].pointer(),NOrb,
411 molecule(),basisSet(),printOp2);
412 }
413
414 if( this->nC >= 2 or not this->iCS ) {
415
416 out << "\n\nCanonical Molecular Orbital Coefficients (Beta)";
417 if( std::is_same<MatsT,dcomplex>::value )
418 out << " Magnitude";
419 if( this->nC == 4 ) out << " for Large component";
420
421 if( this->nC == 1 )
422 prettyMOPrint(out,NB,NOrb,this->eps2,this->mo[1].pointer(),NOrb,
423 molecule(),basisSet(),printOp1);
424 else
425 prettyMOPrint(out,NB,NOrb,this->eps1,this->mo[0].pointer() + (this->nC/2)*NB,
426 NOrb,molecule(),basisSet(),printOp1);
427
428 if( std::is_same<MatsT,dcomplex>::value ) {
429 out << "\n\nCanonical Molecular Orbital Coefficients (Beta) Phase";
430 if( this->nC == 4 ) out << " for Large component";
431
432 if( this->nC == 1 )
433 prettyMOPrint(out,NB,NOrb,this->eps2,this->mo[1].pointer(),NOrb,
434 molecule(),basisSet(),printOp2);
435 else
436 prettyMOPrint(out,NB,NOrb,this->eps1,this->mo[0].pointer() + (this->nC/2)*NB,
437 NOrb,molecule(),basisSet(),printOp2);
438 }
439 }
440
441 if( this->nC == 4 ) {
442
443 out << "\n\nCanonical Molecular Orbital Coefficients (Alpha)";
444 if( std::is_same<MatsT,dcomplex>::value )
445 out << " Magnitude";
446 out << " for Small component";
447 prettyMOPrint(out,NB,NOrb,this->eps1,this->mo[0].pointer() + NB,NOrb,
448 molecule(),basisSet(),printOp1);
449
450 if( std::is_same<MatsT,dcomplex>::value ) {
451 out << "\n\nCanonical Molecular Orbital Coefficients (Alpha) Phase for Small component";
452 prettyMOPrint(out,NB,NOrb,this->eps1,this->mo[0].pointer() + NB,NOrb,
453 molecule(),basisSet(),printOp2);
454 }
455
456 out << "\n\nCanonical Molecular Orbital Coefficients (Beta)";
457 if( std::is_same<MatsT,dcomplex>::value )
458 out << " Magnitude";
459 out << " for Small component";
460 prettyMOPrint(out,NB,NOrb,this->eps1,this->mo[0].pointer() + 3*NB,NOrb,
461 molecule(),basisSet(),printOp1);
462
463 if( std::is_same<MatsT,dcomplex>::value ) {
464 out << "\n\nCanonical Molecular Orbital Coefficients (Beta) Phase for Small component";
465 prettyMOPrint(out,NB,NOrb,this->eps1,this->mo[0].pointer() + 3*NB,NOrb,
466 molecule(),basisSet(),printOp2);
467 }
468
469 }
470 out << "\n" << BannerEnd << "\n\n";
471
472 }
473
474 if ( analyzeMO ) {
475
476 // Analysis angular momentum components of MOs
477 bool groupAtm = (printLevel/2) % 2 == 0;
478 bool groupAB = printLevel/2 >= 3;
479
480 MatsT * MO2 = nullptr;
481 if( groupAB ) {
482 out << "\n\nCanonical Molecular Orbital based Mulliken Population Analysis (Alpha + Beta)";
483 if (this->nC == 1 and not this->iCS) MO2 = this->mo[1].pointer();
484 else MO2 = this->mo[0].pointer() + (this->nC/2)*NB;
485 }
486 else if( this->nC >= 2 ) {
487 out << "\n *** NOTICE: Alpha and Beta Analysis refer to the SAME "
488 << "Canonical MOs ***";
489 out << "\n\nCanonical Molecular Orbital based Mulliken Population Analysis (Alpha)";
490 }
491 if( this->nC == 4 ) out << " for Large component";
492 analyzeMOPrint(out, NB, NOrb, aoints.overlap->pointer(), this->mo[0].pointer(),
493 NOrb, molecule(), basisSet(), this->memManager, groupAtm, MO2);
494
495 if( not groupAB and (this->nC >= 2 or not this->iCS) ) {
496 out << "\n\nCanonical Molecular Orbital based Mulliken Population Analysis (Beta)";
497 if( this->nC == 4 ) out << " for Large component";
498 if( this->nC == 1 )
499 analyzeMOPrint(out, NB, NOrb, aoints.overlap->pointer(),
500 this->mo[1].pointer(), NOrb, molecule(), basisSet(), this->memManager, groupAtm);

502 analyzeMOPrint(out, NB, NOrb, aoints.overlap->pointer(),
503 this->mo[0].pointer() + (this->nC/2)*NB, NOrb, molecule(), basisSet(),
504 this->memManager, groupAtm);
505
506 }
507
508 if( this->nC == 4 ) {
509 IntsT* ssOverlap = this->memManager.template malloc<IntsT>(NB*NB);
510 SetMat('N',NB,NB,1./(2*SpeedOfLight*SpeedOfLight),this->aoints.kinetic->pointer(),
511 NB,ssOverlap,NB);

513 analyzeMOPrint(out, NB, NOrb, ssOverlap, this->mo[0].pointer() + NB,
514 NOrb, molecule(), basisSet(), this->memManager, groupAtm, MO2);
515 if( not groupAB ) {
516 out << "\n\nCanonical Molecular Orbital based Mulliken Population Analysis (Beta) for Small component";
517 analyzeMOPrint(out, NB, NOrb, ssOverlap, this->mo[0].pointer() + 3*NB,
518 NOrb, molecule(), basisSet(), this->memManager, groupAtm);
519 }
520 this->memManager.free(ssOverlap);
521
522 }
523
524
525 out << "\n" << BannerEnd << "\n\n";
526
527 }
