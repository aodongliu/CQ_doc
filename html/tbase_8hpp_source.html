236 inline virtual void run() {
237
238 // Synchronize MPI processes
239 MPI_Barrier(comm_);
240
241 if( alreadyRan_ ) CErr("RESPONSE module only designed to be run once!");
242
243 ProgramTimer::tick("Response Total");
244
245 genSettings.matIsHer = genSettings.doTDA; // TDA always implies hermetian matrix
246 configOptions();
247
248 bool rootHasFullMat = (MPIRank(comm_) == 0) and fullMatrix_;
249#ifdef CQ_ENABLE_MPI
250 MPIBCast(&rootHasFullMat,1,0,comm_);
251#endif
252
253 writeMeta();
254
255 if( (genSettings.doFull or genSettings.formFullMat) and
256 not rootHasFullMat ) formFullMatrix();
257
258
259 allocResults();
260
261 bool isRoot = MPIRank(comm_) == 0;
262
263 if( isRoot and genSettings.printLevel >= 0 )
264 std::cout << " ** STARTING RESPONSE CALCULATION\n\n";
265
266 auto topResp = tick();
267
268 if( genSettings.jobType == RESIDUE ) runResidue();
269 else runFDR();
270
271 double botResp = tock(topResp);
272
273 if( isRoot and genSettings.printLevel >= 0 )
274 std::cout << "\n\n ** RESPONSE CALCULATION FINISHED ("
275 << std::fixed << botResp << " s)\n";
276
277
278 // Synchronize MPI processes
279 MPI_Barrier(comm_);
280 alreadyRan_ = true;
281
282 ProgramTimer::tock("Response Total");
283
284 };

319 virtual void postLinearSolve(){
320
321
322 ROOT_ONLY(comm_);
323
324 bool noDamp = (fdrSettings.dampFactor == 0.) and
325 not fdrSettings.forceDamp;
326
327 size_t N = nSingleDim_;
328 size_t nOmega = fdrSettings.bFreq.size();
329 size_t nRHS = fdrSettings.nRHS;
330
331 // Write Solutions to disk
332 if( savFile.exists() ) {
333
334 savFile.safeWriteData("/RESP/FDR/OMEGA",&fdrSettings.bFreq[0],
335 {nOmega});
336
337
338 savFile.safeWriteData("/RESP/FDR/DAMP",&fdrSettings.dampFactor,
339 {1});
340
341
342 if( noDamp ) {
343
344 savFile.safeWriteData("/RESP/FDR/RHS",fdrResults.RHS, {nRHS,N});
345
346 savFile.safeWriteData("/RESP/FDR/SOLUTION",fdrResults.SOL,
347 {nRHS*nOmega,N});
348
349 } else {
350
351 savFile.safeWriteData("/RESP/FDR/RHS",dfdrResults.RHS, {nRHS,N});
352
353 savFile.safeWriteData("/RESP/FDR/SOLUTION",dfdrResults.SOL,
354 {nRHS*nOmega,N});
355 }
356
357 };

582 template <typename T>
583 template <typename U>
584 void ResponseTBase<T>::iterLinearTrans(size_t nVec, SolverVectors<U> &V, SolverVectors<U> &AV) {
585
586 std::vector<RESPONSE_CONTRACTION<U>> cList(1);
587 cList.back().nVec = nVec;
588 cList.back().N = nSingleDim_;
589 cList.back().X = V.getPtr();
590 cList.back().AX = AV.getPtr();
591
592#ifdef CQ_ENABLE_MPI
593 CB_INT MLoc, NLoc;
594 if( genSettings.isDist() ) {
595 std::tie(MLoc,NLoc) = fullMatGrid_->getLocalDims(nSingleDim_,nVec);
596 if( MLoc and NLoc ) {
597
598 cList.back().X = this->memManager_.template malloc<U>(MLoc*NLoc);
599 cList.back().AX = this->memManager_.template malloc<U>(MLoc*NLoc);
600
601 } else {
602
603 cList.back().X = nullptr;
604 cList.back().AX = nullptr;
605
606 }
607
608 cList.back().DescX = fullMatGrid_->descInit(nSingleDim_,nVec,0,0,MLoc);
609 cList.back().DescAX = cList.back().DescX;
610
611 fullMatGrid_->Scatter(nSingleDim_,nVec,V.getPtr(),nSingleDim_,cList.back().X,
612 MLoc,0,0);
613 }
614#endif
615
616 this->formLinearTrans( cList );
617
618#ifdef CQ_ENABLE_MPI
619 if( genSettings.isDist() ) {
620
621 fullMatGrid_->Gather(nSingleDim_,nVec,AV.getPtr(),nSingleDim_,cList.back().AX,
622 MLoc,0,0);
623
624 if( cList.back().X ) this->memManager_.free(cList.back().X );
625 if( cList.back().AX ) this->memManager_.free(cList.back().AX);
626
627 }
628#endif
629
630 }
