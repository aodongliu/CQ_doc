117 template <typename T>
118 void ResponseTBase<T>::runIterResidue() {
119
120 if((not genSettings.isDist()) and genSettings.formFullMat)
121 ROOT_ONLY(comm_);
122
123 bool isRoot = MPIRank(comm_) == 0;
124 bool isDist = this->genSettings.isDist();
125
126
127 typename GPLHR<T>::LinearTrans_t lt = [&](size_t nVec, SolverVectors<T> &V, SolverVectors<T> &AV) {
128
129 iterLinearTrans(nVec,V,AV);
130
131 };
132
133 ProgramTimer::tick("Iter Diagonalize");
134
135
136 typename GPLHR<T>::Shift_t pc =
137 bool(PC_) ? PC_ :
138 [&](size_t nVec, T shift, SolverVectors<T> &V, SolverVectors<T> &AV) {
139
140 //if( not this->fullMatrix_ ) CErr();
141
142 AV.set_data(0, nVec, V, 0);
143 };
144
145
146 MPI_Comm gplhrComm = (isDist or not genSettings.formFullMat)
147 ? comm_ : rcomm_;
148
149 GPLHR<T> gplhr(gplhrComm,this->memManager_,nSingleDim_,
150 genSettings.maxIter,genSettings.convCrit,
151 resSettings.nRoots,lt,pc);
152
153 gplhr.setM(resSettings.gplhr_m);
154 gplhr.sigma = resSettings.gplhr_sigma;
155 gplhr.hardLim = resSettings.deMin;
156
157 if( hasResGuess_ )
158 gplhr.setGuess(resSettings.nRoots,
159 [&](size_t nG, SolverVectors<T> & G, size_t LDG){ this->resGuess(nG,G,LDG); });
160
161
162
163 gplhr.run();
164
165 if( isRoot ) {
166
167 // Extract data from GPLHR storage
168 for(auto k = 0; k < resSettings.nRoots; k++)
169 resResults.W[k] = std::real(gplhr.eigVal()[k]);
170
171 std::copy_n(gplhr.VR()->getPtr(),
172 this->nSingleDim_ * resSettings.nRoots,
173 resResults.VR);
174
175 }
176
177 ProgramTimer::tock("Iter Diagonalize");
178
179 };
