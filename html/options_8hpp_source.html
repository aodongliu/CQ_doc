46 // Type of Job
47 enum JobType {
48 SCF,
49 RT,
50 RESP,
51 CC,
52 MR,
53 BOMD,
54 EHRENFEST
55 };
56
57 // Tedious, but there isn't an easier way to do this
58 inline JobType parseJob(std::string jobStr) {
59 JobType job;
60 if( jobStr == "SCF" ) {
61 job = SCF;
62 }
63 else if( jobStr == "RT" ) {
64 job = RT;
65 }
66 else if( jobStr == "RESP" ) {
67 job = RESP;
68 }
69 else if( jobStr == "CC" ) {
70 job = CC;
71 }
72 else if( jobStr == "BOMD" ) {
73 job = BOMD;
74 }
75 else if( jobStr == "EHRENFEST" ) {
76 job = EHRENFEST;
77 }
78 else if( jobStr == "MCSCF" ) {
79 job = MR;
80 }
81 else {
82 jobStr = "Unrecognized job type \"" + jobStr + "\"!";
83 CErr(jobStr);
84 }
85 return job;
86 };
