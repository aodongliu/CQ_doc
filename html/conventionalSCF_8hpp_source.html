31class ConventionalSCF : public OptimizeOrbitals<MatsT> {

33protected:
34 // Orthogonal Fock/Density Matrices

38

43 std::vector<double> diisEnergy;
44 std::shared_ptr<SquareMatrix<double>> diisBMat;
47
48 // Damping Matrices
49 std::vector<SquareMatrix<MatsT>> prevFock;
50 std::vector<SquareMatrix<MatsT>> prevOnePDM;
51
52public:
53 // Constructor
54 ConventionalSCF() = delete;
55 ConventionalSCF(SCFControls sC, MPI_Comm comm, ModifyOrbitalsOptions<MatsT> mod, CQMemManager& mem): OptimizeOrbitals<MatsT>(sC, comm, mod, mem) {
56
57 // Allocate ortho Fock and Den
58 VecShrdPtrMat<MatsT> fock = this->modOrbOpt.getFock();
59 for( auto& f : fock )
60 fockMatrixOrtho.emplace_back(f->memManager(), f->dimension());
61 VecShrdPtrMat<MatsT> den = this->modOrbOpt.getOnePDM();
62 for( auto& d : den )
63 onePDMOrtho.emplace_back(d->memManager(), d->dimension());
64 for( auto& d : den )
65 orbGrad.emplace_back(d->memManager(), d->dimension());
66
67 if( this->scfControls.doExtrap ) allocExtrapStorage();
68 };
69
70 // Destructor
71 ~ConventionalSCF() {
72 diisFock.clear();
73 diisOnePDM.clear();
74 diisError.clear();
75 diisEnergy.clear();
76 prevFock.clear();
77 prevOnePDM.clear();
78 fockMatrixOrtho.clear();
79 onePDMOrtho.clear();
80 diisBMat = nullptr;
81 };
