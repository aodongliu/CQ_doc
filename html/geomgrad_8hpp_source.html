87 void doGrad(size_t acc = 2) {
88
89 FiniteDifferencer<double,double> diff;
90
91 diff.setDerOrder(1);
92 diff.setAccOrder(acc);
93
94 diff.setStepSize(1e-2);
95
96
97 diff.setValFunc(std::bind(&NumGradient::getEnergies, this));
98
99 bool cInt = goodType<dcomplex,dcomplex>();
100 bool cMat = cInt or goodType<dcomplex,double>();
101
102 for ( auto iAtm = 0; iAtm < nAtoms; iAtm++ ) {
103 for ( auto iXYZ = 0; iXYZ < 3; iXYZ++ ) {
104
105 if(cInt)
106 diff.addChangeFunc(
107 std::bind(&NumGradient::moveMol<dcomplex,dcomplex>, this,
108 iAtm, iXYZ, std::placeholders::_1));
109
110 else if(cMat)
111 diff.addChangeFunc(
112 std::bind(&NumGradient::moveMol<dcomplex,double>, this,
113 iAtm, iXYZ, std::placeholders::_1));
114
115 else
116 diff.addChangeFunc(
117 std::bind(&NumGradient::moveMol<double,double>, this,
118 iAtm, iXYZ, std::placeholders::_1));
119
120 }
121 }
122
123 diff.doDifference();
124
125 auto res = diff.getResults();
126
127 printResults("Nuclear Gradient (Eh / Bohr)", res, 0, 1e-12);
128 printResults("OneE Gradient (Eh / Bohr)", res, 1);
129 printResults("TwoE Gradient (Eh / Bohr)", res, 2);
130 printResults("NucRep Gradient (Eh / Bohr)", res, 3);
131 if (res.size() > 4)
132 printResults("XC Gradient (Eh / Bohr)", res, 4);
133
134 };

156 std::bind(&NumGradient::moveMol<dcomplex,dcomplex>, this,
157 iAtm, iXYZ, std::placeholders::_1));
158
159 else if(cMat)
160 diff.addChangeFunc(
161 std::bind(&NumGradient::moveMol<dcomplex,double>, this,
162 iAtm, iXYZ, std::placeholders::_1));
163
164 else
165 diff.addChangeFunc(
166 std::bind(&NumGradient::moveMol<double,double>, this,
167 iAtm, iXYZ, std::placeholders::_1));
168
169 }
170 }
171
172 diff.doDifference();
173
174 auto res = diff.getResults();
175
176 moveMol<double,double>(0, 0, 0.);
177
178 for ( auto iAtom = 0; iAtom < nAtoms; iAtom++ ) {
179 for ( auto iXYZ = 0; iXYZ < 3; iXYZ++ ) {
180 std::cout << "=== Weight Gradients [" << iAtom << "," << iXYZ << "]\n";
181
182 auto anaRes = getGridVals(true, iAtom, iXYZ);
183
184 bool allgood = true;
185
186 for ( auto i = 0; i < anaRes.size(); i++ ) {
187 if ( std::max( res[iAtom*3 + iXYZ][i], anaRes[i] ) < 1e-12 )
188 continue;
189 if ( std::abs( res[iAtom*3 + iXYZ][i] - anaRes[i] ) / anaRes[i] > 1e-8 ) {
190 std::cout << " " << res[iAtom*3 + iXYZ][i];
191 std::cout << " " << anaRes[i];
192 std::cout << " " << (res[iAtom*3 + iXYZ][i] - anaRes[i])/anaRes[i];
193 std::cout << "\n";
194 allgood = false;
195 }
196 }
197
198 if (allgood)
199 std::cout << "All good\n";
200 std::cout << std::endl;
201 }
202 }
