39template<typename MatsT>
40void Orthogonalization<MatsT>::computeOrtho() {
41 if( not overlap ) CErr("Overlap has not been initialized");
42
43 size_t NB = overlap->dimension();
44 size_t nSQ = NB * NB;
45
46 forwardTrans->clear();
47 backwardTrans->clear();
48
49 // Allocate scratch
50 MatsT* SCR1 = overlap->memManager().template malloc<MatsT>(nSQ);
51 std::fill_n(SCR1, nSQ, 0.);
52
53 if( orthoType == LOWDIN ) {
54
55 // Allocate more scratch
56 MatsT* sE = overlap->memManager().template malloc<MatsT>(NB);
57 MatsT* SCR2 = overlap->memManager().template malloc<MatsT>(nSQ);
58
59 // Diagonalize the overlap in scratch S = V * s * V**T
60 std::copy_n(overlap->pointer(),nSQ,SCR1);
61 HermetianEigen('V', 'U', NB, SCR1, NB, sE, overlap->memManager());
62
63 if( std::abs(sE[0]) < 1e-10 ) CErr("Contracted Basis Set is Linearly Dependent!");
64
65 // Compute X = V * s^{-1/2}
66 for( auto j = 0; j < NB; j++ )
67 for( auto i = 0; i < NB; i++ )
68 SCR2[i + j * NB] = SCR1[i + j * NB] / std::sqrt(sE[j]);
69
70 // Compute O1 = X * V**T
71 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans, blas::Op::ConjTrans, NB, NB, NB, MatsT(1.), SCR2, NB, SCR1, NB, MatsT(0.), forwardTrans->pointer(), NB);
72
73 // Compute X = V * s^{1/2} in place (by multiplying by s)
74 for( auto j = 0; j < NB; j++ )
75 for( auto i = 0; i < NB; i++ )
76 SCR2[i + j * NB] = SCR2[i + j * NB] * sE[j];
77
78 // Compute O2 = X * V**T
79 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::ConjTrans, NB, NB, NB, MatsT(1.), SCR2, NB, SCR1, NB, MatsT(0.), backwardTrans->pointer(), NB);
80
81#ifdef _DEBUGORTHO
82 // Debug code to validate the Lowdin orthogonalization
83 prettyPrintSmart(std::cout,"Overlap", overlap->pointer(),NB,NB,NB);
84 prettyPrintSmart(std::cout,"Operator Nonortho->Ortho Transformation", forwardTrans->pointer(),NB,NB,NB);
85 prettyPrintSmart(std::cout,"Operator Ortho->Nonortho Transformation", backwardTrans->pointer(),NB,NB,NB);
86
87 std::cerr << "Debugging Lowdin Orthogonalization" << std::endl;
88 double maxDiff(-10000000);
89
90 // Check that ortho1 and ortho2 are inverses of eachother
91 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans, NB, NB, NB, MatsT(1.), forwardTrans->pointer(), NB, backwardTrans->pointer(), NB, MatsT(0.), SCR1, NB);
92
93 for( auto j = 0; j < NB; j++ )
94 for( auto i = 0; i < NB; i++ ) {
95
96 if( i == j )
97 maxDiff = std::max(maxDiff, std::abs(1. - SCR1[i + j * NB]));
98 else
99 maxDiff = std::max(maxDiff, std::abs(SCR1[i + j * NB]));
100 }
101
102 std::cerr << " Ortho1 * Ortho2 = I: " << maxDiff << std::endl;
103
104 // Check that ortho2 * ortho2 is the overlap
105 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans, NB, NB, NB, MatsT(1.), backwardTrans->pointer(), NB, backwardTrans->pointer(), NB, MatsT(0.), SCR1, NB);
106
107 maxDiff = -100000;
108
109 for( auto j = 0; j < NB; j++ )
110 for( auto i = 0; i < NB; i++ ) {
111
112 maxDiff = std::max(maxDiff, std::abs(SCR1[i + j * NB] - overlap->pointer()[i + j * NB]));
113 }
114
115 std::cerr << " Ortho2 * Ortho2 = S: " << maxDiff << std::endl;
116
117 // Check that ortho1 * ortho1 is the inverse of the overlap
118 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans, NB, NB, NB, MatsT(1.), forwardTrans->pointer(), NB, forwardTrans->pointer(), NB, MatsT(0.), SCR1, NB);
119 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans, NB, NB, NB, MatsT(1.), SCR1, NB, overlap->pointer(), NB, MatsT(0.), SCR2, NB);
120
121 maxDiff = -10000;
122 for( auto j = 0; j < NB; j++ )
123 for( auto i = 0; i < NB; i++ ) {
124
125 if( i == j )
126 maxDiff = std::max(maxDiff, std::abs(1. - SCR2[i + j * NB]));
127 else
128 maxDiff = std::max(maxDiff, std::abs(SCR2[i + j * NB]));
129 }
130
131 std::cerr << " Ortho1 * Ortho1 * S = I: " << maxDiff << std::endl;
132
133#endif
134
135 // Free Scratch Space
136 overlap->memManager().free(sE, SCR2);
137
138 } else if( orthoType == CHOLESKY ) {
139
140 std::cout << "*** WARNING: Cholesky orthogonalization has not yet been confirmed ***" << std::endl;
141 CErr("Cholesky orthogonalization should be checked thoroughly before using it");
142
143 // Compute the Cholesky factorization of the overlap S = L * L**T
144 lapack::potrf(lapack::Uplo::Lower,NB,SCR1,NB);
145
146 // Copy the lower triangle to ortho2 (O2 = L)
147 for( auto j = 0; j < NB; j++ )
148 for( auto i = j; i < NB; i++ )
149 backwardTrans->pointer()[i + j * NB] = SCR1[i + j * NB];
150
151 // Compute the inverse of the overlap using the Cholesky factors
152 lapack::potri(lapack::Uplo::Lower,NB,SCR1,NB);
153
154 // O1 = O2**T * S^{-1}
155 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans, NB, NB, NB, MatsT(1.), backwardTrans->pointer(), NB, SCR1, NB, MatsT(0.), forwardTrans->pointer(), NB);
156
157 // Remove upper triangle junk from O1
158 for( auto j = 0; j < NB; j++ )
159 for( auto i = 0; i < j; i++ )
160 forwardTrans->pointer()[i + j * NB] = 0.;
161
162#ifdef _DEBUGORTHO
163 // Debug code to validate the Lowdin orthogonalization
164
165 std::cerr << "Debugging Cholesky Orthogonalization" << std::endl;
166
167 // Debug code to validate the Cholesky orthogonalization
168 MatsT* SCR2 = overlap->memManager().template malloc<MatsT>(nSQ);
169
170 double maxDiff = -1000;
171 blas::gemm(blas::Layout::ColMajor, blas::Op::Trans, blas::Op::NoTrans, NB, NB, NB, MatsT(1.), forwardTrans->pointer(), NB, overlap->pointer(), NB, MatsT(0.), SCR1, NB);
172 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::NoTrans, NB, NB, NB, MatsT(1.), SCR1, NB, forwardTrans->pointer(), NB, MatsT(0.), SCR2, NB);
173
174 for( auto j = 0; j < NB; j++ )
175 for( auto i = 0; i < NB; i++ ) {
176
177 if( i == j )
178 maxDiff = std::max(maxDiff, std::abs(1. - SCR2[i + j * NB]));
179 else
180 maxDiff = std::max(maxDiff, std::abs(SCR2[i + j * NB]));
181 }
182
183 std::cerr << "Ortho1**T * S ** Ortho1 = I: " << maxDiff << std::endl;
184
185 overlap->memManager().free(SCR2); // Free SCR2
186#endif
187 }

338 template <typename MatsT>
339 void Orthogonalization<MatsT> :: getOrthogonalizationGradients(
340 std::vector<SquareMatrix<MatsT>>& gradOrtho,
341 std::vector<SquareMatrix<MatsT>>& operators)
342 {
343
344 if( operators.size() == 0 )
345 CErr("No operators to transform in getOrthogonalizationGradients");
346 if( operators.size() != gradOrtho.size() )
347 CErr("Different number of operators and output matrices in getOrthogonalizationGradients");
348
349 size_t NB = operators[0].dimension();
350 size_t nSQ = NB*NB;
351 CQMemManager& memManager = overlap->memManager();
352
353 if( orthoType == LOWDIN ) {
354 MatsT* sVecs = memManager.malloc<MatsT>(nSQ);
355 MatsT* sE = memManager.malloc<MatsT>(NB);
356 MatsT* weights = memManager.malloc<MatsT>(nSQ);
357 MatsT* SCR1 = memManager.malloc<MatsT>(nSQ);
358 MatsT* SCR2 = memManager.malloc<MatsT>(nSQ);
359
360 // Diagonalize the overlap in scratch S = V * s * V**T
361 std::copy_n(overlap->pointer(),nSQ,sVecs);
362 HermetianEigen('V','U',NB,sVecs,NB,sE,memManager);
363
364 if( std::abs( sE[0] ) < 1e-10 )
365 CErr("Contracted Basis Set is Linearly Dependent!");
366
367
368 // Compute weights = (sqrt(si) + sqrt(sj))^-1
369 for( auto i = 0; i < NB; i++ )
370 for( auto j = 0; j < NB; j++ ) {
371 weights[i*NB + j] = MatsT(1.) / (std::sqrt(sE[i]) + std::sqrt(sE[j]));
372 }
373
374 // Loop over gradient components
375 for( auto iGrad = 0; iGrad < operators.size(); iGrad++ ) {
376
377 //
378 // dV/dR = V . (weights x V**T . O . V) . V**T
379 //
380
381 // O . V
382 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,
383 NB,NB,NB,MatsT(1.),operators[iGrad].pointer(),NB,sVecs,NB,MatsT(0.),SCR1,NB);
384 // V**T . O . V
385 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans,blas::Op::NoTrans,
386 NB,NB,NB,MatsT(1.),sVecs,NB,SCR1,NB,MatsT(0.),SCR2,NB);
387 // weights x V**T . O . V
388 std::transform(SCR2, SCR2+nSQ, weights, SCR2, std::multiplies<>());
389 // (weights x V**T . O . V) . V**T
390 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::ConjTrans,
391 NB,NB,NB,MatsT(1.),SCR2,NB,sVecs,NB,MatsT(0.),SCR1,NB);
392 // dV/dR = V . (weights x V**T . O . V) . V**T
393 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans,blas::Op::NoTrans,
394 NB,NB,NB,MatsT(1.),sVecs,NB,SCR1,NB,MatsT(0.),gradOrtho[iGrad].pointer(),NB);
395
396 }
397
398 memManager.free(sVecs, sE, weights, SCR1, SCR2);
399
400 }
401 else if( orthoType == CHOLESKY ) {
402 CErr("Cholesky orthogonalization gradients not yet implemented");
403 }
404
405 }
