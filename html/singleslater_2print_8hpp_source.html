113
114 out << "\nCharge Analysis:\n" << bannerTop << "\n\n";
115
116 out << std::setw(20) << std::left << " Atom";
117 out << std::setw(20) << std::right << "Mulliken Charges";
118 out << std::setw(20) << std::right << "Lowdin Charges" << std::endl;
119
120 out << std::right << bannerMid << std::endl;
121
122 Molecule & mol = this->molecule();
123 for(auto iAtm = 0; iAtm < mol.nAtoms; iAtm++) {
124
125 // Get symbol
126 std::map<std::string,Atom>::const_iterator it =
127 std::find_if(atomicReference.begin(),atomicReference.end(),
128 [&](const std::pair<std::string,Atom> &st){
129 return (st.second.atomicNumber == mol.atoms[iAtm].atomicNumber) and
130 (st.second.massNumber == mol.atoms[iAtm].massNumber);}
131 );
132
133 out << " " << std::setw(18) << std::left <<
134 (it == atomicReference.end() ? "X" : it->first);
135
136 out << std::setprecision(5) << std::right;
137
138 out << std::setw(20) << mullikenCharges[iAtm];
139 out << std::setw(20) << lowdinCharges[iAtm];
140
141 out << std::endl;
142 }

144 out << std::endl << bannerEnd << std::endl;

153 template <typename MatsT, typename IntsT>
154 void SingleSlater<MatsT,IntsT>::printEPS(std::ostream &out) {
155
156 // List MO eigenenergies
157
158 out << std::scientific << std::setprecision(4);
159 out << "Orbital Eigenenergies " << (this->nC == 1 ? "(Alpha) " : "" )
160 << "/ Eh\n" << bannerTop << "\n";
161
162 // Set number of occupied orbitals
163 size_t NO = (this->nC == 1 ? this->nOA : this->nO);
164 if ( this->nC == 4 ) NO = this->nO + this->nC * this->nAlphaOrbital()/2;
165
166 //for(auto i = 0ul; i < this->nC * this->nAlphaOrbital(); i++) {
167
168 // if( i == 0 )
169 // out << "Occupied:\n";
170 // else if( i == NO )
171 // out << "\n\nVirtual:\n";
172
173 // out << std::setw(13) << this->eps1[i];
174
175 // if( i < NO and (i + 1) % 5 == 0 ) out << "\n";
176 // else if( i >= NO and ((i - NO) + 1) % 5 == 0 ) out << "\n";
177 //}
178
179 if( nC != 4 ) {
180 for(auto i = 0ul; i < this->nC * this->nAlphaOrbital(); i++) {
181
182 if( i == 0 )
183 out << "Occupied:\n";
184 else if( i == NO )
185 out << "\n\nVirtual:\n";
186
187 out << std::setw(13) << this->eps1[i];
188
189 if( i < NO and (i + 1) % 5 == 0 ) out << "\n";
190 else if( i >= NO and ((i - NO) + 1) % 5 == 0 ) out << "\n";
191 }
192 } else {
193 for(auto i = this->nC * this->nAlphaOrbital()/2; i < this->nC * this->nAlphaOrbital(); i++) {
194
195 if( i == this->nC * this->nAlphaOrbital()/2 )
196 out << "Occupied:\n";
197 else if( i == NO )
198 out << "\n\nVirtual:\n";
199
200 out << std::setw(13) << this->eps1[i];
201
202 if( i < NO and (i + 1 - this->nC * this->nAlphaOrbital()/2) % 5 == 0 ) out << "\n";
203 else if( i >= NO and ((i - NO) + 1) % 5 == 0 ) out << "\n";
204 }
205
206 // Negative Energy States
207 for(auto i = 0ul; i < this->nC * this->nAlphaOrbital()/2; i++) {
208
209 if( i == 0 )
210 out << "\n\nNegative Energy States:\n";
211
212 out << std::setw(13) << this->eps1[i];
213
214 if( i < NO and (i + 1) % 5 == 0 ) out << "\n";
215 else if( i >= NO and ((i - NO) + 1) % 5 == 0 ) out << "\n";
216 }
217 }
218
219 out << "\n" << bannerEnd << "\n";
220
221 if( this->nC == 1 and not this->iCS ) {
222 out << "\n\nOrbital Eigenenergies (Beta) / Eh\n" << bannerTop << "\n";
223
224 for(auto i = 0ul; i < this->nBetaOrbital(); i++) {
225
226 if( i == 0 )
227 out << "Occupied:\n";
228 if( i == this->nOB )
229 out << "\n\nVirtual:\n";
230
231 out << std::setw(13) << this->eps2[i];
232
233 if( i < this->nOB and (i + 1) % 5 == 0 ) out << "\n";
234 else if( i >= this->nOB and ((i - this->nOB) + 1) % 5 == 0 ) out << "\n";
235 }
236
237 out << "\n" << bannerEnd << "\n";
238 }
239
240 } // SingleSlater::printEPS
