51 void orderCheck(size_t order) const {
52 if (order > highOrder())
53 CErr("Order of XYZ components exceeds highest order in this MultipoleInts.");
54 if (order <= 0)
55 CErr("No zeroth-order components in MultipoleInts.");
56 }
57
58 size_t cumeComponents(size_t order) const {
59 size_t count = 0;
60 for (size_t i = 1; i <= order; i++)
61 count += VectorInts<IntsT>::nComponentsOnOrder(i, symmetric());
62 return count;
63 }
64
65 std::pair<size_t, size_t> index(std::vector<size_t> comps) const {
66 size_t order = comps.size();
67 orderCheck(order--);
68 return std::make_pair(order, components_[order].index(comps));
69 }
70
71 std::pair<size_t, size_t> index(size_t i) const {
72 size_t order = 1, orderNComp = 3;
73 while (i >= orderNComp) {
74 i -= orderNComp;
75 orderNComp = VectorInts<IntsT>::nComponentsOnOrder(++order, symmetric());
76 }
77 return std::make_pair(order-1, i);
78 }
79
80 std::string label(size_t i) const {
81 std::pair<size_t, size_t> indices = index(i);
82 return VectorInts<IntsT>::indexToLabel(
83 indices.second, indices.first+1, symmetric());
84 }

200 virtual void output(std::ostream &out, const std::string &s = "",
201 bool printFull = false) const {
202 if (printFull) {
203 std::string oeiStr;
204 if (s == "")
205 oeiStr = "MultipoleInts.";
206 else
207 oeiStr = "MultipoleInts[" + s + "].";
208 for (size_t i = 0; i < size(); i++) {
209 std::string label_i = label(i);
210 prettyPrintSmart(out, oeiStr+label_i, operator[](label_i).pointer(),
211 this->nBasis(), this->nBasis(), this->nBasis());
212 }
213 } else {
214 std::string oeiStr;
215 if (s == "")
216 oeiStr = "Multipole integral";
217 else
218 oeiStr = "MultipoleInts[" + s + "]";
219 out << oeiStr << " up to ";
220 switch (highOrder()) {
221 case 1:
222 out << "Dipole";
223 break;
224 case 2:
225 out << "Quadrupole";
226 break;
227 case 3:
228 out << "Octupole";
229 break;
230 default:
231 out << highOrder() << "-th order";
232 break;
233 }
234 out << std::endl;
235 }

238 virtual void broadcast(MPI_Comm comm = MPI_COMM_WORLD, int root = 0) override {
239 ParticleIntegrals::broadcast(comm, root);
240
241#ifdef CQ_ENABLE_MPI
242 if( MPISize(comm) > 1 ) {
243 size_t highOrder_bcast = highOrder_;
244 bool symmetric_bcast = symmetric_;
245 MPIBCast(highOrder_bcast,root,comm);
246 MPIBCast(symmetric_bcast,root,comm);
247
248 if (highOrder_bcast != highOrder_ or symmetric_bcast != symmetric_) {
249 highOrder_ = highOrder_bcast;
250 symmetric_ = symmetric_bcast;
251 components_.clear();
252 components_.reserve(highOrder_);
253 for (size_t i = 1; i <= highOrder_; i++) {
254 components_.emplace_back(memManager_, NB, i, symmetric_);
255 }
256 }
257
258 for (VectorInts<IntsT>& comp : components_)
259 comp.broadcast(comm, root);
260 }
261#endif
262 }

264 template <typename IntsU>
265 MultipoleInts<IntsU> spatialToSpinBlock() const {
266 MultipoleInts<IntsU> spinBlockInts(memManager_, NB * 2, highOrder_, symmetric_);
267 size_t size = components_.size();
268 for (size_t i = 0; i < size; i++) {
269 spinBlockInts.components_[i] = components_[i].template spatialToSpinBlock<IntsU>();
270 }
271 return spinBlockInts;
272 }

280 MultipoleInts<typename std::conditional<
