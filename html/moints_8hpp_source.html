47 template <typename MatsT, typename IntsT>
48 void MCWaveFunction<MatsT,IntsT>::transformInts(EMPerturbation & pert) {
49
50 // TODO: expand to RI
51 // build object and allocate memory
52 size_t nTOrb = this->MOPartition.nMO;
53 size_t nCorrO = this->MOPartition.nCorrO;
54 size_t nInact = this->MOPartition.nInact;
55 size_t nInact2 = nInact * nInact;
56
57 auto & mem = this->memManager;
58
59 /*
60 * compute inactive core energy
61 */
62 MatsT * h1e_ii = mem.template malloc<MatsT>(nInact);
63 MatsT * GDjj_ii = mem.template malloc<MatsT>(nInact);
64
65 double fc1C = (this->reference().nC == 1) ? 2.0 : 1.0;
66
67 mointsTF->transformHCore(pert, h1e_ii, "ii", true);
68 mointsTF->transformGD(pert, 'i', GDjj_ii, "jj", true, true, "WithInactive-i");
69
70 MatsT ECore = 0.;
71 // compute core enenrgy
72 for (auto i = 0ul; i < nInact; i++) {
73 ECore += h1e_ii[i] + 0.5 * GDjj_ii[i];
74 }
75
76 mem.free(h1e_ii, GDjj_ii);
77
78 this->InactEnergy = std::real(ECore) * fc1C;
79
80 /*
81 * compute hCore and ERI in correlated space
82 */
83 OnePInts<MatsT> hCore_tu(mem, nCorrO);
84 OnePInts<MatsT> hCoreP_tu(mem, nCorrO);
85 InCore4indexTPI<MatsT> ERI_tuvw(mem, nCorrO);
86
87 mointsTF->transformHCore(pert, hCore_tu.pointer(), "tu", false, 'i');
88 mointsTF->transformTPI(pert, ERI_tuvw.pointer(), "tuvw", this->cacheHalfTransTPI_);
89
90 /*
91 * compute hCoreP
92 */
93 hCoreP_tu.clear();
94 if (this->MOPartition.scheme == RAS) { // form hCoreP for RAS case
95 std::cout<<"RAS hcore prime forming starts:"<<std::endl;
96 std::vector<size_t> nActO = this->MOPartition.nActOs;
97 std::vector<std::pair<size_t, size_t>> rasloop;
98 rasloop.push_back(std::make_pair(0, nActO[0]));
99 rasloop.push_back(std::make_pair(nActO[0], nActO[0]+nActO[1]));
100 rasloop.push_back(std::make_pair(nActO[0]+nActO[1], nCorrO));
101
102 for (auto i = 0; i < nCorrO; i++)
103 for (auto j = 0; j < nCorrO; j++)
104 hCoreP_tu(i, j) = hCore_tu(i, j);
105 double symmFc;
106 for (auto pblk = 0; pblk < 3; pblk++)
107 for (auto sblk = 0; sblk < 3; sblk++)
108 for (auto qrblk = 0; qrblk < 3; qrblk++) {
109 if ((pblk + 3*qrblk) < (qrblk + 3*sblk)) symmFc = 1.0;
110 else if ((pblk + 3*qrblk) == (qrblk + 3*sblk)) symmFc = 0.5;
111 else continue;
112
113 for (auto p = rasloop[pblk].first; p < rasloop[pblk].second; p++)
114 for (auto s = rasloop[sblk].first; s < rasloop[sblk].second; s++)
115 for (auto qr = rasloop[qrblk].first; qr < rasloop[qrblk].second; qr++)
116 if (pblk == qrblk or sblk == qrblk)
117 hCoreP_tu(p, s) -= symmFc * ERI_tuvw(p, qr, qr, s);
118 else if (pblk < qrblk)
119 hCoreP_tu(p, s) = hCoreP_tu(p, s) - ERI_tuvw(p, qr, qr, s) + ERI_tuvw(qr, qr, p, s);
120 }
121 } else {
122
123#pragma omp parallel for schedule(static) collapse(2) default(shared)
124 for (auto u = 0ul; u < nCorrO; u++)
125 for (auto t = 0ul; t < nCorrO; t++) {
126
127 MatsT tmp = 0.;
128 for (auto v = 0ul; v < nCorrO; v++)
129 tmp += 0.5 * ERI_tuvw(t, v, v, u);
130
131 hCoreP_tu(t, u) = hCore_tu(t, u) - tmp;
132 }
133 }
134
135 this->moints.addIntegral("hCore_Correlated_Space",
136 std::make_shared<OnePInts<MatsT>>(hCore_tu));
137 this->moints.addIntegral("hCoreP_Correlated_Space",
138 std::make_shared<OnePInts<MatsT>>(hCoreP_tu));
139 this->moints.addIntegral("ERI_Correlated_Space",
140 std::make_shared<InCore4indexTPI<MatsT>>(ERI_tuvw));
141
142 }; // MCWaveFunction::transformInts
