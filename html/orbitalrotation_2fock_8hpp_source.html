36 /*
37 * form generalized Fock matrix 1
38 *
39 * F1_pq = factor * (hCore_pq + \sum_tu 1PDM_{tu} [(pq|tu) - xFC * (pu|tq)])
40 *
41 * for 1C, factor = 2.0, xFC = 0.5
42 * for 2C and 4C, factor = 1.0, xFC = 1.0
43 *
44 */
45 template <typename MatsT, typename IntsT>
46 void OrbitalRotation<MatsT, IntsT>::formGeneralizedFock1(EMPerturbation & pert,
47 SquareMatrix<MatsT> & oneRDM, MatsT * F1, const std::string & moType,
48 bool deltaPQ) {
49
50 // unpack pq_size
51 auto pq_sizes = mcwfn_.mointsTF->parseMOType(moType);
52 size_t np = pq_sizes[0].second;
53 size_t nq = pq_sizes[1].second;
54 size_t npq = np * nq;
55 size_t npqDim = deltaPQ ? np: npq;
56 size_t nCorrO = mcwfn_.MOPartition.nCorrO;
57 size_t nCorrO2 = nCorrO * nCorrO;
58
59 // populate F1 as hCore_pq
60 mcwfn_.mointsTF->transformHCore(pert, F1, moType, deltaPQ, 'i');
61
62 auto & mem = mcwfn_.memManager;
63
64 auto tsize = mcwfn_.mointsTF->parseMOType("t");
65 size_t toff = tsize[0].first;
66 size_t nt = tsize[0].second;
67 size_t nAO = mcwfn_.reference().nAlphaOrbital() * mcwfn_.reference().nC;
68
69 SquareMatrix<MatsT> Den(mem, nAO);
70 MatsT * SCR = mem.template malloc<MatsT>(std::max(nAO*nt, npqDim));
71 MatsT * tMO = mcwfn_.reference().mo[0].pointer() + toff*nAO;
72
73 // form densty D(nu,mu) = C(nu, u) 1PDM_{tu} C(mu, t)^*
74 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::Trans,
75 nAO, nt, nt, MatsT(1.), tMO, nAO, oneRDM.pointer(), nt, MatsT(0.), SCR, nAO);
76
77 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::ConjTrans,
78 nAO, nAO, nt, MatsT(1.), SCR, nAO, tMO, nAO, MatsT(0.), Den.pointer(), nAO);
79
80 // get G(Dtu)_pq
81 mcwfn_.mointsTF->transformGD(pert, Den, true, SCR, moType, deltaPQ, true, "tu");
82
83 // add to F1
84 if (mcwfn_.reference().nC == 1) blas::scal(npqDim, MatsT(2.), F1, 1);
85 blas::axpy(npqDim, MatsT(1.), SCR, 1, F1, 1);
86
87 mem.free(SCR);
88
89 }; // OrbitalRotation<MatsT>::formGeneralizedFock1

91 /*
92 * form generalized Fock matrix 2
93 *
94 * F2_tq = \sum_u 1PDM_tu hCore_qu + \sum_uvw 2PDM_tuvw (qu|vw)
95 *
96 */
97 template <typename MatsT, typename IntsT>
98 void OrbitalRotation<MatsT, IntsT>::formGeneralizedFock2(EMPerturbation & pert,
99 SquareMatrix<MatsT> & oneRDM, InCore4indexTPI<MatsT> & twoRDM, MatsT * F2,
100 const std::string & moType) {
101
102 // unpack q_size
103 auto q_size = mcwfn_.mointsTF->parseMOType(std::string(moType));
104 size_t nq = q_size[0].second;
105 size_t nCorrO = mcwfn_.MOPartition.nCorrO;
106 size_t nCorrO2 = nCorrO * nCorrO;
107 size_t nCorrO3 = nCorrO2 * nCorrO;
108
109 MatsT * SCR = mcwfn_.memManager.template malloc<MatsT>(nq * nCorrO3);
110
111 // populate SCR as hCore_qu
112 mcwfn_.mointsTF->transformHCore(pert, SCR, moType + "u", false, 'i');
113
114 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::Trans,
115 nCorrO, nq, nCorrO, MatsT(1.), oneRDM.pointer(), nCorrO,
116 SCR, nq, MatsT(0.), F2, nCorrO);
117
118 // populate ERI as (qu|vw)
119 mcwfn_.mointsTF->transformTPI(pert, SCR, moType + "uvw", true, false);
120
121 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::Trans,
122 nCorrO, nq, nCorrO3, MatsT(1.), twoRDM.pointer(), nCorrO,
123 SCR, nq, MatsT(1.), F2, nCorrO);
124
125 mcwfn_.memManager.free(SCR);
126
127 }; // OrbitalRotation<MatsT>::formGeneralizedFock2
