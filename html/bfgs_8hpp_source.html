42 /*
43 * Brief: This class computes the BFGS quasi-Newton step from the positions,
44 * and gradients. The inverse Hession is updated in place for each
45 * step. At the end of the optimization, xBFGS will contain the new set of
46 * positions.
47 *
48 * NOTE: The objFunc and gradFunc are called at each step and should be overloaded
49 * by the inheriting object
50 *
51 * N = number of positions/gradients i.e. dimension of the problem
52 * xBFGS = pointer to the current set of positions (should be set with setXPointer)
53 * gBFGS = pointer to the gradient at the current position (should be set with setGPointer)
54 */

115 inline void makeUnit(VT *invHess)
116 {
117 for (size_t i = 0; i < N * N; i++)

124 template <typename VT>
125 bool BFGS<VT>::optimize()
126 {
127 VT *invHess = memManager.template malloc<VT>(N * N);
128 VT *gPrev = memManager.template malloc<VT>(N);
129 VT *dx = memManager.template malloc<VT>(N);
130 VT *dg = memManager.template malloc<VT>(N);
131 VT *SCR = memManager.template malloc<VT>(N * N);
132 VT *U = memManager.template malloc<VT>(N * N);
133 VT *hessUpdate = memManager.template malloc<VT>(N * N);
134 VT alpha = VT(0.);
135 VT yPrev = VT(0.);
136 double ymin;
137 VT amin;
138
139 bool result = false;
140
141 // Compute initial/guess values before loop
142 this->gradFunc();
143 ymin = this->objFunc();
144
145 if (this->printVariables)
146 {
147 std::cout << "BFGS: Initial Parameters:" << std::endl;
148 std::cout << std::fixed << std::setprecision(printPrecision);
149 for (size_t i = 0; i < N; i++)
150 std::cout << xBFGS[i] << std::endl;
151 std::cout << std::endl;
152 }
153
154 if (this->printGrad)
155 {
156 std::cout << "BFGS: Initial Gradient:" << std::endl;
157 std::cout << std::fixed << std::setprecision(printPrecision);
158 for (size_t i = 0; i < N; i++)
159 std::cout << gBFGS[i] << std::endl;
160 std::cout << std::endl;
161 }
162
163 if (this->printOptOutput)
164 {
165 //std::cout << std::endl;
166 std::cout << std::scientific << std::setprecision(printPrecision) << "BFGS: Step 0: Function = " << ymin << std::endl;
167 //std::cout << std::endl;
168 yPrev = ymin;
169 }
170
171 // BFGS optimization loop
172 for (size_t iStep = 1; iStep <= this->nStep; iStep++)
173 {
174 // Compute inverse Hessian
175 if (iStep == 1 or ((iStep + 1) % N == 0 and resetInvHess))
176 {
177 makeUnit(invHess);
178 }
179 else if (updateInvHess)
180 {
181 // Compute dx dg inner product
182 VT dxdg = blas::dot(N, dx, 1, dg, 1);
183
184 // Using a scaling for invHessian. See following for details:
185 // Luenberger, D. G.; Ye, Y. Linear and Nonlinear Programming; Hillier, F. S., Ed.; Springer Science+Business Media.
186 VT scale = VT(1.);
187 if (scaleInvHess)
188 {
189 VT dgHdg = VT(0.);
190 for (size_t j = 0; j < N; j++)
191 for (size_t i = 0; i < N; i++)
192 dgHdg += dg[i] * invHess[i + j * N] * dg[j];
193 scale = dxdg / dgHdg;
194 if (std::isnan(scale))
195 scale = VT(1.);
196 }
197
198 // Compute inverse Hessian transformation ( I-( dg*dx^T / dxdg) )
199 makeUnit(U);
200
201 for (size_t j = 0; j < N; j++)
202 for (size_t i = 0; i < N; i++)
203 U[i + j * N] -= dg[i] * dx[j] / dxdg;
204
205 // Transform invHess => U^C * invHess * U
206 blas::gemm(blas::Layout::ColMajor,blas::Op::ConjTrans, blas::Op::NoTrans, N, N, N, VT(scale), U, N, invHess, N, VT(0.), SCR, N);
207 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans, blas::Op::NoTrans, N, N, N, VT(1.), SCR, N, U, N, VT(0.), invHess, N);
208
209 // add (dx*dx^T/dotProd) to invHess
210 for (size_t j = 0; j < N; j++)
211 for (size_t i = 0; i < N; i++)
212 invHess[i + j * N] += dx[i] * dx[j] / dxdg;
213
214 // Check invHess for nan
215 for (size_t i = 0; i < N * N; i++)
216 if (std::isnan(invHess[i]))
217 {
218 std::cout << "WARNING: Found a NaN in BFGS inverse Hessian" << std::endl;
219 makeUnit(invHess);
220 break;
221 }
222 }
223
224 // Compute search direction
225 blas::gemm(blas::Layout::ColMajor,blas::Op::NoTrans, blas::Op::NoTrans, N, 1, N, VT(-1.), invHess, N, gBFGS, N, VT(0.), dx, N);
226
227 if (printSearchDirection)
228 {
229 std::cout << "DX:" << std::endl;
230 for (size_t i = 0; i < N; i++)
231 std::cout << dx[i] << std::endl;
232 std::cout << std::endl;
233 }
234
235 // Perform Line Search
236 std::copy_n(xBFGS, N, SCR);
237 BFGS<VT>::lineSearch(SCR, dx, ymin, amin);
238
239 for (size_t i = 0; i < N; i++)
240 dx[i] = amin * dx[i];
241
242 // Compute new Gradient and positions
243 for (size_t i = 0; i < N; i++)
244 xBFGS[i] = SCR[i] + dx[i];
245
246 ymin = this->objFunc();
247 std::copy_n(gBFGS, N, gPrev);
248 this->gradFunc(); // This overwrites gBFGS
249 for (size_t i = 0; i < N; i++)
250 dg[i] = gBFGS[i] - gPrev[i];
251
252 // Print output
253 if (this->printVariables)
254 {
255 std::cout << "BFGS: Parameters:" << std::endl;
256 std::cout << std::scientific << std::setprecision(printPrecision);
257 for (size_t i = 0; i < N; i++)
258 std::cout << xBFGS[i] << std::endl;
259 std::cout << std::endl;
260 }
261
262 if (this->printGrad)
263 {
264 std::cout << "BFGS: Gradient:" << std::endl;
265 std::cout << std::scientific << std::setprecision(printPrecision);
266 for (size_t i = 0; i < N; i++)
267 std::cout << gBFGS[i] << std::endl;
268 std::cout << std::endl;
269 }
270
271 if (this->printOptOutput)
272 {
273 //std::cout << std::endl;
274 std::cout << std::scientific << std::setprecision(printPrecision) << "BFGS: Step " << iStep << ": Function = " << ymin << " Change = " << ymin - yPrev << std::endl;
275 //std::cout << std::endl;
276 }
277
278 // Evaluate Convergence
279 if (std::abs(yPrev - ymin) < convergence)
280 {
281 result = true;
282 break;
283 }
284 yPrev = ymin;
285 }
286
287 memManager.free(invHess, gPrev, dx, dg, SCR, hessUpdate, U);
288 return result;
289 };

291 template <typename VT>
292 void BFGS<VT>::lineSearch(VT *initX, VT *dx, double &ymin, VT &amin)
293 {
294 double yPrev = ymin;
295 switch (lineAlg)
296 {
297 case GoldenSection:
298 {
299 GoldenSectionSearch<VT>::optimize(initX, dx, VT(convergence), VT(goldenSectionMaxDisp), ymin, amin);
300 // if step size is too large then decrease
301 // the step size. Here we are using percent
302 // difference to determine a large increase
303 if (100. * (ymin - yPrev) / std::abs(yPrev) > percentFailSD)
304 {
305 GoldenSectionSearch<VT>::optimize(initX, dx, VT(convergence), VT(scaleFailedSD * goldenSectionMaxDisp), ymin, amin);
306 }
307
308 break;
309 }
310 case SteapestDescent:
311 {
312 amin = sdMaxDisp;
313 for (size_t i = 0; i < N; i++)
314 xBFGS[i] = initX[i] + amin * dx[i];
315 ymin = this->objFunc();
316
317 // if step size is too large then decrease
318 // the step size. Here we are using percent
319 // difference to determine a large increase
320 if (100. * (ymin - yPrev) / std::abs(yPrev) > percentFailSD)
321 {
322 amin *= scaleFailedSD;
323 for (size_t i = 0; i < N; i++)
324 xBFGS[i] = initX[i] + amin * dx[i];
325 ymin = this->objFunc();
326 }
327 break;
328 }
329 case ParticleSwarm:
330 {
331 CErr("Particle Swarmm Line Search has not been implemented yet");
332 }
333 }
334 };
