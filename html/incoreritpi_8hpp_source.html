93 InCoreRITPI& operator=( const InCoreRITPI &other ) {
94 if (this != &other) { // self-assignment check expected
95 if (this->nBasis() != other.nBasis()) {
96 this->NB = other.NB;
97 NBRI = other.NBRI;
98 NBNBRI = other.NBNBRI;
99 malloc(); // reallocate memory
100 }
101 std::copy_n(other.ERI3J, this->nBasis()*NBNBRI, ERI3J);
102 }
103 return *this;
104 }
105 InCoreRITPI& operator=( InCoreRITPI &&other ) {
106 if (this != &other) { // self-assignment check expected
107 this->memManager().free(ERI3J);
108 this->NB = other.NB;
109 NBRI = other.NBRI;
110 NBNBRI = other.NBNBRI;
111 ERI3J = other.ERI3J;
112 other.ERI3J = nullptr;
113 }
114 return *this;
115 }

184 virtual void broadcast(MPI_Comm comm = MPI_COMM_WORLD, int root = 0) {
185 TwoPInts<IntsT>::broadcast(comm, root);
186
187#ifdef CQ_ENABLE_MPI
188 if( MPISize(comm) > 1 ) {
189 size_t NBRI_bcast = NBRI;
190 size_t NBNBRI_bcast = NBNBRI;
191 MPIBCast(NBRI_bcast,root,comm);
192 MPIBCast(NBNBRI_bcast,root,comm);
193
194 if (NBRI_bcast != NBRI or NBNBRI_bcast != NBNBRI) {
195 NBRI = NBRI_bcast;
196 NBNBRI = NBNBRI_bcast;
197 malloc();
198 }
199
200 MPIBCast(ERI3J,this->nBasis()*NBNBRI,root,comm);
201 }
202#endif

230 void malloc() {
231 size_t NB3 = this->nBasis()*NBNBRI;
232 if (ERI3J) {
233 if (this->memManager().getSize(ERI3J) == NB3)
234 return;
235 this->memManager().free(ERI3J);
236 }
237 try { ERI3J = this->memManager().template malloc<IntsT>(NB3); }
238 catch(...) {
239 std::cout << std::fixed;
240 std::cout << "Insufficient memory for the full RI-ERI tensor ("
241 << (NB3/1e9) * sizeof(double) << " GB)" << std::endl;
242 std::cout << std::endl << this->memManager() << std::endl;
243 CErr();
244 }
