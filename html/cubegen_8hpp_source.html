119 void evalCDCube(){
120
121 size_t NB = ref_->nAlphaOrbital();
122
123 std::vector<libint2::Shell> &shells = ref_->basisSet().shells;
124 std::vector<double> BASIS(NB,0.);
125 std::vector<T> SCR(NB,0.);
126
127 for(auto ix = 0l; ix < nPts[0]; ix++)
128 for(auto iy = 0l; iy < nPts[1]; iy++) {
129 for(auto iz = 0l; iz < nPts[2]; iz++) {
130
131 std::array<double,3> pt =
132 {(ix-(int)nPts[0]/2) * L[0],
133 (iy-(int)nPts[1]/2) * L[1],
134 (iz-(int)nPts[2]/2) * L[2]};
135
136 evalShellSet(ref_->memManager,NOGRAD,shells,&pt[0],1,&BASIS[0],false);
137
138 blas::gemm(blas::Layout::ColMajor,'T',blas::Op::NoTrans,1,NB,NB,T(1.),&BASIS[0],NB,ref_->onePDM->S().pointer(),NB,
139 T(0.),&SCR[0],1);
140
141 double val = blas::dot(NB,&SCR[0],1,&BASIS[0],1);
142
143 *cubeFile << std::right << std::setw(15) << std::setprecision(5)
144 << std::scientific << val;
145
146 if( iz % 6 == 5 ) *cubeFile << "\n";

156 void evalSDCube() {
157
158 size_t NB = ref_->nAlphaOrbital();
159
160 std::vector<libint2::Shell> &shells = ref_->basisSet().shells;
161 std::vector<double> BASIS(NB,0.);
162 std::vector<T> SCR(NB,0.);
163
164 for(auto ix = 0l; ix < nPts[0]; ix++)
165 for(auto iy = 0l; iy < nPts[1]; iy++) {
166 for(auto iz = 0l; iz < nPts[2]; iz++) {
167
168 std::array<double,3> pt =
169 {(ix-(int)nPts[0]/2) * L[0],
170 (iy-(int)nPts[1]/2) * L[1],
171 (iz-(int)nPts[2]/2) * L[2]};
172
173 double val = 0;
174
175 if( ref_->onePDM.hasZ() ) {
176
177 evalShellSet(ref_->memManager,NOGRAD,shells,&pt[0],1,&BASIS[0],false);
178
179 for(size_t i = 1; i < ref_->onePDM->nComponent(); i++) {
180 blas::gemm(blas::Layout::ColMajor,'T',blas::Op::NoTrans,1,NB,NB,T(1.),&BASIS[0],NB,
181 (*ref_->onePDM)[static_cast<PAULI_SPINOR_COMPS>(i)].pointer()],NB,
182 T(0.),&SCR[0],1);
183
184 double tmp = blas::dot(NB,&SCR[0],1,&BASIS[0],1);
185 val += tmp*tmp;
186 }
187
188 val = std::sqrt(val);
189
190 }
191 *cubeFile << std::right << std::setw(15) << std::setprecision(5)
192 << std::scientific << val;
193
194 if( iz % 6 == 5 ) *cubeFile << "\n";
195
196
197 }
198 *cubeFile << "\n";
199 }

205 void evalEPCube(){
206
207
208 size_t NB = ref_->nAlphaOrbital();
209
210 std::vector<libint2::Shell> &shells = ref_->basisSet().shells;
211
212 std::vector<T> SCR(NB*NB,0.);
213
214 // Determine the maximum angular momentum of the passed shell set
215 int maxL = std::max_element(shells.begin(), shells.end(),
216 [](libint2::Shell &sh1, libint2::Shell &sh2){
217 return sh1.contr[0].l < sh2.contr[0].l;
218 }
219 )->contr[0].l;
220
221 // Determine the maximum contraction depth of the passed shell set
222 int maxPrim = std::max_element(shells.begin(), shells.end(),
223 [](libint2::Shell &sh1, libint2::Shell &sh2){
224 return sh1.alpha.size() < sh2.alpha.size();
225 }
226 )->alpha.size();
227
228 libint2::Engine engine(libint2::Operator::nuclear,maxPrim,maxL,0);
229 engine.set_precision(0.);
230
231 for(auto ix = 0l; ix < nPts[0]; ix++)
232 for(auto iy = 0l; iy < nPts[1]; iy++) {
233 for(auto iz = 0l; iz < nPts[2]; iz++) {
234
235 std::array<double,3> pt =
236 {(ix-(int)nPts[0]/2) * L[0],
237 (iy-(int)nPts[1]/2) * L[1],
238 (iz-(int)nPts[2]/2) * L[2]};
239
240 std::vector<std::pair<double,std::array<double,3>>> q;
241 q.push_back( {1., pt} );
242 engine.set_params(q);
243
244 const auto& buf_vec = engine.results();
245 size_t n1,n2;
246
247 // Loop over unique shell pairs
248 for(size_t s1(0), bf1_s(0), s12(0); s1 < shells.size();
249 bf1_s+=n1, s1++){
250 n1 = shells[s1].size(); // Size of Shell 1
251 for(size_t s2(0), bf2_s(0); s2 <= s1; bf2_s+=n2, s2++, s12++) {
252 n2 = shells[s2].size(); // Size of Shell 2
253
254
255 engine.compute(shells[s1],shells[s2]);
256
257 // If the integrals were screened, move on to the next batch
258 if(buf_vec[0] == nullptr) continue;
259
260 SetMat('N',n1,n2,1.,const_cast<double*>(buf_vec[0]),n1,
261 &SCR[bf1_s + bf2_s*NB],NB);
262
263 }
264 }
265
266 HerMat('L',NB,&SCR[0],NB);
267 prettyPrintSmart(std::cout,"V",&SCR[0],NB,NB,NB);
268
269 double val = blas::dot(NB*NB,&SCR[0],1,ref_->onePDM->S().pointer(),1);
270
271 for( auto &atom : ref_->molecule().atoms ) {
272
273 auto &c = atom.coord;
274 val += atom.atomicNumber /
275 std::sqrt( std::pow(pt[0] - c[0],2.) +
276 std::pow(pt[1] - c[1],2.) +
277 std::pow(pt[2] - c[2],2.) );
278
279 }
280
281
282 *cubeFile << std::right << std::setw(15) << std::setprecision(5)
283 << std::scientific << val;
284
285 if( iz % 6 == 5 ) *cubeFile << "\n";
286
287
288 }
289 *cubeFile << "\n";
290 }
