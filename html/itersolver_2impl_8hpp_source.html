60 template <typename _F>
61 void RawVectors<_F>::dot_product(size_t shiftA, const SolverVectors<_F> &B, size_t shiftB,
62 int64_t m, int64_t n, _F *C, int64_t ldc, bool conjA) const {
63
64 if (MPIRank(comm_) == 0){
65 if (m + shiftA > size() or n + shiftB > B.size())
66 CErr("RawVectors does not contain enough number of vectors"
67 " to match requested dot product.", std::cout);
68
69 if (length() != B.length())
70 CErr("Lengths of vectors does not match for dot product.");
71
72 blas::gemm(blas::Layout::ColMajor,
73 conjA ? blas::Op::ConjTrans : blas::Op::Trans,
74 blas::Op::NoTrans,
75 m,n,length(),_F(1.),getPtr(shiftA),length(),
76 B.getPtr(shiftB),length(),_F(0.),C,ldc);
77 }
78
79#ifdef CQ_ENABLE_MPI
80 if (MPISize(comm_) > 1) {
81 bool localC = C == nullptr;
82 if (localC)
83 C = memManager_.template malloc<_F>(ldc * n);
84 if (ldc == m)
85 MPIBCast(C, m*n, 0, comm_);
86 else
87 for (size_t i = 0; i < n; i++)
88 MPIBCast(C + i * ldc, m, 0, comm_);
89 if (localC)
90 memManager_.free(C);
91 }
92#endif
93
94 }

183 template <typename _F>
184 size_t SolverVectors<_F>::GramSchmidt(size_t shift, size_t Mold, size_t Mnew, CQMemManager &mem,
185 size_t NRe, double eps) {
186
187 _F * SCR = mem.template malloc<_F>(Mold + Mnew);
188
189 if( Mold == 0 ) {
190 // Normalize the first vector
191 double inner = norm2F(shift, 1);
192 if(std::abs(inner) < eps) CErr("Zero inner product incurred!");
193 scale(1.0/inner, shift, 1);
194 }
195
196 // Orthonormalize the rest of the matrix using GS
197 size_t iOrtho = (Mold == 0) ? Mold + 1: Mold;
198 for(auto k = iOrtho; k < (Mold + Mnew); k++) {
199
200 if( k != iOrtho ) set_data(iOrtho + shift, 1, *this, k + shift);
201
202 // Project out the inner products
203 for(auto iRe = 0; iRe < (NRe+1); iRe++) {
204 dot_product(shift, *this, shift + iOrtho, iOrtho,1, SCR, iOrtho);
205 multiply_matrix(shift, blas::Op::NoTrans, 1, iOrtho, _F(-1.0), SCR, iOrtho, _F(1.0), *this, shift + iOrtho);
206 }
207
208 // Normalize the new vector
209 double inner = norm2F(iOrtho + shift, 1);
210 std::cout << k << " " << inner << std::endl;
211 if(std::abs(inner) < length()*eps) {
212 std::cout << "Zero inner product incurred! " << k << "\n";
213 scale(0.0, iOrtho + shift, 1);
214 } else {
215 scale(1.0 / inner, iOrtho + shift, 1);
216 iOrtho++;
217 }
218 }
219
220 mem.free(SCR);
221
222 return iOrtho;
223
224 }
