76 template<typename MatsT, typename IntsT>
77 void NEOSS<MatsT, IntsT>::runModifyOrbitals(EMPerturbation& pert) {
78 using SubSSPtr = std::shared_ptr<SingleSlater<MatsT,IntsT>>;
79
80 // Initialize properties
81 applyToEach([&pert](SubSSPtr& ss){
82 ss->getNewOrbitals();
83 ss->computeProperties(pert);
84 });
85
86 // Setup MO reference vector
87 std::vector<std::reference_wrapper<SquareMatrix<MatsT>>> moRefs;
88 applyToEach([&moRefs](SubSSPtr& ss) {
89 for( auto& m: ss->mo )
90 moRefs.emplace_back(m);
91 });
92
93 // Setup Eigenvalue vector
94 std::vector<double*> epsVec;
95 applyToEach([&epsVec](SubSSPtr& ss) {
96 epsVec.push_back(ss->eps1);
97 if( ss->nC == 1 and !ss->iCS )
98 epsVec.push_back(ss->eps2);
99 });
100
101 this->modifyOrbitals->runModifyOrbitals(pert, moRefs, epsVec);
102
103 applyToEach([](SubSSPtr& ss) {
104 ss->ao2orthoFock();
105 ss->MOFOCK();
106 });
107
108 saveCurrentState();
109
110#ifdef CQ_ENABLE_MPI
111 // Broadcast the updated MOs to all MPI processes
112 applyToEach([&](SubSSPtr& ss) {
113 if( MPISize(ss->comm) > 1 ) {
114 std::cerr << " *** Scattering the AO-MOs ***\n";
115 size_t Nmo = ss->mo[0].dimension();
116 MPIBCast(ss->mo[0].pointer(),Nmo*Nmo,0,ss->comm);
117 if( ss->nC == 1 and not ss->iCS )
118 MPIBCast(ss->mo[1].pointer(),Nmo*Nmo,0,ss->comm);
119
120 std::cerr << " *** Scattering EPS ***\n";
121 MPIBCast(ss->eps1,Nmo,0,ss->comm);
122 if( ss->nC == 1 and not ss->iCS )
123 MPIBCast(ss->eps2,Nmo,0,ss->comm);
124
125 std::cerr << " *** Scattering FOCK ***\n";
126 size_t fockDim = ss->fockMatrix->dimension();
127 for(MatsT *mat : ss->fockMatrix->SZYXPointers())
128 MPIBCast(mat,fockDim*fockDim,0,ss->comm);
129
130 std::cerr << " *** Scattering the 1PDM ***\n";
131 size_t denDim = ss->onePDM->dimension();
132 for(auto p : ss->onePDM->SZYXPointers())
133 MPIBCast(p,denDim*denDim,0,ss->comm);
134 }
135 });
136
137#endif
138
139 };
