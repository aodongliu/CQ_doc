170 inline void update() {
171
172 // Compute the total number of
173 nTotalE = std::accumulate(atoms.begin(),atoms.end(),-charge,
174 [&](int c, const Atom &a){ return a.atomicNumber + c; }
175 );
176
177 if(not ((nTotalE % 2) != 0 xor (multip % 2) != 0) or
178 multip > nTotalE + 1) {
179 std::stringstream ss;
180 ss << "Multiplicity = " << multip << " is not compatible with "
181 << "total electrons = " << nTotalE;
182 CErr(ss.str(),std::cout);
183 }
184
185 // Compute the total number of quantum protons
186 atomsQ.clear();
187 atomsC.clear();
188
189 nTotalP = 0;
190 size_t ind = 0;
191 for ( Atom& atom : atoms ) {
192 if ( atom.quantum ) {
193 // return an error if not hydrogen
194 if ( atom.atomicNumber != 1 )
195 CErr("Non-Hydrogen quantum nuclei NYI.");
196
197 nTotalP += 1;
198 atomsQ.push_back(ind);
199 }
200 else
201 atomsC.push_back(ind);
202 ind += 1;
203 }
204
205 // assume high-spin open-shell for protons
206 multip_proton = (size_t)(2 * nTotalP * 0.5 + 1);
207
208 computeRIJ();
209 computeNNRep();
210 computeNNX();
211 computeCOM();
212 computeCOC();
213 computeMOI();
214 computeCDist();
215
216 }
