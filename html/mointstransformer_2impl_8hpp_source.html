144 template <typename MatsT, typename IntsT>
145 void MOIntsTransformer<MatsT,IntsT>::transformTPI(EMPerturbation & pert,
146 MatsT* MOTPI, const std::string & moType, bool cacheIntermediates,
147 bool withExchange, TPI_TRANS_DELTA_TYPE delta) {
148
149 // check delta
150 if (delta != NO_KRONECKER_DELTA) {
151 if ((delta == KRONECKER_DELTA_PQ and getUniqueSymbol(moType[0]) != getUniqueSymbol(moType[1])) or
152 (delta == KRONECKER_DELTA_RS and getUniqueSymbol(moType[2]) != getUniqueSymbol(moType[3])) or
153 (delta == KRONECKER_DELTA_PS and getUniqueSymbol(moType[0]) != getUniqueSymbol(moType[3])) or
154 (delta == KRONECKER_DELTA_RQ and getUniqueSymbol(moType[1]) != getUniqueSymbol(moType[2])) or
155 (delta == KRONECKER_DELTA_PQ_RS and (getUniqueSymbol(moType[0]) != getUniqueSymbol(moType[1])
156 or getUniqueSymbol(moType[2]) != getUniqueSymbol(moType[3]))) or
157 (delta == KRONECKER_DELTA_PS_RQ and (getUniqueSymbol(moType[0]) != getUniqueSymbol(moType[3])
158 or getUniqueSymbol(moType[1]) != getUniqueSymbol(moType[2]))))
159 CErr("Specifiying KRONECKER_DELTA on different dimensions");
160 }
161
162 std::string C_moType = moType;
163 std::string X_moType = "";
164 TPI_TRANS_DELTA_TYPE C_delta = delta;
165 TPI_TRANS_DELTA_TYPE X_delta = NO_KRONECKER_DELTA;
166 bool swapCX = false;
167
168 if (withExchange) {
169 X_moType += moType[0];
170 X_moType += moType[3];
171 X_moType += moType[2];
172 X_moType += moType[1];
173
174 if (delta == KRONECKER_DELTA_PQ) X_delta = KRONECKER_DELTA_PS;
175 else if (delta == KRONECKER_DELTA_RS) X_delta = KRONECKER_DELTA_RQ;
176 else if (delta == KRONECKER_DELTA_PS) X_delta = KRONECKER_DELTA_PQ;
177 else if (delta == KRONECKER_DELTA_RQ) X_delta = KRONECKER_DELTA_RS;
178 else if (delta == KRONECKER_DELTA_PQ_RS) X_delta = KRONECKER_DELTA_PS_RQ;
179 else if (delta == KRONECKER_DELTA_PS_RQ) X_delta = KRONECKER_DELTA_PQ_RS;
180
181 // KRONECKER_DELTA_PS -> KRONECKER_DELTA_PQ
182 // KRONECKER_DELTA_RQ -> KRONECKER_DELTA_RS
183 // KRONECKER_DELTA_PS_RQ -> KRONECKER_DELTA_PQ_RS
184 if (delta == KRONECKER_DELTA_PS or
185 delta == KRONECKER_DELTA_RQ or
186 delta == KRONECKER_DELTA_PS_RQ) {
187 std::swap(C_moType, X_moType);
188 std::swap(C_delta, X_delta);
189 swapCX = true;
190 }
191
192 //std::cout << "C_moType = " << C_moType << ", C_delta =" << C_delta << std::endl;
193 //std::cout << "X_moType = " << X_moType << ", X_delta =" << X_delta << std::endl;
194 }
195
196 auto off_sizes = parseMOType(C_moType);
197
198 // get the Coulomb part
199 if (TPITransAlg_ == DIRECT_N6 or TPITransAlg_ == INCORE_N6) {
200 subsetTransformTPISSFockN6(pert, off_sizes, MOTPI, C_moType, cacheIntermediates, C_delta);
201 } else if (TPITransAlg_ == INCORE_N5) {
202 subsetTransformTPIInCoreN5(off_sizes, MOTPI, cacheIntermediates, C_delta);
203 } else {
204 CErr("DIRECT_N5 NYI");
205 }
206
207 // get exchange part if needed
208 // for 2c/4c: it's antisymmetrized integrals (pq|rs) - (ps|rq)
209 // for 1c: it's scaled for spins, so exchange part will be scaled with 0.5.
210 // So (pq|rs) - 0.5 * (ps|rq) will be computed
211 if (withExchange) {
212
213 size_t np = off_sizes[0].second;
214 size_t nq = off_sizes[1].second;
215 size_t nr = off_sizes[2].second;
216 size_t ns = off_sizes[3].second;
217 size_t npq = np * nq;
218 size_t nps = np * ns;
219 size_t npr = np * nr;
220 size_t npqr = npq * nr;
221 size_t nprs = nps * nr;
222
223 MatsT fc = ss_.nC == 1 ? 0.5: 1.0;
224
225 size_t qoff = off_sizes[1].first;
226 size_t soff = off_sizes[3].first;
227 bool qsSymm = nq == ns and qoff == soff;
228
229 if (C_delta == NO_KRONECKER_DELTA and qsSymm) {
230 #pragma omp parallel for schedule(static) collapse(2) default(shared)
231 for (auto r = 0ul; r < nr; r++)
232 for (auto p = 0ul; p < np; p++) {
233 MatsT tmp1, tmp2;
234 size_t pqrs, psrq;
235 size_t prnpq = p + r*npq;
236 size_t prnps = p + r*nps;
237 for (auto s = 0ul; s < ns; s++)
238 for (auto q = 0ul; q <= s; q++) {
239 pqrs = prnpq + q*np + s*npqr;
240 psrq = prnps + s*np + q*nprs;
241 tmp1 = MOTPI[pqrs];
242 tmp2 = MOTPI[psrq];
243 MOTPI[pqrs] = tmp1 - fc * tmp2;
244 MOTPI[psrq] = tmp2 - fc * tmp1;
245 }
246 }
247 } else {
248
249 size_t SCRSize;
250 if (C_delta == NO_KRONECKER_DELTA) SCRSize = npqr*ns;
251 else if (C_delta == KRONECKER_DELTA_RS) SCRSize = npqr;
252 else if (C_delta == KRONECKER_DELTA_PQ) SCRSize = nprs;
253 else if (C_delta == KRONECKER_DELTA_PQ_RS) SCRSize = npr;
254 MatsT * SCR = memManager_.malloc<MatsT>(SCRSize);
255
256 transformTPI(pert, SCR, X_moType, true, false, X_delta);
257
258 if (C_delta == NO_KRONECKER_DELTA) {
259 // (p,q,r,s) -= (p,s,q,r)
260 #pragma omp parallel for schedule(static) collapse(2) default(shared)
261 for (auto s = 0ul; s < ns; s++)
262 for (auto r = 0ul; r < nr; r++)
263 for (auto q = 0ul; q < nq; q++)
264 for (auto p = 0ul; p < np; p++) {
265 MOTPI[p + q*np + r*npq + s*npqr] -= fc * SCR[p + s*np + r*nps + q*nprs];
266 }
267 } else if (C_delta == KRONECKER_DELTA_PQ or C_delta == KRONECKER_DELTA_RS) {
268
269 // for KRONECKER_DELTA_PQ: (p,r,s) -= (p,s,r)
270 // for KRONECKER_DELTA_RS: (p,q,r) -= (p,r,q)
271 size_t nt = KRONECKER_DELTA_PQ ? ns: nq;
272 size_t npt = np*nt;
273 MatsT fc_inv = swapCX ? 1./ fc: fc;
274
275 #pragma omp parallel for schedule(static) collapse(2) default(shared)
276 for (auto t = 0ul; t < nt; t++)
277 for (auto r = 0ul; r < nr; r++)
278 for (auto p = 0ul; p < np; p++) {
279 MOTPI[p + r*np + t*npr] -= fc_inv * SCR[p + t*np + r*npt];
280 }
281
282 // (p,r,t) --> - (p,t,r) if swapCX
283 if (swapCX) {
284 #pragma omp parallel for
285 for (auto p = 0ul; p < np; p++) {
286 MatsT tmp;
287 size_t prt, ptr;
288 for (auto t = 0ul; t < nt; t++)
289 for (auto r = 0ul; r <= t; r++) {
290 prt = p + r*np + t*npr;
291 ptr = p + t*np + r*npt;
292 tmp = MOTPI[prt];
293 MOTPI[prt] = - MOTPI[ptr] * fc;
294 MOTPI[ptr] = - tmp * fc;
295 }
296 }
297 }
298 } else if (C_delta == KRONECKER_DELTA_PQ_RS) {
299 // (p,p,r,r) -= (p,r,r,p)
300 MatsT * C_term = MOTPI, *X_term = SCR;
301 if (swapCX) std::swap(X_term, C_term);
302
303 MatAdd('N', 'N', np, nr, MatsT(1.), C_term, np, MatsT(-fc), X_term, np, MOTPI, np);
304
305 }// delta
306
307 memManager_.free(SCR);
308 }
309 } // with Exchange
310
311 }; // MOIntsTransformer::transformTPI
