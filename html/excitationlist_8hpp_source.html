135 for (auto i = 0ul; i < nStr1_; i++)

138 } else {
139 std::cout << " for total number of string1 and string2 = (" << nStr1_

141 for (auto i = 0ul; i < nStr1_; i++)

143 prettyPrintSmart(out,
144 "(iStr, jStr) = (" + std::to_string(i) + "," + std::to_string(j) + ")",
145 pointerAtDet(i, j), nElement_, nNonZero_, nElement_);
