45 /*
46 * \brief solve CI Hamiltonian for N States
47 *
48 * Options:
49 * 1. build full matrix and then direct diagonalize it
50 * 2. pass functions of build sigma to davidson iterative solver
51 *
52 */
53
54 template <typename MatsT, typename IntsT>
55 void CISolver<MatsT, IntsT>::solveCI(MCWaveFunction<MatsT, IntsT> & mcwfn) {
56
57 size_t NDet = mcwfn.NDet;
58 size_t nR = mcwfn.NStates;
59 auto & StateEnergy = mcwfn.StateEnergy;
60 auto & CIVecs = mcwfn.CIVecs;
61 auto & mem = mcwfn.memManager;
62
63 if (alg_ == CI_FULL_MATRIX) {
64
65 std::cout << " Diagonalize CI Full Hamitonian Matrix ... \n" << std::endl;
66 dcomplex * Energy = mem.template malloc<dcomplex>(NDet);
67 MatsT * fullH = mem.template malloc<MatsT>(NDet*NDet);
68 MatsT * EigVec = mem.template malloc<MatsT>(NDet*NDet);
69 MatsT * dummy = nullptr;
70
71 ProgramTimer::tick("Full Matrix");
72 mcwfn.ciBuilder->buildFullH(mcwfn, fullH);
73 ProgramTimer::tock("Full Matrix");
74
75#ifdef _DEBUG_CIENGINE_IMPL
76 std::cout << " <0|H|0> = " << std::setprecision(16)
77 << *fullH + mcwfn.reference().molecule().nucRepEnergy + mcwfn.InactEnergy
78 << std::endl;
79#endif
80
81
82 GeneralEigen('N','V', NDet, fullH, NDet, Energy, dummy, 1, EigVec, NDet);
83 //HermetianEigen('V', 'L', NDet, fullH, NDet, Energy, mem);
84
85#ifdef _DEBUG_CIENGINE_IMPL
86 prettyPrintSmart(std::cout,"HH full H", fullH, NDet, NDet, NDet);
87 prettyPrintSmart(std::cout,"HH Eigenvectors", EigVec, NDet, NDet, NDet);
88 prettyPrintSmart(std::cout,"HH Eigenvalues", Energy, NDet, 1, NDet);
89#endif
90
91 // copy over eigenvalues and eigenvectors
92 for (auto i = 0ul; i < nR; i++) {
93 StateEnergy[i] = std::real(Energy[i]);
94 std::copy_n(EigVec+i*NDet, NDet, CIVecs[i]);
95 }
96 mem.free(Energy, fullH, EigVec);
97
98 } else if (alg_ == CI_DAVIDSON) {
99
100 // build diagonal H
101 MatsT * diagH = mem.template malloc<MatsT>(NDet);
102 mcwfn.ciBuilder->buildDiagH(mcwfn, diagH);
103
104#ifdef _DEBUG_CISOLVER_IMPL
105 prettyPrintSmart(std::cout,"HH Diagonal H ", diagH, NDet, 1, NDet);
106#endif
107
108 // set davidson parameters
109 size_t kG = nDavidsonGuess_;
110 size_t m = std::max(maxDavidsonSpace_, kG);
111 size_t nG = kG*nR;
112
113 dcomplex * curEig = mem.template malloc<dcomplex>(nG);
114
115 using LinearTrans_t = typename IterDiagonalizer<MatsT>::LinearTrans_t;
116
117 // define linear transformation
118 LinearTrans_t func = [&] (size_t nVec, SolverVectors<MatsT> &V, SolverVectors<MatsT> &AV) {
119
120#ifdef CQ_ENABLE_MPI
121 // disable MPI for now
122 ROOT_ONLY(mcwfn.comm);
123#endif
124
125 ProgramTimer::tick("Sigma");
126 mcwfn.ciBuilder->buildSigma(mcwfn, nVec, V.getPtr(), AV.getPtr());
127 ProgramTimer::tock("Sigma");
128
129 };
130
131 // define preconditioner
132 LinearTrans_t PC = [&] (size_t nVec, SolverVectors<MatsT> &S, SolverVectors<MatsT> &R) {
133
134#ifdef CQ_ENABLE_MPI
135 // disable MPI for now
136 ROOT_ONLY(mcwfn.comm);
137#endif
138 this->davidsonPC(nVec, NDet, S.getPtr(), R.getPtr(), diagH, curEig);
139 };
140
141 std::cout << " Use Davidson Diagonalization ... \n" << std::endl;
142
143 Davidson<MatsT> davidson(mcwfn.comm, mem, NDet, 5, maxIter_,
144 vectorConv_, nR, func, PC);
145
146 davidson.setM(m);
147 davidson.setkG(kG);
148 davidson.setEigForT(curEig);
149 davidson.setGuess(nG, [&] (size_t nGuess, SolverVectors<MatsT> &Guess, size_t N) {
150 this->davidsonGS(nGuess, N, diagH, Guess.getPtr());
151 });
152
153 davidson.run();
154
155 // copy over eigenvalues and eigenvectors
156 auto davidsonEig = davidson.eigVal();
157 auto davidsonVec = davidson.VR()->getPtr();
158 for (auto i = 0ul; i < nR; i++) {
159 StateEnergy[i] = std::real(davidsonEig[i]);
160 std::copy_n(davidsonVec+i*NDet, NDet, CIVecs[i]);
161 }
162
163 mem.free(curEig, diagH);
164
165 } else{
166 CErr("Haven't Inplement Other Diagonalization yet");
167 }
168
169 // add other parts of the energy
170 double EOther = mcwfn.reference().molecule().nucRepEnergy + mcwfn.InactEnergy;
171 for (auto i = 0ul; i < nR; i++) StateEnergy[i] += EOther;
172
173 } // CISolver::solveCI

177 template <typename MatsT, typename IntsT>
178 void CISolver<MatsT,IntsT>::davidsonGS(size_t nGuess, size_t N, MatsT * HDiag, MatsT * Guess) {
179
180 std::cout << " * use unit vector guess based on diagonal elements:" << std::endl;
181
182 std::vector<int> indx(N);
183 std::iota(indx.begin(), indx.end(), 0);
184
185 std::stable_sort(indx.begin(), indx.end(),
186 [&] (size_t i , size_t j) {
187 return std::real(HDiag[i]) < std::real(HDiag[j]);
188 }
189 );
190
191 std::fill_n(Guess, nGuess*N, MatsT(0.));
192 for(auto i = 0ul; i < nGuess; i++) {
193 Guess[i*N+indx[i]] = 1.0;
194 std::cout << " " << std::setw(9) << std::left << indx[i]
195 << std::setw(40) << std::left << HDiag[indx[i]] << std::endl;
196 }
197
198 } // CISolver::davidsonGS

200 // davidson preconditioner
201 template <typename MatsT, typename IntsT>
202 void CISolver<MatsT,IntsT>::davidsonPC(size_t nVec, size_t N, MatsT * R, MatsT * S,
203 MatsT * HDiag, dcomplex * curEig) {
204
205 auto D = HDiag;
206 auto L = curEig;
207 double small = 1e-12;
208 double P;
209
210 for (auto i = 0ul; i < nVec; i++) {
211 auto Ri = R + i*N;
212 auto Si = S + i*N;
213 for (auto j = 0ul; j < N; j++) {
214 P = std::real(D[j]) - std::real(L[i]);
215 if(std::abs(P) > small) Si[j] = Ri[j] / P;
216 }
217 }
218
219 } // CISolver::davidsonPC
