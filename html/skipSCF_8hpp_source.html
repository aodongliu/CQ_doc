40 // RunModifyOrbitals is a stripped down version from OptimizeOrbitals
41 void runModifyOrbitals(EMPerturbation& pert, VecMORef<MatsT>& mo, VecEPtr& eps) {
42 ProgramTimer::tick("SCF Total");
43
44 // Form the Fock matrix D(k) -> F(k)
45 ProgramTimer::timeOp("Form Fock", [&]() { this->modOrbOpt.formFock(pert); });
46 this->computeEigenvalues(mo,eps);
47
48 // Compute initial properties
49 this->modOrbOpt.computeProperties(pert);
50
51 if( this->scfControls.printLevel > 0 and MPIRank(this->comm) == 0 ) {
52 this->printRunHeader(std::cout, pert);
53 this->printIteration(std::cout, false);
54 }
55
56 // Save current state of the wave function (method specific)
57 this->modOrbOpt.saveCurrentState();
58
59 // printSCFFooter(isConverged);
60 if( this->scfControls.printLevel > 0 ) {
61 std::cout << std::endl
62 << "SCF Completed: E(" << this->scfControls.refShortName_ << ") = " << std::fixed << std::setprecision(10)
63 << this->modOrbOpt.getTotalEnergy() << " Eh after " << this->scfConv.nSCFIter << " SCF Iterations" << std::endl;
64 }
65
66 if( this->scfControls.printLevel > 0 ) std::cout << BannerEnd << std::endl;
67
68 if( this->scfControls.printLevel > 1 ) { this->modOrbOpt.printProperties(); }
69
70 ProgramTimer::tock("SCF Total");
71
72 }; // SkipSCF<MatsT>::RunModifyOrbitals()
