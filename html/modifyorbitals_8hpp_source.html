95/*
96 * Brief: Abstract Base class for the modifyOrbitals object. This allows
97 * us to abstract the algorithms that modify the orbitals into one
98 * interface. This is the interface that the object that owns ModifyOrbitals
99 * runs the ModifyOrbitals algorithms.
100 */
101template<typename MatsT>
102class ModifyOrbitals {
103
104 protected:
105 MPI_Comm comm;
106 CQMemManager& memManager;
107 ModifyOrbitalsOptions<MatsT> modOrbOpt;
108
109 public:
110 ModifyOrbitals() = delete;
111 ModifyOrbitals(MPI_Comm c, ModifyOrbitalsOptions<MatsT> m, CQMemManager& mem): comm(c), modOrbOpt(m), memManager(mem) {
112 // Check that Essential functions are bound
113 if( not modOrbOpt.printProperties ) CErr("printProperties was not bound in ModifyOrbitalOptions");
114 if( not modOrbOpt.saveCurrentState ) CErr("saveCurrentState was not bound in ModifyOrbitalOptions");
115 if( not modOrbOpt.formFock ) CErr("formFock was not bound in ModifyOrbitalOptions");
116 if( not modOrbOpt.formDensity ) CErr("formDensity was not bound in ModifyOrbitalOptions");
117 if( not modOrbOpt.computeProperties ) CErr("computeProperties was not bound in ModifyOrbitalOptions");
118 if( not modOrbOpt.getTotalEnergy ) CErr("getTotalEnergy was not bound in ModifyOrbitalOptions");
119 if( not modOrbOpt.getFock ) CErr("getFock was not bound in ModifyOrbitalOptions");
120 if( not modOrbOpt.getOnePDM ) CErr("getOnePDM was not bound in ModifyOrbitalOptions");
121 if( not modOrbOpt.getOrtho ) CErr("getOrtho was not bound in ModifyOrbitalOptions");
122 };
123
124 // Getter/Setter functions for options struct
125 ModifyOrbitalsOptions<MatsT> getModOpt() { return modOrbOpt; };
126 void setModOpt(const ModifyOrbitalsOptions<MatsT>& mod) { modOrbOpt = mod; };
127
128 // runModifyOrbitals is the driver function that performs
129 // the whole optimization/simulation
130 virtual void runModifyOrbitals(EMPerturbation& pert, VecMORef<MatsT>&, VecEPtr&) = 0;
131
132 // getNewOrbitals performs only a single step in the
133 // optimization/simulation
134 virtual void getNewOrbitals(EMPerturbation& pert, VecMORef<MatsT>&, VecEPtr&) = 0;
135
136 // Printing functions
137 virtual void printRunHeader(std::ostream& out, EMPerturbation&) const = 0;
138 virtual void printIteration(std::ostream& out = std::cout, bool printDiff = true) const = 0;
139};
