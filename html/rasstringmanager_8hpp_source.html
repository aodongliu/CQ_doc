49 /*
50 * find the Excitation list corresponding to a set of input:
51 * p, q ...... p, q blocks
52 * L, K ...... Categories for strings L and K
53 * return a pointer to the ExcitationList
54 */
55 std::shared_ptr<const ExcitationList> RASStringManager::find_exlist(
56 size_t p, size_t q, size_t L, size_t K) const {
57
58 std::string map_key = this->mapKey(p, q, L, K);
59 auto search = this->exList_.find(map_key);

65 /*
66 * Generate the offset of each space for each category
67 * nStr: total number of strings for RAS case.
68 * LCat: nCat_ x 4
69 * RAS 1 offset ... 1
70 * RAS 2 offset ... N(RAS1)
71 * RAS 3 offset ... N(RAS1)*N(RAS2)
72 * total number of string for that category iCat ... N(RAS1)*NRAS2)*N(RAS3)
73 */
74 void RASStringManager::genLCat() {
75
76 LCat_ = std::vector<std::vector<size_t>>(nCat_, std::vector<size_t>(4, 0));
77
78 this->nStr_ = 0ul;
79 for (auto iE = 0ul, iCat = 0ul; iE <= mxElec_; iE++)
80 for (auto iH = 0ul; iH <= mxHole_; iH++, iCat++) {
81 if((nE_+iH) < (nActO_[0]+iE) or (nE_+iH) > (nActO_[0]+iE+nActO_[1])) continue;
82 size_t nEras2 = nE_ - nActO_[0] + iH - iE;
83 LCat_[iCat][0] = 1;
84 LCat_[iCat][1] = Comb(nActO_[0], iH);
85 LCat_[iCat][2] = LCat_[iCat][1] * Comb(nActO_[1], nEras2);
86 LCat_[iCat][3] = LCat_[iCat][2] * Comb(nActO_[2], iE);
87 this->nStr_ += LCat_[iCat][3];
88 }
89
90 } // RASStringManager::genLCat

92 void RASStringManager::computeList() {
93
94// auto RASStringSt = tick();
95
96 if(this->scheme_ == PRECOMPUTED_CONFIGURATION_DRIVEN_LIST){
97 for (auto pBlk = 0ul; pBlk < 3; pBlk++)
98 for (auto qBlk = 0ul; qBlk < 3; qBlk++) {
99 int iCatL = 0;
100 std::unordered_map<size_t, std::shared_ptr<ExcitationList>> LCache;
101
102 for (auto iE = 0ul; iE <= mxElec_; iE++)
103 for (auto iH = 0ul; iH <= mxHole_; iH++, iCatL++) {
104
105 if (LCat_[iCatL][3] ==0) continue;
106
107 int_matrix nEras = this->init_int_matrix(3, 2, 0);
108 int_matrix nDras = this->init_int_matrix(3, 2, 0);
109 std::shared_ptr<ExcitationList> ras_exl;
110
111 nEras[0][0] = nActO_[0] - iH;
112 nEras[2][0] = iE;
113 nEras[1][0] = nE_ - nEras[0][0] - nEras[2][0];
114
115 nDras[0][0] = LCat_[iCatL][1];
116 nDras[1][0] = LCat_[iCatL][2]/LCat_[iCatL][1];
117 nDras[2][0] = LCat_[iCatL][3]/LCat_[iCatL][2];
118
119 if(pBlk == qBlk) { // diagonal block
120 int nVp = nActO_[pBlk] - nEras[pBlk][0];
121 int nNZ = nEras[pBlk][0] * (nVp + 1);
122
123 if(nNZ == 0) continue;
124 else {
125 auto search = LCache.find(nEras[pBlk][0]);
126 if (search != LCache.end() ) ras_exl = search->second;
127 else {
128 int_matrix ras_addr_array =
129 this->buildAddressingArray(nEras[pBlk][0], nActO_[pBlk]);
130 ras_exl = std::make_shared<ExcitationList>(
131 this->computeIntraCASExcitationList(nActO_[pBlk],
132 nEras[pBlk][0], ras_addr_array));
133 LCache.insert({nEras[pBlk][0], ras_exl});
134 }
135 exList_.insert({mapKey(pBlk, qBlk, iCatL, iCatL), ras_exl});
136#ifdef _DEBUG_RASSTRING
137 std::cout << "p: " << pBlk+1 << " and q: " << qBlk+1 << std::endl;
138 std::cout << "LCat: " << iCatL+1 << " and KCat: " << iCatL + 1 << std::endl;
139#endif
140 }
141
142 } else { // off-diagonal block
143 for (auto i = 0ul; i < 3; i++)
144 nEras[i][1] = nEras[i][0];
145 nEras[pBlk][1] += 1;
146 nEras[qBlk][1] -= 1;
147
148 bool checko = false;
149 for (auto i = 0ul; i < 3; i++) {
150 if(nEras[i][1] < 0 or nEras[i][1] > nActO_[i]) {
151 checko = true;
152 break;
153 }
154 }
155 if(not checko) {
156 int iHK = nActO_[0] - nEras[0][1];
157 int iEK = nEras[2][1];
158 if(iHK > mxHole_ or iEK > mxElec_) continue;
159 int iCatK = iHK + (mxHole_ + 1) * iEK;
160 if(LCat_[iCatK][3] == 0) continue;
161
162 nDras[0][1] = LCat_[iCatK][1];
163 nDras[1][1] = LCat_[iCatK][2]/LCat_[iCatK][1];
164 nDras[2][1] = LCat_[iCatK][3]/LCat_[iCatK][2];
165 size_t nVpL = nActO_[pBlk] - nEras[pBlk][0];
166 size_t nNZL = nEras[qBlk][0] * nVpL;
167 size_t nVqK = nActO_[qBlk] - nEras[qBlk][1];
168 size_t nNZK = nEras[pBlk][1] * nVqK;
169
170 if(nNZL == 0 or nNZK == 0) continue;
171
172#ifdef _DEBUG_RASSTRING
173 std::cout << "p: " << pBlk+1 << " and q: " << qBlk+1 << std::endl;
174 std::cout << "LCat: " << iCatL+1 << " and KCat: " << iCatK + 1 << std::endl;
175#endif
176
177 int Sgnoff = 1;
178 if((pBlk == 2 and qBlk == 0) or (pBlk == 0 and qBlk == 2))
179 if(nEras[1][1] % 2 == 1) Sgnoff = -1;
180 bool pqseq = (pBlk < qBlk) ? true : false;
181
182 // form addressing array
183 int_matrix addr_array_pK = buildAddressingArray(nEras[pBlk][1], nActO_[pBlk]);
184 int_matrix addr_array_qK = buildAddressingArray(nEras[qBlk][1], nActO_[qBlk]);
185
186 ras_exl = std::make_shared<ExcitationList>(
187 this->computeInterCASExcitationList(nActO_[pBlk], nEras[pBlk][0],
188 nActO_[qBlk], nEras[qBlk][0], Sgnoff, pqseq,
189 addr_array_pK, addr_array_qK));
190 exList_.insert({mapKey(pBlk, qBlk, iCatL, iCatK),
191 ras_exl});
192 }
193 }
194 }
195 }

199// << RASStringdur << " s." << std::endl;

201#ifdef _DEBUG_RASSTRING
202 std::cout << "RAS string all finished." << std::endl;
203#endif
204
205 } else CErr("Not Implemented yet");

207 } // RASString::buildConfDrivenExcitationList
