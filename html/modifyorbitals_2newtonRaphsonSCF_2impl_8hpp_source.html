34template<typename MatsT>
35void NewtonRaphsonSCF<MatsT>::getNewOrbitals(EMPerturbation& pert, VecMORef<MatsT>& mo,
36 VecEPtr& eps) {
37
38 // Form the Fock matrix D(k) -> F(k)
39 ProgramTimer::timeOp("Form Fock", [&]() { this->modOrbOpt.formFock(pert); });
40
41 if( not storedRef ) saveRefMOs(mo);
42
43 computeGradient(mo);
44 computeDiagHess(eps);
45
46 // Compute NR step and update orbRot
47 NewtonRaphsonIteration();
48
49 // Rotate reference MO's to new MO's
50 rotateMOs(mo);
51
52 // Compute new Eigenvalues
53 this->computeEigenvalues(mo, eps);
54
55 this->modOrbOpt.formDensity();
56}

62template<typename MatsT>
63void NewtonRaphsonSCF<MatsT>::NewtonRaphsonIteration() {
64 if( this->scfControls.nrAlg == FULL_NR ) {
65
66 fullNRStep();
67
68 } else if( this->scfControls.nrAlg == QUASI_BFGS ) {
69
70 qnBFGSStep();
71
72 } else if( this->scfControls.nrAlg == QUASI_SR1 ) {
73
74 qnSR1Step();
75
76 } else if( this->scfControls.nrAlg == GRAD_DESCENT ) {
77
78 gradDescentStep();
79
80 } else {
81
82 CErr("Requested Newton-Raphson Step not yet implemented");
83
84 }
85}

87template<typename MatsT>
88void NewtonRaphsonSCF<MatsT>::gradDescentStep(){
89
90 MatsT* dx = this->memManager.template malloc<MatsT>(nParam);
91 for( size_t i=0; i<nParam; ++i )
92 dx[i] = orbGrad[i] / orbDiagHess[i];
93 takeStep(dx);
94 this->memManager.free(dx);
95
96}

99/*
100 * Brief: Computes the gradient convergence criteria for optimization
101 * In this case it is the maximum element of the orbital gradient
102 */
103template<typename MatsT>
104double NewtonRaphsonSCF<MatsT>::computeFDCConv() {
105 // Compute the Maximum of the gradient or norm of gradient
106 double maxGrad = 0.;
107 for( size_t i = 0; i < nParam; ++i )
108 if( maxGrad < std::abs(orbGrad[i]) ) maxGrad = std::abs(orbGrad[i]);
109 return maxGrad;
110};// NewtonRaphsonSCF<MatsT> :: computeFDCConv

143/*
144 * Brief: Perform sanity checks
145 */
146template<typename MatsT>
147void NewtonRaphsonSCF<MatsT>::sanityChecks(){
148
149 // Check self-rotations
150 for( auto& rO : rotOpt )
151 for( auto& iO : rO.rotIndices.first )
152 for( auto& iV : rO.rotIndices.second )
153 if( iO == iV ) CErr("Cannot rotate an orbital into itself");
154
155 // Check that there are no repeated rotations
156 std::unordered_map<std::string,bool> checkMap;
157 for( auto& rO : rotOpt )
158 for( auto& iO : rO.rotIndices.first )
159 for( auto& iV : rO.rotIndices.second ){
160 std::string indStr = std::to_string(rO.spaceIndex) + "_" + std::to_string(iO)
161 + "_" + std::to_string(iV);
162 if( checkMap.find(indStr) != checkMap.end() ){
163 CErr("Rotation parameters were included twice");
164 } else {
165 checkMap.insert(std::pair<std::string,bool>(indStr,true));
166 }
167 }
168
169 // Check that the spaceIndex is correct
170 VecShrdPtrMat<MatsT> den = this->modOrbOpt.getOnePDM();
171 size_t nMat = den.size();
172 for( auto& rO : rotOpt )
173 if( rO.spaceIndex >= nMat ) CErr("Space Index in NRRotOptions is larger than the input vectors");
174}

187/*
188 * Brief: Allocates the parameter vectors
189 */
190template<typename MatsT>
191void NewtonRaphsonSCF<MatsT>::alloc() {
192 orbRot = this->memManager.template malloc<MatsT>(nParam);
193 orbGrad = this->memManager.template malloc<MatsT>(nParam);
194 orbDiagHess = this->memManager.template malloc<MatsT>(nParam);
195 std::fill_n(orbRot, nParam, MatsT(0.));
196 std::fill_n(orbGrad, nParam, MatsT(0.));
197 std::fill_n(orbDiagHess, nParam, MatsT(0.));
198 if( this->scfControls.nrAlg == QUASI_BFGS or this->scfControls.nrAlg == QUASI_SR1 ) {
199 for( size_t i = 0; i < this->scfControls.nKeep; i++ ) {
200 qnOrbRot.emplace_back(this->memManager.template malloc<MatsT>(nParam));
201 qnOrbGrad.emplace_back(this->memManager.template malloc<MatsT>(nParam));
202 std::fill_n(qnOrbRot[i], nParam, MatsT(0.));
203 std::fill_n(qnOrbGrad[i], nParam, MatsT(0.));
204 }
205 }
206};
