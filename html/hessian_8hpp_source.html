37 /*
38 * Build orbital hessian diagonals block-wisely:
39 * IN -> inactive, indexed by i, j, k, l
40 * CO -> correlated, indexed by t, u, v, w
41 * FV -> frozen virtual indexed by a, b, c, d
42 * NA -> negative energy orbitals, indexed by m, n
43 */
44 template <typename MatsT, typename IntsT>
45 void OrbitalRotation<MatsT, IntsT>::computeOrbOrbHessianDiag(EMPerturbation & pert,
46 SquareMatrix<MatsT> & oneRDM, InCore4indexTPI<MatsT> & twoRDM, MatsT * H) {
47
48 bool oneETermApproximation = settings.alg == ORB_ROT_APPROX_QUASI_2ND_ORDER;
49 MatsT notRotatedHessian = MatsT (1 / settings.hessianDiagScale);
50
51 auto & mopart = mcwfn_.MOPartition;
52 auto & mem = mcwfn_.memManager;
53
54 size_t nTOrb = mopart.nMO;
55 size_t nCorrO = mopart.nCorrO;
56 size_t nInact = mopart.nInact;
57 size_t nFVirt = mopart.nFVirt;
58 size_t nNegMO = mopart.nNegMO;
59
60 size_t nINCO = nInact + nCorrO;
61 size_t SCRDim = std::max(nCorrO, std::max(nInact, nFVirt));
62 size_t SCRDim2 = SCRDim * SCRDim;
63 double fc = (mcwfn_.reference().nC > 1) ? 1.0: 2.0;
64
65 // Allocate SCR
66 MatsT * SCR = mem.template malloc<MatsT>(SCRDim2);
67 MatsT * F1_ii = nullptr, * F1_aa = nullptr, * F1_nn = nullptr;
68 MatsT * RDM1_tt = mem.template malloc<MatsT>(nCorrO);
69 MatsT * F1_tt = mem.template malloc<MatsT>(nCorrO);
70 MatsT * F2_tt = mem.template malloc<MatsT>(nCorrO);
71
72 // populate One electron terms
73 for(auto t = 0ul; t < nCorrO; t++) RDM1_tt[t] = oneRDM(t,t) / fc;
74 // F1_tt -> SCR
75 this->formGeneralizedFock1(pert, oneRDM, F1_tt, "tt", true);
76
77 // F2_tt -> SCR
78 this->formGeneralizedFock2(pert, oneRDM, twoRDM, SCR, "t");
79 for(auto t = 0ul; t < nCorrO; t++) F2_tt[t] = SCR[t + t * nCorrO];
80
81 if (nInact > 0) {
82 // F1_ii -> SCR
83 F1_ii = mem.template malloc<MatsT>(nInact);
84 this->formGeneralizedFock1(pert, oneRDM, F1_ii, "ii", true);
85 }
86
87 if (nFVirt > 0) {
88 // F1_aa -> SCR
89 F1_aa = mem.template malloc<MatsT>(nFVirt);
90 this->formGeneralizedFock1(pert, oneRDM, F1_aa, "aa", true);
91 }
92
93 if (nNegMO > 0) {
94 // F1_nn -> SCR
95 F1_nn = mem.template malloc<MatsT>(nNegMO);
96 this->formGeneralizedFock1(pert, oneRDM, F1_nn, "nn", true);
97 }
98
99 // intialize H as all one
100 std::fill_n(H, nTOrb*nTOrb, notRotatedHessian);
101
102 // CO-CO block
103 if (settings.rotate_within_correlated) {
104 CErr("rotate MOs within correlated orbitals not implemented");
105 }
106
107 // point to start of the positive energy MO part
108 MatsT * HP = H + nNegMO * (nTOrb + 1);
109
110 // IN-CO block
111 if (settings.rotate_inact_correlated) {
112 // H_{it,it} = F1_tt + 1RDM_tt * F1_ii - F1_ii - F2_tt
113#pragma omp parallel for schedule(static) default(shared)
114 for (auto t = 0ul; t < nCorrO; t++)
115 for (auto i = 0ul; i < nInact; i++)
116 SCR[i + t * nInact] = F1_tt[t] + RDM1_tt[t] * F1_ii[i] - F2_tt[t]- F1_ii[i];
117
118 SetMat('N', nInact, nCorrO, MatsT(1.), SCR, nInact, HP + nTOrb * nInact, nTOrb);
119 SetMat('C', nInact, nCorrO, MatsT(1.), SCR, nInact, HP + nInact, nTOrb);
120 }
121
122
123 // IN-FV block
124 if (settings.rotate_inact_virtual) {
125 // H_{ia,ia} = F1_aa - F1_ii
126#pragma omp parallel for schedule(static) default(shared)
127 for (auto a = 0ul; a < nFVirt; a++)
128 for (auto i = 0ul; i < nInact; i++)
129 SCR[i + a * nInact] = F1_aa[a] - F1_ii[i];
130
131 SetMat('N', nInact, nFVirt, MatsT(1.), SCR, nInact, HP + nTOrb * nINCO, nTOrb);
132 SetMat('C', nInact, nFVirt, MatsT(1.), SCR, nInact, HP + nINCO, nTOrb);
133 }
134
135
136 // CO-FV block
137 if (settings.rotate_correlated_virtual) {
138 // H_{ta,ta} = 1RDM_tt * F1_aa - F2_tt
139#pragma omp parallel for schedule(static) default(shared)
140 for (auto a = 0ul; a < nFVirt; a++)
141 for (auto t = 0ul; t < nCorrO; t++)
142 SCR[t + a * nCorrO] = RDM1_tt[t] * F1_aa[a] - F2_tt[t];
143
144 SetMat('N', nCorrO, nFVirt, MatsT(1.), SCR, nCorrO, HP + nTOrb * nINCO + nInact, nTOrb);
145 SetMat('C', nCorrO, nFVirt, MatsT(1.), SCR, nCorrO, HP + nTOrb * nInact + nINCO, nTOrb);
146 }
147
148 if(SCR) mem.free(SCR);
149
150 // NA-PO block
151 if (settings.rotate_negative_positive) {
152
153 SCR = mem.template malloc<MatsT>(nNegMO * std::max(nInact, nCorrO));
154
155 // IN-NA block
156 if (nInact > 0) {
157#pragma omp parallel for schedule(static) default(shared)
158 for (auto n = 0ul; n < nNegMO; n++)
159 for (auto i = 0ul; i < nInact; i++)
160 SCR[i + n * nInact] = F1_nn[n] - F1_ii[i];
161
162 SetMat('N', nInact, nNegMO, MatsT(1.), SCR, nInact, H + nNegMO, nTOrb);
163 SetMat('C', nInact, nNegMO, MatsT(1.), SCR, nInact, H + nNegMO * nTOrb, nTOrb);
164 }
165
166#pragma omp parallel for schedule(static) default(shared)
167 for (auto n = 0ul; n < nNegMO; n++)
168 for (auto t = 0ul; t < nCorrO; t++)
169 SCR[t + n * nCorrO] = RDM1_tt[t] * F1_nn[n] - F2_tt[t];
170
171 SetMat('N', nCorrO, nNegMO, MatsT(1.), SCR, nCorrO, H + nNegMO + nInact, nTOrb);
172 SetMat('C', nCorrO, nNegMO, MatsT(1.), SCR, nCorrO, H + (nNegMO + nInact) * nTOrb, nTOrb);
173
174 mem.free(SCR);
175 }
176
177 // Free SCRs
178 if(F1_ii) mem.free(F1_ii);
179 if(F2_tt) mem.free(F2_tt);
180 if(F1_tt) mem.free(F1_tt);
181 if(F1_aa) mem.free(F1_aa);
182 if(F1_nn) mem.free(F1_nn);
183 if(RDM1_tt) mem.free(RDM1_tt);
184
185 // Scale Hessian
186 blas::scal(nTOrb * nTOrb, MatsT(settings.hessianDiagScale), H, 1);
187
188/*
189 * TODO: Test Damping
190 */
191 // Dampping and taking care of negative or small eigenvalues
192#pragma omp parallel for schedule(static) default(shared)
193 for (auto pq = 0ul; pq < nTOrb * nTOrb; pq++) {
194// if (std::abs(OOHessianD[i]) > this->hessianDiagDampingThreshold)
195// OOHessianD[i] /= this->hessianDiagDamping;
196// else
197 if (std::abs(H[pq]) < settings.hessianDiagMinTol) {
198 if ( std::real(H[pq]) > 0.) H[pq] = MatsT(1.0e6);
199 else H[pq] = -MatsT(1.0e6);
200 }
201 }
202//*/
203
204#ifdef DEBUG_ORBITALROTATION_HESSIAN
205 prettyPrintSmart(std::cout, "OR Orbital-Orbital Hessian Approximated Diagonal",
206 H, nTOrb, nTOrb, nTOrb);
207 double HDNorm = lapack::lange(lapack::Norm::Fro, nTOrb, nTOrb, H, nTOrb);
208 std::cout << "Hessian Approximated Diagonal Norm = " << HDNorm << std::endl;
209#endif
210
211 }; // OrbitalRotation<MatsT>::computeOrbOrbHessianDiag
