68 /*
69 * \brief Compute oscillator strength for MC wavefunction
70 * using AO dipole and MO coefficients and MO TDM
71 * Only for 1C and 2C
72 * s1: initial state
73 * s2: final state
74 */
75 template <typename MatsT, typename IntsT>
76 double MCWaveFunction<MatsT,IntsT>::oscillator_strength(size_t s2, size_t s1) {
77
78 if (referenceWaveFunction().nC == 4) CErr("4C has no dipole.");
79
80 auto &mem = memManager;
81 size_t nAO = reference().nAlphaOrbital() * reference().nC;;
82 size_t nCorrO = MOPartition.nCorrO;
83 size_t nInact = MOPartition.nInact;
84
85 // compute transition density matrix for specific state
86 SquareMatrix<MatsT> tmpTDM1(mem,nCorrO);
87 SquareMatrix<MatsT> tmpTDM2(mem,nCorrO);
88 ciBuilder->computeTDM(*this, CIVecs[s1], CIVecs[s2], tmpTDM1);
89 ciBuilder->computeTDM(*this, CIVecs[s2], CIVecs[s1], tmpTDM2);
90
91 MatsT D = MatsT(0.);
92
93 // dipole AO -> MO transformation
94 auto MOdipole = moints.getIntegral<VectorInts,MatsT>("MOdipole");
95
96 if (not MOdipole) {
97 std::shared_ptr<VectorInts<IntsT>> AOdipole =
98 std::make_shared<VectorInts<IntsT>>(mem, nAO, 1, true);
99 std::shared_ptr<VectorInts<MatsT>> MOdipole_scr =
100 std::make_shared<VectorInts<MatsT>>(mem, nCorrO, 1, true);
101
102 std::vector<std::pair<size_t, size_t>> active(2, {MOPartition.nFCore+nInact, nCorrO});
103
104 for(auto iXYZ = 0; iXYZ < 3; iXYZ++) {
105 if (referenceWaveFunction().nC == 1)
106 (*AOdipole)[iXYZ] = (*reference().aoints.lenElectric)[iXYZ];
107 else if (referenceWaveFunction().nC == 2)
108 (*AOdipole)[iXYZ] = (*reference().aoints.lenElectric)[iXYZ]
109 .template spatialToSpinBlock<IntsT>();
110 (*AOdipole)[iXYZ].subsetTransform('N',reference().mo[0].pointer(),
111 nAO, active, (*MOdipole_scr)[iXYZ].pointer(), false);
112 }
113
114 moints.addIntegral("MOdipole", MOdipole_scr);
115 }
116
117 MOdipole = moints.getIntegral<VectorInts,MatsT>("MOdipole");
118
119 // dipole strength D = Tr(TDM \dot MOdiple) Tr(TDM^* \dot MOdipole)
120 for(auto iXYZ = 0; iXYZ < 3; iXYZ++) {
121 D += blas::dot(nCorrO*nCorrO,tmpTDM1.pointer(),1,(*MOdipole)[iXYZ].pointer(),1)
122 *blas::dot(nCorrO*nCorrO,tmpTDM2.pointer(),1,(*MOdipole)[iXYZ].pointer(),1);
123 }
124
125 // oscillator strength f = 2/3 (E2 - E1) D.
126 double f = (2./3.) * (StateEnergy[s2] - StateEnergy[s1]) * std::real(D);
127
128 // output
129 std::cout << "Excited State: " << std::setw(3) << std::right << s2+1
130 << " to state: " << std::setw(3) << std::right << s1+1 << ":";
131 std::cout << std::setw(15) << std::right << "E(Eh) = "
132 << std::setprecision(8) << std::fixed << (StateEnergy[s2] - StateEnergy[s1]);
133 std::cout << std::setw(15) << std::right << "f = "
134 << std::setprecision(6) << std::fixed << f << std::endl;
135
136 tmpTDM1.clear();
137 tmpTDM2.clear();
138
139 return f;
140
141 } // MCWaveFunction::oscillator_strength
