31template<typename MatsT>
32void NewtonRaphsonSCF<MatsT>::saveRefMOs(VecMORef<MatsT>& mo) {
33
34 // Store Reference MO's from which to compute rotations
35 size_t nMO = mo.size();
36
37 // Allocate Matrices
38 if( not refMOsAllocated ) {
39 refMO.reserve(nMO);
40 for( size_t i = 0; i < nMO; i++ ) {
41 size_t NBC = mo[i].get().dimension();
42 refMO.emplace_back(this->memManager, NBC);
43 }
44 }
45
46 // Copy MO's
47 for( size_t i = 0; i < nMO; i++ ) {
48 size_t NBC = mo[i].get().dimension();
49 refMO[i] = mo[i].get();
50 }
51
52 storedRef = true;
53}

80template<typename MatsT>
81std::vector<SquareMatrix<MatsT>> NewtonRaphsonSCF<MatsT>::computeUnitary(){
82
83 VecShrdPtrMat<MatsT> den = this->modOrbOpt.getOnePDM();
84 size_t nMat = den.size();
85
86 // Initialize Anti-hermitian matrix
87 std::vector<SquareMatrix<MatsT>> A;
88 A.reserve(nMat);
89 for( size_t i=0; i<nMat; ++i )
90 A.emplace_back(this->memManager,den[i]->dimension());
91 for( auto& a : A ) a.clear();
92
93 // Parse parameters to make antisymmetric matrices
94 size_t disp = 0;
95 for( auto& rO : rotOpt ){
96 for( auto& iO : rO.rotIndices.first ){
97 for( auto& iV : rO.rotIndices.second ){
98 A[rO.spaceIndex](iO, iV) += -orbRot[disp];
99 A[rO.spaceIndex](iV, iO) += SmartConj(orbRot[disp]);
100 ++disp;
101 }
102 }
103 }
104
105 // Compute the Unitary matrices
106 std::vector<SquareMatrix<MatsT>> U;
107 U.reserve(nMat);
108 for( size_t i=0; i<nMat; ++i )
109 U.emplace_back(this->memManager,den[i]->dimension());
110 for( auto& u : U ) u.clear();
111
112 // Compute either exp(A) or Cayley transform
113 for( size_t i=0; i<nMat; ++i ){
114 size_t N = U[i].dimension();
115 // Try to compute exp(A) or use Cayley transform if failed
116 try{
117 MatExp(N,A[i].pointer(),N,U[i].pointer(),N,this->memManager);
118 }
119 catch(...){
120 CErr("Matrix Exponential failed to converge");
121 }
122#ifdef _NRSCF_DEBUG_UNITARY
123 prettyPrintSmart(std::cout, "Unitary Matrix", U[i].pointer(),N,N,N);
124 SquareMatrix<MatsT> SCR(this->memManager,N);
125 blas::gemm(blas::Layout::ColMajor, blas::Op::ConjTrans, blas::Op::NoTrans,
126 N, N, N,
127 MatsT(1.), U[i].pointer(), N,
128 U[i].pointer(), N,
129 MatsT(0.), SCR.pointer(), N);
130 for( size_t p = 0; p < N; p++ )
131 SCR(p, p) -= MatsT(1.);
132 double sum = 0.;
133 for( size_t p = 0; p < N * N; p++ )
134 sum += std::abs(SCR.pointer()[p]);
135 std::cerr << "Unitary Test: U^H * U = 1 -> Error = " << sum << std::endl;
136#endif
137 }
138 return U;
139}

142/*
143 * Brief: Computes the orbital gradients and the diagonal Hessian and stores
144 * them in their vectors. The total gradient is the concatonation of
145 * orbital gradients for each Fock matrix. e.g. UHF->[Grad_alpha,Grad_beta]
146 */
147template<typename MatsT>
148void NewtonRaphsonSCF<MatsT>::computeGradient(VecMORef<MatsT>& mo) {
149
150 // Compute the Gradient of orbital rotations
151 if( this->modOrbOpt.computeNROrbGrad ) {
152 // Compute user-defined gradient
153 this->modOrbOpt.computeNROrbGrad(orbGrad);
154 } else {
155 // Compute default gradient from fock and mo's
156
157
158 // Compute MO Fock
159 size_t nMat = mo.size();
160 VecShrdPtrMat<MatsT> fock = this->modOrbOpt.getFock();
161 std::vector<SquareMatrix<MatsT>> moFock;
162 moFock.reserve(nMat);
163 for( size_t i=0; i<nMat; ++i ) {
164 size_t NB = mo[i].get().dimension();
165 moFock.emplace_back( fock[i]->transform('N',mo[i].get().pointer(),NB,NB) );
166 }
167
168 // Compute Gradient for parameters
169 size_t disp = 0;
170 for( auto& rO : rotOpt ){
171 for( auto& iO : rO.rotIndices.first ){
172 for( auto& iV : rO.rotIndices.second ){
173 orbGrad[disp] = moFock[rO.spaceIndex](iO,iV);
174 ++disp;
175 }
176 }
177 }
178
179 // Conjugate the gradient for complex wave functions.
180 // this is necessary to make quasi-newton and gradient descent
181 // consistent with the full NR implementation
182 /*
183 if( std::is_same<MatsT, dcomplex>::value ){
184 for(size_t iP=0; iP<nParam; ++iP)
185 orbGrad[iP] = SmartConj(orbGrad[iP]);
186 }
187 */
188 } // if( computeNROrbGrad ) else
189#ifdef _NRSCF_DEBUG_GRAD
190 printParamVec("Gradient",orbGrad);
191#endif
192}; // NewtonRaphsonSCF<MatsT> :: computeGradient

194template<typename MatsT>
195void NewtonRaphsonSCF<MatsT>::computeDiagHess(VecEPtr& eps){
196
197 size_t disp = 0;
198 for( auto& rO : rotOpt ){
199 for( auto& iO : rO.rotIndices.first ){
200 double eO = eps[rO.spaceIndex][iO] + this->scfControls.nrLevelShift;
201 for( auto& iV : rO.rotIndices.second ){
202 orbDiagHess[disp] = eps[rO.spaceIndex][iV] - eO;
203 ++disp;
204 }
205 }
206 }
207#ifdef _NRSCF_DEBUG_GRAD
208 printParamVec("Diagonal Orbital Hessian",orbDiagHess);
209#endif
210}

212/*
213 * Brief: This computes the new set of parameters from
214 * a given search direction (dx)
215 */
216template<typename MatsT>
217void NewtonRaphsonSCF<MatsT>::takeStep(MatsT* dx) {
218
219 // Compute norm of step
220 double norm = blas::nrm2(nParam, dx, 1);
221#ifdef _NRSCF_DEBUG
222 std::cout << "Search Direction Norm = " << norm << std::endl;
223 printParamVec("Search Direction", dx);
224#endif
225
226 double scale = norm > this->scfControls.nrTrust ? this->scfControls.nrTrust / norm : 1.;
227
228 // Perform Damping
229 if( this->doingDamp )
230 scale *= double(1. - this->scfControls.dampParam);
231
232 // take Step
233 blas::axpy(nParam, MatsT(-scale), dx, 1, orbRot, 1);
234};
