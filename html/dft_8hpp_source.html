34 class DFTFunctional {
35 // (polarized) rho[2*np], vrho[2*np], sigma[3*np], vsigma[3*np], v2rho2[3*np], v2rhosigma[6*np], v2sigma2[6*np]
36 // in rho [0] = UP rho
37 // in rho [1] = DOWN rho
38 // in sigma[0] = UP - UP Del N dot Del N
39 // in sigma[1] = UP - DOWN Del N dot Del N
40 // in sigma[2] = DOWN - DOWN Del N dot Del N
41 // out eps = the energy per unit particle
42 // out vxc/vrho[0] = UP first part der of the energy per unit volume in terms of the dens
43 // out vxc/vrho[1] = DOWN first part der of the energy per unit volume in terms of the dens N
44 // out vsigma[0] = UP UP first part der of the energy per unit volume in terms of sigma
45 // out vsigma[1] = UP DOWN first part der of the energy per unit volume in terms of sigma
46 // out vsigma[2] = DOWN DOWN first part der of the energy per unit volume in terms of sigma
47 // out fxc/v2rho2[0] = UP UP second part der of the energy per unit volume in terms of the dens
48 // out fxc/v2rho2[1] = UP UP second part der of the energy per unit volume in terms of the dens
49 // out fxc/v2rho2[2] = UP UP second part der of the energy per unit volume in terms of the dens
50 // out v2rhosigma[0] = UP - UP UP second part der of the energy per unit volume in terms of the dens and sigma
51 // out v2rhosigma[1] = UP - UP DOWN second part der of the energy per unit volume in terms of the dens and sigma
52 // out v2rhosigma[2] = UP - DOWN DOWN second part der of the energy per unit volume in terms of the dens and sigma
53 // out v2rhosigma[3] = DOWN - UP UP second part der of the energy per unit volume in terms of the dens and sigma
54 // out v2rhosigma[4] = DOWN - UP DOWN second part der of the energy per unit volume in terms of the dens and sigma
55 // out v2rhosigma[5] = DOWN - DOWN DOWN second part der of the energy per unit volume in terms of the dens and sigma
56 // out v2sigma2[0] = UP UP - UP UP second part der of the energy per unit volume in terms of sigma sigma
57 // out v2sigma2[1] = UP UP - UP DOWN second part der of the energy per unit volume in terms of sigma sigma
58 // out v2sigma2[2] = UP UP - DOWN DOWN second part der of the energy per unit volume in terms of sigma sigma
59 // out v2sigma2[3] = UP DONW - UP DONW second part der of the energy per unit volume in terms of sigma sigma
60 // out v2sigma2[4] = UP DONW - DOWN DOWN second part der of the energy per unit volume in terms of sigma sigma
61 // out v2sigma2[5] = DOWN DOWN - DOWN DOWN second part der of the energy per unit volume in terms of sigma sigma
