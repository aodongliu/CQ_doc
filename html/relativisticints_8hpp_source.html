169 virtual void broadcast(MPI_Comm comm = MPI_COMM_WORLD, int root = 0) override {
170 OnePInts<IntsT>::broadcast(comm, root);
171
172#ifdef CQ_ENABLE_MPI
173 if( MPISize(comm) > 1 ) {
174 size_t nRel = components_.size();
175 MPIBCast(nRel,root,comm);
176
177 if (components_.size() != nRel) {
178 components_.clear();
179 components_.reserve(nRel);
180 for (size_t i = 0; i < nRel; i++) {
181 components_.emplace_back(this->memManager_, this->NB);
182 }
183 }
184
185 for (OnePInts<IntsT>& comp : components_)
186 comp.broadcast(comm, root);
187 }
188#endif
189 }

191 template <typename TransT>
192 OnePRelInts<typename std::conditional<
193 (std::is_same<IntsT, dcomplex>::value or
194 std::is_same<TransT, dcomplex>::value),
195 dcomplex, double>::type> transform(
196 char TRANS, const TransT* T, int NT, int LDT) const {
197 OnePRelInts<typename std::conditional<
198 (std::is_same<IntsT, dcomplex>::value or
199 std::is_same<TransT, dcomplex>::value),
200 dcomplex, double>::type> transInts(
201 this->ParticleIntegrals::memManager(), NT, hasSpinOrbit());
202 transInts[REL_INTS_COMPS::O] =
203 (*this)[REL_INTS_COMPS::O].transform(TRANS, T, NT, LDT);
204 transInts.scalar() = scalar().transform(TRANS, T, NT, LDT);
205 if (hasSpinOrbit()) {
206 transInts.SOX() = SOX().transform(TRANS, T, NT, LDT);
207 transInts.SOY() = SOY().transform(TRANS, T, NT, LDT);
208 transInts.SOZ() = SOZ().transform(TRANS, T, NT, LDT);
209 }
210 return transInts;
211 }
