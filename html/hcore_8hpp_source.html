33 template <typename MatsT, typename IntsT>
34 std::shared_ptr<SquareMatrix<MatsT>> MOIntsTransformer<MatsT,IntsT>::formInactDen(
35 const char coreIndex) {
36
37 size_t nAO = ss_.nAlphaOrbital() * ss_.nC;
38 auto Den = std::make_shared<SquareMatrix<MatsT>>(memManager_, nAO);
39
40 auto off_size = parseMOType(coreIndex);
41 size_t ioff = off_size.first;
42 size_t ni = off_size.second;
43
44 if (ni != 0) {
45 blas::gemm(blas::Layout::ColMajor, blas::Op::NoTrans, blas::Op::ConjTrans,
46 nAO, nAO, ni, MatsT(1.), ss_.mo[0].pointer() + ioff*nAO, nAO,
47 ss_.mo[0].pointer() + ioff*nAO, nAO, MatsT(0.), Den->pointer(), nAO);
48 } else {
49 Den->clear();
50 }
51
52 return Den;
53
54 }; // form inactive core density

103 template <typename MatsT, typename IntsT>
104 std::shared_ptr<OnePInts<MatsT>> MOIntsTransformer<MatsT,IntsT>::formAOHCore(
105 EMPerturbation & pert, bool cacheAOHCore, const char coreIndex) {
106
107 bool withInactiveCore = coreIndex != '\0';
108
109 std::string cacheAOHCoreStr = "AOHCore";
110 if (withInactiveCore) {
111 cacheAOHCoreStr += "-WithInactive-";
112 cacheAOHCoreStr += coreIndex;
113 }
114
115 auto AOHCore = ints_cache_.template getIntegral<OnePInts, MatsT>(cacheAOHCoreStr);
116
117#ifdef _DEBUG_MOINTSTRANSFORMER_CACHE
118 std::cout << "AOCache = " << std::setw(20) << cacheAOHCoreStr;
119 if (AOHCore) std::cout << "----Find cache!!!" << std::endl;
120 else {
121 std::cout << "----Not find cache, do transformation" << std::endl;
122#else
123 if (not AOHCore) {
124#endif
125
126 AOHCore = ints_cache_.template getIntegral<OnePInts, MatsT>("AOHCore");
127
128 if (not AOHCore) {
129 if(ss_.nC == 1) {
130 AOHCore = std::make_shared<OnePInts<MatsT>>(0.5*ss_.coreH->S());
131 } else {
132 AOHCore = std::make_shared<OnePInts<MatsT>>(
133 ss_.coreH->template spinGather<MatsT>());
134 }
135 if (cacheAOHCore) ints_cache_.addIntegral("AOHCore", AOHCore);
136 }
137
138 if (withInactiveCore) {
139
140 auto AOH1e = ints_cache_.template getIntegral<OnePInts, MatsT>("AOHCore");
141
142 auto Den = formInactDen(coreIndex);
143 std::string cacheAOGDStr = "WithInactive-";
144 cacheAOGDStr += coreIndex;
145 auto AOGD = formAOGD(pert, *Den, true, true, cacheAOGDStr);
146 size_t nAO = ss_.nAlphaOrbital() * ss_.nC;
147
148 AOHCore = std::make_shared<OnePInts<MatsT>>(memManager_, nAO);
149 AOHCore->matrix() = AOGD->matrix() + AOH1e->matrix();
150
151 if (cacheAOHCore) ints_cache_.addIntegral(cacheAOHCoreStr, AOHCore);
152 }
153 }
154
155 return AOHCore;
156 }; // MOIntsTransformer::getAOHCoreInCore

161 template <typename MatsT, typename IntsT>
162 void MOIntsTransformer<MatsT,IntsT>::subsetTransformOPI(
163 const std::vector<std::pair<size_t,size_t>> &off_sizes,
164 const OnePInts<MatsT> & AOOPI, MatsT* MOOPI, bool deltaPQ) {
165
166 size_t nAO = ss_.nAlphaOrbital() * ss_.nC;
167 if (not deltaPQ) {
168 AOOPI.subsetTransform('N', ss_.mo[0].pointer(), nAO, off_sizes, MOOPI, false);
169 } else {
170
171 size_t poff = off_sizes[0].first;
172 size_t np = off_sizes[0].second;
173 MatsT * SCR = memManager_.malloc<MatsT>(nAO);
174
175 for (auto p = 0ul; p < np; p++) {
176
177 auto pMO = ss_.mo[0].pointer() + (p + poff) * nAO;
178
179 // SCR(nu) = MO(mu, p)^H AOHCOre(mu, nu)
180 blas::gemm(blas::Layout::ColMajor, blas::Op::ConjTrans, blas::Op::NoTrans,
181 1, nAO, nAO, MatsT(1.), pMO, nAO, AOOPI.pointer(), nAO,
182 MatsT(0.), SCR, 1);
183
184 // MOTPI(p,p) = SCR(nu) MO(nu, p)
185 MOOPI[p] = blas::dotu(nAO, SCR, 1, pMO, 1);
186 }
187 memManager_.free(SCR);
188 }
189
190 }; // MOIntsTransformer::subsetTransformOPI
